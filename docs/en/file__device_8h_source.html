155 explicit basic_file_device(const std::string& file_name, file_open_flag flags = file_open_flag::none)
156 {
157 const char* mode = fopen_mode(flags);
158 m_file.reset(std::fopen(file_name.c_str(), mode));
159 if (!m_file)
160 throw device_error("cannot open file \"" + file_name + "\" (errno=" + std::to_string(errno) + ")");
161
162 if (fseek_64(m_file.get(), 0, SEEK_END) != 0)
163 throw device_error("cannot get file length for \"" + file_name + "\"");
164
165 int64_t len = ftell_64(m_file.get());
166 if (len < 0)
167 throw device_error("cannot get file length for \"" + file_name + "\"");
168 else if (!std::in_range<size_t>(len))
169 throw device_error("file too large for this platform");
170
171 if (fseek_64(m_file.get(), 0, SEEK_SET) != 0)
172 throw device_error("cannot reset the file for \"" + file_name + "\"");
173
174 m_file_len = static_cast<size_t>(len);
175 }

438 void dput(const CharType* ch, size_t n)
439 requires (IsOut)
440 {
441 if (n == 0) return;
442 if (ch == nullptr && n > 0)
443 throw device_error("file_device::dput fail: null buffer");
444 if (!is_open())
445 throw device_error("file_device::dput fail: file closed.");
446 size_t written = std::fwrite(ch, sizeof(CharType), n, m_file.get());
447
448 auto res = ftell_64(m_file.get());
449 if (res >= 0)
450 {
451 if (!std::in_range<size_t>(res))
452 throw device_error("file_device::dput fail: file size overflow");
453 if (static_cast<size_t>(res) > m_file_len)
454 m_file_len = static_cast<size_t>(res);
455 }
456
457 if (written != n)
458 throw device_error("file_device::dput fail: partial success.");
459 }

485 static const char* fopen_mode(file_open_flag flags)
486 {
487 using enum file_open_flag;
488
489 file_open_flag checkres = flags & (trunc | binary | noreplace);
490
491 if constexpr (IsIn && IsOut)
492 {
493 if (checkres == none)
494 return "r+";
495 else if (checkres == trunc)
496 return "w+";
497 else if (checkres == binary)
498 return "r+b";
499 else if (checkres == (trunc | binary))
500 return "w+b";
501 // noreplace (O_EXCL) implies creating a new file, so it's always empty.
502 // Therefore, trunc is implicit in this mode.
503 else if (checkres == noreplace || checkres == (trunc | noreplace))
504 return "w+x";
505 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
506 return "w+bx";
507 else
508 throw device_error("Invalid open mode combination for read-write file");
509 }
510 else if constexpr (IsIn)
511 {
512 if (checkres == none)
513 return "r";
514 else if (checkres == binary)
515 return "rb";
516 else
517 throw device_error("Invalid open mode combination for read-only file");
518 }
519 else if constexpr (IsOut)
520 {
521 // In write-only mode, "w" already truncates by default if noreplace is not present.
522 if (checkres == none || checkres == trunc)
523 return "w";
524 else if (checkres == binary || checkres == (trunc | binary))
525 return "wb";
526 // noreplace implies creation of a new file.
527 else if (checkres == noreplace || checkres == (trunc | noreplace))
528 return "wx";
529 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
530 return "wbx";
531 else
532 throw device_error("Invalid open mode combination for write-only file");
533 }
534 else
535 {
536 static_assert(dependent_false_nttp_v<IsIn, IsOut>, "file device cannot open with neither in nor out mode");
537 }
538 }

562 // Cached file length, in CharType units.
563 //
564 // Type is `size_t` (not `uint64_t`) because the public API — `dsize()`,
565 // `dseek(size_t)`, `drseek(size_t)` — is `size_t`-typed; using `size_t`
566 // here keeps the value lossless across the boundary on every platform
567 // (including 32-bit, where a `uint64_t` field could hold values that
568 // `dsize()` cannot return without truncation).
569 //
570 // Invariant: `m_file_len <= INT64_MAX`. All writes to this field source
571 // from a non-negative `int64_t` (the result of `ftell_64()`), narrowed
572 // through `std::in_range<size_t>` in the constructor and `dput()`.
573 // This invariant is what makes `-static_cast<int64_t>(offset)` in
574 // `drseek()` safe: any `offset <= m_file_len` fits in `int64_t`'s
575 // positive range, so the negation cannot overflow. Do not introduce
576 // setters or alternate sources that bypass the `int64_t`-derived path.
577 size_t m_file_len = 0;
