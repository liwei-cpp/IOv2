162 explicit basic_file_device(const std::string& file_name, file_open_flag flags = file_open_flag::none)
163 {
164 const char* mode = fopen_mode(flags);
165 m_file.reset(std::fopen(file_name.c_str(), mode));
166 if (!m_file)
167 throw device_error("cannot open file \"" + file_name + "\" (errno=" + std::to_string(errno) + ")");
168
169 if (fseek_64(m_file.get(), 0, SEEK_END) != 0)
170 throw device_error("cannot get file length for \"" + file_name + "\"");
171
172 int64_t len = ftell_64(m_file.get());
173 if (len < 0)
174 throw device_error("cannot get file length for \"" + file_name + "\"");
175 else if (!std::in_range<size_t>(len))
176 throw device_error("file too large for this platform");
177
178 if (fseek_64(m_file.get(), 0, SEEK_SET) != 0)
179 throw device_error("cannot reset the file for \"" + file_name + "\"");
180
181 m_file_len = static_cast<size_t>(len);
182 }

389 void dseek(size_t v)
390 {
391 if (!is_open())
392 throw device_error("file_device::dseek fail: file is closed");
393
394 // In input-enabled modes, seeking is restricted to the current file length.
395 // This ensures that any subsequent read operation (dget) starts from a valid
396 // position within the existing data bounds.
397 if constexpr (IsIn)
398 {
399 if (v > m_file_len)
400 throw device_error("file_device::dseek fail: invalid parameter");
401 }
402
403 if (v > std::numeric_limits<int64_t>::max())
404 throw device_error("file_device::dseek fail: position exceeds maximum seekable range");
405
406 if (fseek_64(m_file.get(), static_cast<int64_t>(v), SEEK_SET) != 0)
407 throw device_error("file_device::dseek fail: fseek invalid response");
408 }

452 void dput(const CharType* ch, size_t n)
453 requires (IsOut)
454 {
455 if (n == 0) return;
456 if (ch == nullptr)
457 throw device_error("file_device::dput fail: null buffer");
458 if (!is_open())
459 throw device_error("file_device::dput fail: file closed.");
460 size_t written = std::fwrite(ch, sizeof(CharType), n, m_file.get());
461
462 auto res = ftell_64(m_file.get());
463 if (res >= 0)
464 {
465 if (!std::in_range<size_t>(res))
466 throw device_error("file_device::dput fail: file size overflow");
467 if (static_cast<size_t>(res) > m_file_len)
468 m_file_len = static_cast<size_t>(res);
469 }
470 else
471 throw device_error("file_device::dput fail: cannot determine file position");
472
473 if (written != n)
474 throw device_error("file_device::dput fail: partial success.");
475 }

501 static const char* fopen_mode(file_open_flag flags)
502 {
503 using enum file_open_flag;
504
505 file_open_flag checkres = flags & (trunc | binary | noreplace);
506
507 if constexpr (IsIn && IsOut)
508 {
509 if (checkres == none)
510 return "r+";
511 else if (checkres == trunc)
512 return "w+";
513 else if (checkres == binary)
514 return "r+b";
515 else if (checkres == (trunc | binary))
516 return "w+b";
517 // noreplace (O_EXCL) implies creating a new file, so it's always empty.
518 // Therefore, trunc is implicit in this mode.
519 else if (checkres == noreplace || checkres == (trunc | noreplace))
520 return "w+x";
521 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
522 return "w+bx";
523 else
524 throw device_error("Invalid open mode combination for read-write file");
525 }
526 else if constexpr (IsIn)
527 {
528 if (checkres == none)
529 return "r";
530 else if (checkres == binary)
531 return "rb";
532 else
533 throw device_error("Invalid open mode combination for read-only file");
534 }
535 else if constexpr (IsOut)
536 {
537 // In write-only mode, "w" already truncates by default if noreplace is not present.
538 if (checkres == none || checkres == trunc)
539 return "w";
540 else if (checkres == binary || checkres == (trunc | binary))
541 return "wb";
542 // noreplace implies creation of a new file.
543 else if (checkres == noreplace || checkres == (trunc | noreplace))
544 return "wx";
545 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
546 return "wbx";
547 else
548 throw device_error("Invalid open mode combination for write-only file");
549 }
550 else
551 {
552 static_assert(dependent_false_nttp_v<IsIn, IsOut>, "file device cannot open with neither in nor out mode");
553 }
554 }

578 // Cached file length, in CharType units.
579 //
580 // Type is `size_t` (not `uint64_t`) because the public API — `dsize()`,
581 // `dseek(size_t)`, `drseek(size_t)` — is `size_t`-typed; using `size_t`
582 // here keeps the value lossless across the boundary on every platform
583 // (including 32-bit, where a `uint64_t` field could hold values that
584 // `dsize()` cannot return without truncation).
585 //
586 // Invariant: `m_file_len <= INT64_MAX`. All writes to this field source
587 // from a non-negative `int64_t` (the result of `ftell_64()`), narrowed
588 // through `std::in_range<size_t>` in the constructor and `dput()`.
589 // This invariant is what makes `-static_cast<int64_t>(offset)` in
590 // `drseek()` safe: any `offset <= m_file_len` fits in `int64_t`'s
591 // positive range, so the negation cannot overflow. Do not introduce
592 // setters or alternate sources that bypass the `int64_t`-derived path.
593 size_t m_file_len = 0;
