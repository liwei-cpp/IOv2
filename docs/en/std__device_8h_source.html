138 size_t dget(char* s, size_t n)
139 requires (ID == STDIN_FILENO)
140 {
141 if (n == 0 || m_eof_hit) return 0;
142 if (s == nullptr)
143 throw device_error("std_device::dget fail: null buffer");
144
145 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
146 ssize_t ret = 0;
147 while (true)
148 {
149 ret = read(ID, s, std::min(n, max_read));
150 if (ret != -1) break;
151
152 if (errno == EINTR) continue;
153 if (errno == EAGAIN || errno == EWOULDBLOCK)
154 {
155 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
156 if (poll(&pfd, 1, -1) == -1)
157 {
158 if (errno == EINTR) continue;
159 throw device_error("std_device::dget fail: poll error");
160 }
161 else if (pfd.revents & (POLLERR | POLLNVAL))
162 throw device_error("std_device::dget fail: poll revents error");
163 else if (pfd.revents & POLLIN) continue;
164 else if (pfd.revents & POLLHUP)
165 {
166 m_eof_hit = true;
167 return 0;
168 }
169 else
170 throw device_error("std_device::dget fail: unexpected poll revents");
171 }
172 else
173 throw device_error("std_device::dget fail: read error");
174 }
175
176 if (ret == 0)
177 m_eof_hit = true;
178
179 return static_cast<size_t>(ret);
180 }

197 void dput(const char* ch, size_t n)
198 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
199 {
200 if (n == 0) return;
201 if (ch == nullptr)
202 throw device_error("std_device::dput fail: null buffer");
203
204 bool put_res = false;
205 if constexpr (ID == STDOUT_FILENO)
206 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
