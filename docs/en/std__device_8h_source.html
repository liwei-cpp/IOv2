152 size_t dget(char* s, size_t n)
153 requires (ID == STDIN_FILENO)
154 {
155 if (n == 0 || m_eof_hit) return 0;
156 if (s == nullptr)
157 throw device_error("std_device::dget fail: null buffer");
158
159 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
160 ssize_t ret = 0;
161 while (true)
162 {
163 ret = read(ID, s, std::min(n, max_read));
164 if (ret != -1) break;
165
166 if (errno == EINTR) continue;
167 if (errno == EAGAIN || errno == EWOULDBLOCK)
168 {
169 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
170 if (poll(&pfd, 1, -1) == -1)
171 {
172 if (errno == EINTR) continue;
173 throw device_error("std_device::dget fail: poll error");
174 }
175 else if (pfd.revents & (POLLERR | POLLNVAL))
176 throw device_error("std_device::dget fail: poll revents error");
177 else if (pfd.revents & POLLIN) continue;
178 else if (pfd.revents & POLLHUP)
179 {
180 m_eof_hit = true;
181 return 0;
182 }
183 else
184 throw device_error("std_device::dget fail: unexpected poll revents");
185 }
186 else
187 throw device_error("std_device::dget fail: read error");
188 }
189
190 if (ret == 0)
191 m_eof_hit = true;
192
193 return static_cast<size_t>(ret);
194 }

211 void dput(const char* ch, size_t n)
212 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
213 {
214 if (n == 0) return;
215 if (ch == nullptr)
216 throw device_error("std_device::dput fail: null buffer");
217
218 bool put_res = false;
219 if constexpr (ID == STDOUT_FILENO)
220 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
221 else
222 put_res = (std::fwrite(ch, sizeof(char), n, stderr) == n);
223
224 if (!put_res)
225 {
226 std::clearerr(ID == STDOUT_FILENO ? stdout : stderr);
227 throw device_error("std_device::dput fail: partial write");
228 }
229 }
