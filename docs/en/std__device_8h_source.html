129 size_t dget(char* s, size_t n)
130 requires (ID == STDIN_FILENO)
131 {
132 if (n == 0 || m_eof_hit) return 0;
133 if (s == nullptr && n > 0)
134 throw device_error("std_device::dget fail: null buffer");
135
136 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
137 ssize_t ret = 0;
138 while (true)
139 {
140 ret = read(ID, s, std::min(n, max_read));
141 if (ret != -1) break;
142
143 if (errno == EINTR) continue;
144 if (errno == EAGAIN || errno == EWOULDBLOCK)
145 {
146 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
147 if (poll(&pfd, 1, -1) == -1)
148 {
149 if (errno == EINTR) continue;
150 throw device_error("std_device::dget fail: poll error");
151 }
152 if (pfd.revents & (POLLERR | POLLNVAL))
153 throw device_error("std_device::dget fail: poll revents error");
154 // POLLHUP is intentionally not treated as EOF: the peer may have
155 // closed while bytes remain buffered. Re-issue read(); it returns 0
156 // only when both the buffer is drained and the peer is gone.
157 continue;
158 }
159 throw device_error("std_device::dget fail: read error");
160 }
161
162 if (ret == 0)
163 m_eof_hit = true;
164
165 return static_cast<size_t>(ret);
166 }

183 void dput(const char* ch, size_t n)
184 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
185 {
186 if (n == 0) return;
187 if (ch == nullptr && n > 0)
188 throw device_error("std_device::dput fail: null buffer");
189
190 bool put_res = false;
191 if constexpr (ID == STDOUT_FILENO)
192 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
193 else
194 put_res = (std::fwrite(ch, sizeof(char), n, stderr) == n);
195
196 if (!put_res)
197 throw device_error("std_device::dput fail: partial success.");
198 }

211 void dflush()
212 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
213 {
214 bool flush_res = false;
215 if constexpr (ID == STDOUT_FILENO)
216 flush_res = (std::fflush(stdout) != EOF);
217 else

219
220 if (!flush_res)
221 throw device_error("std_device::dflush fail: fflush failed.");
222 }
