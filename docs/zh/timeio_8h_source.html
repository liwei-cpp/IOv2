1173 else if (modifier) goto bad_parse_format;
1174 /* We recognize four formats:
1175 1. Two digits specify hours.
1176 2. Four digits specify hours and minutes.
1177 3. Two digits, ':', and two digits specify hours and minutes.
1178 4. 'Z' is equivalent to +0000.
1179 */

1181 // so we just omit the parse result

1197 {

1207 if (*rp == static_cast<CharT>(':') && n == 2) ++rp;
1208 if (rp == rp_end) break;
1209 }

1211 if (n == 2) val *= 100;
1212 else if (n != 4)
1213 {
1214 /* Only two or four digits recognized. */
1215 succ = false; return rp;
1216 }
1217 else if (val % 100 >= 60)
1218 {
1219 succ = false; return rp;
1220 }
1221 }

1234 succ = false;
1235 return rp;
1236 }
1237 else if ((*zone_res)[0] != (CharT)'*')
1238 ctx.m_zone_name = *zone_res;
1239 }
1240 break;

1242 default:

1244 {
1245 if ((rp == rp_end) || (*rp != static_cast<CharT>('%')))
1246 {
1247 succ = false;
1248 return rp;
1249 }
1250
1251 ++rp;
1252 if (modifier)
1253 {
1254 if ((rp == rp_end) || (*rp != modifier))
1255 {
1256 succ = false;
1257 return rp;
1258 }
1259 ++rp;
1260 }
1261

1263 {
1264 succ = false;
1265 return rp;
1266 }
1267 ++rp;
1268 }
1269 }
1270 ++fmt;
1271 }
1272
1273 if ((fmt != _fmt.cend()) && (rp == rp_end))
1274 succ = false;
1275 return rp;

1784 template <size_t n, CharT def = static_cast<CharT>('0'), typename OutIt>
1785 OutIt put_dec(OutIt out, int val) const
1786 {
1787 if (val < 0) val = 0;
1788
1789 if constexpr (n == 0)
1790 {
1791 int buf[128]; // 128 is large enough
1792
1793 buf[0] = 0;
1794 int i = 0;
1795
1796 while(val != 0)
1797 {
1798 buf[i++] = val % 10;
1799 val /= 10;
1800 }
1801 if (i == 0) i = 1;
1802 for (int j = i - 1; j >=0; --j)
1803 {
1804 if (buf[j] == 0) *out++ = def;
1805 else *out++ = static_cast<CharT>(buf[j] + '0');
1806 }
1807 }
1808 else
1809 {
1810 int buf[n];
1811 for (size_t i = 0; i < n; ++i)
1812 {
1813 buf[n - i -1] = val % 10;
1814 val /= 10;
1815 }
1816 bool leading = true;
1817 for (size_t i = 0; i < n; ++i)
1818 {
1819 if (buf[i] != 0) leading = false;
1820 if (leading && (i < n - 1)) *out++ = def;
1821 else *out++ = static_cast<CharT>(buf[i] + '0');
1822 }
1823 }
1824 return out;
1825 }

1827 template <typename TIter, std::sentinel_for<TIter> TSent>
1828 static TIter extract_num(TIter beg, TSent end, int& member, int min_val, int max_val, size_t len, bool& succ)

1833 {

1838 if (value > max_val) break;
