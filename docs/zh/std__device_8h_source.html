160 size_t dget(char* s, size_t n)
161 requires (ID == STDIN_FILENO)
162 {
163 if (n == 0 || m_eof_hit) return 0;
164 if (s == nullptr)
165 throw device_error("std_device::dget fail: null buffer");
166
167 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
168 ssize_t ret = 0;
169 while (true)
170 {
171 ret = read(ID, s, std::min(n, max_read));
172 if (ret != -1) break;
173
174 if (errno == EINTR) continue;
175 if (errno == EAGAIN || errno == EWOULDBLOCK)
176 {
177 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
178 if (poll(&pfd, 1, -1) == -1)
179 {
180 if (errno == EINTR) continue;
181 throw device_error("std_device::dget fail: poll error");
182 }
183 else if (pfd.revents & (POLLERR | POLLNVAL))
184 throw device_error("std_device::dget fail: poll revents error");
185 else if (pfd.revents & POLLIN) continue;
186 else if (pfd.revents & POLLHUP)
187 {
188 m_eof_hit = true;
189 return 0;
190 }
191 else
192 throw device_error("std_device::dget fail: unexpected poll revents");
193 }
194 else
195 throw device_error("std_device::dget fail: read error");
196 }
197
198 if (ret == 0)
199 m_eof_hit = true;
200
201 return static_cast<size_t>(ret);
202 }

219 void dput(const char* ch, size_t n)
220 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
221 {
222 if (n == 0) return;
223 if (ch == nullptr)
224 throw device_error("std_device::dput fail: null buffer");
225
226 bool put_res = false;
227 if constexpr (ID == STDOUT_FILENO)
228 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
229 else
230 put_res = (std::fwrite(ch, sizeof(char), n, stderr) == n);
231
232 if (!put_res)
233 {
234 std::clearerr(ID == STDOUT_FILENO ? stdout : stderr);
235 throw device_error("std_device::dput fail: partial write");
236 }
237 }

250 void dflush()
251 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
252 {
253 bool flush_res = false;
254 if constexpr (ID == STDOUT_FILENO)
255 flush_res = (std::fflush(stdout) != EOF);
256 else
257 flush_res = (std::fflush(stderr) != EOF);
258
259 if (!flush_res)
260 {
261 std::clearerr(ID == STDOUT_FILENO ? stdout : stderr);
262 throw device_error("std_device::dflush fail: fflush error");
263 }
264 }
