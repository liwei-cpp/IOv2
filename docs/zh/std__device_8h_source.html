135 size_t dget(char* s, size_t n)
136 requires (ID == STDIN_FILENO)
137 {
138 if (n == 0 || m_eof_hit) return 0;
139 if (s == nullptr && n > 0)
140 throw device_error("std_device::dget fail: null buffer");
141
142 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
143 ssize_t ret = 0;
144 while (true)
145 {
146 ret = read(ID, s, std::min(n, max_read));
147 if (ret != -1) break;
148
149 if (errno == EINTR) continue;
150 if (errno == EAGAIN || errno == EWOULDBLOCK)
151 {
152 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
153 if (poll(&pfd, 1, -1) == -1)
154 {
155 if (errno == EINTR) continue;
156 throw device_error("std_device::dget fail: poll error");
157 }
158 if (pfd.revents & (POLLERR | POLLNVAL))
159 throw device_error("std_device::dget fail: poll revents error");
160 // POLLHUP is intentionally not treated as EOF: the peer may have
161 // closed while bytes remain buffered. Re-issue read(); it returns 0
162 // only when both the buffer is drained and the peer is gone.
163 continue;
164 }
165 throw device_error("std_device::dget fail: read error");
166 }
167
168 if (ret == 0)
169 m_eof_hit = true;
170
171 return static_cast<size_t>(ret);
172 }

189 void dput(const char* ch, size_t n)
190 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
191 {
192 if (n == 0) return;
193 if (ch == nullptr && n > 0)
194 throw device_error("std_device::dput fail: null buffer");
195
196 bool put_res = false;
197 if constexpr (ID == STDOUT_FILENO)
198 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
199 else
200 put_res = (std::fwrite(ch, sizeof(char), n, stderr) == n);
201
202 if (!put_res)
203 throw device_error("std_device::dput fail: partial success.");
204 }

217 void dflush()
218 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
219 {
220 bool flush_res = false;
221 if constexpr (ID == STDOUT_FILENO)
222 flush_res = (std::fflush(stdout) != EOF);
223 else
224 flush_res = (std::fflush(stderr) != EOF);
225
226 if (!flush_res)
227 throw device_error("std_device::dflush fail: fflush failed.");
228 }
