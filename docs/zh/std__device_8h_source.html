120 size_t dget(char* s, size_t n)
121 requires (ID == STDIN_FILENO)
122 {
123 if (s == nullptr && n > 0)
124 throw device_error("std_device::dget fail: null buffer");
125 if (n == 0 || m_eof_hit) return 0;
126
127 constexpr size_t max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
128 ssize_t ret = 0;
129 while (true)
130 {
131 ret = read(ID, s, std::min(n, max_read));
132 if (ret != -1) break;
133
134 if (errno == EINTR) continue;
135 if (errno == EAGAIN || errno == EWOULDBLOCK)
136 {
137 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
138 if (poll(&pfd, 1, -1) == -1)
139 {
140 if (errno == EINTR) continue;
141 throw device_error("std_device::dget fail: poll error");
142 }
143 if (pfd.revents & (POLLERR | POLLNVAL))
144 throw device_error("std_device::dget fail: poll revents error");
145 if (pfd.revents & POLLHUP)
146 {
147 m_eof_hit = true;
148 return 0;
149 }
150 continue;
151 }
152 throw device_error("std_device::dget fail: read error");
153 }
154
155 if (ret == 0)
156 m_eof_hit = true;
157
158 return static_cast<size_t>(ret);
159 }

176 void dput(const char* ch, size_t n)
177 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
178 {
179 if (ch == nullptr && n > 0)
180 throw device_error("std_device::dput fail: null buffer");
181 if (n == 0) return;
182
183 bool put_res = false;
184 if constexpr (ID == STDOUT_FILENO)
185 put_res = (std::fwrite(ch, sizeof(char), n, stdout) == n);
186 else
187 put_res = (std::fwrite(ch, sizeof(char), n, stderr) == n);
188
189 if (!put_res)
190 throw device_error("std_device::dput fail: partial success.");
191 }

204 void dflush()
205 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
206 {
207 bool flush_res = false;
208 if constexpr (ID == STDOUT_FILENO)
209 flush_res = (std::fflush(stdout) != EOF);
210 else
211 flush_res = (std::fflush(stderr) != EOF);
212
213 if (!flush_res)
214 throw device_error("std_device::dflush fail: fflush failed.");
