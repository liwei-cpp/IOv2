73 void clear(ios_defs::iostate s = ios_defs::goodbit)
74 {
75 m_stream_state = s;
76 if ((m_stream_state & ios_defs::devfailbit) == ios_defs::goodbit) m_exp_dev_fail = std::exception_ptr{};
77 if ((m_stream_state & ios_defs::cvtfailbit) == ios_defs::goodbit) m_exp_cvt_fail = std::exception_ptr{};
78 if ((m_stream_state & ios_defs::strfailbit) == ios_defs::goodbit) m_exp_str_fail = std::exception_ptr{};
79 if ((m_stream_state & ios_defs::otherfailbit) == ios_defs::goodbit) m_exp_other_fail = std::exception_ptr{};
80
81 ios_defs::iostate state_in_exp = m_exception & m_stream_state;
82 bool need_throw_exp = false;
83 if (state_in_exp & ios_defs::devfailbit)
84 {
85 need_throw_exp = true;
86 if (m_exp_dev_fail)
87 std::rethrow_exception(std::exchange(m_exp_dev_fail, nullptr));
88 }
89 else if (state_in_exp & ios_defs::cvtfailbit)
90 {
91 need_throw_exp = true;
92 if (m_exp_cvt_fail)
93 std::rethrow_exception(std::exchange(m_exp_cvt_fail, nullptr));
94 }
95 else if (state_in_exp & ios_defs::strfailbit)
96 {
97 need_throw_exp = true;
98 if (m_exp_str_fail)
99 std::rethrow_exception(std::exchange(m_exp_str_fail, nullptr));
100 }
101 else if (state_in_exp & ios_defs::otherfailbit)
102 {
103 need_throw_exp = true;
104 if (m_exp_other_fail)
105 std::rethrow_exception(std::exchange(m_exp_other_fail, nullptr));
106 }
107 else if (state_in_exp & ios_defs::eofbit)
108 {
109 if (!std::current_exception())
110 throw eof_error{};
111 }
112
113 if (need_throw_exp)
114 throw ios_defs::failure("failure bit has been set");
115 }

118 bool good() const { return rdstate() == 0; }
119 bool dev_fail() const { return rdstate() & ios_defs::devfailbit; }
120 bool cvt_fail() const { return rdstate() & ios_defs::cvtfailbit; }
121 bool str_fail() const { return rdstate() & ios_defs::strfailbit; }
122 bool other_fail() const { return rdstate() & ios_defs::otherfailbit; }

138 {
139 try
140 {
141 std::rethrow_exception(ex);
142 }
143 catch (device_error&)
144 {

148 }
149 catch (cvt_error&)

154 }
155 catch (stream_error&)

157 if (!m_exp_str_fail)

159 setstate(ios_defs::strfailbit);
160 }
161 catch (eof_error&)

163 setstate(ios_defs::eofbit);
164 }
165 catch(...)

168 m_exp_other_fail = std::current_exception();

170 }
171 }
