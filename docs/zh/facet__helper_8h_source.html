29 inline CharT* add_grouping(CharT* s, CharT sep, const std::vector<uint8_t>& grouping,
30 const CharT* first, const CharT* last)
31 {
32 size_t idx = 0;
33 size_t ctr = 0;
34 size_t gsize = grouping.size();
35
36 while (last - first > grouping[idx]
37 && (grouping[idx] > 0))
38 {
39 last -= grouping[idx];
40 idx < gsize - 1 ? ++idx : ++ctr;
41 }
42
43 while (first != last)
44 *s++ = *first++;
45
46 while (ctr--)
47 {
48 *s++ = sep;
49 for (char i = grouping[idx]; i > 0; --i)
50 *s++ = *first++;
51 }
52
53 while (idx--)
54 {
55 *s++ = sep;
56 for (char i = grouping[idx]; i > 0; --i)
57 *s++ = *first++;
58 }
59 return s;
60 }

62 inline bool verify_grouping(const std::vector<uint8_t>& grouping,
63 const std::string& grouping_tmp)
64 {
65 const size_t n = grouping_tmp.size() - 1;
66 const size_t min_val = std::min(n, size_t(grouping.size() - 1));
67 size_t i = n;
68 bool test = true;
69
70 // Parsed number groupings have to match the
71 // numpunct::grouping string exactly, starting at the
72 // right-most point of the parsed sequence of elements ...
73 for (size_t j = 0; j < min_val && test; --i, ++j)
74 test = grouping_tmp[i] == grouping[j];
75 for (; i && test; --i)
76 test = grouping_tmp[i] == grouping[min_val];
77 // ... but the first parsed grouping can be <= numpunct
78 // grouping (only do the check if the numpunct char is > 0
79 // because <= 0 means any size is ok).
80 if (static_cast<signed char>(grouping[min_val]) > 0
81 && grouping[min_val] != std::numeric_limits<char>::max())
82 test &= grouping_tmp[0] <= grouping[min_val];
83 return test;
84 }

103 inline CharT nl_langinfo_char(nl_item item, const std::string& locale_name, CharT default_char)
104 {
