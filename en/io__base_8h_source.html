201 void clear(ios_defs::iostate s = ios_defs::goodbit)
202 {
203 std::lock_guard guard(m_state_mutex);
204 m_stream_state.store(s);
205 if ((s & ios_defs::devfailbit) == ios_defs::goodbit) m_exp_dev_fail = std::exception_ptr{};
206 if ((s & ios_defs::cvtfailbit) == ios_defs::goodbit) m_exp_cvt_fail = std::exception_ptr{};
207 if ((s & ios_defs::strfailbit) == ios_defs::goodbit) m_exp_str_fail = std::exception_ptr{};
208 if ((s & ios_defs::otherfailbit) == ios_defs::goodbit) m_exp_other_fail = std::exception_ptr{};
209

211 {
212 ios_defs::iostate state_in_exp = m_exception & s;
213 if (state_in_exp & ios_defs::devfailbit)
214 {
215 if (m_exp_dev_fail)

217 else
218 throw device_error("device failure bit has been set");

429 template <bool ignore_exception_mask = false>
430 void handle_exception(const std::exception_ptr& ex, bool at_eof = false)
431 {
432 if (!ex) return;
433 std::lock_guard guard(m_state_mutex);
434 const ios_defs::iostate eof = at_eof ? ios_defs::eofbit : ios_defs::goodbit;
435 try
436 {
437 std::rethrow_exception(ex);
438 }
439 catch (device_error&)
440 {
441 if (!m_exp_dev_fail)
442 m_exp_dev_fail = std::current_exception();
443 setstate<ignore_exception_mask>(ios_defs::devfailbit | eof);
444 }
445 catch (cvt_error&)
446 {
447 if (!m_exp_cvt_fail)
448 m_exp_cvt_fail = std::current_exception();
449 setstate<ignore_exception_mask>(ios_defs::cvtfailbit | eof);
450 }
451 catch (stream_error&)
452 {
453 if (!m_exp_str_fail)
454 m_exp_str_fail = std::current_exception();
455 setstate<ignore_exception_mask>(ios_defs::strfailbit | eof);
456 }
457 catch (eof_error&)
458 {
459 setstate<ignore_exception_mask>(ios_defs::eofbit);
460 }
461 catch(...)
462 {
463 if (!m_exp_other_fail)
464 m_exp_other_fail = std::current_exception();
465 setstate<ignore_exception_mask>(ios_defs::otherfailbit | eof);
466 }
467 }

924 void access_callbacks(const locale<TChar>& new_loc)
925 {
926 std::exception_ptr throw_exception = nullptr;
927
928 for (const auto& [cb, id] : m_callbacks)
929 {
930 try
931 {
932 auto pre = m_pwords.find(id);
933 std::shared_ptr<void> old_data =
934 (pre != m_pwords.end()) ? pre->second : nullptr;
935
936 std::shared_ptr<void> new_data = cb(new_loc, old_data);
937
938 // Re-locate after the callback: cb may reentrantly mutate
939 // m_pwords (via set_pword), which can rehash/erase and would
940 // invalidate an iterator held across the call.
941 auto it = m_pwords.find(id);
942 if (new_data)
943 {
944 if (it != m_pwords.end()) it->second = std::move(new_data);
945 else m_pwords.insert({id, std::move(new_data)});
946 }
947 else if (it != m_pwords.end())
948 {
949 m_pwords.erase(it);
950 }
951 }
952 catch(...)
953 {
954 if (!throw_exception)
955 throw_exception = std::current_exception();
956 }
957 }
958
959 if (throw_exception)
960 std::rethrow_exception(throw_exception);
961 }
