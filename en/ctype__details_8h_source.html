27// The ctype lookup tables in this header and in ctype.h are sized as
28// std::numeric_limits<unsigned char>::max() + 1, on the assumption that a
29// byte is 8 bits and that the table covers all values an unsigned char can
30// hold. Exotic platforms with CHAR_BIT != 8 (some DSPs, historical mainframes)
31// would silently grow every per-byte table by 2^(CHAR_BIT-8)x. Reject them at
32// compile time rather than ship a non-obviously-broken binary.
33static_assert(CHAR_BIT == 8,
34 "facet/ctype tables assume an 8-bit byte; see ctype.h::s_len");

48 //
49 // When adding a new primitive mask bit, update ALL of:
50 // 1) the constants below
51 // 2) ctype_conf<CharT>::m_wmask_<name> (private member)
52 // 3) ctype_conf<CharT> constructor (wctype_wrapper assignment)

63 constexpr static mask cntrl = 0x0080;

197 // Matches std::ctype<wchar_t>::widen() semantics: only required to be
198 // correct for the basic source character set. For multi-byte locales
199 // (e.g. UTF-8), high bytes are not standalone characters and btowc()
200 // returns WEOF for them at table-build time; the corresponding entries
201 // in m_widen hold WEOF cast to CharT (a sentinel-looking but
202 // unspecified value). Callers must not widen() such bytes.
203 [[nodiscard]] virtual CharT widen(char c) const
204 {
205 return static_cast<CharT>(m_widen[static_cast<unsigned char>(c)]);
206 }

208 // Implementation note: unlike is/toupper/tolower (which call *_l
209 // functions with an explicit, immutable locale_t), wctob has no _l
210 // variant -- it reads whatever locale is currently active on the
211 // calling thread. We therefore install m_inter_locale via clocale_user
212 // for the duration of the wctob call, and restore on scope exit.
213 //
214 // Visible side effect for the window of this call: the calling
215 // thread's locale is m_inter_locale, not whatever it was on entry.
216 // Anything reached during this call -- in particular, a callback
217 // dispatched through a user-derived override of this function --
218 // that performs locale-sensitive I/O (std::ostream numeric
219 // formatting, std::format, third-party locale-aware code) will
220 // observe m_inter_locale instead of the thread's outer locale.
221 // Derivations should keep this function pure-computational and
222 // avoid locale-sensitive side effects inside the override; the same
223 // guidance the std::ctype facet conventions imply.
224 [[nodiscard]] virtual std::optional<char> narrow(CharT wc) const
225 {
226 if (out_of_wchar_range(wc)) return std::nullopt;
227 clocale_user guard(m_inter_locale);
228 const int c = wctob(wc);
229 if (c == EOF) return std::nullopt;
230 else return static_cast<char>(c);
231 }

253// Specialisation for UTF-8 code units (char8_t). The design rests on two
254// structural guarantees of the UTF-8 encoding: (1) no byte in [0x00, 0x7F]
255// ever appears as a leading or continuation byte of a multi-byte sequence,
256// so code units in that range are ASCII-compatible and can be classified
257// exactly like the corresponding char value in the locale; (2) every byte
258// in [0x80, 0xFF] is either a leading byte (0xC0–0xFF) or a continuation
259// byte (0x80–0xBF) of a multi-byte sequence, and therefore does not
260// represent a standalone Unicode code point.
261//
262// For [0x00, 0x7F]: all operations delegate to an internal ctype_conf<char>,
263// which builds locale-aware lookup tables for all 256 char values. Only the
264// lower half of that table is meaningful here; the high half is never used.
265//
266// For [0x80, 0xFF]: each API applies the conservative treatment documented
267// on its individual method below.
268template <>
269class ctype_conf<char8_t> : public ft_basic<ctype<char8_t>>
270{
271public:
272 ctype_conf(const std::string& name)
273 : ft_basic<ctype<char8_t>>()
274 , m_internal(name)
275 {}
276
277public:
278 [[nodiscard]] virtual mask is(char8_t c) const
279 {
280 if (c & 0x80)
281 return static_cast<mask>(0); // not a standalone code point; no classification applies
282 return m_internal.is(static_cast<char>(c));
283 }
284
285 [[nodiscard]] virtual char8_t toupper(char8_t c) const
286 {
287 if (c & 0x80) return c; // not a standalone code point; no case information
288 return static_cast<char8_t>(m_internal.toupper(static_cast<char>(c)));
289 }
290
291 [[nodiscard]] virtual char8_t tolower(char8_t c) const
292 {
293 if (c & 0x80) return c; // not a standalone code point; no case information
294 return static_cast<char8_t>(m_internal.tolower(static_cast<char>(c)));
295 }
296
297 // char8_t has the same range as unsigned char, so every char value maps
298 // to exactly one char8_t code unit with no loss and no WEOF sentinel (cf.
299 // the wchar_t/char32_t specialisation, where btowc() may return WEOF for
300 // high bytes).
301 [[nodiscard]] virtual char8_t widen(char c) const { return static_cast<char8_t>(c); }
302
303 [[nodiscard]] virtual std::optional<char> narrow(char8_t c) const
304 {
305 if (c & 0x80) return std::nullopt; // not a standalone code point; no char equivalent
306 return static_cast<char>(c);
307 }
308
309private:
310 ctype_conf<char> m_internal;
311};
