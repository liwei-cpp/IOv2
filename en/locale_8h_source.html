31template <typename T>
32static const size_t type_id_v = std::bit_cast<size_t>(&(type_id<T>::s_id));
33
34// type_id_v keys m_facets by the address of type_id<T>::s_id. This mirrors
35// ft_basic::id() (facet_common.h), which guards the same idiom. bit_cast already
36// requires the pointer and size_t to have equal size; the assert documents the
37// invariant and yields a clear diagnostic so a platform with
38// sizeof(void*) > sizeof(size_t) fails to compile instead of silently aliasing
39// distinct facet types in m_facets.
40static_assert(sizeof(void*) <= sizeof(size_t),
41 "type_id_v relies on a pointer fitting losslessly into size_t");
