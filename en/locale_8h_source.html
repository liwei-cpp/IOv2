31template <typename T>
32static const size_t type_id_v = std::bit_cast<size_t>(&(type_id<T>::s_id));
33
34// type_id_v keys m_facets by the address of type_id<T>::s_id. This mirrors
35// ft_basic::id() (facet_common.h), which guards the same idiom. std::bit_cast is
36// well-formed only when sizeof(To) == sizeof(From); since we bit_cast a pointer
37// into size_t, the assert must require *equal* size, not merely "fits". With the
38// correct condition a platform where the sizes differ trips this assert with a
39// clear diagnostic instead of an opaque error on the bit_cast line below.
40static_assert(sizeof(void*) == sizeof(size_t),
41 "type_id_v uses std::bit_cast<size_t>(pointer), which is well-formed "
42 "only when sizeof(void*) == sizeof(size_t)");
