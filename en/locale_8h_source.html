312 // Assignment is the only mutating operation on a locale instance. Per the
313 // class threading contract (see the class doc), the caller must ensure no
314 // other thread accesses *this same instance* (read or write) during an
315 // assignment. Two-phase locking: first snapshot the source under a shared
316 // lock (concurrent const readers of the source are not blocked), then publish
317 // into *this under an exclusive lock. The two locks are never held at once, so
318 // this is deadlock-free by construction. They keep the source consistent and
319 // the maps free of torn reads, but they do NOT make a concurrent get<TF>() on
320 // the assigned-into object atomic against the assignment.
321 locale& operator=(const locale& val)
322 {
323 if (this == &val) return *this;
324 decltype(m_facet_confs) confs;
325 decltype(m_facets) facets;
326 {
327 std::shared_lock src(val.m_facet_mutex);
328 confs = val.m_facet_confs;
329 facets = val.m_facets;
330 }
331 {
332 std::unique_lock dst(m_facet_mutex);
333 m_facet_confs = std::move(confs);
334 m_facets = std::move(facets);
335 }
336 return *this;
337 }

424 locale involve_msg(const std::string& domain, const std::string& lang = "",
425 bool throw_if_fail = false) const requires (!std::is_same_v<TChar, char>)
426 {
427 // messages_conf<TChar> is only specialized for char8_t, char32_t and
428 // UTF-32 wchar_t (plus the char overload above); for char16_t or a
429 // non-UTF-32 wchar_t it stays incomplete. Reject those here with a clear
430 // message instead of an opaque "incomplete type" error from make_shared
431 // below. Safe as a completeness check because messages_conf<TChar>'s
432 // completeness is fixed program-wide (a type either has a specialization
433 // everywhere or nowhere).
434 static_assert(requires { sizeof(messages_conf<TChar>); },
435 "involve_msg: messages translation is unsupported for this character "
436 "type -- no messages_conf<TChar> specialization exists (e.g. char16_t, "
437 "or wchar_t on a non-UTF-32 platform).");
438 const std::string filtered_lang = base_ft<messages>::filter_lang(domain, lang);
439 const std::string dirname = base_ft<messages>::get_dirname(domain);
440 auto ft = s_ori_facet_buf.try_get_msg<TChar>(domain, filtered_lang, dirname);
441 if (!ft)
442 {
443 // Build strictly so a load failure surfaces as an exception, and intern only
444 // on success: a degraded (load-failed) facet is never cached, so a later
445 // throw_if_fail=true call re-attempts the load and can still throw.
446 try
447 {
448 ft = std::make_shared<messages_conf<TChar>>(domain, filtered_lang, dirname, true);
449 // Caching is separate from the load: a cache-insert failure (e.g. an
450 // allocation failure while interning) must never degrade or drop a
451 // successfully-loaded facet. On such a failure keep it, just uncached.
452 try { ft = s_ori_facet_buf.put_msg<TChar>(ft, domain, filtered_lang, dirname); }
453 catch (...) {} // NOLINT(bugprone-empty-catch)
454 }
455 catch (...)
456 {
457 if (throw_if_fail) throw;
458 ft = std::make_shared<messages_conf<TChar>>(domain, filtered_lang, dirname, false);
459 }
460 }

462 // At most one messages facet per locale (keyed by the single messages_conf
463 // type id). If *this already carries one, this replaces it in the returned
464 // copy -- matching involve()'s same-id replace semantics; *this is unchanged.
465 locale res(*this);
466 res.m_facet_confs[ft->id()] = std::move(ft);
467 res.m_facets.clear();
468 return res;

600 // The `!std::derived_from` conjunct excludes the pathological "both a conf and a
601 // composite facet" type, so that for such a type only the guard overload below is
602 // viable (it would otherwise tie with this one and make the call ambiguous --
603 // std::derived_from is a concept and subsumes, but the bare is_nonempty_... variable
604 // template re-typed in two places forms distinct, non-subsuming atomic constraints).
605 template <typename TF>
606 requires (is_nonempty_facet_create_rule<typename TF::create_rules>
607 && !std::derived_from<TF, abs_ft>)
608 bool has() const
609 {
610 {
611 std::shared_lock g(m_facet_mutex);
612 if (auto it = m_facets.find(type_id_v<TF>()); it != m_facets.end())
613 return true;
614 }
615
616 ft_wrapper<typename TF::create_rules> obj(*this);
617 return obj.has();
618 }

620 // Guard overload. A type that is *both* a facet conf (derived from abs_ft) and a
621 // composite facet (with a non-empty create_rules) would satisfy the constraints of
622 // both has() overloads above and make the call ambiguous. By design that never
623 // happens -- confs derive from abs_ft while facets carry create_rules, and they are
624 // distinct types. This overload's constraint is the conjunction of the other two's,
625 // so it subsumes both and is selected unambiguously for such a pathological type,
626 // turning an opaque "ambiguous call" into the precise static_assert below.
627 template <typename TF>
628 requires (std::derived_from<TF, abs_ft>
629 && is_nonempty_facet_create_rule<typename TF::create_rules>)
630 bool has() const
631 {
632 static_assert(dependent_false_v<TF>,
633 "has<TF>(): TF is both a facet conf (derived from abs_ft) and a "
634 "composite facet (with a non-empty create_rules). These roles are "
635 "mutually exclusive by design -- confs derive from abs_ft, facets "
636 "carry create_rules -- so such a type is a definition error.");
637 return false;
