39// ft_basic::id() (facet_common.h), which guards the same idiom. std::bit_cast is

335 // Assignment is the only mutating operation on a locale instance. Per the
336 // class threading contract (see the class doc), the caller must ensure no
337 // other thread accesses *this same instance* (read or write) during an
338 // assignment. The locks below keep the source consistent and the maps free
339 // of torn reads, but they do NOT make a concurrent get<TF>() on the
340 // assigned-into object atomic against the assignment.

445 // The `!std::derived_from` conjunct excludes the pathological "both a conf and a
446 // composite facet" type, so that for such a type only the guard overload below is
447 // viable (it would otherwise tie with this one and make the call ambiguous --
448 // std::derived_from is a concept and subsumes, but the bare is_nonempty_... variable
449 // template re-typed in two places forms distinct, non-subsuming atomic constraints).
450 template <typename TF>
451 requires (is_nonempty_facet_create_rule<typename TF::create_rules>
452 && !std::derived_from<TF, abs_ft>)
453 bool has() const
454 {
455 {
456 std::shared_lock g(m_facet_mutex);
457 if (auto it = m_facets.find(type_id_v<TF>()); it != m_facets.end())
458 return true;
459 }
460
461 ft_wrapper<typename TF::create_rules> obj(*this);
462 return obj.has();
463 }

465 // Guard overload. A type that is *both* a facet conf (derived from abs_ft) and a
466 // composite facet (with a non-empty create_rules) would satisfy the constraints of
467 // both has() overloads above and make the call ambiguous. By design that never
468 // happens -- confs derive from abs_ft while facets carry create_rules, and they are
469 // distinct types. This overload's constraint is the conjunction of the other two's,
470 // so it subsumes both and is selected unambiguously for such a pathological type,
471 // turning an opaque "ambiguous call" into the precise static_assert below.
472 template <typename TF>
473 requires (std::derived_from<TF, abs_ft>
474 && is_nonempty_facet_create_rule<typename TF::create_rules>)
475 bool has() const
476 {
477 static_assert(dependent_false_v<TF>,
478 "has<TF>(): TF is both a facet conf (derived from abs_ft) and a "
479 "composite facet (with a non-empty create_rules). These roles are "
480 "mutually exclusive by design -- confs derive from abs_ft, facets "
481 "carry create_rules -- so such a type is a definition error.");
482 return false;
483 }

500 std::shared_ptr<TF> get() const
501 {
502 {

505 return std::static_pointer_cast<TF>(it->second);

510 if (res)
511 {

514 if (!inserted)
