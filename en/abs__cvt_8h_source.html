226 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);

232 {

234 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
235 if (new_read_size == 0)
236 throw cvt_error("get_buf<Saturate> fail: meet eos");
237 m_end_pos += new_read_size;
238 read_size += new_read_size;
239 }
240 }
241
242 if constexpr (!Saturate)
243 {
244 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
245 m_cur_pos = m_end_pos;

1279 size_t get(internal_type* to, size_t to_max)
1280 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1281 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1282 }
1283 {
1284 // Read failures do not taint: they only advance the kernel cursor
1285 // and possibly leave a per-call mbstate undefined; the underlying
1286 // stream content is unchanged, and callers can reseek to recover.
1287 // We still honour any pre-existing taint so that a writer that
1288 // corrupted the stream cannot then read back a confused image.
1289 assert_not_tainted();
1290
1291 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1292
1293 if (!m_is_bos_done)
1294 {
1295 if (to_max == 0) return 0;
1296
1297 constexpr size_t ext_size = sizeof(external_type);
1298
1299 reader.reset(s_bos_chunk);
1300 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1301 char* dest_bytes = reinterpret_cast<char*>(to);
1302 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1303 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1304
1305 // BOS phase requires complete data: if the stream ends unexpectedly
1306 // before providing enough bytes, this is a format error and we throw.
1307 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1308
1309 // Read full external_type chunks
1310 while (dest_bytes + ext_size <= dest_bytes_end)
1311 {
1312 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1313
1314 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1315 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1316 dest_bytes += units_to_read_now * ext_size;
1317 }
1318
1319 // Read the final partial internal_type if needed
1320 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1321 if (final_remainder_bytes > 0)
1322 {
1323 const auto* src_buf = reader.template get_buf<true>(1);
1324 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1325 dest_bytes += final_remainder_bytes;
1326 }
1327
1328 return to_max;
1329 }
1330 else
1331 {
1332 if (m_io_status != io_status::input)
1333 {
1334 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1335 static_cast<CurrentType*>(this)->switch_to_get();
1336 else
1337 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1338 }
1339 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1340 }

1388 void put(const internal_type* to, size_t to_size)
1389 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1390 { t.put_main(w, data, len) } -> std::same_as<void>;
1391 }
1392 {
1393 assert_not_tainted();
1394
1395 // IO-direction switching does not write bytes to the output stream:
1396 // a failed switch_to_put() leaves the stream unchanged, so the
1397 // converter is not corrupted and must not be tainted. Perform the
1398 // switch BEFORE entering the taint-on-exception region so that only
1399 // actual kernel writes can cause tainting.
1400 if (m_is_bos_done && m_io_status != io_status::output)
1401 {
1402 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1403 static_cast<CurrentType*>(this)->switch_to_put();
1404 else
1405 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1406 }
1407
1408 try
1409 {
1410 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1411 if (!m_is_bos_done)
1412 {
1413 if (to_size == 0) return;
1414
1415 constexpr size_t ext_size = sizeof(external_type);
1416
1417 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1418 auto to_bytes = reinterpret_cast<const char*>(to);
1419 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1420 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1421
1422 // BOS phase writes complete external_type units. If the input data
1423 // is not a multiple of ext_size, pad the final unit with zeros.
1424
1425 writer.reset(s_bos_chunk);
1426 while (to_bytes + ext_size <= to_bytes_end)
1427 {
1428 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1429 auto ptr = writer.put_buf(dest_count);
1430 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1431 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1432 to_bytes += dest_count * ext_size;
1433 }
1434
1435 if (to_bytes < to_bytes_end)
1436 {
1437 // The while loop above guarantees remaining < ext_size.
1438 // The modulo is logically redundant but helps the compiler's
1439 // static analyzer prove the bound for memset size calculation.
1440 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1441 auto ptr = writer.put_buf(1);
1442 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1443 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1444 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1445 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1446 }
1447
1448 writer.commit();
1449 }
1450 else
1451 {
1452 // switch_to_put() was already called above if needed;
1453 // m_io_status is guaranteed to be output at this point.
1454 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1455 // commit() is the sole responsibility of abs_cvt::put: derived
1456 // put_main implementations must NOT call writer.commit() themselves.
1457 // If put_main throws, this commit is skipped and any uncommitted
1458 // buffer contents are dropped at writer destruction; the catch
1459 // below taints the converter.
1460 writer.commit();
1461 }
1462 }
1463 catch (...)
1464 {
1465 m_is_tainted = true;
1466 throw;
1467 }
