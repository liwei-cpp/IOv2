1134 size_t get(internal_type* to, size_t to_max)
1135 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1136 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1137 }
1138 {
1139 // Read failures do not taint: they only advance the kernel cursor
1140 // and possibly leave a per-call mbstate undefined; the underlying
1141 // stream content is unchanged, and callers can reseek to recover.
1142 // We still honour any pre-existing taint so that a writer that
1143 // corrupted the stream cannot then read back a confused image.
1144 assert_not_tainted();
1145
1146 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1147
1148 if (!m_is_bos_done)
1149 {
1150 if (to_max == 0) return 0;
1151
1152 constexpr size_t ext_size = sizeof(external_type);
1153
1154 reader.reset(s_bos_chunk);
1155 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1156 char* dest_bytes = reinterpret_cast<char*>(to);
1157 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1158 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1159
1160 // BOS phase requires complete data: if the stream ends unexpectedly
1161 // before providing enough bytes, this is a format error and we throw.
1162 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1163
1164 // Read full external_type chunks
1165 while (dest_bytes + ext_size <= dest_bytes_end)
1166 {
1167 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1168
1169 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1170 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1171 dest_bytes += units_to_read_now * ext_size;
1172 }
1173
1174 // Read the final partial internal_type if needed
1175 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1176 if (final_remainder_bytes > 0)
1177 {
1178 const auto* src_buf = reader.template get_buf<true>(1);
1179 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1180 dest_bytes += final_remainder_bytes;
1181 }
1182
1183 return to_max;
1184 }
1185 else
1186 {
1187 if (m_io_status != io_status::input)
1188 {
1189 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1190 static_cast<CurrentType*>(this)->switch_to_get();
1191 else
1192 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1193 }
1194 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1195 }

1243 void put(const internal_type* to, size_t to_size)
1244 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1245 { t.put_main(w, data, len) } -> std::same_as<void>;
1246 }
1247 {
1248 assert_not_tainted();
1249
1250 // IO-direction switching does not write bytes to the output stream:
1251 // a failed switch_to_put() leaves the stream unchanged, so the
1252 // converter is not corrupted and must not be tainted. Perform the
1253 // switch BEFORE entering the taint-on-exception region so that only
1254 // actual kernel writes can cause tainting.
1255 if (m_is_bos_done && m_io_status != io_status::output)
1256 {
1257 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1258 static_cast<CurrentType*>(this)->switch_to_put();
1259 else
1260 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1261 }
1262
1263 try
1264 {
1265 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1266 if (!m_is_bos_done)
1267 {
1268 if (to_size == 0) return;
1269
1270 constexpr size_t ext_size = sizeof(external_type);
1271
1272 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1273 auto to_bytes = reinterpret_cast<const char*>(to);
1274 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1275 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1276
1277 // BOS phase writes complete external_type units. If the input data
1278 // is not a multiple of ext_size, pad the final unit with zeros.
1279
1280 writer.reset(s_bos_chunk);
1281 while (to_bytes + ext_size <= to_bytes_end)
1282 {
1283 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1284 auto ptr = writer.put_buf(dest_count);
1285 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1286 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1287 to_bytes += dest_count * ext_size;
1288 }
1289
1290 if (to_bytes < to_bytes_end)
1291 {
1292 // The while loop above guarantees remaining < ext_size.
1293 // The modulo is logically redundant but helps the compiler's
1294 // static analyzer prove the bound for memset size calculation.
1295 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1296 auto ptr = writer.put_buf(1);
1297 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1298 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1299 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1300 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1301 }
1302
1303 writer.commit();
1304 }
1305 else
1306 {
1307 // switch_to_put() was already called above if needed;
1308 // m_io_status is guaranteed to be output at this point.
1309 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1310 // commit() is the sole responsibility of abs_cvt::put: derived
1311 // put_main implementations must NOT call writer.commit() themselves.
1312 // If put_main throws, this commit is skipped and any uncommitted
1313 // buffer contents are dropped at writer destruction; the catch
1314 // below taints the converter.
1315 writer.commit();
1316 }
1317 }
1318 catch (...)
1319 {
1320 m_is_tainted = true;
1321 throw;
1322 }
