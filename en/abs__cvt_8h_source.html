1393 // and possibly leave a per-call mbstate undefined; the underlying
1394 // stream content is unchanged, and callers can reseek to recover.

1396 // corrupted the stream cannot then read back a confused image.

1400
1401 if (!m_is_bos_done)
1402 {

1405 constexpr size_t ext_size = sizeof(external_type);
1406
1407 reader.reset(s_bos_chunk);
1408 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1409 char* dest_bytes = reinterpret_cast<char*>(to);
1410 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)

1414 // before providing enough bytes, this is a format error and we throw.
1415 // Unlike get_main() which may return partial data, BOS is all-or-nothing.

1419 {

1421

1423 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1424 dest_bytes += units_to_read_now * ext_size;
1425 }
1426

1428 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1429 if (final_remainder_bytes > 0)
1430 {
1431 const auto* src_buf = reader.template get_buf<true>(1);
1432 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);

1496 void put(const internal_type* to, size_t to_size)

1504 // a failed switch_to_put() leaves the stream unchanged, so the
1505 // converter is not corrupted and must not be tainted. Perform the
1506 // switch BEFORE entering the taint-on-exception region so that only
1507 // actual kernel writes can cause tainting.

1511 static_cast<CurrentType*>(this)->switch_to_put();

1513 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");

1517 {
1518 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1519 if (!m_is_bos_done)
1520 {

1523 constexpr size_t ext_size = sizeof(external_type);
1524
1525 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1526 auto to_bytes = reinterpret_cast<const char*>(to);
1527 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1528 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1529
1530 // BOS phase writes complete external_type units. If the input data
1531 // is not a multiple of ext_size, pad the final unit with zeros.
1532
1533 writer.reset(s_bos_chunk);
1534 while (to_bytes + ext_size <= to_bytes_end)
1535 {
1536 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1537 auto ptr = writer.put_buf(dest_count);
1538 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1539 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1540 to_bytes += dest_count * ext_size;
1541 }
1542
1543 if (to_bytes < to_bytes_end)
1544 {
1545 // The while loop above guarantees remaining < ext_size.
1546 // The modulo is logically redundant but helps the compiler's
1547 // static analyzer prove the bound for memset size calculation.
1548 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1549 auto ptr = writer.put_buf(1);
1550 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1551 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1552 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1553 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
