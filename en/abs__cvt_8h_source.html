199 auto get_buf(size_t to_max)
200 {
201 if (to_max == 0)
202 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero.");
203 if (to_max > m_buffer.size())
204 throw cvt_error("cvt_reader::get_buf fail, read size too large.");
205
206 const auto rollback_size = m_end_pos - m_cur_pos;
207 if (to_max <= rollback_size)
208 {
209 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
210 m_cur_pos += to_max;
211 if constexpr (!Saturate)
212 return res;
213 else
214 return res.first;
215 }
216
217 if (to_max > m_buffer.size() - m_cur_pos)
218 {
219 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
220 m_cur_pos = 0;
221 m_end_pos = rollback_size;
222 }
223 const size_t aim_size = to_max - rollback_size;
224 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
225 m_end_pos += read_size;
226
227 if constexpr (Saturate)
228 {
229 while (read_size < aim_size)
230 {
231 const size_t new_aim_size = aim_size - read_size;
232 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
233 if (new_read_size == 0)
234 throw cvt_error("get_buf<Saturate> fail: meet eos");
235 m_end_pos += new_read_size;
236 read_size += new_read_size;
237 }
238 }
239
240 if constexpr (!Saturate)
241 {
242 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
243 m_cur_pos = m_end_pos;
244 return res;
245 }
246 else
247 {
248 const char_type* res = m_buffer.data() + m_cur_pos;
249 m_cur_pos = m_end_pos;
250 return res;
251 }
