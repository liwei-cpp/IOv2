199 auto get_buf(size_t to_max)
200 {
201 if (to_max == 0)
202 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
203 if (to_max > m_buffer.size())
204 throw cvt_error("cvt_reader::get_buf fail, read size too large");
205
206 const auto rollback_size = m_end_pos - m_cur_pos;
207 if (to_max <= rollback_size)
208 {
209 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
210 m_cur_pos += to_max;
211 if constexpr (!Saturate)
212 return res;
213 else
214 return res.first;
215 }
216
217 if (to_max > m_buffer.size() - m_cur_pos)
218 {
219 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
220 m_cur_pos = 0;
221 m_end_pos = rollback_size;
222 }
223 const size_t aim_size = to_max - rollback_size;
224 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
225 m_end_pos += read_size;
226
227 if constexpr (Saturate)
228 {
229 while (read_size < aim_size)
230 {
231 const size_t new_aim_size = aim_size - read_size;
232 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
233 if (new_read_size == 0)
234 throw cvt_error("get_buf<Saturate> fail: meet eos");
235 m_end_pos += new_read_size;
236 read_size += new_read_size;
237 }
238 }
239
240 if constexpr (!Saturate)
241 {
242 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
243 m_cur_pos = m_end_pos;
244 return res;
245 }
246 else
247 {
248 const char_type* res = m_buffer.data() + m_cur_pos;
249 m_cur_pos = m_end_pos;
250 return res;
251 }

1053 {

1057
1058 reader.reset(s_bos_chunk);
1059 char* dest_bytes = reinterpret_cast<char*>(to);
1060 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1061
1062 // BOS phase requires complete data: if the stream ends unexpectedly
1063 // before providing enough bytes, this is a format error and we throw.
1064 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1065
1066 // Read full external_type chunks
1067 while (dest_bytes + ext_size <= dest_bytes_end)
1068 {

1070
1071 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1072 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1073 dest_bytes += units_to_read_now * ext_size;
1074 }
1075
1076 // Read the final partial internal_type if needed
1077 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1078 if (final_remainder_bytes > 0)

1081 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1082 dest_bytes += final_remainder_bytes;
1083 }

1085 return to_max;
1086 }

1145 void put(const internal_type* to, size_t to_size)
1146 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1147 { t.put_main(w, data, len) } -> std::same_as<void>;
1148 }
1149 {
1150 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1151 if (!m_is_bos_done)
1152 {
1153 if (to_size == 0) return;
1154
1155 constexpr size_t ext_size = sizeof(external_type);
1156
1157 auto to_bytes = reinterpret_cast<const char*>(to);
1158 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1159
1160 // BOS phase writes complete external_type units. If the input data
1161 // is not a multiple of ext_size, pad the final unit with zeros.
1162
1163 writer.reset(s_bos_chunk);
1164 while (to_bytes + ext_size <= to_bytes_end)
1165 {
1166 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1167 auto ptr = writer.put_buf(dest_count);
1168 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1169 to_bytes += dest_count * ext_size;
1170 }
1171
1172 if (to_bytes < to_bytes_end)
1173 {
1174 // The while loop above guarantees remaining < ext_size.
1175 // The modulo is logically redundant but helps the compiler's
1176 // static analyzer prove the bound for memset size calculation.
1177 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1178 auto ptr = writer.put_buf(1);
1179 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1180 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1181 }
1182
1183 writer.commit();
1184 }
1185 else
