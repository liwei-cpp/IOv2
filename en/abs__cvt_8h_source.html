1304 size_t get(internal_type* to, size_t to_max)
1305 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1306 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1307 }
1308 {
1309 // Read failures do not taint: they only advance the kernel cursor
1310 // and possibly leave a per-call mbstate undefined; the underlying
1311 // stream content is unchanged, and callers can reseek to recover.
1312 // We still honour any pre-existing taint so that a writer that
1313 // corrupted the stream cannot then read back a confused image.
1314 assert_not_tainted();
1315
1316 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1317
1318 if (!m_is_bos_done)
1319 {
1320 if (to_max == 0) return 0;
1321
1322 constexpr size_t ext_size = sizeof(external_type);
1323
1324 reader.reset(s_bos_chunk);
1325 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1326 char* dest_bytes = reinterpret_cast<char*>(to);
1327 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1328 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1329
1330 // BOS phase requires complete data: if the stream ends unexpectedly
1331 // before providing enough bytes, this is a format error and we throw.
1332 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1333
1334 // Read full external_type chunks
1335 while (dest_bytes + ext_size <= dest_bytes_end)
1336 {
1337 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1338
1339 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1340 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1341 dest_bytes += units_to_read_now * ext_size;
1342 }
1343
1344 // Read the final partial internal_type if needed
1345 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1346 if (final_remainder_bytes > 0)
1347 {
1348 const auto* src_buf = reader.template get_buf<true>(1);
1349 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1350 dest_bytes += final_remainder_bytes;
1351 }
1352
1353 return to_max;
1354 }
1355 else
1356 {
1357 if (m_io_status != io_status::input)
1358 {
1359 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1360 static_cast<CurrentType*>(this)->switch_to_get();
1361 else
1362 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1363 }
1364 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1365 }

1413 void put(const internal_type* to, size_t to_size)
1414 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1415 { t.put_main(w, data, len) } -> std::same_as<void>;
1416 }
1417 {
1418 assert_not_tainted();
1419
1420 // IO-direction switching does not write bytes to the output stream:
1421 // a failed switch_to_put() leaves the stream unchanged, so the
1422 // converter is not corrupted and must not be tainted. Perform the
1423 // switch BEFORE entering the taint-on-exception region so that only
1424 // actual kernel writes can cause tainting.
1425 if (m_is_bos_done && m_io_status != io_status::output)
1426 {
1427 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1428 static_cast<CurrentType*>(this)->switch_to_put();
1429 else
1430 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1431 }
1432
1433 try
1434 {
1435 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1436 if (!m_is_bos_done)
1437 {
1438 if (to_size == 0) return;
1439
1440 constexpr size_t ext_size = sizeof(external_type);
1441
1442 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1443 auto to_bytes = reinterpret_cast<const char*>(to);
1444 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1445 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1446
1447 // BOS phase writes complete external_type units. If the input data
1448 // is not a multiple of ext_size, pad the final unit with zeros.
1449
1450 writer.reset(s_bos_chunk);
1451 while (to_bytes + ext_size <= to_bytes_end)
1452 {
1453 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1454 auto ptr = writer.put_buf(dest_count);
1455 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1456 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1457 to_bytes += dest_count * ext_size;
1458 }
1459
1460 if (to_bytes < to_bytes_end)
1461 {
1462 // The while loop above guarantees remaining < ext_size.
1463 // The modulo is logically redundant but helps the compiler's
1464 // static analyzer prove the bound for memset size calculation.
1465 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1466 auto ptr = writer.put_buf(1);
1467 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1468 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1469 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1470 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1471 }
1472
1473 writer.commit();
1474 }
1475 else
1476 {
1477 // switch_to_put() was already called above if needed;
1478 // m_io_status is guaranteed to be output at this point.
1479 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1480 // commit() is the sole responsibility of abs_cvt::put: derived
1481 // put_main implementations must NOT call writer.commit() themselves.
1482 // If put_main throws, this commit is skipped and any uncommitted
1483 // buffer contents are dropped at writer destruction; the catch
1484 // below taints the converter.
1485 writer.commit();
1486 }
1487 }
1488 catch (...)
1489 {
1490 m_is_tainted = true;
1491 throw;
1492 }
