1127 // and possibly leave a per-call mbstate undefined; the underlying
1128 // stream content is unchanged, and callers can reseek to recover.

1130 // corrupted the stream cannot then read back a confused image.

1148 // before providing enough bytes, this is a format error and we throw.
1149 // Unlike get_main() which may return partial data, BOS is all-or-nothing.

1157 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1158 dest_bytes += units_to_read_now * ext_size;
1159 }
1160

1162 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1163 if (final_remainder_bytes > 0)
1164 {
1165 const auto* src_buf = reader.template get_buf<true>(1);
1166 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1167 dest_bytes += final_remainder_bytes;
1168 }

1171 }

1230 void put(const internal_type* to, size_t to_size)
1231 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1232 { t.put_main(w, data, len) } -> std::same_as<void>;
1233 }
1234 {
1235 assert_not_tainted();
1236
1237 // IO-direction switching does not write bytes to the output stream:
1238 // a failed switch_to_put() leaves the stream unchanged, so the
1239 // converter is not corrupted and must not be tainted. Perform the
1240 // switch BEFORE entering the taint-on-exception region so that only
1241 // actual kernel writes can cause tainting.
1242 if (m_is_bos_done && m_io_status != io_status::output)
1243 {
1244 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1245 static_cast<CurrentType*>(this)->switch_to_put();
1246 else
1247 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1248 }
1249
1250 try
1251 {
1252 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1253 if (!m_is_bos_done)
1254 {
1255 if (to_size == 0) return;
1256
1257 constexpr size_t ext_size = sizeof(external_type);
1258
1259 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1260 auto to_bytes = reinterpret_cast<const char*>(to);
1261 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1262 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1263
1264 // BOS phase writes complete external_type units. If the input data
1265 // is not a multiple of ext_size, pad the final unit with zeros.
1266
1267 writer.reset(s_bos_chunk);
1268 while (to_bytes + ext_size <= to_bytes_end)
1269 {
1270 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1271 auto ptr = writer.put_buf(dest_count);
1272 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1273 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1274 to_bytes += dest_count * ext_size;
1275 }
1276
1277 if (to_bytes < to_bytes_end)
1278 {
1279 // The while loop above guarantees remaining < ext_size.
1280 // The modulo is logically redundant but helps the compiler's
1281 // static analyzer prove the bound for memset size calculation.
1282 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1283 auto ptr = writer.put_buf(1);
1284 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1285 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1286 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1287 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1288 }
1289
1290 writer.commit();
1291 }
1292 else
1293 {
1294 // switch_to_put() was already called above if needed;
1295 // m_io_status is guaranteed to be output at this point.
1296 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1297 // commit() is the sole responsibility of abs_cvt::put: derived
1298 // put_main implementations must NOT call writer.commit() themselves.
1299 // If put_main throws, this commit is skipped and any uncommitted
1300 // buffer contents are dropped at writer destruction; the catch
1301 // below taints the converter.

1303 }
1304 }
