199 auto get_buf(size_t to_max)
200 {
201 if (to_max == 0)
202 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
203 if (to_max > m_buffer.size())
204 throw cvt_error("cvt_reader::get_buf fail, read size too large");
205
206 const auto rollback_size = m_end_pos - m_cur_pos;
207 if (to_max <= rollback_size)
208 {
209 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
210 m_cur_pos += to_max;
211 if constexpr (!Saturate)
212 return res;
213 else
214 return res.first;
215 }
216
217 if (to_max > m_buffer.size() - m_cur_pos)
218 {
219 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
220 m_cur_pos = 0;
221 m_end_pos = rollback_size;
222 }
223 const size_t aim_size = to_max - rollback_size;
224 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
225 m_end_pos += read_size;
226
227 if constexpr (Saturate)
228 {
229 while (read_size < aim_size)
230 {
231 const size_t new_aim_size = aim_size - read_size;
232 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
233 if (new_read_size == 0)
234 throw cvt_error("get_buf<Saturate> fail: meet eos");
235 m_end_pos += new_read_size;
236 read_size += new_read_size;
237 }
238 }
239
240 if constexpr (!Saturate)
241 {
242 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
243 m_cur_pos = m_end_pos;
244 return res;
245 }
246 else
247 {
248 const char_type* res = m_buffer.data() + m_cur_pos;
249 m_cur_pos = m_end_pos;

251 }

1110 size_t get(internal_type* to, size_t to_max)
1111 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1112 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1113 }
1114 {
1115 // Read failures do not taint: they only advance the kernel cursor
1116 // and possibly leave a per-call mbstate undefined; the underlying
1117 // stream content is unchanged, and callers can reseek to recover.
1118 // We still honour any pre-existing taint so that a writer that
1119 // corrupted the stream cannot then read back a confused image.
1120 assert_not_tainted();
1121
1122 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1123
1124 if (!m_is_bos_done)
1125 {
1126 if (to_max == 0) return 0;
1127
1128 constexpr size_t ext_size = sizeof(external_type);
1129
1130 reader.reset(s_bos_chunk);
1131 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1132 char* dest_bytes = reinterpret_cast<char*>(to);
1133 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1134 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1135
1136 // BOS phase requires complete data: if the stream ends unexpectedly
1137 // before providing enough bytes, this is a format error and we throw.
1138 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1139
1140 // Read full external_type chunks
1141 while (dest_bytes + ext_size <= dest_bytes_end)
1142 {
1143 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1144
1145 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1146 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1147 dest_bytes += units_to_read_now * ext_size;
1148 }
1149
1150 // Read the final partial internal_type if needed
1151 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1152 if (final_remainder_bytes > 0)
1153 {
1154 const auto* src_buf = reader.template get_buf<true>(1);
1155 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1156 dest_bytes += final_remainder_bytes;
1157 }
1158
1159 return to_max;
1160 }
1161 else
1162 {
1163 if (m_io_status != io_status::input)
1164 {

1167 else
1168 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1169 }

1219 void put(const internal_type* to, size_t to_size)
1220 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1221 { t.put_main(w, data, len) } -> std::same_as<void>;
1222 }
1223 {
1224 assert_not_tainted();
1225
1226 // IO-direction switching does not write bytes to the output stream:
1227 // a failed switch_to_put() leaves the stream unchanged, so the
1228 // converter is not corrupted and must not be tainted. Perform the
1229 // switch BEFORE entering the taint-on-exception region so that only
1230 // actual kernel writes can cause tainting.
1231 if (m_is_bos_done && m_io_status != io_status::output)
1232 {
1233 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1234 static_cast<CurrentType*>(this)->switch_to_put();
1235 else
1236 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1237 }
1238
1239 try
1240 {
1241 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1242 if (!m_is_bos_done)
1243 {
1244 if (to_size == 0) return;
1245
1246 constexpr size_t ext_size = sizeof(external_type);
1247
1248 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1249 auto to_bytes = reinterpret_cast<const char*>(to);
1250 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1251 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1252
1253 // BOS phase writes complete external_type units. If the input data
1254 // is not a multiple of ext_size, pad the final unit with zeros.
1255
1256 writer.reset(s_bos_chunk);
1257 while (to_bytes + ext_size <= to_bytes_end)
1258 {
1259 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1260 auto ptr = writer.put_buf(dest_count);
1261 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1262 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1263 to_bytes += dest_count * ext_size;
1264 }
1265
1266 if (to_bytes < to_bytes_end)
1267 {
1268 // The while loop above guarantees remaining < ext_size.
1269 // The modulo is logically redundant but helps the compiler's
1270 // static analyzer prove the bound for memset size calculation.
1271 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1272 auto ptr = writer.put_buf(1);
1273 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1274 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1275 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1276 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1277 }
1278
1279 writer.commit();
1280 }
1281 else
1282 {
1283 // switch_to_put() was already called above if needed;
1284 // m_io_status is guaranteed to be output at this point.
1285 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1286 // commit() is the sole responsibility of abs_cvt::put: derived
1287 // put_main implementations must NOT call writer.commit() themselves.
1288 // If put_main throws, this commit is skipped and any uncommitted
1289 // buffer contents are dropped at writer destruction; the catch
1290 // below taints the converter.
1291 writer.commit();
1292 }
1293 }
1294 catch (...)
1295 {
1296 m_is_tainted = true;
1297 throw;
1298 }
