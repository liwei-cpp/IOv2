1335 size_t get(internal_type* to, size_t to_max)
1336 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1337 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1338 }
1339 {
1340 // Read failures do not taint: they only advance the kernel cursor
1341 // and possibly leave a per-call mbstate undefined; the underlying
1342 // stream content is unchanged, and callers can reseek to recover.
1343 // We still honour any pre-existing taint so that a writer that
1344 // corrupted the stream cannot then read back a confused image.
1345 assert_not_tainted();
1346
1347 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1348
1349 if (!m_is_bos_done)
1350 {
1351 if (to_max == 0) return 0;
1352
1353 constexpr size_t ext_size = sizeof(external_type);
1354
1355 reader.reset(s_bos_chunk);
1356 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1357 char* dest_bytes = reinterpret_cast<char*>(to);
1358 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1359 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1360
1361 // BOS phase requires complete data: if the stream ends unexpectedly
1362 // before providing enough bytes, this is a format error and we throw.
1363 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1364
1365 // Read full external_type chunks
1366 while (dest_bytes + ext_size <= dest_bytes_end)
1367 {
1368 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1369
1370 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1371 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1372 dest_bytes += units_to_read_now * ext_size;
1373 }
1374
1375 // Read the final partial internal_type if needed
1376 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1377 if (final_remainder_bytes > 0)
1378 {
1379 const auto* src_buf = reader.template get_buf<true>(1);
1380 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1381 dest_bytes += final_remainder_bytes;
1382 }
1383
1384 return to_max;
1385 }
1386 else
1387 {
1388 if (m_io_status != io_status::input)
1389 {
1390 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1391 static_cast<CurrentType*>(this)->switch_to_get();
1392 else
1393 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1394 }

1444 void put(const internal_type* to, size_t to_size)
1445 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1446 { t.put_main(w, data, len) } -> std::same_as<void>;

1451 // IO-direction switching does not write bytes to the output stream:
1452 // a failed switch_to_put() leaves the stream unchanged, so the
1453 // converter is not corrupted and must not be tainted. Perform the
1454 // switch BEFORE entering the taint-on-exception region so that only
1455 // actual kernel writes can cause tainting.
1456 if (m_is_bos_done && m_io_status != io_status::output)
1457 {
1458 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1459 static_cast<CurrentType*>(this)->switch_to_put();
1460 else
1461 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1462 }

1465 {
1466 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1467 if (!m_is_bos_done)
1468 {
1469 if (to_size == 0) return;
1470
1471 constexpr size_t ext_size = sizeof(external_type);
1472
1473 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1474 auto to_bytes = reinterpret_cast<const char*>(to);
1475 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1476 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1477
1478 // BOS phase writes complete external_type units. If the input data
1479 // is not a multiple of ext_size, pad the final unit with zeros.
1480
1481 writer.reset(s_bos_chunk);
1482 while (to_bytes + ext_size <= to_bytes_end)
1483 {
1484 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1485 auto ptr = writer.put_buf(dest_count);
1486 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1487 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1488 to_bytes += dest_count * ext_size;
1489 }
1490
1491 if (to_bytes < to_bytes_end)
1492 {
1493 // The while loop above guarantees remaining < ext_size.
1494 // The modulo is logically redundant but helps the compiler's
1495 // static analyzer prove the bound for memset size calculation.
1496 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1497 auto ptr = writer.put_buf(1);
1498 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1499 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1500 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1501 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1502 }
1503
1504 writer.commit();

1509 // m_io_status is guaranteed to be output at this point.
