87inline Botan::secure_vector<uint8_t> key_gen(std::string_view key)
88{
89 // Reject empty input: hashing an empty string would silently yield the
90 // publicly known constant SHA-256("") and use it as the ChaCha20 key,
91 // providing zero confidentiality. The secure_vector-based constructor
92 // already rejects invalid key lengths; mirror that contract here.
93 if (key.empty())
94 throw cvt_error("chacha20 key generation fail: key must not be empty");
95
96 auto hash = Botan::HashFunction::create("SHA-256");
97 if (!hash)
98 throw cvt_error("chacha20 key generation fail: cannot create SHA-256 hash");
99
100 hash->update(reinterpret_cast<const uint8_t*>(key.data()), key.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
101 return hash->final();
102}
103}

254 // Move ops are written by hand (not = default) so the source's m_leftover
255 // ciphertext fragment is zeroed after transfer. std::array's "move" is a

393 void bos_impl()
394 {
395 if (!m_cipher || m_key.empty())
396 throw cvt_error("chacha20_cvt::bos fail: cipher or key missing (moved-from object?)");
397
398 // Reset any carry-over from a previous stream. bos() establishes a
399 // fresh stream, so the leftover buffer from a prior main-content phase
400 // must not bleed across the boundary.
401 m_leftover.fill(0);
402 m_leftover_len = 0;
403
404 const size_t iv_len = m_cipher->default_iv_length();
405 if (iv_len == 0)
406 throw cvt_error("chacha20_cvt::bos fail: cipher reports zero IV length");
407 std::vector<external_type> iv_buf(iv_len);
408
409 if (BT::m_io_status == io_status::input)
410 {
411 if constexpr (cvt_cpt::support_get<KernelType>)
412 {
413 // Contract: kernel.get() on BOS path guarantees to return exactly the
414 // requested byte count, or throw an exception if insufficient data is
415 // available. The check below is purely defensive, verifying this
416 // invariant at runtime.
417 const size_t n = BT::m_kernel.get(iv_buf.data(), iv_len);
418 if (n != iv_len)
419 throw cvt_error("chacha20_cvt::bos fail: incomplete IV read");
420
421 m_cipher->set_key(m_key);
422 m_cipher->set_iv(reinterpret_cast<const uint8_t*>(iv_buf.data()), iv_len); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
423 }
424 else
425 throw cvt_error("chacha20_cvt::bos fail: input mode but kernel does not support get");
426 }
427 else if (BT::m_io_status == io_status::output)
428 {
429 if constexpr (cvt_cpt::support_put<KernelType>)
430 {
431 Botan::AutoSeeded_RNG rng;
432 rng.randomize(reinterpret_cast<uint8_t*>(iv_buf.data()), iv_len); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
433
434 m_cipher->set_key(m_key);
435 m_cipher->set_iv(reinterpret_cast<const uint8_t*>(iv_buf.data()), iv_len); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
436
437 BT::m_kernel.put(iv_buf.data(), iv_len);
438 }
439 else
440 throw cvt_error("chacha20_cvt::bos fail: output mode but kernel does not support put");
441 }
442 else
443 throw cvt_error("chacha20_cvt::bos fail: neither in input nor output mode");
444 }

510 size_t get_main(cvt_reader<KernelType>& reader, internal_type* _to, size_t to_max)
511 requires (cvt_cpt::support_get<KernelType>)
512 {
513 if (to_max == 0) return 0;
514 if (!m_cipher)
515 throw cvt_error("chacha20_cvt::get_main fail: cipher not initialized (moved-from object?)");
516
517 constexpr size_t isize = sizeof(internal_type);
518 constexpr size_t bulk_chunk = (block_size / isize) * isize;
519 static_assert(bulk_chunk >= isize,
520 "block_size must accommodate at least one internal_type");
521
522 auto* to = reinterpret_cast<uint8_t*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
523 size_t elements_delivered = 0;
524
525 reader.reset(block_size);
526
527 while (elements_delivered < to_max)
528 {
529 if (m_leftover_len > 0)
530 {
531 // Splice path: finish the partial element carried over from a
532 // previous short read before consuming any new bulk data.
533 const size_t want = isize - m_leftover_len;
534 auto [ptr, cur_len] = reader.get_buf(want);
535 if (cur_len == 0) break;
536 std::memcpy(m_leftover.data() + m_leftover_len, ptr, cur_len);
537 m_leftover_len += cur_len;
538 if (m_leftover_len == isize)
539 {
540 try { m_cipher->cipher(m_leftover.data(), to, isize); }
541 catch (...) { BT::set_tainted(); throw; }
542 to += isize;
543 elements_delivered += 1;
544 m_leftover_len = 0;
545 }
546 }
547 else
548 {
549 // Bulk path: cipher only the aligned prefix into the user buffer;
550 // stash any trailing sub-element ciphertext for the next call.
551 const size_t elements_remaining = to_max - elements_delivered;
552 const size_t want = (elements_remaining > bulk_chunk / isize)
553 ? bulk_chunk
554 : elements_remaining * isize;
555 auto [ptr, cur_len] = reader.get_buf(want);
556 if (cur_len == 0) break;
557
558 const size_t complete_bytes = (cur_len / isize) * isize;
559 const size_t tail = cur_len - complete_bytes;
560
561 if (complete_bytes > 0)
562 {
563 try
564 {
565 m_cipher->cipher(reinterpret_cast<const uint8_t*>(ptr), to, complete_bytes); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
566 }
567 catch (...) { BT::set_tainted(); throw; }
568 to += complete_bytes;
569 elements_delivered += complete_bytes / isize;
570 }
571 if (tail > 0)
572 {
573 std::memcpy(m_leftover.data(), ptr + complete_bytes, tail);
574 m_leftover_len = tail;
575 }
576 }
577 }
578
579 // A short read with no real EOF is a recoverable state: m_leftover is
580 // preserved and the next get_main call will complete the partial
581 // element via the splice path above. Only flag taint when we could not
582 // satisfy the request, the stream is genuinely at EOF, and we still
583 // hold a partial element — that misalignment can never be resolved.
584 //
585 // Gating on `elements_delivered < to_max` is essential: when the loop
586 // exits because the request was fully satisfied, the leftover is an
587 // in-flight artifact of the final bulk read, not a failure indicator —
588 // throwing here would discard elements we just successfully decrypted.
589 if (elements_delivered < to_max && m_leftover_len > 0 && BT::is_eof())
590 {
591 BT::set_tainted();
592 throw cvt_error("chacha20_cvt::get_main fail: stream ended on a non-aligned boundary");
593 }
594
595 return elements_delivered;
596 }

631 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
632 requires (cvt_cpt::support_put<KernelType>)
633 {
634 if (to_size == 0) return;
635 if (!m_cipher)
636 throw cvt_error("chacha20_cvt::put_main fail: cipher not initialized (moved-from object?)");
637
638 constexpr size_t max_type_limit = std::numeric_limits<size_t>::max();
639 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
640
641 const auto* to = reinterpret_cast<const uint8_t*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
642
643 writer.reset(block_size);
644 while (to_size > 0)
645 {
646 size_t aim_output = (max_chunk / sizeof(internal_type) > to_size)
647 ? (to_size * sizeof(internal_type))
648 : max_chunk;
649 size_t to_i = 0;
650 while (to_i != aim_output)
651 {
652 size_t dest_size = std::min<size_t>(block_size, aim_output - to_i);
653 auto ptr = writer.put_buf(dest_size);
654 // No local try/catch here: abs_cvt::put already wraps put_main in
655 // a catch-all that sets the taint flag on any exception. If cipher
656 // ever throws, that outer guard correctly marks the cvt unusable.
657 // See the get_main doxygen for the full design rationale.
658 m_cipher->cipher(to + to_i, reinterpret_cast<uint8_t*>(ptr), dest_size); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
659 to_i += dest_size;
660 }
661 to += aim_output;
662 to_size -= aim_output / sizeof(internal_type);
663 }
664 // commit() is the responsibility of abs_cvt::put.
665 }
666private:
667 std::unique_ptr<Botan::StreamCipher> m_cipher;
668 Botan::secure_vector<uint8_t> m_key;
669 // Cross-call carry-over for sub-element ciphertext. Holds raw (un-ciphered)
670 // bytes from a short read that did not complete one internal_type; the next
671 // get_main call splices them with fresh bytes before feeding cipher() so the
672 // keystream stays aligned with the byte stream.
673 std::array<uint8_t, sizeof(internal_type)> m_leftover{};
674 size_t m_leftover_len{0};
675};
