207 size_t dget(char* s, size_t n)
208 requires (ID == STDIN_FILENO)
209 {
210 if (n == 0 || m_eof_hit) return 0;
211 if (s == nullptr)
212 throw device_error("std_device::dget fail: null buffer");
213
214 if (m_c.has_value())
215 {

217 if (n == 1)

223 // the look-ahead byte so the next dget() redelivers it (no loss).
224 const size_t got = do_get(s + 1, n - 1);
225 m_c.reset();

246 void dput(const char* ch, size_t n)
247 requires ((ID == STDOUT_FILENO) || (ID == STDERR_FILENO))
248 {

294 size_t do_get(char* s, size_t n)
295 {
296 assert((s != nullptr) && (n != 0));
297
298 constexpr auto max_read = static_cast<size_t>(std::numeric_limits<ssize_t>::max());
299 ssize_t ret = 0;
300 while (true)
301 {
302 ret = read(ID, s, std::min(n, max_read));
303 if (ret != -1) break;
304
305 if (errno == EINTR) continue;
306 if (errno == EAGAIN || errno == EWOULDBLOCK)
307 {
308 struct pollfd pfd{ .fd = ID, .events = POLLIN, .revents = 0 };
309 if (poll(&pfd, 1, -1) == -1)
310 {
311 if (errno == EINTR) continue;
312 throw device_error("std_device::do_get fail: poll error");
313 }
314 else if (pfd.revents & (POLLERR | POLLNVAL))
315 throw device_error("std_device::do_get fail: poll revents error");
316 else if (pfd.revents & POLLIN) continue;
317 else if (pfd.revents & POLLHUP)
318 {
319 ret = 0;
320 break;
321 }
322 else
323 throw device_error("std_device::do_get fail: unexpected poll revents");
324 }
325 else
326 throw device_error("std_device::do_get fail: read error");
327 }
328
329 if (ret == 0)
330 m_eof_hit = true;
331 return static_cast<size_t>(ret);
332 }
