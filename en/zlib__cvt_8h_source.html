418 // finalized on its device; it leaves *this's in-memory zlib state

615 void bos_impl()
616 {
617 if (BT::m_io_status == io_status::input)
618 {
619 if constexpr (cvt_cpt::support_get<KernelType>)
620 {
621 m_strm = std::make_unique<z_stream>();
622 m_strm->zalloc = Z_NULL;
623 m_strm->zfree = Z_NULL;
624 m_strm->opaque = Z_NULL;
625 m_strm->avail_in = 0;
626 m_strm->next_in = Z_NULL;
627 m_strm->state = Z_NULL;
628 auto ret = inflateInit(m_strm.get());
629 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
630 inflate_guard zlib_guard(m_strm);
631
632 // Contract: kernel.get() on BOS path guarantees to return exactly the
633 // requested byte count, or throw an exception if insufficient data is
634 // available. The assert below is purely defensive, verifying this
635 // invariant during development. It is intentionally compiled out in
636 // release builds.
637 std::array<external_type, zlib_header_size> header_buf{};
638 [[maybe_unused]] const size_t n = BT::m_kernel.get(header_buf.data(), zlib_header_size);
639 assert(n == zlib_header_size);
640
641 m_strm->avail_in = zlib_header_size;
642 m_strm->next_in = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
643
644 unsigned char ch = 0;
645 m_strm->avail_out = 1;
646 m_strm->next_out = &ch;
647
648 ret = inflate(m_strm.get(), Z_NO_FLUSH);
649 if (ret == Z_NEED_DICT)
650 throw cvt_error("zlib_cvt::bos fail: preset dictionary not supported");
651 zerr("zlib_cvt::bos fail", ret);
652
653 if ((m_strm->avail_in != 0) || (m_strm->avail_out != 1)) throw cvt_error("zlib_cvt::bos fail: Invalid zlib header");
654 m_strm->next_in = nullptr;
655 m_strm->avail_out = 0;
656 m_strm->next_out = nullptr;
657
658 zlib_guard.release();
659 }
660 else
661 throw cvt_error("zlib_cvt::bos fail: BT returned input but KernelType lacks get support");
662 }
663 else if (BT::m_io_status == io_status::output)
664 {
665 if constexpr (cvt_cpt::support_put<KernelType>)
666 {
667 m_strm = std::make_unique<z_stream>();
668 m_strm->zalloc = Z_NULL;
669 m_strm->zfree = Z_NULL;
670 m_strm->opaque = Z_NULL;
671 m_strm->state = Z_NULL;
672
673 auto ret = deflateInit(m_strm.get(), static_cast<int>(m_put_level));
674 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
675 deflate_guard zlib_guard(m_strm);
676
677 // Contract enforced below: after this deflate() call we require
678 // exactly zlib_header_size bytes in the output buffer and zero
679 // bytes/bits still pending inside zlib. Z_NO_FLUSH on a fresh
680 // stream with no input is expected to flush the queued header
681 // and nothing else. The post-checks turn that expectation into
682 // a hard invariant: any future zlib behavior change (more
683 // bytes, fewer bytes, or bytes left buffered) surfaces here as
684 // a loud failure rather than silent framing corruption.
685 std::array<external_type, zlib_header_size + 1> header_buf{};
686
687 m_strm->avail_in = 0;
688 m_strm->next_in = nullptr;
689 m_strm->avail_out = zlib_header_size + 1;
690 m_strm->next_out = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
691 ret = deflate(m_strm.get(), Z_NO_FLUSH);
692 zerr("zlib_cvt::bos fail", ret);
693
694 if (m_strm->avail_out != 1)
695 throw cvt_error("zlib_cvt::bos fail: zlib did not emit exactly the header");
696
697 unsigned pending_bytes = 0;
698 int pending_bits = 0;
699 if (deflatePending(m_strm.get(), &pending_bytes, &pending_bits) != Z_OK ||
700 pending_bytes != 0 || pending_bits != 0)
701 throw cvt_error("zlib_cvt::bos fail: zlib has unflushed bytes after header");
702
703 BT::m_kernel.put(header_buf.data(), zlib_header_size);
704
705 m_strm->avail_out = 0;
706 m_strm->next_out = nullptr;
707
708 zlib_guard.release();
709 }
710 else
711 throw cvt_error("zlib_cvt::bos fail: BT returned output but KernelType lacks put support");
712 }
713 else
714 throw cvt_error("zlib_cvt::bos fail: invalid response value.");
715 }

790 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
791 requires (cvt_cpt::support_get<KernelType>)
792 {
793 // Once inflate() has reported Z_STREAM_END on this stream, further
794 // inflate() calls on the same z_stream would either fail with
795 // Z_DATA_ERROR or attempt to interpret trailing bytes as a new
796 // concatenated zlib stream. Both are wrong for a single-stream
797 // contract. Surface end-of-stream as a zero-byte read so callers
798 // see a clean EOF instead. Reset by close_stream's state_guard.
799 if (m_stream_ended) return 0;
800
801 // Note: We read one byte at a time intentionally. Decompression has
802 // unpredictable expansion ratio - a few compressed bytes could expand
803 // to overflow the output buffer. This keeps the code simple and correct.
804 // The underlying converter already buffers, so this doesn't cause syscalls.
805 reader.reset(1);
806
807 // Ensures m_strm's in/out pointers and counts are cleared on every exit
808 // path, so a throw cannot leave next_in pointing into reader's buffer
809 // after the reader is destroyed.
810 io_buf_guard buf_guard{*m_strm};
811
812 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
813 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
814
815 auto aim_output = (max_chunk / sizeof(internal_type) > to_max)
816 ? static_cast<decltype(m_strm->avail_out)>(to_max * sizeof(internal_type))
817 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
818
819 m_strm->avail_out = aim_output;
820 m_strm->next_out = reinterpret_cast<unsigned char*>(to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
821 auto ret = inflate(m_strm.get(), Z_NO_FLUSH);
822 zerr<true>("zlib_cvt::get fail", ret);
823
824 while (m_strm->avail_out && ret != Z_STREAM_END)
825 {
826 auto [ptr, len] = reader.get_buf(1);
827 if (len == 0) break;
828 m_strm->next_in = const_cast<unsigned char*>(reinterpret_cast<const unsigned char*>(ptr)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast,cppcoreguidelines-pro-type-const-cast)
829 m_strm->avail_in = 1;
830 // Snapshot before the call so we can prove zlib made progress on at
831 // least one axis (input consumed or output produced). The Z_OK
832 // contract guarantees this, but enforcing it here turns a
833 // hypothetical infinite loop into an explicit failure if the
834 // library ever diverges from its contract.
835 const auto prev_avail_in = m_strm->avail_in;
836 const auto prev_avail_out = m_strm->avail_out;
837 ret = inflate(m_strm.get(), Z_NO_FLUSH);
838 // Invariant: When output space is available, inflate() always consumes
839 // all provided input into its internal state, even if no output is
840 // produced yet. This assert guards against zlib misbehavior or memory
841 // corruption during development.
842 assert(m_strm->avail_in == 0);
843 zerr("zlib_cvt::get fail", ret);
844 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
845 throw cvt_error("zlib_cvt::get fail: zlib made no progress");
846 }
847
848 // Latch end-of-stream as soon as inflate() observes it, before any
849 // post-condition checks below. If a check throws afterwards, this
850 // ensures is_eof() still reports true and any future get_main call
851 // short-circuits to 0 instead of re-entering inflate() on a finished
852 // stream. The latch is a pure value write and cannot throw.
853 if (ret == Z_STREAM_END)
854 m_stream_ended = true;
855
856 // Exiting the loop with output space still available means the underlying
857 // kernel ran out of bytes before zlib reached Z_STREAM_END. The compressed
858 // stream is truncated and partial output would silently corrupt caller data.
859 if (ret != Z_STREAM_END && m_strm->avail_out != 0)
860 throw cvt_error("zlib_cvt::get fail: compressed stream truncated");
861
862 auto res = aim_output - m_strm->avail_out;
863 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
864
865 if (res % sizeof(internal_type))
866 throw cvt_error("zlib_cvt::get fail: partial sequence");
867
868 return res / sizeof(internal_type);
869 }

897 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
898 requires (cvt_cpt::support_put<KernelType>)
899 {
900 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
901 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
902
903 auto to = reinterpret_cast<const unsigned char*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
904
905 writer.reset(CHUNK);
906
907 // Ensures m_strm's in/out pointers and counts are cleared on every exit
908 // path, so a throw cannot leave next_in / next_out pointing into the
909 // caller's input or writer's buffer after they go out of scope.
910 io_buf_guard buf_guard{*m_strm};
911
912 while (to_size > 0)
913 {
914 auto aim_output = (max_chunk / sizeof(internal_type) > to_size)
915 ? static_cast<decltype(m_strm->avail_out)>(to_size * sizeof(internal_type))
916 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
917
918 size_t write_size = 0;
919 while (aim_output != write_size)
920 {
921 m_strm->next_in = const_cast<unsigned char*>(to + write_size); // NOLINT(cppcoreguidelines-pro-type-const-cast)
922 auto cur_put_size = static_cast<size_t>(aim_output - write_size);
923 m_strm->avail_in = static_cast<decltype(m_strm->avail_in)>(cur_put_size);
924 m_strm->next_out = reinterpret_cast<unsigned char*>(writer.put_buf(CHUNK)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
925 m_strm->avail_out = CHUNK;
926
927 // put_buf reserved CHUNK bytes directly inside the kernel's
928 // persistent output buffer (root_cvt / mem_device specializations
929 // of cvt_writer commit at put_buf, not at commit()). If anything
930 // below throws, the trailing avail_out bytes are uninitialized
931 // memory still inside the kernel's cursor range and would be
932 // flushed to the device on the next put / close_stream / dtor.
933 // Roll the tail back on the failure path so the kernel buffer
934 // ends exactly at the last valid deflate output byte.
935 try
936 {
937 // Snapshot before the call so we can prove zlib made progress on
938 // at least one axis (input consumed or output produced). The
939 // Z_OK contract guarantees this, but enforcing it here turns a
940 // hypothetical infinite loop into an explicit failure if the
941 // library ever diverges from its contract.
942 const auto prev_avail_in = m_strm->avail_in;
943 const auto prev_avail_out = m_strm->avail_out;
944 auto ret = deflate(m_strm.get(), Z_NO_FLUSH);
945 zerr("zlib_cvt::put fail", ret);
946 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
947 throw cvt_error("zlib_cvt::put fail: zlib made no progress");
948 write_size += cur_put_size - m_strm->avail_in;
949 }
950 catch (...)
951 {
952 // rollback's throw conditions (len==0, len>reserved) are
953 // unreachable here: we just reserved CHUNK and avail_out<=CHUNK.
954 // The catch is defensive cleanup-path policy - swallow any
955 // future regression in rollback rather than mask the original
956 // exception.
957 if (m_strm->avail_out > 0)
958 {
959 try { writer.rollback(m_strm->avail_out); } catch (...) {} // NOLINT(bugprone-empty-catch)
960 }
961 throw;
962 }
963
964 if (m_strm->avail_out)
965 writer.rollback(m_strm->avail_out);
966 }
967 to += aim_output;
968 to_size -= aim_output / sizeof(internal_type);
969 }
970 // commit() is the responsibility of abs_cvt::put.
971 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
972 }

995 void flush_impl()
996 requires (cvt_cpt::support_put<KernelType>)
997 {
998 if (m_sync_flush)
999 {
1000 io_buf_guard buf_guard{*m_strm};
1001
1002 std::array<external_type, CHUNK> local_buf{};
1003 m_strm->next_in = nullptr;
1004 m_strm->avail_in = 0;
1005 while (true)
1006 {
1007 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1008 m_strm->avail_out = CHUNK;
1009 zerr("zlib_cvt::flush_impl fail", deflate(m_strm.get(), Z_SYNC_FLUSH));
1010 const size_t written = CHUNK - m_strm->avail_out;
1011 if (written > 0)
1012 BT::m_kernel.put(local_buf.data(), written);
1013 if (m_strm->avail_out)
1014 break;
1015 }
1016 }

1066 template <bool IgnoreBufError = false>
1067 void zerr(const char* info, int ret)
1068 {
1069 switch (ret)
1070 {
1071 case Z_ERRNO:
1072 throw cvt_error(std::string(info) + ": zlib i/o error");
1073 case Z_STREAM_ERROR:
1074 // Z_STREAM_ERROR has two meanings depending on the call site:
1075 // deflateInit returns it for an invalid compression level argument;
1076 // inflate / deflate / inflateEnd / deflateEnd return it when zlib
1077 // detects an inconsistent internal stream state (often a symptom
1078 // of memory corruption, ABI mismatch, or concurrent access).
1079 throw cvt_error(std::string(info) + ": invalid argument or inconsistent stream state (Z_STREAM_ERROR)");
1080 case Z_DATA_ERROR:
1081 throw cvt_error(std::string(info) + ": invalid or incomplete deflate data");
1082 case Z_MEM_ERROR:
1083 throw cvt_error(std::string(info) + ": out of memory");
1084 case Z_VERSION_ERROR:
1085 throw cvt_error(std::string(info) + ": zlib version mismatch");
1086 case Z_BUF_ERROR:
1087 if constexpr (!IgnoreBufError)
1088 throw cvt_error(std::string(info) + ": no progress possible (Z_BUF_ERROR)");
1089 break;
1090 default:
1091 if (ret < 0)
1092 throw cvt_error(std::string(info) + ": unknown zlib error " + std::to_string(ret));
1093 }
