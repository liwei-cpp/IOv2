333 size_t sgetn(char_type* s, size_t n) requires (IsIn)
334 {
335 if ((s == nullptr) || (n == 0)) return 0;
336
337 if constexpr (IsOut)
338 switch_to_get();
339
340 size_t res = 0;
341 while (!m_read_buf.empty())
342 {
343 *s++ = m_read_buf.front();
344 m_read_buf.pop_front();
345 if (++res == n) break;
346 }
347
348 while (res < n)
349 {
350 size_t c = m_cvt.get(s, n - res);
351 assert(c <= n - res);
352 if (c == 0) break;
353 res += c;
354 s += c;
355 }
356 return res;
357 }

748 std::pair<device_type, std::exception_ptr> detach() noexcept
749 {
750 if constexpr (IsIn)
751 {
752 if (!m_read_buf.empty())
753 {
754 try
755 {
756 // On over-pushback (more put-back than actually read), tell()
757 // saturates at 0, so the device is handed back positioned at the
758 // stream origin. That is consistent with the logical read cursor
759 // being at 0; see tell()/sputbackc() and switch_to_put().
760 const size_t pos = tell();
761 m_cvt.seek(pos);
762 }
763 catch (...) // NOLINT(bugprone-empty-catch)
764 {
765 // Swallowed on purpose: a device that doesn't support positioning
766 // (e.g. a pipe/tty-backed stdin) can never honor this reposition,
767 // and that is an inherent property of the device, not an
768 // actionable failure - the caller could not have repositioned it
769 // either after getting it back. Reporting this error would make
770 // routine detach()/attach() cycles on such devices start throwing
771 // (e.g. sync_with_stdio() right after a formatted read has left
772 // one buffered lookahead character), even though nothing is
773 // actually broken; losing that lookahead character is the
774 // accepted, unavoidable cost for non-positionable devices.
775 }
776 m_read_buf.clear();
777 }
778 }
779 return m_cvt.detach();
