314 std::pair<bool, size_t> in_helper(const char*& from, const char* from_end,
315 TInt*& to, TInt* to_end)
316 {
317 if (std::greater<>{}(from, from_end) || std::greater<>{}(to, to_end)) [[unlikely]]
318 throw cvt_error("codecvt_kernel::in_helper fail: invalid pointer range");
319 clocale_user guard(m_inter_locale);
320 wchar_t wch = 0;
321 size_t i_count = 0;
322
323 const size_t to_max = to_end - to;
324 while (from < from_end && (i_count < to_max))
325 {
326 auto tmp_state = m_state;
327 size_t conv = mbrtowc(&wch, from, from_end - from, &tmp_state);
328 if (conv == static_cast<size_t>(-1))
329 return std::pair{false, i_count};
330 else if (conv == static_cast<size_t>(-2))
331 {
332 from = from_end;
333 m_state = tmp_state;
334 break;
335 }
336 else if (conv == 0)
337 {
338 // Find the actual byte length of the null character encoding
339 // before writing to output buffer to ensure consistency
340 size_t n = 1;
341 const auto max_n = static_cast<size_t>(from_end - from);
342 for (; n <= max_n; ++n)
343 {
344 auto tmp_state2(m_state);
345 if (mbrtowc(nullptr, from, n, &tmp_state2) == 0)
346 {
347 m_state = tmp_state2;
348 break;
349 }
350 }
351 if (n > max_n)
352 return std::pair{false, i_count};
353
354 // Only write to output after validation succeeds
355 *to++ = static_cast<TInt>(0);
356 from += n;
357 ++i_count;
358 }
359 else
360 {
361 // mbrtowc returning > 0 guarantees a complete character was converted;
362 // no partial character can exist in m_state at this point
363 m_state = tmp_state;
364 *to++ = wch;
365 from += conv;
366 ++i_count;
367 }
368 }
369
370 return std::pair{true, i_count};
371 }
