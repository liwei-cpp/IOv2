286 inline static const prefix_tree<CharT, std::string> s_timezone_tree =
287 []()
288 {
289 prefix_tree<CharT, std::string> res;
290
291 try
292 {
293 const auto& tzdb = std::chrono::get_tzdb();
294
295 // Collect all abbreviations (deduplicated across zones).
296 std::set<std::string> abbrevs;
297 for (const auto& zone : tzdb.zones)
298 {
299 std::string abbr = zone.get_info(std::chrono::sys_time<std::chrono::seconds>{}).abbrev;
300 if (!abbr.empty())
301 abbrevs.insert(std::move(abbr));
302 }
303
304 // Resolve each abbreviation: if locate_zone succeeds (e.g. "EST"
305 // is an IANA link), store the canonical name; otherwise mark "*".
306 for (const auto& abbr : abbrevs)
307 {
308 try
309 {
310 std::string canonical{std::chrono::locate_zone(abbr)->name()};
311 res.add(abbr.begin(), abbr.end(), canonical);
312 }
313 catch (...)
314 {
315 res.add(abbr.begin(), abbr.end(), "*" + abbr);
316 }
317 }
318
319 // Add full zone names that differ from their own abbreviation.
320 for (const auto& zone : tzdb.zones)
321 {
322 std::string full_name{zone.name()};
323 std::string abbr_name = zone.get_info(std::chrono::sys_time<std::chrono::seconds>{}).abbrev;
324 if (full_name != abbr_name)
325 res.add(full_name.begin(), full_name.end(), full_name);
326 }
327 }
328 catch (...) // NOLINT(bugprone-empty-catch)
329 {
330 // tz database unavailable or malformed at static-init time:
331 // degrade to an empty/partial tree instead of letting the
332 // exception escape this static initializer (which would call
333 // std::terminate). %Z then simply fails to match and reports a
334 // catchable stream_error at parse time, matching the defensive
335 // behaviour of time_zone_parse_helper.
336 }
337 return res;
338 }();
339 };

984 static std::vector<era_entry> parse_glibc_era_entries()
985 {
986 std::vector<era_entry> items;
987
988 const int32_t era_item_num = static_cast<int32_t>(
989 reinterpret_cast<uintptr_t>(nl_langinfo(_NL_TIME_ERA_NUM_ENTRIES)));
990 if (era_item_num <= 0)
991 return items;
992
993 items.reserve(era_item_num);
994 const char *ptr = reinterpret_cast<const char*>(nl_langinfo(_NL_TIME_ERA_ENTRIES));
995 for (int32_t cnt = 0; cnt < era_item_num; ++cnt)
996 {
997 const char *base_ptr = ptr;
998 era_entry cur_entry;
999
1000 int32_t buf[8];
1001 std::memcpy(static_cast<void*>(buf), static_cast<const void*>(ptr), sizeof(int32_t) * 8);
1002 ptr += sizeof(uint32_t) * 8;
1003
1004 if (buf[2] > std::numeric_limits<int32_t>::max() - 1900)
1005 cur_entry.from_year = std::numeric_limits<int32_t>::max();
1006 else
1007 cur_entry.from_year = buf[2] + 1900;
1008 cur_entry.from_month = buf[3] + 1;
1009 cur_entry.from_day = buf[4];
1010 if (buf[5] > std::numeric_limits<int32_t>::max() - 1900)
1011 {
1012 cur_entry.to_year = std::numeric_limits<int32_t>::max();
1013 cur_entry.to_month = 12;
1014 cur_entry.to_day = 31;
1015 }
1016 else
1017 {
1018 cur_entry.to_year = buf[5] + 1900;
1019 cur_entry.to_month = buf[6] + 1;
1020 cur_entry.to_day = buf[7];
1021 }
1022
1023 // Normalise direction to match glibc's absolute_direction (era.c ~line 98).
1024 // Forward era (from <= to): marker maps directly.
1025 // Backward era (from > to): marker is flipped so that the linear formula
1026 // era_year = offset + (calendar - from_year) * direction
1027 // stays consistent; see OUTPUT INVARIANTS above for the full rationale.
1028 if (TimeioHelper::era_small_or_equal(cur_entry.from_year, cur_entry.from_month, cur_entry.from_day,
1029 cur_entry.to_year, cur_entry.to_month, cur_entry.to_day))
1030 {
1031 if (buf[0] == (uint32_t) '+') cur_entry.direction = 1;
1032 else cur_entry.direction = -1;
1033 }
1034 else
1035 {
1036 if (buf[0] == (uint32_t) '+') cur_entry.direction = -1;
1037 else cur_entry.direction = 1;
1038 }
1039 cur_entry.offset = buf[1];
1040
1041 cur_entry.name = ptr; ptr = strchr(ptr, '\0') + 1;
1042 cur_entry.format = ptr; ptr = strchr(ptr, '\0') + 1;
1043
1044 // skip wchar_t name and format
1045 ptr += 3 - (((ptr - base_ptr) + 3) & 3);
1046 ptr = reinterpret_cast<const char*>(wcschr(reinterpret_cast<const wchar_t*>(ptr), L'\0') + 1);
1047 ptr = reinterpret_cast<const char*>(wcschr(reinterpret_cast<const wchar_t*>(ptr), L'\0') + 1);
1048
1049 items.push_back(std::move(cur_entry));
1050 }
1051 return items;
1052 }
