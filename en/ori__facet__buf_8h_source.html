74 template <typename TF>
75 std::shared_ptr<abs_ft> try_get(const std::string& name)
76 {
77 const std::size_t id = TF::id();
78
79 {
80 std::lock_guard guard(m_mutex);
81 if (auto cached = m_cache[id].get(name))
82 return *cached;
83 }
84
85 // Construct the facet *outside* the lock: a facet constructor may be
86 // expensive and -- crucially -- may itself call back into ori_facet_buf, so
87 // building under m_mutex would both serialize all construction and risk a
88 // self-deadlock on this non-recursive mutex. Re-lock only to insert, and
89 // double-check: if another thread (or a re-entrant call) inserted the same
90 // (id, name) meanwhile, return that entry and discard the object built here.
91 auto obj = std::make_shared<TF>(name);
92
93 std::lock_guard guard(m_mutex);
94 auto& sub_cache = m_cache[id];
95 if (auto cached = sub_cache.get(name))
96 return *cached;
97 // The get() above missed under this same lock, so the key is absent and put()
98 // simply inserts (no existing value to overwrite).
99 sub_cache.put(name, obj);
100 return obj;
101 }

217 static std::string resolve_locale(const char* category_var)
218 {
219 std::string name = "C";
220 if (const char* all = std::getenv("LC_ALL"); all && *all)
221 name = all;
222 else if (const char* cat = std::getenv(category_var); cat && *cat)
223 name = cat;
224 else if (const char* lang = std::getenv("LANG"); lang && *lang)
225 name = lang;
226
227 // Validate the resolved name against the C library: clocale_wrapper builds a
228 // locale_t via newlocale and throws if the name cannot be instantiated. An
229 // unusable name (e.g. a bogus LANG) falls back to the neutral "C". Done here,
230 // once per category at construction, so locale_name stays a plain O(1) read.
231 try
232 {
233 clocale_wrapper probe(name.c_str());
234 return name;
235 }
236 catch (...)
237 {
238 return "C";
239 }
240 }

244 // (see resolve_locale / locale_name). Written once during static init, then
245 // immutable -- safe to read concurrently without locking.
246 std::string m_ctype;
247 std::string m_collate;
248 std::string m_monetary;

252 // Upper bound on the number of distinct entries cached *per facet type* (per name
253 // for m_cache, per (domain, lang, cvt) for m_msg_cache). Past this, the lru_cache
254 // evicts the least-recently-used entry, bounding memory under workloads that derive
255 // locale names / message keys from variable (e.g. external) input. The outer map's
256 // key space -- one entry per facet type id -- is already bounded by the program's
257 // instantiated facet types. Eviction only forces a rebuild of an equivalent,
258 // immutable facet on the next miss; nothing relies on facet identity, so this is
259 // purely an interning optimization, not a correctness guarantee.
260 static constexpr std::size_t s_cache_capacity = 256;
261
262 std::unordered_map<std::size_t, lru_cache<std::string, std::shared_ptr<abs_ft>, s_cache_capacity>> m_cache;
263 std::unordered_map<std::size_t, lru_cache<detail::msg_key, std::shared_ptr<abs_ft>, s_cache_capacity>> m_msg_cache;
264 // Guards m_cache / m_msg_cache. Facet construction in try_get happens *outside*
265 // this lock (it is taken only to look up and to insert), so the mutex is never
266 // held across a facet constructor. That keeps construction unserialized and
267 // makes a re-entrant facet ctor safe even though the mutex is non-recursive.
268 //
269 // The lock must also cover every *lookup*: lru_cache::get() reorders the LRU list
270 // (it is a mutating "touch"), so concurrent gets would race -- there is no
271 // lock-free read path here.
272 std::mutex m_mutex;
