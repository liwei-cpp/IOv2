25 std::shared_ptr<abs_ft> try_get(const std::string& name)
26 {
27 const std::size_t id = TF::id();
28
29 {
30 std::lock_guard guard(m_mutex);
31 auto& sub_cache = m_cache[id];
32 auto it = sub_cache.find(name);
33 if (it != sub_cache.end())
34 return it->second;
35 }
36
37 // Construct the facet *outside* the lock: a facet constructor may be
38 // expensive and -- crucially -- may itself call back into ori_facet_buf, so
39 // building under m_mutex would both serialize all construction and risk a
40 // self-deadlock on this non-recursive mutex. Re-lock only to insert, and
41 // double-check via emplace: if another thread (or a re-entrant call)
42 // inserted the same (id, name) meanwhile, emplace no-ops and returns the
43 // existing entry, so the object built here is simply discarded.
44 auto obj = std::make_shared<TF>(name);
45
46 std::lock_guard guard(m_mutex);
47 auto& sub_cache = m_cache[id];
48 return sub_cache.emplace(name, std::move(obj)).first->second;
49 }

178 // Validate the resolved name against the C library: clocale_wrapper builds a
179 // locale_t via newlocale and throws if the name cannot be instantiated. An
180 // unusable name (e.g. a bogus LANG) falls back to the neutral "C". Done here,
181 // once per category at construction, so locale_name stays a plain O(1) read.
182 try
183 {
184 clocale_wrapper probe(name.c_str());

216 seed ^= h(std::get<2>(k)) + 0x9e3779b9 + (seed << 6) + (seed >> 2);

218 }
