65template <>
66struct hash<IOv2::detail::msg_key>
67{
68 std::size_t operator()(const IOv2::detail::msg_key& k) const noexcept
69 {
70 // Golden-ratio constant sized to std::size_t (2^bits / phi). The 64-bit
71 // 0x9e3779b97f4a7c15 mixes the high half of a 64-bit seed that the 32-bit
72 // 0x9e3779b9 leaves largely untouched; chosen at compile time so the mixer
73 // matches the actual width of std::size_t on every platform. (The unselected
74 // ternary branch is a well-defined explicit truncation, valid on either width.)
75 constexpr std::size_t golden = sizeof(std::size_t) >= 8
76 ? static_cast<std::size_t>(0x9e3779b97f4a7c15ULL)
77 : static_cast<std::size_t>(0x9e3779b9UL);
78
79 const std::hash<std::string> h;
80 std::size_t seed = h(k.domain);
81 seed ^= h(k.lang) + golden + (seed << 6) + (seed >> 2);
82 seed ^= h(k.dirname) + golden + (seed << 6) + (seed >> 2);
83 seed ^= h(k.cvt) + golden + (seed << 6) + (seed >> 2);
84 return seed;
85 }
86};

96 template <typename TF>
97 std::shared_ptr<abs_ft> try_get(const std::string& name)
98 {
99 const facet_id_t id = TF::id();
100
101 {
102 std::scoped_lock guard(m_mutex);
103 if (auto it = m_cache.find(id); it != m_cache.end())
104 {
105 if (auto cached = it->second.get(name))
106 return *cached;
107 }
108 }
109
110 // Construct the facet *outside* the lock: a facet constructor may be
111 // expensive and -- crucially -- may itself call back into ori_facet_buf, so
112 // building under m_mutex would both serialize all construction and risk a
113 // self-deadlock on this non-recursive mutex. Re-lock only to insert, and
114 // double-check: if another thread (or a re-entrant call) inserted the same
115 // (id, name) meanwhile, return that entry and discard the object built here.
116 auto obj = std::make_shared<TF>(name);
117
118 std::scoped_lock guard(m_mutex);
119 auto& sub_cache = m_cache[id];
120 if (auto cached = sub_cache.get(name))
121 return *cached;
122 // The get() above missed under this same lock, so the key is absent and put()
123 // simply inserts (no existing value to overwrite).
124 sub_cache.put(name, obj);
125 return obj;
126 }

272 static std::string resolve_locale(const char* category_var)
273 {
274 std::string name = "C";
275 if (const char* all = std::getenv("LC_ALL"); all && *all)
276 name = all;
277 else if (const char* cat = std::getenv(category_var); cat && *cat)
278 name = cat;
279 else if (const char* lang = std::getenv("LANG"); lang && *lang)
280 name = lang;
281
282 // Validate the resolved name against the C library: clocale_wrapper builds a
283 // locale_t via newlocale and throws if the name cannot be instantiated. An
284 // unusable name (e.g. a bogus LANG) falls back to the neutral "C". Done here,
285 // once per category at construction, so locale_name stays a plain O(1) read.
286 try
287 {
288 clocale_wrapper probe(name.c_str());
289 return name;
290 }
291 catch (...)
292 {
293 return "C";
294 }
295 }

307 // Upper bound on the number of distinct entries cached *per facet type* (per name
308 // for m_cache, per (domain, lang, dirname, cvt) for m_msg_cache). Past this, the lru_cache
309 // evicts the least-recently-used entry, bounding memory under workloads that derive
310 // locale names / message keys from variable (e.g. external) input. The outer map's
311 // key space -- one entry per facet type id -- is already bounded by the program's
312 // instantiated facet types. Eviction only forces a rebuild of an equivalent,
313 // immutable facet on the next miss; nothing relies on facet identity, so this is
314 // purely an interning optimization, not a correctness guarantee.
315 static constexpr std::size_t s_cache_capacity = 256;
316
317 std::unordered_map<facet_id_t, lru_cache<std::string, std::shared_ptr<abs_ft>, s_cache_capacity>> m_cache;
318 std::unordered_map<facet_id_t, lru_cache<detail::msg_key, std::shared_ptr<abs_ft>, s_cache_capacity>> m_msg_cache;
319 // Guards m_cache / m_msg_cache. Facet construction in try_get happens *outside*
320 // this lock (it is taken only to look up and to insert), so the mutex is never
321 // held across a facet constructor. That keeps construction unserialized and
322 // makes a re-entrant facet ctor safe even though the mutex is non-recursive.
323 //
324 // The lock must also cover every *lookup*: lru_cache::get() reorders the LRU list
325 // (it is a mutating "touch"), so concurrent gets would race -- there is no
326 // lock-free read path here.
327 std::mutex m_mutex;
