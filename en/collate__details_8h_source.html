179 std::vector<CharT> buf1; bool extra_eos1 = false;
180 std::vector<CharT> buf2; bool extra_eos2 = false;
181
182 clocale_user guard(m_inter_locale);
183
184 while ((low1 != high1) && (low2 != high2))
185 {
186 const CharT* cl1 = low1;
187 if (auto ch1 = std::find(low1, high1, static_cast<CharT>(0)); ch1 == high1)
188 {
189 auto data_len = high1 - low1;

192 buf1[data_len] = static_cast<CharT>(0);
193 extra_eos1 = true;
194 cl1 = buf1.data();
195 low1 = high1;
196 }
197 else low1 = ch1 + 1;
198
199 const CharT* cl2 = low2;
200 if (auto ch2 = std::find(low2, high2, static_cast<CharT>(0)); ch2 == high2)

205 buf2[data_len] = static_cast<CharT>(0);

212 int c_res = 0;
213 if constexpr (std::is_same_v<CharT, char>)

223 c_res = std::wcscoll(reinterpret_cast<const wchar_t*>(cl1), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)

227 }
228 else

230
231 if (c_res < 0) return std::strong_ordering::less;
232 if (c_res > 0) return std::strong_ordering::greater;

236 if (low2 != high2) return std::strong_ordering::less;
237 if (extra_eos1 && !extra_eos2) return std::strong_ordering::less;

356 virtual size_t transform(const CharT* low, const CharT* high, CharT* dest, size_t mx_len = 0) const
357 {
358 size_t trans_count = 0;
359 // All buffers are hoisted out of the loop so resize() reuses their
360 // capacity instead of reallocating per segment.
361 // buf : input staging — null-terminated copy of a segment that has
362 // no embedded '\0' (so cur can point at a terminated string).
363 // buf2 : output staging for strxfrm/wcsxfrm.
364 std::vector<CharT> buf;
365 std::vector<CharT> buf2;
366 bool extra_eos = false;
367
368 clocale_user guard(m_inter_locale);
369 while ((low != high) && ((mx_len == 0) || (trans_count < mx_len)))
370 {
371 const CharT* cur = low;
372 if (auto next = std::find(low, high, static_cast<CharT>(0)); next == high)
373 {
374 auto data_len = high - low;
375 buf.resize(data_len + 1 + SIMD_PADDING_BYTES / sizeof(CharT));
376 std::copy(low, high, buf.data());
377 buf[data_len] = static_cast<CharT>(0);
378 cur = buf.data();
379 low = high;
380 extra_eos = true;
381 }
382 else
383 low = next + 1;
384
385 {
386 // strxfrm/wcsxfrm always append a terminating '\0' (writing
387 // cur_trans + 1 elements). Transform into the reused buffer
388 // first, then copy exactly cur_trans elements to dest, so the
389 // appended '\0' is discarded. Writing it straight into dest
390 // would overflow the caller's buffer by one on the trailing
391 // segment, for which transform_length only accounts for
392 // cur_trans (no terminator).
393 size_t trans_len = 0;
394 if constexpr (std::is_same_v<CharT, char>)
395 trans_len = strxfrm(nullptr, cur, 0);
396 else if constexpr (std::is_same_v<CharT, wchar_t>)
397 trans_len = wcsxfrm(nullptr, cur, 0);
398 else if constexpr (std::is_same_v<CharT, char8_t>)
399 trans_len = strxfrm(nullptr, reinterpret_cast<const char*>(cur), 0); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
400 else if constexpr ((std::is_same_v<CharT, char32_t> &&
401 wchar_t_is_utf32))
402 trans_len = wcsxfrm(nullptr, reinterpret_cast<const wchar_t*>(cur), 0); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
403 else
404 static_assert(dependent_false_v<CharT>, "collate_conf::transform is not implemented.");
405
406 if (trans_len == xfrm_failed)
407 throw cvt_error("collate_conf::transform: strxfrm/wcsxfrm failed");
408 buf2.resize(trans_len + 1);
409
410 size_t cur_trans = 0;
411 if constexpr (std::is_same_v<CharT, char>)
412 cur_trans = strxfrm(buf2.data(), cur, buf2.size());
413 else if constexpr (std::is_same_v<CharT, wchar_t>)
414 cur_trans = wcsxfrm(buf2.data(), cur, buf2.size());
415 else if constexpr (std::is_same_v<CharT, char8_t>)
416 cur_trans = strxfrm(reinterpret_cast<char*>(buf2.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
417 reinterpret_cast<const char*>(cur), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
418 buf2.size());
419 else if constexpr ((std::is_same_v<CharT, char32_t> &&
420 wchar_t_is_utf32))
421 cur_trans = wcsxfrm(reinterpret_cast<wchar_t*>(buf2.data()), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
422 reinterpret_cast<const wchar_t*>(cur), // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
423 buf2.size());
424 else
425 static_assert(dependent_false_v<CharT>, "collate_conf::transform is not implemented.");
426
427 if (cur_trans == xfrm_failed)
428 throw cvt_error("collate_conf::transform: strxfrm/wcsxfrm failed");
429 if (mx_len != 0)
430 cur_trans = std::min(cur_trans, mx_len - trans_count);
431 dest = std::copy(buf2.data(), buf2.data() + cur_trans, dest);
432 trans_count += cur_trans;
433 }
434
435 if ((!extra_eos) && ((mx_len == 0) || (trans_count < mx_len)))
436 {
437 *dest++ = '\0';
438 ++trans_count;
439 }
440 }
441 return trans_count;
442 }
