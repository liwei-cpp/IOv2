104 inline char string_to_char_convert(const std::string& narrow,
105 const std::string& locale_name)
106 {
107 if (narrow.empty()) return '\0';
108 if (narrow.size() == 1) return narrow[0];
109
110 // Convert (multi-byte) narrow to wchar_t then narrow down
111 std::wstring wide_str = detail::to_wstring(narrow.c_str(), locale_name);
112 if (wide_str.empty()) return '\0';
113
114 // Narrow the first wide character back to a single byte in the target
115 // locale. This reproduces ctype_conf<wchar_t>::narrow() (wctob() under
116 // a thread-local locale switch) directly, instead of building a whole
117 // ctype_conf<wchar_t> facet whose 256-entry widen table and wctype
118 // masks this path never reads. The out_of_wchar_range guard used there
119 // is unnecessary here: the value is a wchar_t and is therefore always
120 // representable as wchar_t.
121 clocale_wrapper loc(locale_name.c_str());
122 clocale_user guard(loc);
123 const int c = wctob(static_cast<wint_t>(wide_str[0]));
124 return c == EOF ? '\0' : static_cast<char>(c);
125 }

300 inline CharT* add_grouping(CharT* s, CharT sep, const std::vector<uint8_t>& grouping,
301 const CharT* first, const CharT* last)
302 {
303 assert(!grouping.empty());
304 assert(first <= last);
305
306 size_t idx = 0;
307 size_t ctr = 0;
308 const size_t max_idx = grouping.size() - 1;
309
310 while (last - first > grouping[idx]
311 && (grouping[idx] > 0))
312 {
313 last -= grouping[idx];
314 idx < max_idx ? ++idx : ++ctr;
315 }
316
317 s = std::copy(first, last, s);
318 first = last;
319
320 const uint8_t n = grouping[idx];
321 while (ctr--)
322 {
323 *s++ = sep;
324 s = std::copy_n(first, n, s);
325 first += n;
326 }
327
328 while (idx--)
329 {
330 *s++ = sep;
331 const uint8_t m = grouping[idx];
332 s = std::copy_n(first, m, s);
333 first += m;
334 }
335 return s;
336 }

385 inline void adjust_grouping(std::vector<uint8_t>& grouping)
386 {
387 // Step 1: POSIX "0 = repeat previous" → drop the 0 and the tail.
388 auto zero = std::ranges::find(grouping, uint8_t{0});
389 grouping.erase(zero, grouping.end());
390
391 // Step 2: POSIX CHAR_MAX (= stop) → internal 0 sentinel, plus
392 // canonical-form cleanup. After step 1 there are no 0s in the vector,
393 // so the FIRST CHAR_MAX is the only point at which we introduce one,
394 // and three cases fully cover the outcome:
395 // - at begin(): grouping would start with the stop sentinel,
396 // i.e. no usable leading group ⇒ discard.
397 // (This also subsumes the empty-vector case left
398 // by step 1, since find on an empty range returns
399 // end() == begin().)
400 // - in the tail: rewrite to 0 and drop everything after it; those
401 // bytes are dead code anyway because add_grouping
402 // short-circuits on 0.
403 // - not found: nothing to do.
404 constexpr auto posix_sentinel =
405 static_cast<uint8_t>(std::numeric_limits<char>::max());
406 auto stop = std::ranges::find(grouping, posix_sentinel);
407 if (stop == grouping.begin())
408 grouping.clear();
409 else if (stop != grouping.end())
410 {
411 *stop = 0;
412 grouping.erase(std::next(stop), grouping.end());
413 }
414 }

462 inline bool verify_grouping(const std::vector<uint8_t>& grouping,
463 const std::vector<uint8_t>& grouping_tmp)
464 {
465 assert(!grouping.empty());
466 assert(!grouping_tmp.empty());
467
468 size_t i = grouping_tmp.size() - 1;
469 const size_t min_val = std::min(i, grouping.size() - 1);
470 bool test = true;
471
472 // Parsed number groupings have to match the
473 // numpunct::grouping string exactly, starting at the
474 // right-most point of the parsed sequence of elements ...
475 for (size_t j = 0; j < min_val && test; --i, ++j)
476 test = grouping_tmp[i] == grouping[j];
477 for (; i && test; --i)
478 test = grouping_tmp[i] == grouping[min_val];
479 // ... but the first parsed grouping can be <= numpunct grouping.
480 // Skip this check when grouping[min_val] is 0 (stop sentinel,
481 // no upper bound on the first parsed group).
482 if (grouping[min_val] > 0)
483 test &= grouping_tmp[0] <= grouping[min_val];
484 return test;
485 }
