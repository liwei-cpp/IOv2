16 // to a narrow char in the given locale, returning '\0' if conversion

18 //
19 // Preconditions (not validated):
20 // - `ptr` is non-null.
21 // - `ptr` points to a NUL-terminated C string.
22 inline char string_to_char_convert(const char* ptr, const std::string& locale_name)
23 {
24 assert(ptr != nullptr);

39 // Convert the first character of a (possibly multibyte) narrow string
40 // `narrow` to a wide character `CharT` in the given locale, returning
41 // `default_char` when `narrow` is empty or conversion produces no
42 // characters.
43 //
44 // Suitable for fields read out of `lconv` / `nl_langinfo()` that
45 // semantically represent a single user-visible character (e.g.
46 // decimal_point, thousands_sep, mon_decimal_point, mon_thousands_sep).
47 //
48 // Why `const std::string&` and not `const char*`: `lconv*` and
49 // `nl_langinfo()` pointers may be invalidated by any setlocale() /
50 // uselocale() call. The conversion below transitively performs such
51 // calls (via detail::to_wstring/to_u32string), so the input must be a
52 // caller-owned snapshot rather than a borrowed pointer into the
53 // libc-owned locale data. Taking std::string makes this contract
54 // unmissable at every call site.
55 template <typename CharT>
56 requires std::is_same_v<CharT, wchar_t> ||
57 (std::is_same_v<CharT, char32_t> &&
58 (sizeof(char32_t) == sizeof(wchar_t)) &&
59 (static_cast<wchar_t>(U'李') == L'李') &&
60 (static_cast<char32_t>(L'伟') == U'伟'))
61 inline CharT string_to_widechar_convert(const std::string& narrow,
62 const std::string& locale_name,
63 CharT default_char)
64 {
65 if (narrow.empty()) return default_char;
66 if constexpr (std::is_same_v<CharT, wchar_t>)
67 {
68 auto wide = detail::to_wstring(narrow.c_str(), locale_name);
69 return wide.empty() ? default_char : wide[0];
70 }
71 else
72 {
73 auto wide = detail::to_u32string(narrow.c_str(), locale_name);
74 return wide.empty() ? default_char : wide[0];
75 }
76 }

78 // Internal helper function.
79 //
80 // Preconditions (not validated):
81 // - `grouping` is non-empty (only the first precondition is asserted).
82 // - `[first, last)` is a valid range, i.e. `first <= last`. Passing
83 // `first > last` causes the inner copy loop to never terminate and
84 // write past the output buffer.
85 // - `s` points to an output buffer with sufficient capacity to hold
86 // `(last - first)` characters plus one separator per non-leading
87 // group; callers are responsible for sizing the buffer.
88 template <typename CharT>
89 inline CharT* add_grouping(CharT* s, CharT sep, const std::vector<uint8_t>& grouping,
90 const CharT* first, const CharT* last)
91 {
92 assert(!grouping.empty());
93 assert(first <= last);
94
95 size_t idx = 0;
96 size_t ctr = 0;
97 const size_t max_idx = grouping.size() - 1;
98
99 while (last - first > grouping[idx]
100 && (grouping[idx] > 0))
101 {
102 last -= grouping[idx];
103 idx < max_idx ? ++idx : ++ctr;
104 }
105
106 s = std::copy(first, last, s);
107 first = last;
108
109 const uint8_t n = grouping[idx];
110 while (ctr--)
111 {
112 *s++ = sep;
113 s = std::copy_n(first, n, s);
114 first += n;
115 }
116
117 while (idx--)
118 {
119 *s++ = sep;
120 const uint8_t m = grouping[idx];
121 s = std::copy_n(first, m, s);
122 first += m;
123 }
124 return s;
125 }

127 // Normalise a raw POSIX-style grouping vector to the internal convention,
128 // or discard it if it carries no usable constraint.
129 //
130 // Internal convention (uint8_t, platform-independent):
131 // 1–255 — explicit group size.
132 // 0 — stop: no further grouping after this point (the ONLY sentinel).
133 // The LAST element (if in 1–255) is implicitly repeated for all
134 // remaining groups. If the last element is 0, the explicit stop wins.
135 //
136 // Conversions applied to raw POSIX input:
137 // - POSIX "0 = repeat previous": the 0 and everything after it is
138 // dropped. POSIX [N1, ..., Nk, 0, ...] becomes [N1, ..., Nk] in our
139 // convention, because last-element implicit repeat already covers
140 // that semantic exactly.
141 // - POSIX CHAR_MAX (= stop): rewritten to internal 0. CHAR_MAX is 127
142 // on signed-char platforms and 255 on unsigned-char platforms;
143 // callers do not need to know the platform's char signedness after
144 // this step.
145 //
146 // Order matters: stripping POSIX "0 = repeat" MUST happen before
147 // CHAR_MAX → 0 rewrite, otherwise our newly-introduced internal stop
148 // would be mistaken for a POSIX repeat-previous marker and dropped.
149 inline void adjust_grouping(std::vector<uint8_t>& grouping)
150 {
151 // Step 1: POSIX "0 = repeat previous" → drop the 0 and the tail.
152 auto zero = std::find(grouping.begin(), grouping.end(), uint8_t{0});
153 grouping.erase(zero, grouping.end());
154
155 // Step 2: POSIX CHAR_MAX (= stop) → internal 0 sentinel, plus
156 // canonical-form cleanup. After step 1 there are no 0s in the vector,
157 // so the FIRST CHAR_MAX is the only point at which we introduce one,
158 // and three cases fully cover the outcome:
159 // - at begin(): grouping would start with the stop sentinel,
160 // i.e. no usable leading group ⇒ discard.
161 // (This also subsumes the empty-vector case left
162 // by step 1, since find on an empty range returns
163 // end() == begin().)
164 // - in the tail: rewrite to 0 and drop everything after it; those
165 // bytes are dead code anyway because add_grouping
166 // short-circuits on 0.
167 // - not found: nothing to do.
168 constexpr uint8_t posix_sentinel =
169 static_cast<uint8_t>(std::numeric_limits<char>::max());
170 auto stop = std::find(grouping.begin(), grouping.end(), posix_sentinel);
171 if (stop == grouping.begin())
172 grouping.clear();
173 else if (stop != grouping.end())
174 {
175 *stop = 0;
176 grouping.erase(std::next(stop), grouping.end());
177 }
178 }

180 // Internal helper function. Callers are responsible for ensuring that both
181 // grouping and grouping_tmp are non-empty; this function does not validate
182 // those preconditions. Callers guarantee grouping is non-empty because
183 // grouping_tmp is only populated inside !grouping.empty() branches, so
184 // the non-emptiness of grouping_tmp implies the non-emptiness of grouping.
185 //
186 // grouping uses the internal uint8_t convention established by
187 // adjust_grouping(): 1–255 are explicit group sizes, 0 is the sole
188 // "stop" sentinel, and the last 1–255 element is implicitly repeated
189 // for all remaining groups.
190 inline bool verify_grouping(const std::vector<uint8_t>& grouping,
191 const std::vector<uint8_t>& grouping_tmp)
192 {
193 assert(!grouping.empty());
194 assert(!grouping_tmp.empty());
195
196 size_t i = grouping_tmp.size() - 1;
197 const size_t min_val = std::min(i, grouping.size() - 1);
198 bool test = true;
199
200 // Parsed number groupings have to match the
201 // numpunct::grouping string exactly, starting at the
202 // right-most point of the parsed sequence of elements ...
203 for (size_t j = 0; j < min_val && test; --i, ++j)
204 test = grouping_tmp[i] == grouping[j];
205 for (; i && test; --i)
206 test = grouping_tmp[i] == grouping[min_val];
207 // ... but the first parsed grouping can be <= numpunct grouping.
208 // Skip this check when grouping[min_val] is 0 (stop sentinel,
209 // no upper bound on the first parsed group).
210 if (grouping[min_val] > 0)
211 test &= grouping_tmp[0] <= grouping[min_val];
212 return test;
213 }
