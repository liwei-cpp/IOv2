20 // Convert the first character of a (possibly multibyte) C string `ptr`
21 // to a narrow char in the given locale, returning '\0' if conversion
22 // is not representable.
23 //
24 // Preconditions (not validated):
25 // - `ptr` is non-null.
26 // - `ptr` points to a NUL-terminated C string.
27 inline char string_to_char_convert(const char* ptr, const std::string& locale_name)
28 {
29 assert(ptr != nullptr);
30 if (*ptr == '\0')
31 return '\0';
32 if (ptr[1] == '\0')
33 return *ptr;
34
35 // Convert char to wchar_t then narrow down
36 std::wstring wide_str = detail::to_wstring(ptr, locale_name);
37 if (wide_str.empty()) return '\0';

44 // Internal helper function.
45 //
46 // Preconditions (not validated):
47 // - `grouping` is non-empty (only the first precondition is asserted).
48 // - `[first, last)` is a valid range, i.e. `first <= last`. Passing
49 // `first > last` causes the inner copy loop to never terminate and
50 // write past the output buffer.
51 // - `s` points to an output buffer with sufficient capacity to hold
52 // `(last - first)` characters plus one separator per non-leading
53 // group; callers are responsible for sizing the buffer.
54 template <typename CharT>
55 inline CharT* add_grouping(CharT* s, CharT sep, const std::vector<uint8_t>& grouping,
56 const CharT* first, const CharT* last)
57 {
58 assert(!grouping.empty());
59 assert(first <= last);
60
61 size_t idx = 0;
62 size_t ctr = 0;
63 const size_t max_idx = grouping.size() - 1;
64
65 while (last - first > grouping[idx]
66 && (grouping[idx] > 0))
67 {
68 last -= grouping[idx];
69 idx < max_idx ? ++idx : ++ctr;
70 }
71
72 s = std::copy(first, last, s);
73 first = last;
74
75 const uint8_t n = grouping[idx];
76 while (ctr--)
77 {
78 *s++ = sep;
79 s = std::copy_n(first, n, s);
80 first += n;
81 }
82
83 while (idx--)
84 {
85 *s++ = sep;
86 const uint8_t m = grouping[idx];
87 s = std::copy_n(first, m, s);
88 first += m;
89 }
90 return s;
91 }

93 // Internal helper function. Callers are responsible for ensuring that both
94 // grouping and grouping_tmp are non-empty; this function does not validate
95 // those preconditions. Callers guarantee grouping is non-empty because
96 // grouping_tmp is only populated inside !grouping.empty() branches, so
97 // the non-emptiness of grouping_tmp implies the non-emptiness of grouping.
98 inline bool verify_grouping(const std::vector<uint8_t>& grouping,
99 const std::string& grouping_tmp)
100 {
101 assert(!grouping.empty());
102 assert(!grouping_tmp.empty());
103
104 size_t i = grouping_tmp.size() - 1;
105 const size_t min_val = std::min(i, grouping.size() - 1);
106 bool test = true;
107
108 // Parsed number groupings have to match the
109 // numpunct::grouping string exactly, starting at the
110 // right-most point of the parsed sequence of elements ...
111 for (size_t j = 0; j < min_val && test; --i, ++j)
112 test = grouping_tmp[i] == grouping[j];
113 for (; i && test; --i)
114 test = grouping_tmp[i] == grouping[min_val];
115 // ... but the first parsed grouping can be <= numpunct
116 // grouping (only do the check if the numpunct char is > 0
117 // because <= 0 means any size is ok).
118 if (static_cast<signed char>(grouping[min_val]) > 0
119 && grouping[min_val] != std::numeric_limits<char>::max())
120 test &= grouping_tmp[0] <= grouping[min_val];
121 return test;
122 }
