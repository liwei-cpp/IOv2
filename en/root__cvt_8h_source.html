471 void main_cont_beg()
472 {
473 // bos() must be called before main_cont_beg() to establish io_status
474 assert(m_io_status != io_status::neutral);
475
476 if constexpr (dev_cpt::support_put<device_type>)
477 {

483 {
484 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
485 // which advances the device position by the amount read. Consuming data
486 // from the buffer only moves m_buf_cur forward, never increasing the
487 // buffered amount beyond what was read from the device.
488 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
489 assert(m_device.dtell() >= buffered);

757 [[nodiscard]] size_t tell() const
758 requires (dev_cpt::support_positioning<device_type>)
759 {
760 const size_t device_tell = m_device.dtell();
761
762 switch(m_io_status)
763 {
764 case io_status::output:
765 {
766 // Invariant: device_tell >= m_bos_len. The device position only advances
767 // via flush operations, never moving backward from the stream start.
768 assert(device_tell >= m_bos_len);
769 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());

771 }
772 case io_status::input:
773 {
774 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
775 // main_cont_beg(): buffer data comes from dget(), which advances device
776 // position by the amount read.
777 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
778 assert(device_tell >= m_bos_len);
779 assert(device_tell - m_bos_len >= buffered);
780 return device_tell - m_bos_len - buffered;
781 }
782 default:
783 throw cvt_error("root_cvt::tell fails: invalid io status");
784 }

1631 // need to read, then read at most as it can
1632 if (remain < to_max)
