496 void main_cont_beg()
497 {
498 // bos() must be called before main_cont_beg() to establish io_status
499 assert(m_io_status != io_status::neutral);
500
501 if constexpr (dev_cpt::support_put<device_type>)
502 {
503 if (m_io_status == io_status::output)
504 flush();
505 }
506
507 if constexpr (dev_cpt::support_positioning<device_type>)
508 {
509 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
510 // which advances the device position by the amount read. Consuming data
511 // from the buffer only moves m_buf_cur forward, never increasing the
512 // buffered amount beyond what was read from the device.
513 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
514 assert(m_device.dtell() >= buffered);
515 m_bos_len = m_device.dtell() - buffered;
516 }

802 [[nodiscard]] size_t tell() const
803 requires (dev_cpt::support_positioning<device_type>)
804 {
805 const size_t device_tell = m_device.dtell();
806
807 switch(m_io_status)
808 {
809 case io_status::output:
810 {
811 // Invariant: device_tell >= m_bos_len. The device position only advances
812 // via flush operations, never moving backward from the stream start.
813 assert(device_tell >= m_bos_len);
814 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
815 return device_tell - m_bos_len + buf_used;
816 }
817 case io_status::input:
818 {
819 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
820 // main_cont_beg(): buffer data comes from dget(), which advances device
821 // position by the amount read.
822 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
823 assert(device_tell >= m_bos_len);
824 assert(device_tell - m_bos_len >= buffered);
825 return device_tell - m_bos_len - buffered;
826 }
827 default:
828 throw cvt_error("root_cvt::tell fails: invalid io status");
829 }

1707 if (to_max > KernelType::s_buffer_length)

1709
1710 if constexpr (dev_cpt::support_put<device_type>)
1711 {

1713 m_kernel.switch_to_get();

1719 // need to read, then read at most as it can
1720 if (remain < to_max)
1721 {
1722 if (remain != 0)
1723 m_kernel.m_buf_end = std::copy(m_kernel.m_buf_cur, m_kernel.m_buf_end, m_kernel.m_buffer.data());
1724 else

1729 }
