524 void main_cont_beg()
525 {
526 // bos() must be called before main_cont_beg() to establish io_status
527 assert(m_io_status != io_status::neutral);
528
529 if constexpr (dev_cpt::support_put<device_type>)
530 {
531 if (m_io_status == io_status::output)
532 flush();
533 }
534
535 if constexpr (dev_cpt::support_positioning<device_type>)
536 {
537 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
538 // which advances the device position by the amount read. Consuming data
539 // from the buffer only moves m_buf_cur forward, never increasing the
540 // buffered amount beyond what was read from the device.
541 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
542 assert(m_device.dtell() >= buffered);
543 m_bos_len = m_device.dtell() - buffered;
544 }

830 [[nodiscard]] size_t tell() const
831 requires (dev_cpt::support_positioning<device_type>)
832 {
833 const size_t device_tell = m_device.dtell();
834
835 switch(m_io_status)
836 {
837 case io_status::output:
838 {
839 // Invariant: device_tell >= m_bos_len. The device position only advances
840 // via flush operations, never moving backward from the stream start.
841 assert(device_tell >= m_bos_len);
842 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
843 return device_tell - m_bos_len + buf_used;
844 }
845 case io_status::input:
846 {
847 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
848 // main_cont_beg(): buffer data comes from dget(), which advances device
849 // position by the amount read.
850 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
851 assert(device_tell >= m_bos_len);
852 assert(device_tell - m_bos_len >= buffered);
853 return device_tell - m_bos_len - buffered;
854 }
855 default:
856 throw cvt_error("root_cvt::tell fails: invalid io status");
857 }
