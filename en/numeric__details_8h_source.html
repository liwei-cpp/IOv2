37 // active. The lconv* and nl_langinfo() pointers may be invalidated
38 // by any subsequent setlocale()/uselocale() call (e.g. inside
39 // FacetHelper::string_to_char_convert), so copy first
40 // and convert after.
41 std::string dp_raw, ts_raw, grp_raw, yes_raw, no_raw;
42 bool yes_set = false, no_set = false;
43 {
44 clocale_wrapper inter_locale(name.c_str());
45 clocale_user guard(inter_locale);
46 const lconv* lc = localeconv();
47 if (lc->decimal_point) dp_raw = lc->decimal_point;
48 if (lc->thousands_sep) ts_raw = lc->thousands_sep;
49 if (lc->grouping) grp_raw = lc->grouping;
50
51 // Treat empty YESSTR/NOSTR the same as a missing key: an empty
52 // m_true_name / m_false_name would make numeric::get(bool&) fail
53 // unconditionally (zero-length prefix never matches) and let
54 // numeric::put(bool) write nothing but padding.
55 if (auto* p = nl_langinfo(YESSTR); p && *p) { yes_raw = p; yes_set = true; }
56 if (auto* p = nl_langinfo(NOSTR); p && *p) { no_raw = p; no_set = true; }
57 }
58
59 m_decimal_point = FacetHelper::string_to_char_convert(dp_raw, name);
60 m_thousands_sep = FacetHelper::string_to_char_convert(ts_raw, name);
61
62 // string_to_char_convert hard-codes '\0' as its failure sentinel. A
63 // decimal point of '\0' would later be mistaken for an in-stream
64 // null byte during parsing; fall back to '.' so the contract that
65 // m_decimal_point is a real, distinguishable character is preserved.
66 if (m_decimal_point == '\0') m_decimal_point = '.';
67
68 if (m_thousands_sep != '\0' && !grp_raw.empty())
69 {
70 // Copy raw POSIX grouping bytes, then normalise into the
71 // internal convention here at the POSIX boundary.
72 // Downstream (numeric, user-derived numeric_conf) sees
73 // only the internal form.
74 m_grouping.resize(grp_raw.size());
75 for (size_t i = 0; i < grp_raw.size(); ++i)
76 m_grouping[i] = static_cast<uint8_t>(grp_raw[i]);
77 FacetHelper::adjust_grouping(m_grouping);
78 }
79
80 // If a locale (or a user-derived numeric_conf that overrides
81 // thousands_sep() / decimal_point()) ends up with the same
82 // character for both separators, downstream extract_int /
83 // extract_float in facet/numeric.h check thousands_sep BEFORE
84 // decimal_point against each input character. An equal pair
85 // makes the decimal-point branch unreachable and groups would
86 // be parsed where a decimal mark was intended. Drop grouping
87 // so parsing falls back to the no-grouping path. Mirrors the
88 // same guard in numeric_conf<char8_t>, which can hit this case
89 // after multi-byte UTF-8 separators collapse onto single-byte

91 if (m_thousands_sep == m_decimal_point)
92 m_grouping.clear();
93
94 if (yes_set) m_true_name = yes_raw;
95 else m_true_name = "true";
96 if (no_set) m_false_name = no_raw;
97 else m_false_name = "false";
98 }
