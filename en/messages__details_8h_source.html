451 static std::unordered_map<std::u8string, std::u8string> get_translate_dictionary(const std::string& filename)
452 {
453 std::FILE* fp = fopen(filename.c_str(), "rb"); // NOLINT(cppcoreguidelines-owning-memory)
454 if (!fp)
455 throw stream_error("get_translate_dictionary fail: cannot open file " + filename);
456
457 file_closer guard(fp);
458
459 if (std::fseek(fp, 0, SEEK_END) != 0)
460 throw stream_error("get_translate_dictionary fail: invalid format");
461 const long file_size = std::ftell(fp);
462 if (file_size < 0)
463 throw stream_error("get_translate_dictionary fail: file inconsistent");
464 if (std::fseek(fp, 0, SEEK_SET) != 0)
465 throw stream_error("get_translate_dictionary fail: invalid format");
466
467 auto read_num = [fp](unsigned char* buf, bool need_swap)
468 {
469 if (std::fread(buf, 1, 4, fp) != 4)
470 throw stream_error("get_translate_dictionary fail: invalid format");
471
472 std::uint32_t res = 0;
473 std::memcpy(&res, buf, 4);
474 if (need_swap)
475 res = std::byteswap(res);
476 return res;
477 };
478
479 std::array<unsigned char, 8> buff{};
480 if (std::fread(buff.data(), 1, 8, fp) != 8)
481 throw stream_error("get_translate_dictionary fail: invalid format");
482
483 bool need_swap = false;
484 if ((buff[0] == 0x95) && (buff[1] == 0x04) && (buff[2] == 0x12) && (buff[3] == 0xde))
485 need_swap = (std::endian::native == std::endian::little);
486 else if ((buff[0] == 0xde) && (buff[1] == 0x12) && (buff[2] == 0x04) && (buff[3] == 0x95))
487 need_swap = (std::endian::native == std::endian::big);
488 else throw stream_error("get_translate_dictionary fail: invalid format");
489
490 // Accept any .mo whose major revision (the high 16 bits) is 0, regardless of
491 // minor revision. Only minor 0 (basic) and minor 1 (system-dependent strings)
492 // are defined today, but minor revisions are backward-compatible additions:
493 // they only add separate tables (sysdep strings, hash table) and never change
494 // the regular orig/trans tables this parser reads. Sysdep strings live in their
495 // own orig/trans_sysdep tables we never touch, so a sysdep string is simply
496 // absent (lookup misses and the caller keeps the original) rather than returned
497 // with its segment markers unexpanded. A non-zero major revision uses a layout
498 // this basic parser does not commit to, so it is rejected.
499 std::uint32_t revision = 0;
500 std::memcpy(&revision, buff.data() + 4, 4);
501 if (need_swap)
502 revision = std::byteswap(revision);
503 if ((revision >> 16) != 0)
504 throw stream_error("get_translate_dictionary fail: unsupported .mo major revision");
505
506 auto str_num = read_num(buff.data(), need_swap);
507 auto ori_offset = read_num(buff.data(), need_swap);
508 auto aim_offset = read_num(buff.data(), need_swap);
509
510 // Each entry needs an 8-byte (length, offset) descriptor in both the
511 // original and translation tables, so a valid str_num cannot exceed
512 // file_size / 16. Rejecting larger counts caps the O(str_num) growth of
513 // oris and the result map, which the per-string length cap does not.
514 if (str_num > static_cast<std::uint64_t>(file_size) / 16u)
515 throw stream_error("get_translate_dictionary fail: implausible string count");
516
517 // The .mo file offsets (and string lengths) are unsigned 32-bit, but
518 // std::fseek takes a signed long. Where long is 64-bit (e.g. LP64
519 // Linux) every uint32_t is representable, so all seeks below are exact.
520 // On a 32-bit-long platform (LLP64/ILP32) an offset >= 2 GiB would
521 // convert to a negative long and seek wrongly — but that can only arise
522 // from a > 2 GiB or corrupt .mo, and every fseek/fread return value is
523 // checked, so the worst case is a clean "invalid format" throw, never a
524 // misread or out-of-bounds access.
525 if (std::fseek(fp, ori_offset, SEEK_SET) != 0)
526 throw stream_error("get_translate_dictionary fail: invalid format");
527
528 // The .mo format stores each string length as an unbounded 32-bit
529 // integer, so the buffer is grown to fit each entry rather than
530 // assuming a fixed maximum. A sanity cap guards against a corrupt or
531 // malicious file requesting a huge allocation.
532 constexpr std::uint32_t k_max_str_len = 64u * 1024u * 1024u; // 64 MiB
533 std::uint64_t total_str_bytes = 0;
534 std::vector<char8_t> str_buf;
535 std::vector<std::u8string> oris;
536 for (size_t i = 0; i < str_num; ++i)
537 {
538 auto length = read_num(buff.data(), need_swap);
539 auto offset = read_num(buff.data(), need_swap);
540 auto cur_pos = std::ftell(fp);
541 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
542
543 if (length > k_max_str_len)
544 throw stream_error("get_translate_dictionary fail: string too long");
545 total_str_bytes += length;
546 if (total_str_bytes > static_cast<std::uint64_t>(file_size))
547 throw stream_error("get_translate_dictionary fail: total string bytes exceed file size");
548 if (str_buf.size() < static_cast<size_t>(length) + 1)
549 str_buf.resize(static_cast<size_t>(length) + 1);
550
551 if (std::fseek(fp, offset, SEEK_SET) != 0)
552 throw stream_error("get_translate_dictionary fail: invalid format");
553
554 if (std::fread(str_buf.data(), 1, length, fp) != length)
555 throw stream_error("get_translate_dictionary fail: invalid format");
556
557 str_buf[length] = u8'\0';
558 oris.emplace_back(str_buf.data());
559
560 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
561 throw stream_error("get_translate_dictionary fail: invalid format");
562 }
563
564 if (std::fseek(fp, aim_offset, SEEK_SET) != 0)
565 throw stream_error("get_translate_dictionary fail: invalid format");
566 std::unordered_map<std::u8string, std::u8string> res;
567
568 for (size_t i = 0; i < str_num; ++i)
569 {
570 auto length = read_num(buff.data(), need_swap);
571 auto offset = read_num(buff.data(), need_swap);
572 auto cur_pos = std::ftell(fp);
573 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
574
575 if (length > k_max_str_len)
576 throw stream_error("get_translate_dictionary fail: string too long");
577 total_str_bytes += length;
578 if (total_str_bytes > static_cast<std::uint64_t>(file_size))
579 throw stream_error("get_translate_dictionary fail: total string bytes exceed file size");
580 if (str_buf.size() < static_cast<size_t>(length) + 1)
581 str_buf.resize(static_cast<size_t>(length) + 1);
582
583 if (std::fseek(fp, offset, SEEK_SET) != 0)
584 throw stream_error("get_translate_dictionary fail: invalid format");
585
586 if (std::fread(str_buf.data(), 1, length, fp) != length)
587 throw stream_error("get_translate_dictionary fail: invalid format");
588
589 str_buf[length] = u8'\0';
590 res.insert({oris[i], str_buf.data()});
591
592 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
593 throw stream_error("get_translate_dictionary fail: invalid format");
594 }
595
596 return res;
597 }

681 static std::string filter_lang(const text_domain& td, const std::string& p_lang)
682 {
683 auto match_lang = [&td](const std::string& str_lang)
684 {
685 std::size_t start = 0;
686 std::size_t end = 0;
687 while ((end = str_lang.find(':', start)) != std::string::npos)
688 {
689 auto cur_lang = str_lang.substr(start, end - start);
690 if ((!cur_lang.empty()) && (base_ft<messages>::available(td, cur_lang)))
691 return cur_lang;
692 start = end + 1;
693 }
694
695 auto last_str = str_lang.substr(start);
696 if ((!last_str.empty()) && (base_ft<messages>::available(td, last_str)))
697 return last_str;
698
699 return std::string{};
700 };
701
702 if (p_lang.empty())
703 {
704 // Environment fallback (mirrors gettext): LANGUAGE, then
705 // LC_ALL / LC_MESSAGES / LANG. std::getenv is NOT synchronised
706 // against a concurrent setenv()/putenv() on another thread — the
707 // returned pointer and the bytes it addresses may be invalidated
708 // mid-read — and neither C++ nor POSIX offers a portable
709 // thread-safe environment read, so a lock here could not close the
710 // race against mutators elsewhere. This code therefore assumes the
711 // process environment is not modified concurrently with messages
712 // facet construction (true for the usual "configure locale once at
713 // startup" usage). Each value is copied into a std::string at once
714 // to keep the read window minimal.
715 std::string res;
716 if (const char* p = std::getenv("LANGUAGE"))
717 {
718 res = match_lang(p);
719 if (!res.empty()) return res;
720 }
721
722 for (const char* var : {"LC_ALL", "LC_MESSAGES", "LANG"})
723 {
724 if (const char* p = std::getenv(var); p && p[0] != '\0')
725 {
726 res = p;
727 // Unlike LANGUAGE, these are a single locale name, not a
728 // ':'-separated list (gettext never splits them). A value
729 // containing ':' is therefore not a valid locale name:
730 // skip it and try the next variable.
731 if (res.find(':') == std::string::npos && base_ft<messages>::available(td, res))
732 return res;
733 }
734 }
735
736 return "";
737 }
738 else
739 return match_lang(p_lang);
740 }
