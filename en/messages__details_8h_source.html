409 static std::unordered_map<std::u8string, std::u8string> get_translate_dictionary(const std::string& filename)
410 {
411 std::FILE* fp = fopen(filename.c_str(), "rb"); // NOLINT(cppcoreguidelines-owning-memory)
412 if (!fp)
413 throw stream_error("get_translate_dictionary fail: cannot open file " + filename);
414
415 file_closer guard(fp);
416
417 if (std::fseek(fp, 0, SEEK_END) != 0)
418 throw stream_error("get_translate_dictionary fail: invalid format");
419 const long file_size = std::ftell(fp);
420 if (file_size < 0)
421 throw stream_error("get_translate_dictionary fail: file inconsistent");
422 if (std::fseek(fp, 0, SEEK_SET) != 0)
423 throw stream_error("get_translate_dictionary fail: invalid format");
424
425 auto read_num = [fp](unsigned char* buf, bool need_swap)
426 {
427 if (std::fread(buf, 1, 4, fp) != 4)
428 throw stream_error("get_translate_dictionary fail: invalid format");
429
430 std::uint32_t res = 0;
431 std::memcpy(&res, buf, 4);
432 if (need_swap)
433 res = std::byteswap(res);
434 return res;
435 };
436
437 std::array<unsigned char, 8> buff{};
438 if (std::fread(buff.data(), 1, 8, fp) != 8)
439 throw stream_error("get_translate_dictionary fail: invalid format");
440
441 bool need_swap = false;
442 if ((buff[0] == 0x95) && (buff[1] == 0x04) && (buff[2] == 0x12) && (buff[3] == 0xde))
443 need_swap = (std::endian::native == std::endian::little);
444 else if ((buff[0] == 0xde) && (buff[1] == 0x12) && (buff[2] == 0x04) && (buff[3] == 0x95))
445 need_swap = (std::endian::native == std::endian::big);
446 else throw stream_error("get_translate_dictionary fail: invalid format");
447
448 // Accept any .mo whose major revision (the high 16 bits) is 0, regardless of
449 // minor revision. Only minor 0 (basic) and minor 1 (system-dependent strings)
450 // are defined today, but minor revisions are backward-compatible additions:
451 // they only add separate tables (sysdep strings, hash table) and never change
452 // the regular orig/trans tables this parser reads. Sysdep strings live in their
453 // own orig/trans_sysdep tables we never touch, so a sysdep string is simply
454 // absent (lookup misses and the caller keeps the original) rather than returned
455 // with its segment markers unexpanded. A non-zero major revision uses a layout
456 // this basic parser does not commit to, so it is rejected.
457 std::uint32_t revision = 0;
458 std::memcpy(&revision, buff.data() + 4, 4);
459 if (need_swap)
460 revision = std::byteswap(revision);
461 if ((revision >> 16) != 0)
462 throw stream_error("get_translate_dictionary fail: unsupported .mo major revision");
463
464 auto str_num = read_num(buff.data(), need_swap);
465 auto ori_offset = read_num(buff.data(), need_swap);
466 auto aim_offset = read_num(buff.data(), need_swap);
467
468 // Each entry needs an 8-byte (length, offset) descriptor in both the
469 // original and translation tables, so a valid str_num cannot exceed
470 // file_size / 16. Rejecting larger counts caps the O(str_num) growth of
471 // oris and the result map, which the per-string length cap does not.
472 if (str_num > static_cast<std::uint64_t>(file_size) / 16u)
473 throw stream_error("get_translate_dictionary fail: implausible string count");
474
475 // The .mo file offsets (and string lengths) are unsigned 32-bit, but
476 // std::fseek takes a signed long. Where long is 64-bit (e.g. LP64
477 // Linux) every uint32_t is representable, so all seeks below are exact.
478 // On a 32-bit-long platform (LLP64/ILP32) an offset >= 2 GiB would
479 // convert to a negative long and seek wrongly — but that can only arise
480 // from a > 2 GiB or corrupt .mo, and every fseek/fread return value is
481 // checked, so the worst case is a clean "invalid format" throw, never a
482 // misread or out-of-bounds access.
483 if (std::fseek(fp, ori_offset, SEEK_SET) != 0)
484 throw stream_error("get_translate_dictionary fail: invalid format");
485
486 // The .mo format stores each string length as an unbounded 32-bit
487 // integer, so the buffer is grown to fit each entry rather than
488 // assuming a fixed maximum. A sanity cap guards against a corrupt or
489 // malicious file requesting a huge allocation.
490 constexpr std::uint32_t k_max_str_len = 64u * 1024u * 1024u; // 64 MiB
491 std::uint64_t total_str_bytes = 0;
492 std::vector<char8_t> str_buf;
493 std::vector<std::u8string> oris;
494 for (size_t i = 0; i < str_num; ++i)
495 {
496 auto length = read_num(buff.data(), need_swap);
497 auto offset = read_num(buff.data(), need_swap);
498 auto cur_pos = std::ftell(fp);
499 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
500
501 if (length > k_max_str_len)
502 throw stream_error("get_translate_dictionary fail: string too long");
503 total_str_bytes += length;
504 if (total_str_bytes > static_cast<std::uint64_t>(file_size))
505 throw stream_error("get_translate_dictionary fail: total string bytes exceed file size");
506 if (str_buf.size() < static_cast<size_t>(length) + 1)
507 str_buf.resize(static_cast<size_t>(length) + 1);
508
509 if (std::fseek(fp, offset, SEEK_SET) != 0)
510 throw stream_error("get_translate_dictionary fail: invalid format");
511
512 if (std::fread(str_buf.data(), 1, length, fp) != length)
513 throw stream_error("get_translate_dictionary fail: invalid format");
514
515 str_buf[length] = u8'\0';
516 oris.emplace_back(str_buf.data());
517
518 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
519 throw stream_error("get_translate_dictionary fail: invalid format");
520 }
521
522 if (std::fseek(fp, aim_offset, SEEK_SET) != 0)
523 throw stream_error("get_translate_dictionary fail: invalid format");
524 std::unordered_map<std::u8string, std::u8string> res;
525
526 for (size_t i = 0; i < str_num; ++i)
527 {
528 auto length = read_num(buff.data(), need_swap);
529 auto offset = read_num(buff.data(), need_swap);
530 auto cur_pos = std::ftell(fp);
531 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
532
533 if (length > k_max_str_len)
534 throw stream_error("get_translate_dictionary fail: string too long");
535 total_str_bytes += length;
536 if (total_str_bytes > static_cast<std::uint64_t>(file_size))
537 throw stream_error("get_translate_dictionary fail: total string bytes exceed file size");
538 if (str_buf.size() < static_cast<size_t>(length) + 1)
539 str_buf.resize(static_cast<size_t>(length) + 1);
540
541 if (std::fseek(fp, offset, SEEK_SET) != 0)
542 throw stream_error("get_translate_dictionary fail: invalid format");
543
544 if (std::fread(str_buf.data(), 1, length, fp) != length)
545 throw stream_error("get_translate_dictionary fail: invalid format");
546
547 str_buf[length] = u8'\0';
548 res.insert({oris[i], str_buf.data()});
549
550 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
551 throw stream_error("get_translate_dictionary fail: invalid format");
552 }
553
554 return res;
555 }

639 static std::string filter_lang(const text_domain& td, const std::string& p_lang)
640 {
641 auto match_lang = [&td](const std::string& str_lang)
642 {
643 std::size_t start = 0;
644 std::size_t end = 0;
645 while ((end = str_lang.find(':', start)) != std::string::npos)
646 {
647 auto cur_lang = str_lang.substr(start, end - start);
648 if ((!cur_lang.empty()) && (base_ft<messages>::available(td, cur_lang)))
649 return cur_lang;
650 start = end + 1;
651 }
652
653 auto last_str = str_lang.substr(start);
654 if ((!last_str.empty()) && (base_ft<messages>::available(td, last_str)))
655 return last_str;
656
657 return std::string{};
658 };
659
660 if (p_lang.empty())
661 {
662 // Environment fallback (mirrors gettext): LANGUAGE, then
663 // LC_ALL / LC_MESSAGES / LANG. std::getenv is NOT synchronised
664 // against a concurrent setenv()/putenv() on another thread — the
665 // returned pointer and the bytes it addresses may be invalidated
666 // mid-read — and neither C++ nor POSIX offers a portable
667 // thread-safe environment read, so a lock here could not close the
668 // race against mutators elsewhere. This code therefore assumes the
669 // process environment is not modified concurrently with messages
670 // facet construction (true for the usual "configure locale once at
671 // startup" usage). Each value is copied into a std::string at once
672 // to keep the read window minimal.
673 std::string res;
674 if (const char* p = std::getenv("LANGUAGE"))
675 {
676 res = match_lang(p);
677 if (!res.empty()) return res;
678 }
679
680 for (const char* var : {"LC_ALL", "LC_MESSAGES", "LANG"})
681 {
682 if (const char* p = std::getenv(var); p && p[0] != '\0')
683 {
684 res = p;
685 // Unlike LANGUAGE, these are a single locale name, not a
686 // ':'-separated list (gettext never splits them). A value
687 // containing ':' is therefore not a valid locale name:
688 // skip it and try the next variable.
689 if (res.find(':') == std::string::npos && base_ft<messages>::available(td, res))
690 return res;
691 }
692 }
693
694 return "";
695 }
696 else
697 return match_lang(p_lang);
698 }
