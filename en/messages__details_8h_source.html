408 static std::unordered_map<std::u8string, std::u8string> get_translate_dictionary(const std::string& filename)
409 {
410 std::FILE* fp = fopen(filename.c_str(), "rb"); // NOLINT(cppcoreguidelines-owning-memory)
411 if (!fp)
412 throw stream_error("get_translate_dictionary fail: cannot open file " + filename);
413
414 file_closer guard(fp);
415
416 if (std::fseek(fp, 0, SEEK_END) != 0)
417 throw stream_error("get_translate_dictionary fail: invalid format");
418 const long file_size = std::ftell(fp);
419 if (file_size < 0)
420 throw stream_error("get_translate_dictionary fail: file inconsistent");
421 if (std::fseek(fp, 0, SEEK_SET) != 0)
422 throw stream_error("get_translate_dictionary fail: invalid format");
423
424 auto read_num = [fp](unsigned char* buf, bool need_swap)
425 {
426 if (std::fread(buf, 1, 4, fp) != 4)
427 throw stream_error("get_translate_dictionary fail: invalid format");
428
429 std::uint32_t res = static_cast<std::uint32_t>(buf[0])
430 | (static_cast<std::uint32_t>(buf[1]) << 8)
431 | (static_cast<std::uint32_t>(buf[2]) << 16)
432 | (static_cast<std::uint32_t>(buf[3]) << 24);
433 if (need_swap)
434 res = std::byteswap(res);
435 return res;
436 };
437
438 std::array<unsigned char, 8> buff{};
439 if (std::fread(buff.data(), 1, 8, fp) != 8)
440 throw stream_error("get_translate_dictionary fail: invalid format");
441
442 bool need_swap = false;
443 if ((buff[0] == 0x95) && (buff[1] == 0x04) && (buff[2] == 0x12) && (buff[3] == 0xde))
444 need_swap = (std::endian::native == std::endian::little);
445 else if ((buff[0] == 0xde) && (buff[1] == 0x12) && (buff[2] == 0x04) && (buff[3] == 0x95))
446 need_swap = (std::endian::native == std::endian::big);
447 else throw stream_error("get_translate_dictionary fail: invalid format");
448
449 if ((buff[4] != 0) || (buff[5] != 0) || (buff[6] != 0) || (buff[7] != 0))
450 throw stream_error("get_translate_dictionary fail: invalid format");
451
452 auto str_num = read_num(buff.data(), need_swap);
453 auto ori_offset = read_num(buff.data(), need_swap);
454 auto aim_offset = read_num(buff.data(), need_swap);
455
456 // Each entry needs an 8-byte (length, offset) descriptor in both the
457 // original and translation tables, so a valid str_num cannot exceed
458 // file_size / 16. Rejecting larger counts caps the O(str_num) growth of
459 // oris and the result map, which the per-string length cap does not.
460 if (str_num > static_cast<std::uint64_t>(file_size) / 16u)
461 throw stream_error("get_translate_dictionary fail: implausible string count");
462
463 // The .mo file offsets (and string lengths) are unsigned 32-bit, but
464 // std::fseek takes a signed long. Where long is 64-bit (e.g. LP64
465 // Linux) every uint32_t is representable, so all seeks below are exact.
466 // On a 32-bit-long platform (LLP64/ILP32) an offset >= 2 GiB would
467 // convert to a negative long and seek wrongly — but that can only arise
468 // from a > 2 GiB or corrupt .mo, and every fseek/fread return value is
469 // checked, so the worst case is a clean "invalid format" throw, never a
470 // misread or out-of-bounds access.
471 if (std::fseek(fp, ori_offset, SEEK_SET) != 0)
472 throw stream_error("get_translate_dictionary fail: invalid format");
473
474 // The .mo format stores each string length as an unbounded 32-bit
475 // integer, so the buffer is grown to fit each entry rather than
476 // assuming a fixed maximum. A sanity cap guards against a corrupt or
477 // malicious file requesting a huge allocation.
478 constexpr std::uint32_t k_max_str_len = 64u * 1024u * 1024u; // 64 MiB
479 std::vector<char8_t> str_buf;
480 std::vector<std::u8string> oris;
481 for (size_t i = 0; i < str_num; ++i)
482 {
483 auto length = read_num(buff.data(), need_swap);
484 auto offset = read_num(buff.data(), need_swap);
485 auto cur_pos = std::ftell(fp);
486 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
487
488 if (length > k_max_str_len)
489 throw stream_error("get_translate_dictionary fail: string too long");
490 if (str_buf.size() < static_cast<size_t>(length) + 1)
491 str_buf.resize(static_cast<size_t>(length) + 1);
492
493 if (std::fseek(fp, offset, SEEK_SET) != 0)
494 throw stream_error("get_translate_dictionary fail: invalid format");
495
496 if (std::fread(str_buf.data(), 1, length, fp) != length)
497 throw stream_error("get_translate_dictionary fail: invalid format");
498
499 str_buf[length] = u8'\0';
500 oris.emplace_back(str_buf.data());
501
502 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
503 throw stream_error("get_translate_dictionary fail: invalid format");
504 }
505
506 if (std::fseek(fp, aim_offset, SEEK_SET) != 0)
507 throw stream_error("get_translate_dictionary fail: invalid format");
508 std::unordered_map<std::u8string, std::u8string> res;
509
510 for (size_t i = 0; i < str_num; ++i)
511 {
512 auto length = read_num(buff.data(), need_swap);
513 auto offset = read_num(buff.data(), need_swap);
514 auto cur_pos = std::ftell(fp);
515 if (cur_pos == -1L) throw stream_error("get_translate_dictionary fail: file inconsistent");
516
517 if (length > k_max_str_len)
518 throw stream_error("get_translate_dictionary fail: string too long");
519 if (str_buf.size() < static_cast<size_t>(length) + 1)
520 str_buf.resize(static_cast<size_t>(length) + 1);
521
522 if (std::fseek(fp, offset, SEEK_SET) != 0)
523 throw stream_error("get_translate_dictionary fail: invalid format");
524
525 if (std::fread(str_buf.data(), 1, length, fp) != length)
526 throw stream_error("get_translate_dictionary fail: invalid format");
527
528 str_buf[length] = u8'\0';
529 res.insert({oris[i], str_buf.data()});
530
531 if (std::fseek(fp, cur_pos, SEEK_SET) != 0)
532 throw stream_error("get_translate_dictionary fail: invalid format");
533 }
534
535 return res;
536 }

620 static std::string filter_lang(const text_domain& td, const std::string& p_lang)
621 {
622 auto match_lang = [&td](const std::string& str_lang)
623 {
624 std::size_t start = 0;
625 std::size_t end = 0;
626 while ((end = str_lang.find(':', start)) != std::string::npos)
627 {
628 auto cur_lang = str_lang.substr(start, end - start);
629 if ((!cur_lang.empty()) && (base_ft<messages>::available(td, cur_lang)))
630 return cur_lang;
631 start = end + 1;
632 }
633
634 auto last_str = str_lang.substr(start);
635 if ((!last_str.empty()) && (base_ft<messages>::available(td, last_str)))
636 return last_str;
637
638 return std::string{};
639 };
640
641 if (p_lang.empty())
642 {
643 // Environment fallback (mirrors gettext): LANGUAGE, then
644 // LC_ALL / LC_MESSAGES / LANG. std::getenv is NOT synchronised
645 // against a concurrent setenv()/putenv() on another thread — the
646 // returned pointer and the bytes it addresses may be invalidated
647 // mid-read — and neither C++ nor POSIX offers a portable
648 // thread-safe environment read, so a lock here could not close the
649 // race against mutators elsewhere. This code therefore assumes the
650 // process environment is not modified concurrently with messages
651 // facet construction (true for the usual "configure locale once at
652 // startup" usage). Each value is copied into a std::string at once
653 // to keep the read window minimal.
654 std::string res;
655 if (const char* p = std::getenv("LANGUAGE"))
656 {
657 res = match_lang(p);
658 if (!res.empty()) return res;
659 }
660
661 for (const char* var : {"LC_ALL", "LC_MESSAGES", "LANG"})
662 {
663 if (const char* p = std::getenv(var); p && p[0] != '\0')
664 {
665 res = p;
666 // Unlike LANGUAGE, these are a single locale name, not a
667 // ':'-separated list (gettext never splits them). A value
668 // containing ':' is therefore not a valid locale name:
669 // skip it and try the next variable.
670 if (res.find(':') == std::string::npos && base_ft<messages>::available(td, res))
671 return res;
672 }
673 }
674
675 return "";
676 }
677 else
678 return match_lang(p_lang);
679 }
