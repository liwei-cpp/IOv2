435template <istream_type T, typename TValue>
436T& operator>>(T& obj, TValue& value)
437{
438 auto iter = obj.i_iter();
439
440 using TChar = typename T::char_type;
441 using TCtx = typename parse_context_type<TChar, TValue>::type;
442 using sentry_type = typename T::in_sentry_type;
443
444 try
445 {
446 bool skip = bool(obj.flags() & ios_defs::skipws);
447 sentry_type cerb(obj, !skip);
448
449 if constexpr (is_reader_def<TChar, TCtx>)
450 {
451 if constexpr (std::is_same_v<TCtx, TValue>)
452 reader<TChar, TValue>::sread(iter, std::default_sentinel, obj, obj.locale(), value);
453 else
454 {
455 TCtx tmp;
456 reader<TChar, TCtx>::sread(iter, std::default_sentinel, obj, obj.locale(), tmp);
457 value = static_cast<TValue>(tmp);
458 }
459 }
460 else
461 static_assert(dependent_false_v<TValue>, "No parse method provided");
462 }
463 catch(...)
464 {
465 obj.handle_exception(std::current_exception());
466 }
