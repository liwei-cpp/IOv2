201 auto get_buf(size_t to_max)
202 {
203 if (to_max == 0)
204 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
205 if (to_max > m_buffer.size())
206 throw cvt_error("cvt_reader::get_buf fail, read size too large");
207
208 const auto rollback_size = m_end_pos - m_cur_pos;
209 if (to_max <= rollback_size)
210 {
211 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
212 m_cur_pos += to_max;
213 if constexpr (!Saturate)
214 return res;
215 else
216 return res.first;
217 }
218
219 if (to_max > m_buffer.size() - m_cur_pos)
220 {
221 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
222 m_cur_pos = 0;
223 m_end_pos = rollback_size;
224 }
225 const size_t aim_size = to_max - rollback_size;
226 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
227 m_end_pos += read_size;
228
229 if constexpr (Saturate)
230 {
231 while (read_size < aim_size)
232 {
233 const size_t new_aim_size = aim_size - read_size;
234 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
235 if (new_read_size == 0)
236 throw cvt_error("get_buf<Saturate> fail: meet eos");
237 m_end_pos += new_read_size;
238 read_size += new_read_size;
239 }
240 }
241
242 if constexpr (!Saturate)
243 {
244 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
245 m_cur_pos = m_end_pos;
246 return res;
247 }
248 else
249 {
250 const char_type* res = m_buffer.data() + m_cur_pos;
251 m_cur_pos = m_end_pos;
252 return res;
253 }

1142 size_t get(internal_type* to, size_t to_max)
1143 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1144 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1145 }
1146 {
1147 // Read failures do not taint: they only advance the kernel cursor
1148 // and possibly leave a per-call mbstate undefined; the underlying
1149 // stream content is unchanged, and callers can reseek to recover.
1150 // We still honour any pre-existing taint so that a writer that
1151 // corrupted the stream cannot then read back a confused image.
1152 assert_not_tainted();
1153
1154 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1155
1156 if (!m_is_bos_done)
1157 {
1158 if (to_max == 0) return 0;
1159
1160 constexpr size_t ext_size = sizeof(external_type);
1161
1162 reader.reset(s_bos_chunk);
1163 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1164 char* dest_bytes = reinterpret_cast<char*>(to);
1165 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1166 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1167
1168 // BOS phase requires complete data: if the stream ends unexpectedly
1169 // before providing enough bytes, this is a format error and we throw.
1170 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1171
1172 // Read full external_type chunks
1173 while (dest_bytes + ext_size <= dest_bytes_end)
1174 {
1175 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1176
1177 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1178 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1179 dest_bytes += units_to_read_now * ext_size;
1180 }
1181
1182 // Read the final partial internal_type if needed
1183 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1184 if (final_remainder_bytes > 0)
1185 {
1186 const auto* src_buf = reader.template get_buf<true>(1);
1187 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1188 dest_bytes += final_remainder_bytes;
1189 }
1190
1191 return to_max;
1192 }
1193 else
1194 {
1195 if (m_io_status != io_status::input)
1196 {
1197 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1198 static_cast<CurrentType*>(this)->switch_to_get();
1199 else
1200 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1201 }
1202 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1203 }
1204 }

1251 void put(const internal_type* to, size_t to_size)
1252 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1253 { t.put_main(w, data, len) } -> std::same_as<void>;
1254 }
1255 {
1256 assert_not_tainted();
1257
1258 // IO-direction switching does not write bytes to the output stream:
1259 // a failed switch_to_put() leaves the stream unchanged, so the
1260 // converter is not corrupted and must not be tainted. Perform the
1261 // switch BEFORE entering the taint-on-exception region so that only
1262 // actual kernel writes can cause tainting.
1263 if (m_is_bos_done && m_io_status != io_status::output)
1264 {
1265 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1266 static_cast<CurrentType*>(this)->switch_to_put();
1267 else
1268 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1269 }
1270
1271 try
1272 {
1273 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1274 if (!m_is_bos_done)
1275 {
1276 if (to_size == 0) return;
1277
1278 constexpr size_t ext_size = sizeof(external_type);
1279
1280 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1281 auto to_bytes = reinterpret_cast<const char*>(to);
1282 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1283 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1284
1285 // BOS phase writes complete external_type units. If the input data
1286 // is not a multiple of ext_size, pad the final unit with zeros.
1287
1288 writer.reset(s_bos_chunk);
1289 while (to_bytes + ext_size <= to_bytes_end)
1290 {
1291 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1292 auto ptr = writer.put_buf(dest_count);
1293 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1294 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1295 to_bytes += dest_count * ext_size;
1296 }
1297
1298 if (to_bytes < to_bytes_end)
1299 {
1300 // The while loop above guarantees remaining < ext_size.
1301 // The modulo is logically redundant but helps the compiler's
1302 // static analyzer prove the bound for memset size calculation.
1303 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1304 auto ptr = writer.put_buf(1);
1305 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1306 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1307 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1308 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1309 }
1310
1311 writer.commit();
1312 }
1313 else
1314 {
1315 // switch_to_put() was already called above if needed;
1316 // m_io_status is guaranteed to be output at this point.
1317 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1318 // commit() is the sole responsibility of abs_cvt::put: derived
1319 // put_main implementations must NOT call writer.commit() themselves.
1320 // If put_main throws, this commit is skipped and any uncommitted
1321 // buffer contents are dropped at writer destruction; the catch
1322 // below taints the converter.
1323 writer.commit();
1324 }
1325 }
1326 catch (...)
1327 {
1328 m_is_tainted = true;
1329 throw;
1330 }
1331 }
