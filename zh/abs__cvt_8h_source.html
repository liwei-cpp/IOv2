201 auto get_buf(size_t to_max)
202 {
203 if (to_max == 0)
204 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
205 if (to_max > m_buffer.size())
206 throw cvt_error("cvt_reader::get_buf fail, read size too large");
207
208 const auto rollback_size = m_end_pos - m_cur_pos;
209 if (to_max <= rollback_size)
210 {
211 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
212 m_cur_pos += to_max;
213 if constexpr (!Saturate)
214 return res;
215 else
216 return res.first;
217 }
218
219 if (to_max > m_buffer.size() - m_cur_pos)
220 {
221 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
222 m_cur_pos = 0;
223 m_end_pos = rollback_size;
224 }
225 const size_t aim_size = to_max - rollback_size;
226 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
227 m_end_pos += read_size;
228
229 if constexpr (Saturate)
230 {
231 while (read_size < aim_size)
232 {
233 const size_t new_aim_size = aim_size - read_size;
234 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
235 if (new_read_size == 0)
236 throw cvt_error("get_buf<Saturate> fail: meet eos");
237 m_end_pos += new_read_size;
238 read_size += new_read_size;
239 }
240 }
241
242 if constexpr (!Saturate)
243 {
244 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
245 m_cur_pos = m_end_pos;
246 return res;
247 }
248 else
249 {
250 const char_type* res = m_buffer.data() + m_cur_pos;
251 m_cur_pos = m_end_pos;
252 return res;
253 }
254 }

1251 size_t get(internal_type* to, size_t to_max)
1252 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1253 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1254 }
1255 {
1256 // Read failures do not taint: they only advance the kernel cursor
1257 // and possibly leave a per-call mbstate undefined; the underlying
1258 // stream content is unchanged, and callers can reseek to recover.
1259 // We still honour any pre-existing taint so that a writer that
1260 // corrupted the stream cannot then read back a confused image.
1261 assert_not_tainted();
1262
1263 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1264
1265 if (!m_is_bos_done)
1266 {
1267 if (to_max == 0) return 0;
1268
1269 constexpr size_t ext_size = sizeof(external_type);
1270
1271 reader.reset(s_bos_chunk);
1272 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1273 char* dest_bytes = reinterpret_cast<char*>(to);
1274 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1275 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1276
1277 // BOS phase requires complete data: if the stream ends unexpectedly
1278 // before providing enough bytes, this is a format error and we throw.
1279 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1280
1281 // Read full external_type chunks
1282 while (dest_bytes + ext_size <= dest_bytes_end)
1283 {
1284 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1285
1286 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1287 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1288 dest_bytes += units_to_read_now * ext_size;
1289 }
1290
1291 // Read the final partial internal_type if needed
1292 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1293 if (final_remainder_bytes > 0)
1294 {
1295 const auto* src_buf = reader.template get_buf<true>(1);
1296 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1297 dest_bytes += final_remainder_bytes;
1298 }
1299
1300 return to_max;
1301 }
1302 else
1303 {
1304 if (m_io_status != io_status::input)
1305 {
1306 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1307 static_cast<CurrentType*>(this)->switch_to_get();
1308 else
1309 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1310 }
1311 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1312 }
1313 }

1360 void put(const internal_type* to, size_t to_size)
1361 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1362 { t.put_main(w, data, len) } -> std::same_as<void>;
1363 }
1364 {
1365 assert_not_tainted();
1366
1367 // IO-direction switching does not write bytes to the output stream:
1368 // a failed switch_to_put() leaves the stream unchanged, so the
1369 // converter is not corrupted and must not be tainted. Perform the
1370 // switch BEFORE entering the taint-on-exception region so that only
1371 // actual kernel writes can cause tainting.
1372 if (m_is_bos_done && m_io_status != io_status::output)
1373 {
1374 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1375 static_cast<CurrentType*>(this)->switch_to_put();
1376 else
1377 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1378 }
1379
1380 try
1381 {
1382 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1383 if (!m_is_bos_done)
1384 {
1385 if (to_size == 0) return;
1386
1387 constexpr size_t ext_size = sizeof(external_type);
1388
1389 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1390 auto to_bytes = reinterpret_cast<const char*>(to);
1391 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1392 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1393
1394 // BOS phase writes complete external_type units. If the input data
1395 // is not a multiple of ext_size, pad the final unit with zeros.
1396
1397 writer.reset(s_bos_chunk);
1398 while (to_bytes + ext_size <= to_bytes_end)
1399 {
1400 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1401 auto ptr = writer.put_buf(dest_count);
1402 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1403 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1404 to_bytes += dest_count * ext_size;
1405 }
1406
1407 if (to_bytes < to_bytes_end)
1408 {
1409 // The while loop above guarantees remaining < ext_size.
1410 // The modulo is logically redundant but helps the compiler's
1411 // static analyzer prove the bound for memset size calculation.
1412 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1413 auto ptr = writer.put_buf(1);
1414 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1415 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1416 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1417 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1418 }
1419
1420 writer.commit();
1421 }
1422 else
1423 {
1424 // switch_to_put() was already called above if needed;
1425 // m_io_status is guaranteed to be output at this point.
1426 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1427 // commit() is the sole responsibility of abs_cvt::put: derived
1428 // put_main implementations must NOT call writer.commit() themselves.
1429 // If put_main throws, this commit is skipped and any uncommitted
1430 // buffer contents are dropped at writer destruction; the catch
1431 // below taints the converter.
1432 writer.commit();
1433 }
1434 }
1435 catch (...)
1436 {
1437 m_is_tainted = true;
1438 throw;
1439 }
1440 }
