199 auto get_buf(size_t to_max)
200 {
201 if (to_max == 0)
202 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
203 if (to_max > m_buffer.size())
204 throw cvt_error("cvt_reader::get_buf fail, read size too large");
205
206 const auto rollback_size = m_end_pos - m_cur_pos;
207 if (to_max <= rollback_size)
208 {
209 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, to_max};
210 m_cur_pos += to_max;
211 if constexpr (!Saturate)
212 return res;
213 else
214 return res.first;
215 }
216
217 if (to_max > m_buffer.size() - m_cur_pos)
218 {
219 std::copy(m_buffer.data() + m_cur_pos, m_buffer.data() + m_end_pos, m_buffer.data());
220 m_cur_pos = 0;
221 m_end_pos = rollback_size;
222 }
223 const size_t aim_size = to_max - rollback_size;
224 auto read_size = m_kernel.get(m_buffer.data() + m_end_pos, aim_size);
225 m_end_pos += read_size;
226
227 if constexpr (Saturate)
228 {
229 while (read_size < aim_size)
230 {
231 const size_t new_aim_size = aim_size - read_size;
232 const auto new_read_size = m_kernel.get(m_buffer.data() + m_end_pos, new_aim_size);
233 if (new_read_size == 0)
234 throw cvt_error("get_buf<Saturate> fail: meet eos");
235 m_end_pos += new_read_size;
236 read_size += new_read_size;
237 }
238 }
239
240 if constexpr (!Saturate)
241 {
242 std::pair<const char_type*, size_t> res{m_buffer.data() + m_cur_pos, rollback_size + read_size};
243 m_cur_pos = m_end_pos;
244 return res;
245 }
246 else
247 {
248 const char_type* res = m_buffer.data() + m_cur_pos;
249 m_cur_pos = m_end_pos;
250 return res;
251 }

1117 size_t get(internal_type* to, size_t to_max)
1118 requires requires(CurrentType& t, cvt_reader<KernelType>& r, internal_type* data, size_t len) {
1119 { t.get_main(r, data, len) } -> std::same_as<size_t>;
1120 }
1121 {
1122 // Read failures do not taint: they only advance the kernel cursor
1123 // and possibly leave a per-call mbstate undefined; the underlying
1124 // stream content is unchanged, and callers can reseek to recover.
1125 // We still honour any pre-existing taint so that a writer that
1126 // corrupted the stream cannot then read back a confused image.
1127 assert_not_tainted();
1128
1129 cvt_reader<KernelType> reader(m_kernel, m_tmp_io_buffer);
1130
1131 if (!m_is_bos_done)
1132 {
1133 if (to_max == 0) return 0;
1134
1135 constexpr size_t ext_size = sizeof(external_type);
1136
1137 reader.reset(s_bos_chunk);
1138 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1139 char* dest_bytes = reinterpret_cast<char*>(to);
1140 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1141 const char* dest_bytes_end = reinterpret_cast<const char*>(to + to_max);
1142
1143 // BOS phase requires complete data: if the stream ends unexpectedly
1144 // before providing enough bytes, this is a format error and we throw.
1145 // Unlike get_main() which may return partial data, BOS is all-or-nothing.
1146
1147 // Read full external_type chunks
1148 while (dest_bytes + ext_size <= dest_bytes_end)
1149 {
1150 const size_t units_to_read_now = std::min((size_t)((dest_bytes_end - dest_bytes) / ext_size), s_bos_chunk);
1151
1152 const auto* src_buf = reader.template get_buf<true>(units_to_read_now);
1153 std::memcpy(dest_bytes, src_buf, units_to_read_now * ext_size);
1154 dest_bytes += units_to_read_now * ext_size;
1155 }
1156
1157 // Read the final partial internal_type if needed
1158 const size_t final_remainder_bytes = dest_bytes_end - dest_bytes;
1159 if (final_remainder_bytes > 0)
1160 {
1161 const auto* src_buf = reader.template get_buf<true>(1);
1162 std::memcpy(dest_bytes, src_buf, final_remainder_bytes);
1163 dest_bytes += final_remainder_bytes;
1164 }
1165
1166 return to_max;
1167 }
1168 else
1169 {
1170 if (m_io_status != io_status::input)
1171 {
1172 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1173 static_cast<CurrentType*>(this)->switch_to_get();
1174 else
1175 throw cvt_error("abs_cvt::get fail: cannot switch to input mode");
1176 }
1177 return static_cast<CurrentType*>(this)->get_main(reader, to, to_max);
1178 }

1226 void put(const internal_type* to, size_t to_size)
1227 requires requires(CurrentType& t, cvt_writer<KernelType>& w, const internal_type* data, size_t len) {
1228 { t.put_main(w, data, len) } -> std::same_as<void>;
1229 }
1230 {
1231 assert_not_tainted();
1232
1233 // IO-direction switching does not write bytes to the output stream:
1234 // a failed switch_to_put() leaves the stream unchanged, so the
1235 // converter is not corrupted and must not be tainted. Perform the
1236 // switch BEFORE entering the taint-on-exception region so that only
1237 // actual kernel writes can cause tainting.
1238 if (m_is_bos_done && m_io_status != io_status::output)
1239 {
1240 if constexpr (cvt_cpt::support_io_switch<CurrentType>)
1241 static_cast<CurrentType*>(this)->switch_to_put();
1242 else
1243 throw cvt_error("abs_cvt::put fail: cannot switch to output mode");
1244 }
1245
1246 try
1247 {
1248 cvt_writer<KernelType> writer(m_kernel, m_tmp_io_buffer);
1249 if (!m_is_bos_done)
1250 {
1251 if (to_size == 0) return;
1252
1253 constexpr size_t ext_size = sizeof(external_type);
1254
1255 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1256 auto to_bytes = reinterpret_cast<const char*>(to);
1257 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1258 auto to_bytes_end = reinterpret_cast<const char*>(to + to_size);
1259
1260 // BOS phase writes complete external_type units. If the input data
1261 // is not a multiple of ext_size, pad the final unit with zeros.
1262
1263 writer.reset(s_bos_chunk);
1264 while (to_bytes + ext_size <= to_bytes_end)
1265 {
1266 auto dest_count = std::min<size_t>((to_bytes_end - to_bytes) / ext_size, s_bos_chunk);
1267 auto ptr = writer.put_buf(dest_count);
1268 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1269 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, dest_count * ext_size);
1270 to_bytes += dest_count * ext_size;
1271 }
1272
1273 if (to_bytes < to_bytes_end)
1274 {
1275 // The while loop above guarantees remaining < ext_size.
1276 // The modulo is logically redundant but helps the compiler's
1277 // static analyzer prove the bound for memset size calculation.
1278 size_t remaining = static_cast<size_t>(to_bytes_end - to_bytes) % ext_size;
1279 auto ptr = writer.put_buf(1);
1280 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1281 std::memcpy(reinterpret_cast<char*>(ptr), to_bytes, remaining);
1282 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
1283 std::memset(reinterpret_cast<char*>(ptr) + remaining, 0, ext_size - remaining);
1284 }
1285
1286 writer.commit();
1287 }
1288 else
1289 {
1290 // switch_to_put() was already called above if needed;
1291 // m_io_status is guaranteed to be output at this point.
1292 static_cast<CurrentType*>(this)->put_main(writer, to, to_size);
1293 // commit() is the sole responsibility of abs_cvt::put: derived
1294 // put_main implementations must NOT call writer.commit() themselves.
1295 // If put_main throws, this commit is skipped and any uncommitted
1296 // buffer contents are dropped at writer destruction; the catch
1297 // below taints the converter.
1298 writer.commit();
1299 }
1300 }
1301 catch (...)
1302 {
1303 m_is_tainted = true;
1304 throw;
1305 }
1306 }
