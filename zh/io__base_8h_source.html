371 void handle_exception(const std::exception_ptr& ex, bool at_eof = false)
372 {
373 if (!ex) return;

870 void access_callbacks(const locale<TChar>& new_loc)
871 {
872 std::exception_ptr throw_exception = nullptr;
873
874 for (const auto& [cb, id] : m_callbacks)
875 {
876 try
877 {
878 auto pre = m_pwords.find(id);
879 std::shared_ptr<void> old_data =
880 (pre != m_pwords.end()) ? pre->second : nullptr;
881
882 std::shared_ptr<void> new_data = cb(new_loc, old_data);
883
884 // Re-locate after the callback: cb may reentrantly mutate
885 // m_pwords (via set_pword), which can rehash/erase and would
886 // invalidate an iterator held across the call.
887 auto it = m_pwords.find(id);
888 if (new_data)
889 {
890 if (it != m_pwords.end()) it->second = std::move(new_data);
891 else m_pwords.insert({id, std::move(new_data)});
892 }
893 else if (it != m_pwords.end())
894 {
895 m_pwords.erase(it);
896 }
897 }
898 catch(...)
899 {
900 if (!throw_exception)
901 throw_exception = std::current_exception();
902 }
903 }
904
905 if (throw_exception)
906 std::rethrow_exception(throw_exception);
907 }
