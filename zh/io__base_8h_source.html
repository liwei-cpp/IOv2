853 void access_callbacks(const locale<TChar>& new_loc)
854 {
855 std::exception_ptr throw_exception = nullptr;
856
857 for (const auto& [cb, id] : m_callbacks)
858 {
859 try
860 {
861 auto pre = m_pwords.find(id);
862 std::shared_ptr<void> old_data =
863 (pre != m_pwords.end()) ? pre->second : nullptr;
864
865 std::shared_ptr<void> new_data = cb(new_loc, old_data);
866
867 // Re-locate after the callback: cb may reentrantly mutate
868 // m_pwords (via set_pword), which can rehash/erase and would
869 // invalidate an iterator held across the call.
870 auto it = m_pwords.find(id);
871 if (new_data)
872 {
873 if (it != m_pwords.end()) it->second = std::move(new_data);
874 else m_pwords.insert({id, std::move(new_data)});
875 }
876 else if (it != m_pwords.end())
877 {
878 m_pwords.erase(it);
879 }
880 }
881 catch(...)
882 {
883 if (!throw_exception)
884 throw_exception = std::current_exception();
885 }
886 }
887
888 if (throw_exception)
889 std::rethrow_exception(throw_exception);
890 }
