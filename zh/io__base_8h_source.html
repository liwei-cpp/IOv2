188 {
189 std::lock_guard guard(m_state_mutex);
190 m_stream_state.store(s);
191 if ((s & ios_defs::devfailbit) == ios_defs::goodbit) m_exp_dev_fail = std::exception_ptr{};
192 if ((s & ios_defs::cvtfailbit) == ios_defs::goodbit) m_exp_cvt_fail = std::exception_ptr{};
193 if ((s & ios_defs::strfailbit) == ios_defs::goodbit) m_exp_str_fail = std::exception_ptr{};

391 void handle_exception(const std::exception_ptr& ex, bool at_eof = false)
392 {
393 if (!ex) return;
394 std::lock_guard guard(m_state_mutex);
395 const ios_defs::iostate eof = at_eof ? ios_defs::eofbit : ios_defs::goodbit;
396 try
397 {
398 std::rethrow_exception(ex);
399 }
400 catch (device_error&)
401 {
402 if (!m_exp_dev_fail)
403 m_exp_dev_fail = std::current_exception();
404 setstate(ios_defs::devfailbit | eof);
405 }
406 catch (cvt_error&)
407 {
408 if (!m_exp_cvt_fail)
409 m_exp_cvt_fail = std::current_exception();
410 setstate(ios_defs::cvtfailbit | eof);
411 }
412 catch (stream_error&)
413 {
414 if (!m_exp_str_fail)
415 m_exp_str_fail = std::current_exception();
416 setstate(ios_defs::strfailbit | eof);
417 }
418 catch (eof_error&)
419 {
420 setstate(ios_defs::eofbit);
421 }
422 catch(...)
423 {
424 if (!m_exp_other_fail)
425 m_exp_other_fail = std::current_exception();
426 setstate(ios_defs::otherfailbit | eof);
427 }

885 void access_callbacks(const locale<TChar>& new_loc)
886 {
887 std::exception_ptr throw_exception = nullptr;
888
889 for (const auto& [cb, id] : m_callbacks)
890 {
891 try
892 {
893 auto pre = m_pwords.find(id);
894 std::shared_ptr<void> old_data =
895 (pre != m_pwords.end()) ? pre->second : nullptr;
896
897 std::shared_ptr<void> new_data = cb(new_loc, old_data);
898
899 // Re-locate after the callback: cb may reentrantly mutate
900 // m_pwords (via set_pword), which can rehash/erase and would
901 // invalidate an iterator held across the call.
902 auto it = m_pwords.find(id);
903 if (new_data)
904 {
905 if (it != m_pwords.end()) it->second = std::move(new_data);
906 else m_pwords.insert({id, std::move(new_data)});
907 }
908 else if (it != m_pwords.end())
909 {
910 m_pwords.erase(it);
911 }
912 }
913 catch(...)
914 {
915 if (!throw_exception)
916 throw_exception = std::current_exception();
917 }
918 }
919
920 if (throw_exception)
921 std::rethrow_exception(throw_exception);
922 }
