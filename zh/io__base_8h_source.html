187 void clear(ios_defs::iostate s = ios_defs::goodbit)
188 {
189 std::lock_guard guard(m_state_mutex);
190 m_stream_state.store(s);
191 if ((s & ios_defs::devfailbit) == ios_defs::goodbit) m_exp_dev_fail = std::exception_ptr{};
192 if ((s & ios_defs::cvtfailbit) == ios_defs::goodbit) m_exp_cvt_fail = std::exception_ptr{};
193 if ((s & ios_defs::strfailbit) == ios_defs::goodbit) m_exp_str_fail = std::exception_ptr{};
194 if ((s & ios_defs::otherfailbit) == ios_defs::goodbit) m_exp_other_fail = std::exception_ptr{};
195
196 ios_defs::iostate state_in_exp = m_exception & s;
197 if (state_in_exp & ios_defs::devfailbit)
198 {
199 if (m_exp_dev_fail)
200 std::rethrow_exception(std::exchange(m_exp_dev_fail, nullptr));
201 else
202 throw device_error("device failure bit has been set");
203 }

391 void handle_exception(const std::exception_ptr& ex, bool at_eof = false)
392 {
393 if (!ex) return;
394 std::lock_guard guard(m_state_mutex);
395 const ios_defs::iostate eof = at_eof ? ios_defs::eofbit : ios_defs::goodbit;
396 try
397 {
398 std::rethrow_exception(ex);
399 }
400 catch (device_error&)
401 {
402 if (!m_exp_dev_fail)
403 m_exp_dev_fail = std::current_exception();
404 setstate(ios_defs::devfailbit | eof);
405 }
406 catch (cvt_error&)
407 {
408 if (!m_exp_cvt_fail)
409 m_exp_cvt_fail = std::current_exception();
410 setstate(ios_defs::cvtfailbit | eof);
411 }
412 catch (stream_error&)
413 {
414 if (!m_exp_str_fail)
415 m_exp_str_fail = std::current_exception();
416 setstate(ios_defs::strfailbit | eof);
417 }
418 catch (eof_error&)
419 {
420 setstate(ios_defs::eofbit);
421 }
422 catch(...)
423 {
424 if (!m_exp_other_fail)
425 m_exp_other_fail = std::current_exception();
426 setstate(ios_defs::otherfailbit | eof);
427 }
428 }

893 void access_callbacks(const locale<TChar>& new_loc)
894 {
895 std::exception_ptr throw_exception = nullptr;
896
897 for (const auto& [cb, id] : m_callbacks)
898 {
899 try
900 {
901 auto pre = m_pwords.find(id);
902 std::shared_ptr<void> old_data =
903 (pre != m_pwords.end()) ? pre->second : nullptr;
904
905 std::shared_ptr<void> new_data = cb(new_loc, old_data);
906
907 // Re-locate after the callback: cb may reentrantly mutate
908 // m_pwords (via set_pword), which can rehash/erase and would
909 // invalidate an iterator held across the call.
910 auto it = m_pwords.find(id);
911 if (new_data)
912 {
913 if (it != m_pwords.end()) it->second = std::move(new_data);
914 else m_pwords.insert({id, std::move(new_data)});
915 }
916 else if (it != m_pwords.end())
917 {
918 m_pwords.erase(it);
919 }
920 }
921 catch(...)
922 {
923 if (!throw_exception)
924 throw_exception = std::current_exception();
925 }
926 }
927
928 if (throw_exception)
929 std::rethrow_exception(throw_exception);
930 }
