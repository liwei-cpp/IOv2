188 {
189 std::lock_guard guard(m_state_mutex);
190 m_stream_state.store(s);
191 if ((s & ios_defs::devfailbit) == ios_defs::goodbit) m_exp_dev_fail = std::exception_ptr{};
192 if ((s & ios_defs::cvtfailbit) == ios_defs::goodbit) m_exp_cvt_fail = std::exception_ptr{};
193 if ((s & ios_defs::strfailbit) == ios_defs::goodbit) m_exp_str_fail = std::exception_ptr{};
194 if ((s & ios_defs::otherfailbit) == ios_defs::goodbit) m_exp_other_fail = std::exception_ptr{};

385 void handle_exception(const std::exception_ptr& ex, bool at_eof = false)
386 {
387 if (!ex) return;
388 std::lock_guard guard(m_state_mutex);
389 const ios_defs::iostate eof = at_eof ? ios_defs::eofbit : ios_defs::goodbit;
390 try
391 {
392 std::rethrow_exception(ex);
393 }
394 catch (device_error&)
395 {
396 if (!m_exp_dev_fail)
397 m_exp_dev_fail = std::current_exception();
398 setstate(ios_defs::devfailbit | eof);
399 }
400 catch (cvt_error&)
401 {
402 if (!m_exp_cvt_fail)
403 m_exp_cvt_fail = std::current_exception();
404 setstate(ios_defs::cvtfailbit | eof);
405 }
406 catch (stream_error&)
407 {
408 if (!m_exp_str_fail)
409 m_exp_str_fail = std::current_exception();
410 setstate(ios_defs::strfailbit | eof);
411 }
412 catch (eof_error&)
413 {
414 setstate(ios_defs::eofbit);
415 }
416 catch(...)
417 {
418 if (!m_exp_other_fail)
419 m_exp_other_fail = std::current_exception();
420 setstate(ios_defs::otherfailbit | eof);
421 }
422 }

887 void access_callbacks(const locale<TChar>& new_loc)
888 {
889 std::exception_ptr throw_exception = nullptr;
890
891 for (const auto& [cb, id] : m_callbacks)
892 {
893 try
894 {
895 auto pre = m_pwords.find(id);
896 std::shared_ptr<void> old_data =
897 (pre != m_pwords.end()) ? pre->second : nullptr;
898
899 std::shared_ptr<void> new_data = cb(new_loc, old_data);
900
901 // Re-locate after the callback: cb may reentrantly mutate
902 // m_pwords (via set_pword), which can rehash/erase and would
903 // invalidate an iterator held across the call.
904 auto it = m_pwords.find(id);
905 if (new_data)
906 {
907 if (it != m_pwords.end()) it->second = std::move(new_data);
908 else m_pwords.insert({id, std::move(new_data)});
909 }
910 else if (it != m_pwords.end())
911 {
912 m_pwords.erase(it);
913 }
914 }
915 catch(...)
916 {
917 if (!throw_exception)
918 throw_exception = std::current_exception();
919 }
920 }
921
922 if (throw_exception)
923 std::rethrow_exception(throw_exception);
924 }
