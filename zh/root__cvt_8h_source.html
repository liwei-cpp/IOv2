507 void main_cont_beg()
508 {
509 // bos() must be called before main_cont_beg() to establish io_status
510 assert(m_io_status != io_status::neutral);
511
512 if constexpr (dev_cpt::support_put<device_type>)
513 {
514 if (m_io_status == io_status::output)
515 flush();
516 }
517
518 if constexpr (dev_cpt::support_positioning<device_type>)
519 {
520 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
521 // which advances the device position by the amount read. Consuming data
522 // from the buffer only moves m_buf_cur forward, never increasing the
523 // buffered amount beyond what was read from the device.
524 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
525 assert(m_device.dtell() >= buffered);
526 m_bos_len = m_device.dtell() - buffered;
527 }
528 }

729 void put(const internal_type* to, size_t to_size)
730 requires (dev_cpt::support_put<device_type>)
731 {
732 if constexpr (dev_cpt::support_get<device_type>)
733 {
734 if (m_io_status != io_status::output)
735 switch_to_put();
736 }
737 if (m_io_status != io_status::output)
738 throw cvt_error("root_cvt::put fails: invalid io status");
739
740 const size_t buf_used = m_buf_cur - m_buffer.data();
741 const size_t remain = s_buffer_length - buf_used;
742 if (to_size < remain)
743 {
744 m_buf_cur = std::copy(to, to + to_size, m_buf_cur);
745 return;
746 }
747
748 if (to_size == remain)
749 {
750 std::copy(to, to + to_size, m_buf_cur);
751 m_device.dput(m_buffer.data(), s_buffer_length);
752 m_buf_cur = m_buffer.data();
753 return;
754 }
755
756 if (buf_used > 0)
757 m_device.dput(m_buffer.data(), buf_used);
758 if (to_size < s_buffer_length / 2)
759 m_buf_cur = std::copy(to, to + to_size, m_buffer.data());
760 else
761 {
762 m_buf_cur = m_buffer.data();
763 m_device.dput(to, to_size);
764 }
765 }

813 [[nodiscard]] size_t tell() const
814 requires (dev_cpt::support_positioning<device_type>)
815 {
816 const size_t device_tell = m_device.dtell();
817
818 switch(m_io_status)
819 {
820 case io_status::output:
821 {
822 // Invariant: device_tell >= m_bos_len. The device position only advances
823 // via flush operations, never moving backward from the stream start.
824 assert(device_tell >= m_bos_len);
825 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
826 return device_tell - m_bos_len + buf_used;
827 }
828 case io_status::input:
829 {
830 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
831 // main_cont_beg(): buffer data comes from dget(), which advances device
832 // position by the amount read.
833 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
834 assert(device_tell >= m_bos_len);
835 assert(device_tell - m_bos_len >= buffered);
836 return device_tell - m_bos_len - buffered;
837 }
838 default:
839 throw cvt_error("root_cvt::tell fails: invalid io status");
840 }
841 }

1712 template <bool Saturate = false>
1713 auto get_buf(size_t to_max)
1714 {
1715 if (to_max == 0)
1716 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
1717 if (to_max > KernelType::s_buffer_length)
1718 throw cvt_error("cvt_reader::get_buf fail, read size too large");
1719
1720 if constexpr (dev_cpt::support_put<device_type>)
1721 {
1722 if (m_kernel.m_io_status != io_status::input)
1723 m_kernel.switch_to_get();
1724 }
1725 if (m_kernel.m_io_status != io_status::input)
1726 throw cvt_error("cvt_reader::get_buf fail, invalid io status");
1727
1728 const size_t remain = m_kernel.m_buf_end - m_kernel.m_buf_cur;
1729 // need to read, then read at most as it can
1730 if (remain < to_max)
1731 {
1732 if (remain != 0)
1733 m_kernel.m_buf_end = std::copy(m_kernel.m_buf_cur, m_kernel.m_buf_end, m_kernel.m_buffer.data());
1734 else
1735 m_kernel.m_buf_end = m_kernel.m_buffer.data();
1736
1737 m_kernel.m_buf_cur = m_kernel.m_buffer.data();
1738 m_kernel.m_buf_end += m_kernel.m_device.dget(m_kernel.m_buf_end, KernelType::s_buffer_length - remain);
1739 }
1740
1741 if constexpr (Saturate)
1742 {
1743 const auto aim_ptr = to_max + m_kernel.m_buf_cur;
1744 while (m_kernel.m_buf_end < aim_ptr)
1745 {
1746 auto new_size = m_kernel.m_device.dget(m_kernel.m_buf_end, aim_ptr - m_kernel.m_buf_end);
1747 if (new_size == 0)
1748 throw cvt_error("get_buf<Saturate> fail: meet eos");
1749 m_kernel.m_buf_end += new_size;
1750 }
1751 }
1752
1753 if constexpr (!Saturate)
