492 void main_cont_beg()
493 {
494 // bos() must be called before main_cont_beg() to establish io_status
495 assert(m_io_status != io_status::neutral);
496
497 if constexpr (dev_cpt::support_put<device_type>)
498 {
499 if (m_io_status == io_status::output)
500 flush();
501 }
502
503 if constexpr (dev_cpt::support_positioning<device_type>)
504 {
505 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
506 // which advances the device position by the amount read. Consuming data
507 // from the buffer only moves m_buf_cur forward, never increasing the
508 // buffered amount beyond what was read from the device.
509 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
510 assert(m_device.dtell() >= buffered);
511 m_bos_len = m_device.dtell() - buffered;
512 }
513 }

714 void put(const internal_type* to, size_t to_size)
715 requires (dev_cpt::support_put<device_type>)
716 {
717 if constexpr (dev_cpt::support_get<device_type>)
718 {
719 if (m_io_status != io_status::output)
720 switch_to_put();
721 }
722 if (m_io_status != io_status::output)
723 throw cvt_error("root_cvt::put fails: invalid io status");
724
725 const size_t buf_used = m_buf_cur - m_buffer.data();
726 const size_t remain = s_buffer_length - buf_used;
727 if (to_size < remain)
728 {
729 m_buf_cur = std::copy(to, to + to_size, m_buf_cur);
730 return;
731 }
732
733 if (to_size == remain)
734 {
735 std::copy(to, to + to_size, m_buf_cur);
736 m_device.dput(m_buffer.data(), s_buffer_length);
737 m_buf_cur = m_buffer.data();
738 return;
739 }
740
741 if (buf_used > 0)
742 m_device.dput(m_buffer.data(), buf_used);
743 if (to_size < s_buffer_length / 2)
744 m_buf_cur = std::copy(to, to + to_size, m_buffer.data());
745 else
746 {
747 m_buf_cur = m_buffer.data();
748 m_device.dput(to, to_size);
749 }
750 }

798 [[nodiscard]] size_t tell() const
799 requires (dev_cpt::support_positioning<device_type>)
800 {
801 const size_t device_tell = m_device.dtell();
802
803 switch(m_io_status)
804 {
805 case io_status::output:
806 {
807 // Invariant: device_tell >= m_bos_len. The device position only advances
808 // via flush operations, never moving backward from the stream start.
809 assert(device_tell >= m_bos_len);
810 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
811 return device_tell - m_bos_len + buf_used;
812 }
813 case io_status::input:
814 {
815 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
816 // main_cont_beg(): buffer data comes from dget(), which advances device
817 // position by the amount read.
818 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
819 assert(device_tell >= m_bos_len);
820 assert(device_tell - m_bos_len >= buffered);
821 return device_tell - m_bos_len - buffered;
822 }
823 default:
824 throw cvt_error("root_cvt::tell fails: invalid io status");
825 }
826 }

1696 template <bool Saturate = false>
1697 auto get_buf(size_t to_max)
1698 {
1699 if (to_max == 0)
1700 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
1701 if (to_max > KernelType::s_buffer_length)
1702 throw cvt_error("cvt_reader::get_buf fail, read size too large");
1703
1704 if constexpr (dev_cpt::support_put<device_type>)
1705 {
1706 if (m_kernel.m_io_status != io_status::input)
1707 m_kernel.switch_to_get();
1708 }
1709 if (m_kernel.m_io_status != io_status::input)
1710 throw cvt_error("cvt_reader::get_buf fail, invalid io status");
1711
1712 const size_t remain = m_kernel.m_buf_end - m_kernel.m_buf_cur;
1713 // need to read, then read at most as it can
1714 if (remain < to_max)
1715 {
1716 if (remain != 0)
1717 m_kernel.m_buf_end = std::copy(m_kernel.m_buf_cur, m_kernel.m_buf_end, m_kernel.m_buffer.data());
1718 else
1719 m_kernel.m_buf_end = m_kernel.m_buffer.data();
1720
1721 m_kernel.m_buf_cur = m_kernel.m_buffer.data();
1722 m_kernel.m_buf_end += m_kernel.m_device.dget(m_kernel.m_buf_end, KernelType::s_buffer_length - remain);
1723 }
1724
1725 if constexpr (Saturate)
1726 {
1727 const auto aim_ptr = to_max + m_kernel.m_buf_cur;
1728 while (m_kernel.m_buf_end < aim_ptr)
1729 {
1730 auto new_size = m_kernel.m_device.dget(m_kernel.m_buf_end, aim_ptr - m_kernel.m_buf_end);
1731 if (new_size == 0)
1732 throw cvt_error("get_buf<Saturate> fail: meet eos");
1733 m_kernel.m_buf_end += new_size;
1734 }
1735 }
1736
1737 if constexpr (!Saturate)
1738 {
1739 auto res_len = std::min<size_t>(m_kernel.m_buf_end - m_kernel.m_buf_cur, to_max);
1740 std::pair<const char_type*, size_t> res{m_kernel.m_buf_cur, res_len};
1741 m_kernel.m_buf_cur += res_len;
1742 return res;
1743 }
1744 else
1745 {
1746 const char_type* res = m_kernel.m_buf_cur;
1747 m_kernel.m_buf_cur += to_max;
1748 return res;
1749 }
