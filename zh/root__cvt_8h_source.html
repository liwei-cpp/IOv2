476 void main_cont_beg()
477 {
478 // bos() must be called before main_cont_beg() to establish io_status
479 assert(m_io_status != io_status::neutral);
480
481 if constexpr (dev_cpt::support_put<device_type>)
482 {
483 if (m_io_status == io_status::output)
484 flush();
485 }
486
487 if constexpr (dev_cpt::support_positioning<device_type>)
488 {
489 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
490 // which advances the device position by the amount read. Consuming data
491 // from the buffer only moves m_buf_cur forward, never increasing the
492 // buffered amount beyond what was read from the device.
493 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
494 assert(m_device.dtell() >= buffered);
495 m_bos_len = m_device.dtell() - buffered;
496 }

618 size_t get(internal_type* to, size_t to_max)

678 void put(const internal_type* to, size_t to_size)
679 requires (dev_cpt::support_put<device_type>)
680 {
681 if constexpr (dev_cpt::support_get<device_type>)
682 {
683 if (m_io_status != io_status::output)
684 switch_to_put();
685 }
686 if (m_io_status != io_status::output)
687 throw cvt_error("root_cvt::put fails: invalid io status");
688
689 const size_t buf_used = m_buf_cur - m_buffer.data();
690 const size_t remain = s_buffer_length - buf_used;
691 if (to_size < remain)
692 {
693 m_buf_cur = std::copy(to, to + to_size, m_buf_cur);
694 return;
695 }
696
697 if (to_size == remain)
698 {
699 std::copy(to, to + to_size, m_buf_cur);
700 m_device.dput(m_buffer.data(), s_buffer_length);
701 m_buf_cur = m_buffer.data();
702 return;
703 }
704
705 if (buf_used > 0)
706 m_device.dput(m_buffer.data(), buf_used);
707 if (to_size < s_buffer_length / 2)
708 m_buf_cur = std::copy(to, to + to_size, m_buffer.data());
709 else
710 {
711 m_buf_cur = m_buffer.data();
712 m_device.dput(to, to_size);
713 }
714 }

762 [[nodiscard]] size_t tell() const
763 requires (dev_cpt::support_positioning<device_type>)
764 {
765 const size_t device_tell = m_device.dtell();
766
767 switch(m_io_status)
768 {
769 case io_status::output:
770 {
771 // Invariant: device_tell >= m_bos_len. The device position only advances
772 // via flush operations, never moving backward from the stream start.
773 assert(device_tell >= m_bos_len);
774 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
775 return device_tell - m_bos_len + buf_used;
776 }
777 case io_status::input:
778 {
779 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
780 // main_cont_beg(): buffer data comes from dget(), which advances device
781 // position by the amount read.
782 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
783 assert(device_tell >= m_bos_len);
784 assert(device_tell - m_bos_len >= buffered);
785 return device_tell - m_bos_len - buffered;
786 }
787 default:
788 throw cvt_error("root_cvt::tell fails: invalid io status");
789 }
790 }

1623 template <bool Saturate = false>
1624 auto get_buf(size_t to_max)
1625 {
1626 if (to_max == 0)
1627 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero");
1628 if (to_max > KernelType::s_buffer_length)
1629 throw cvt_error("cvt_reader::get_buf fail, read size too large");
1630
1631 if constexpr (dev_cpt::support_put<device_type>)
1632 {
1633 if (m_kernel.m_io_status != io_status::input)
1634 m_kernel.switch_to_get();
1635 }
1636 if (m_kernel.m_io_status != io_status::input)
1637 throw cvt_error("cvt_reader::get_buf fail, invalid io status");
1638
1639 const size_t remain = m_kernel.m_buf_end - m_kernel.m_buf_cur;
1640 // need to read, then read at most as it can
1641 if (remain < to_max)
1642 {
1643 if (remain != 0)
1644 m_kernel.m_buf_end = std::copy(m_kernel.m_buf_cur, m_kernel.m_buf_end, m_kernel.m_buffer.data());
1645 else
1646 m_kernel.m_buf_end = m_kernel.m_buffer.data();
1647
1648 m_kernel.m_buf_cur = m_kernel.m_buffer.data();
1649 m_kernel.m_buf_end += m_kernel.m_device.dget(m_kernel.m_buf_end, KernelType::s_buffer_length - remain);
1650 }
1651
1652 if constexpr (Saturate)
1653 {
1654 const auto aim_ptr = to_max + m_kernel.m_buf_cur;
1655 while (m_kernel.m_buf_end < aim_ptr)
1656 {
1657 auto new_size = m_kernel.m_device.dget(m_kernel.m_buf_end, aim_ptr - m_kernel.m_buf_end);
1658 if (new_size == 0)
1659 throw cvt_error("get_buf<Saturate> fail: meet eos");
1660 m_kernel.m_buf_end += new_size;
1661 }
1662 }
1663
1664 if constexpr (!Saturate)
1665 {
1666 auto res_len = std::min<size_t>(m_kernel.m_buf_end - m_kernel.m_buf_cur, to_max);
1667 std::pair<const char_type*, size_t> res{m_kernel.m_buf_cur, res_len};
1668 m_kernel.m_buf_cur += res_len;
