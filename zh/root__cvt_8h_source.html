470 void main_cont_beg()
471 {
472 // bos() must be called before main_cont_beg() to establish io_status
473 assert(m_io_status != io_status::neutral);
474
475 if constexpr (dev_cpt::support_put<device_type>)
476 {
477 if (m_io_status == io_status::output)
478 flush();
479 }
480
481 if constexpr (dev_cpt::support_positioning<device_type>)
482 {
483 // Invariant: dtell() >= buffered data. The buffer is filled via dget(),
484 // which advances the device position by the amount read. Consuming data
485 // from the buffer only moves m_buf_cur forward, never increasing the
486 // buffered amount beyond what was read from the device.
487 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
488 assert(m_device.dtell() >= buffered);
489 m_bos_len = m_device.dtell() - buffered;
490 }

612 size_t get(internal_type* to, size_t to_max)
613 requires (dev_cpt::support_get<device_type>)

672 void put(const internal_type* to, size_t to_size)
673 requires (dev_cpt::support_put<device_type>)
674 {
675 if constexpr (dev_cpt::support_get<device_type>)
676 {
677 if (m_io_status != io_status::output)
678 switch_to_put();
679 }
680 if (m_io_status != io_status::output)
681 throw cvt_error("root_cvt::put fails: invalid io status");
682
683 const size_t buf_used = m_buf_cur - m_buffer.data();
684 const size_t remain = s_buffer_length - buf_used;
685 if (to_size < remain)
686 {
687 m_buf_cur = std::copy(to, to + to_size, m_buf_cur);
688 return;
689 }
690
691 if (to_size == remain)
692 {
693 std::copy(to, to + to_size, m_buf_cur);
694 m_device.dput(m_buffer.data(), s_buffer_length);
695 m_buf_cur = m_buffer.data();
696 return;
697 }
698
699 if (buf_used > 0)
700 m_device.dput(m_buffer.data(), buf_used);
701 if (to_size < s_buffer_length / 2)
702 m_buf_cur = std::copy(to, to + to_size, m_buffer.data());
703 else
704 {
705 m_buf_cur = m_buffer.data();
706 m_device.dput(to, to_size);
707 }
708 }

756 [[nodiscard]] size_t tell() const
757 requires (dev_cpt::support_positioning<device_type>)
758 {
759 const size_t device_tell = m_device.dtell();
760
761 switch(m_io_status)
762 {
763 case io_status::output:
764 {
765 // Invariant: device_tell >= m_bos_len. The device position only advances
766 // via flush operations, never moving backward from the stream start.
767 assert(device_tell >= m_bos_len);
768 const auto buf_used = static_cast<size_t>(m_buf_cur - m_buffer.data());
769 return device_tell - m_bos_len + buf_used;
770 }
771 case io_status::input:
772 {
773 // Invariant: device_tell - m_bos_len >= buffered. Same invariant as
774 // main_cont_beg(): buffer data comes from dget(), which advances device
775 // position by the amount read.
776 const auto buffered = static_cast<size_t>(m_buf_end - m_buf_cur);
777 assert(device_tell >= m_bos_len);
778 assert(device_tell - m_bos_len >= buffered);
779 return device_tell - m_bos_len - buffered;
780 }
781 default:
782 throw cvt_error("root_cvt::tell fails: invalid io status");
783 }
784 }

1603 template <bool Saturate = false>
1604 auto get_buf(size_t to_max)
1605 {
1606 if (to_max == 0)
1607 throw cvt_error("cvt_reader::get_buf fail, read size cannot be zero.");
1608 if (to_max > KernelType::s_buffer_length)
1609 throw cvt_error("cvt_reader::get_buf fail, read size too large.");
1610
1611 if constexpr (dev_cpt::support_put<device_type>)
1612 {
1613 if (m_kernel.m_io_status != io_status::input)
1614 m_kernel.switch_to_get();
1615 }
1616 if (m_kernel.m_io_status != io_status::input)
1617 throw cvt_error("cvt_reader::get_buf fail, invalid io status");
1618
1619 const size_t remain = m_kernel.m_buf_end - m_kernel.m_buf_cur;
1620 // need to read, then read at most as it can
1621 if (remain < to_max)
1622 {
1623 if (remain != 0)
1624 m_kernel.m_buf_end = std::copy(m_kernel.m_buf_cur, m_kernel.m_buf_end, m_kernel.m_buffer.data());
1625 else
1626 m_kernel.m_buf_end = m_kernel.m_buffer.data();
1627
1628 m_kernel.m_buf_cur = m_kernel.m_buffer.data();
1629 m_kernel.m_buf_end += m_kernel.m_device.dget(m_kernel.m_buf_end, KernelType::s_buffer_length - remain);
1630 }
1631
1632 if constexpr (Saturate)
1633 {
1634 const auto aim_ptr = to_max + m_kernel.m_buf_cur;
1635 while (m_kernel.m_buf_end < aim_ptr)
1636 {
1637 auto new_size = m_kernel.m_device.dget(m_kernel.m_buf_end, aim_ptr - m_kernel.m_buf_end);
1638 if (new_size == 0)
1639 throw cvt_error("get_buf<Saturate> fail: meet eos");
1640 m_kernel.m_buf_end += new_size;
1641 }
1642 }
1643
1644 if constexpr (!Saturate)
1645 {
1646 auto res_len = std::min<size_t>(m_kernel.m_buf_end - m_kernel.m_buf_cur, to_max);
1647 std::pair<const char_type*, size_t> res{m_kernel.m_buf_cur, res_len};
1648 m_kernel.m_buf_cur += res_len;
1649 return res;
1650 }
1651 else
1652 {
1653 const char_type* res = m_kernel.m_buf_cur;
1654 m_kernel.m_buf_cur += to_max;
1655 return res;
1656 }
1657 }
