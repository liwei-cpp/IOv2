23template <typename TInt>
24 requires std::is_same_v<TInt, wchar_t> ||
25 (std::is_same_v<TInt, char32_t> &&
26 (sizeof(char32_t) == sizeof(wchar_t)) &&
27 (static_cast<wchar_t>(U'李') == L'李') &&
28 (static_cast<char32_t>(L'伟') == U'伟'))
29struct codecvt_kernel<char, TInt>
30{
31 codecvt_kernel(const std::string& name)
32 : m_inter_locale(name.c_str())
33 {
34 clocale_user guard(m_inter_locale);
35
36 // there are no known constant length encodings
37 // m_epc == 1 means fix len
38 m_epc = MB_CUR_MAX;
39
40 // Note: mbtowc might not thread-safe. But when checking the glibc source code, it only impact the internal mbstate_t.
41 // This means that the status of internal mbstate_t is undefined when called by multiple threads.
42 // But the return number can be used to determine whether it is state-dependend or not.
43 m_is_state_dep = (std::mbtowc(nullptr, nullptr, MB_CUR_MAX) != 0);
44 init_state();
45 }
46
47 void init_state() { m_state = std::mbstate_t{}; }
48 bool is_init_state() const { return std::mbsinit(&m_state); }
49 unsigned epc() const { return m_epc; }
50 bool is_var_length() const { return m_epc != 1; }
51 bool is_state_dep() const { return m_is_state_dep; }
52
53 bool out_helper(TInt ch, char*& to, char* to_end)
54 {
55 clocale_user guard(m_inter_locale);
56 if (to_end - to < m_epc)
57 return false;
58
59 const size_t conv = std::wcrtomb(to, ch, &m_state);
60 if (conv == static_cast<size_t>(-1))
61 {
62 return false;
63 }
64 to += conv;
65
66 return true;
67 }
68
69 std::pair<bool, size_t> in_helper(const char*& from, const char* from_end,
70 TInt*& to, TInt* to_end)
71 {
72 clocale_user guard(m_inter_locale);
73 wchar_t wch = 0;
74 size_t i_count = 0;
75
76 const size_t to_max = to_end - to;
77 while (from < from_end && (i_count < to_max))
78 {
79 auto tmp_state = m_state;
80 size_t conv = mbrtowc(&wch, from, from_end - from, &tmp_state);
81 if (conv == static_cast<size_t>(-1))
82 return std::pair{false, i_count};
83 else if (conv == static_cast<size_t>(-2))
84 {
85 from = from_end;
86 m_state = tmp_state;
87 break;
88 }
89 else if (conv == 0)
90 {
91 *to++ = static_cast<TInt>(0);
92 size_t n = 1;
93 for (; n <= static_cast<size_t>(from_end - from); ++n)
94 {
95 auto tmp_state2(m_state);
96 if (mbrtowc(nullptr, from, n, &tmp_state2) == 0)
97 {
98 m_state = tmp_state2;
99 break;
100 }
101 }
102 from += n;
103 ++i_count;
104 }
105 else
106 {
107 m_state = tmp_state;
108 assert(is_init_state());
109 *to++ = wch;
110 from += conv;
111 ++i_count;
112 }
113 }
114
115 return std::pair{true, i_count};
116 }
117
118private:
119 clocale_wrapper m_inter_locale;
120 unsigned m_epc;
121 std::mbstate_t m_state;
122 bool m_is_state_dep;

125template <typename TInt>
126 requires std::is_same_v<TInt, char32_t> ||
127 (std::is_same_v<TInt, wchar_t> &&
128 (sizeof(char32_t) == sizeof(wchar_t)) &&
129 (static_cast<wchar_t>(U'李') == L'李') &&
130 (static_cast<char32_t>(L'伟') == U'伟'))
131struct codecvt_kernel<char8_t, TInt>
132{
133 codecvt_kernel() = default;
134
135 void init_state() { return; }
136 bool is_init_state() const { return true; }
137 unsigned epc() const { return 4; }
138 bool is_var_length() const { return true; }
139 bool is_state_dep() const { return false; }
140
141 bool out_helper(TInt ch, char8_t*& to, char8_t* to_end)
142 {
143 if (to_end - to < 4)
144 return false;
145 const uint32_t c = static_cast<uint32_t>(ch);
146 if ((uint32_t)0xD800 <= c && c <= (uint32_t)0xDFFF) [[unlikely]]
147 return false;
148
149 if (c < 0x80)
150 *to++ = static_cast<char8_t>(c);
151 else if (c <= (uint32_t)0x7ff)
152 {
153 if (to_end - to < 2) return false;
154 *to++ = static_cast<char8_t>((c >> 6) + 0xC0);
155 *to++ = static_cast<char8_t>((c & 0x3F) + 0x80);
156 }
157 else if (c <= (uint32_t)0xFFFF)
158 {
159 if (to_end - to < 3) return false;
160 *to++ = static_cast<char8_t>((c >> 12) + 0xE0);
161 *to++ = static_cast<char8_t>(((c >> 6) & 0x3F) + 0x80);
162 *to++ = static_cast<char8_t>((c & 0x3F) + 0x80);
163 }
164 else if (c <= (uint32_t)0x10FFFF)
165 {
166 if (to_end - to < 4) return false;
167 *to++ = static_cast<char8_t>((c >> 18) + 0xF0);
168 *to++ = static_cast<char8_t>(((c >> 12) & 0x3F) + 0x80);
169 *to++ = static_cast<char8_t>(((c >> 6) & 0x3F) + 0x80);
170 *to++ = static_cast<char8_t>((c & 0x3F) + 0x80);
171 }
172 else [[unlikely]]
173 return false;
174 return true;
175 }
176
177 std::pair<bool, size_t> in_helper(const char8_t*& from, const char8_t* from_end,
178 TInt*& to, TInt* to_end)
179 {
180 const TInt* const ori_to = to;
181
182 while ((from != from_end) && (to != to_end))
183 {
184 auto c1 = static_cast<uint32_t>(*from);
185 if (c1 < 0x80) [[likely]]
186 {
187 ++from;
188 *to++ = static_cast<TInt>(c1);
189 }
190 else if (c1 < 0xE0)
191 {
192 if (from_end - from < 2) break;
193 auto c2 = static_cast<uint32_t>(from[1]);
194 if ((c2 & 0xC0) != 0x80) [[unlikely]]
195 return std::pair{false, static_cast<size_t>(to - ori_to)};
196 auto c = (c1 << 6) + c2 - 0x3080;
197 if (c < 0x80) return std::pair{false, static_cast<size_t>(to - ori_to)};
198 *to++ = c;
199 from += 2;
200 }
201 else if (c1 < 0xF0)
202 {
203 if (from_end - from < 3) break;
204 auto c2 = static_cast<uint32_t>(from[1]);
205 auto c3 = static_cast<uint32_t>(from[2]);
206 if (((c2 & 0xC0) != 0x80) || ((c3 & 0xC0) != 0x80)) [[unlikely]]
207 return std::pair{false, static_cast<size_t>(to - ori_to)};
208 auto c = (c1 << 12) + (c2 << 6) + c3 - 0xE2080;
209 if (c < 0x800) return std::pair{false, static_cast<size_t>(to - ori_to)};
210 *to++ = c;
211 from += 3;
212 }
213 else if (c1 < 0xF8)
214 {
215 if (from_end - from < 4) break;
216 auto c2 = static_cast<uint32_t>(from[1]);
217 auto c3 = static_cast<uint32_t>(from[2]);
218 auto c4 = static_cast<uint32_t>(from[3]);
219 if (((c2 & 0xC0) != 0x80) || ((c3 & 0xC0) != 0x80) || ((c4 & 0xC0) != 0x80)) [[unlikely]]
220 return std::pair{false, static_cast<size_t>(to - ori_to)};
221 auto c = (c1 << 18) + (c2 << 12) + (c3 << 6) + c4 - 0x3C82080;
222 if (c < 0x10000 || c> 0x10FFFF) return std::pair{false, static_cast<size_t>(to - ori_to)};
223 *to++ = c;
224 from += 4;
225 }
226 else
227 return std::pair{false, static_cast<size_t>(to - ori_to)};
228 }
229
230 return std::pair{true, static_cast<size_t>(to - ori_to)};
231 }

328 void put_main(cvt_writer<KernelType>& writer, const internal_type* to, size_t to_size)
329 requires (cvt_cpt::support_put<KernelType>)
330 {
331 writer.reset(s_max_buf_size);
332 const size_t buf_len = m_cvt_kernel.epc();
333 for (size_t i = 0; i < to_size; ++i)

350 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
351 requires (cvt_cpt::support_get<KernelType>)
352 {
353 reader.reset(s_max_buf_size);
354 size_t total_size = 0;
355
356 size_t prev_rollback = 0;
357 while (total_size < to_max)
358 {
359 size_t dest_size = std::min<size_t>(to_max - total_size, s_max_buf_size);
360 dest_size = std::max(dest_size, prev_rollback + 1);
361
362 if (dest_size > s_max_buf_size) [[unlikely]]
363 throw cvt_error("code_cvt::get fail, input sequence too long.");
364
365 auto [ptr, cur_size] = reader.get_buf(dest_size);
366 if (cur_size == prev_rollback)
367 {
368 if (cur_size == 0) return total_size;
369 throw cvt_error("code_cvt::get fail, partial input sequence.");
370 }
371
372 auto ext_cur = ptr;
373 auto [succ, int_len] = m_cvt_kernel.in_helper(ext_cur, ptr + cur_size, to, to + to_max - total_size);
374
375 if (!succ)
376 throw cvt_error("code_cvt::get fail, invalid external sequence.");
377 m_accu_len += int_len;
378 total_size += int_len;
379
380 if (ext_cur == ptr + cur_size)
381 prev_rollback = 0;
382 else
383 {
384 prev_rollback = ptr + cur_size - ext_cur;
385 reader.rollback(prev_rollback);
386 }
387 }
388 return total_size;
389 }
