40{
41 static_assert(MB_LEN_MAX <= std::numeric_limits<unsigned>::max(),
42 "MB_LEN_MAX exceeds unsigned range");
43
44 explicit codecvt_kernel(const std::string& name)
45 : m_inter_locale(name.c_str())
46 {
47 clocale_user guard(m_inter_locale);
48
49 // there are no known constant-length encodings
50 // m_epc == 1 means fixed length
51 // Defensive check: the C standard requires MB_CUR_MAX to be a positive
52 // integer, but a misconfigured CRT / exotic libc could report 0, which
53 // would lead to division-by-zero UB in seek / rseek. Reject it here so
54 // m_epc >= 1 holds as a class invariant for any constructed instance.
55 const auto cur_max = MB_CUR_MAX;
56 if (cur_max == 0) [[unlikely]]
57 throw cvt_error("codecvt_kernel: locale reports MB_CUR_MAX == 0");
58 m_epc = static_cast<unsigned>(cur_max);
59
60 // Note: mbtowc uses internal static state, so concurrent construction of
61 // codecvt_kernel instances is NOT thread-safe. However, this is acceptable
62 // because this class does not support multi-threaded construction.
63 // Once constructed, instance-level operations use explicit mbstate_t (m_state)
64 // and are safe for single-instance usage.
65 // NOLINTNEXTLINE(cppcoreguidelines-prefer-member-initializer)
66 m_is_state_dep = (std::mbtowc(nullptr, nullptr, MB_CUR_MAX) != 0);
67 init_state();
68 }

70 void init_state() { m_state = std::mbstate_t{}; }
71 [[nodiscard]] bool is_init_state() const { return std::mbsinit(&m_state); }
72 [[nodiscard]] unsigned epc() const { return m_epc; }
73 [[nodiscard]] bool is_var_length() const { return m_epc != 1; }
74 [[nodiscard]] bool is_state_dep() const { return m_is_state_dep; }

76 // Precondition: `to` and `to_end` must point into the same array.
77 // Behavior is undefined otherwise — std::greater enforces a total order
78 // across pointers regardless of provenance, but the pointer subtraction
79 // below requires same-object provenance per [expr.add]/5.
80 bool out_helper(TInt ch, char*& to, char* to_end)
81 {
82 if (std::greater<>{}(to, to_end)) [[unlikely]]
83 throw cvt_error("codecvt_kernel::out_helper fail: invalid pointer range");
84 clocale_user guard(m_inter_locale);
85 if (static_cast<size_t>(to_end - to) < m_epc)
86 return false;
87
88 const size_t conv = std::wcrtomb(to, ch, &m_state);
89 if (conv == static_cast<size_t>(-1))
90 {
91 init_state(); // Reset to known state per C standard
92 return false;
93 }
94 to += conv;
95
96 return true;
97 }

99 // Precondition: `from`/`from_end` and `to`/`to_end` must each point into
100 // the same array. Behavior is undefined otherwise — std::greater enforces
101 // a total order across pointers regardless of provenance, but the pointer
102 // subtractions below require same-object provenance per [expr.add]/5.
103 std::pair<bool, size_t> in_helper(const char*& from, const char* from_end,
104 TInt*& to, TInt* to_end)
105 {
106 if (std::greater<>{}(from, from_end) || std::greater<>{}(to, to_end)) [[unlikely]]
107 throw cvt_error("codecvt_kernel::in_helper fail: invalid pointer range");
108 clocale_user guard(m_inter_locale);
109 wchar_t wch = 0;
110 size_t i_count = 0;
111
112 const size_t to_max = to_end - to;
113 while (from < from_end && (i_count < to_max))
114 {
115 auto tmp_state = m_state;
116 size_t conv = mbrtowc(&wch, from, from_end - from, &tmp_state);
117 if (conv == static_cast<size_t>(-1))
118 return std::pair{false, i_count};
119 else if (conv == static_cast<size_t>(-2))
120 {
121 from = from_end;
122 m_state = tmp_state;
123 break;
124 }
125 else if (conv == 0)
126 {
127 // Find the actual byte length of the null character encoding
128 // before writing to output buffer to ensure consistency
129 size_t n = 1;
130 const auto max_n = static_cast<size_t>(from_end - from);
131 for (; n <= max_n; ++n)
132 {
133 auto tmp_state2(m_state);
134 if (mbrtowc(nullptr, from, n, &tmp_state2) == 0)
135 {
136 m_state = tmp_state2;
137 break;
138 }
139 }
140 if (n > max_n)
141 return std::pair{false, i_count};
142
143 // Only write to output after validation succeeds
144 *to++ = static_cast<TInt>(0);
145 from += n;
146 ++i_count;
147 }
148 else
149 {
150 // mbrtowc returning > 0 guarantees a complete character was converted;
151 // no partial character can exist in m_state at this point
152 m_state = tmp_state;
153 *to++ = wch;
154 from += conv;
155 ++i_count;
156 }
157 }
158
159 return std::pair{true, i_count};
160 }
161
162private:
163 clocale_wrapper m_inter_locale;
164 unsigned m_epc = 0;
165 std::mbstate_t m_state{};
166 bool m_is_state_dep = false;

186 [[nodiscard]] bool is_init_state() const { return true; }

191 // Precondition: `to` and `to_end` must point into the same array.
192 // Behavior is undefined otherwise — std::greater enforces a total order
193 // across pointers regardless of provenance, but the pointer subtraction
194 // below requires same-object provenance per [expr.add]/5.
195 bool out_helper(TInt ch, char8_t*& to, char8_t* to_end)
196 {
197 if (std::greater<>{}(to, to_end)) [[unlikely]]
198 throw cvt_error("codecvt_kernel::out_helper fail: invalid pointer range");
199 // Check for maximum UTF-8 length (4 bytes) upfront
200 if (to_end - to < 4)
201 return false;
202 const auto c = static_cast<uint32_t>(ch);
203 if (0xD800U <= c && c <= 0xDFFFU) [[unlikely]]
204 return false;
205
206 if (c < 0x80U)
207 *to++ = static_cast<char8_t>(c);
208 else if (c <= 0x7ffU)
209 {
210 *to++ = static_cast<char8_t>((c >> 6) + 0xC0U);
211 *to++ = static_cast<char8_t>((c & 0x3FU) + 0x80U);
212 }
213 else if (c <= 0xFFFFU)
214 {
215 *to++ = static_cast<char8_t>((c >> 12) + 0xE0U);
216 *to++ = static_cast<char8_t>(((c >> 6) & 0x3FU) + 0x80U);
217 *to++ = static_cast<char8_t>((c & 0x3FU) + 0x80U);
218 }
219 else if (c <= 0x10FFFFU)
220 {
221 *to++ = static_cast<char8_t>((c >> 18) + 0xF0U);
222 *to++ = static_cast<char8_t>(((c >> 12) & 0x3FU) + 0x80U);
223 *to++ = static_cast<char8_t>(((c >> 6) & 0x3FU) + 0x80U);
224 *to++ = static_cast<char8_t>((c & 0x3FU) + 0x80U);
225 }
226 else [[unlikely]]
227 return false;
228 return true;
229 }

231 // Precondition: `from`/`from_end` and `to`/`to_end` must each point into
232 // the same array. Behavior is undefined otherwise — std::greater enforces
233 // a total order across pointers regardless of provenance, but the pointer
234 // subtractions below require same-object provenance per [expr.add]/5.
235 std::pair<bool, size_t> in_helper(const char8_t*& from, const char8_t* from_end,
236 TInt*& to, TInt* to_end)
237 {
238 if (std::greater<>{}(from, from_end) || std::greater<>{}(to, to_end)) [[unlikely]]
239 throw cvt_error("codecvt_kernel::in_helper fail: invalid pointer range");
240 const TInt* const ori_to = to;
241
242 while ((from != from_end) && (to != to_end))
243 {
244 auto c1 = static_cast<uint32_t>(*from);
245 if (c1 < 0x80U) [[likely]]
246 {
247 ++from;
248 *to++ = static_cast<TInt>(c1);
249 }
250 else if (c1 < 0xE0U)
251 {
252 if (c1 < 0xC0U) [[unlikely]]
253 return std::pair{false, static_cast<size_t>(to - ori_to)};
254 if (from_end - from < 2) break;
255 auto c2 = static_cast<uint32_t>(from[1]);
256 if ((c2 & 0xC0U) != 0x80U) [[unlikely]]
257 return std::pair{false, static_cast<size_t>(to - ori_to)};
258 auto c = (c1 << 6) + c2 - 0x3080U;
259 if (c < 0x80U) return std::pair{false, static_cast<size_t>(to - ori_to)};
260 *to++ = c;
261 from += 2;
262 }
263 else if (c1 < 0xF0U)
264 {
265 if (from_end - from < 3) break;
266 auto c2 = static_cast<uint32_t>(from[1]);
267 auto c3 = static_cast<uint32_t>(from[2]);
268 if (((c2 & 0xC0U) != 0x80U) || ((c3 & 0xC0U) != 0x80U)) [[unlikely]]
269 return std::pair{false, static_cast<size_t>(to - ori_to)};
270 auto c = (c1 << 12) + (c2 << 6) + c3 - 0xE2080U;
271 if (c < 0x800U) return std::pair{false, static_cast<size_t>(to - ori_to)};
272 if (c >= 0xD800U && c <= 0xDFFFU) [[unlikely]]
273 return std::pair{false, static_cast<size_t>(to - ori_to)};
274 *to++ = c;
275 from += 3;
276 }
277 else if (c1 < 0xF8U)
278 {
279 if (from_end - from < 4) break;
280 auto c2 = static_cast<uint32_t>(from[1]);
281 auto c3 = static_cast<uint32_t>(from[2]);
282 auto c4 = static_cast<uint32_t>(from[3]);
283 if (((c2 & 0xC0U) != 0x80U) || ((c3 & 0xC0U) != 0x80U) || ((c4 & 0xC0U) != 0x80U)) [[unlikely]]
284 return std::pair{false, static_cast<size_t>(to - ori_to)};
285 auto c = (c1 << 18) + (c2 << 12) + (c3 << 6) + c4 - 0x3C82080U;
286 if (c < 0x10000U || c > 0x10FFFFU) return std::pair{false, static_cast<size_t>(to - ori_to)};
287 *to++ = c;
288 from += 4;
289 }
290 else
291 return std::pair{false, static_cast<size_t>(to - ori_to)};
292 }
293
294 return std::pair{true, static_cast<size_t>(to - ori_to)};
295 }

398 void put_main(cvt_writer<KernelType>& writer, const internal_type* to, size_t to_size)
399 requires (cvt_cpt::support_put<KernelType>)
400 {
401 writer.reset(s_max_buf_size);
402 const size_t buf_len = m_cvt_kernel.epc();
403 for (size_t i = 0; i < to_size; ++i)
404 {
405 external_type* out_beg = writer.put_buf(buf_len);
406 external_type* out_next = out_beg;
407
408 internal_type ch = *to++;
409
410 if (!m_cvt_kernel.out_helper(ch, out_next, out_beg + buf_len))
411 throw cvt_error("code_cvt::put fail: input character cannot be encoded");
412
413 if (out_next < out_beg + buf_len)
414 writer.rollback(out_beg + buf_len - out_next);
415 }
416 // commit() is the responsibility of abs_cvt::put; m_accu_len is
417 // updated only on success so that a thrown encoding error leaves
418 // it untouched (the converter will be tainted by abs_cvt::put,
419 // so further reads of m_accu_len are blocked anyway).
420 m_accu_len += to_size;
421 }

423 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
424 requires (cvt_cpt::support_get<KernelType>)
425 {
426 if (to_max == 0) return 0;
427 reader.reset(s_max_buf_size);
428 size_t total_size = 0;
429
430 size_t prev_rollback = 0;
431 while (total_size < to_max)
432 {
433 size_t dest_size = std::min<size_t>(to_max - total_size, s_max_buf_size);
434 dest_size = std::max(dest_size, prev_rollback + 1);
435
436 if (dest_size > s_max_buf_size) [[unlikely]]
437 throw cvt_error("code_cvt::get fail: input sequence too long");
438
439 auto [ptr, cur_size] = reader.get_buf(dest_size);
440 if (cur_size == prev_rollback)
441 {
442 if (cur_size == 0) return total_size;
443 throw cvt_error("code_cvt::get fail: partial input sequence");
444 }
445
446 auto ext_cur = ptr;
447 auto [succ, int_len] = m_cvt_kernel.in_helper(ext_cur, ptr + cur_size, to, to + to_max - total_size);
448
449 // Update accumulated length BEFORE the failure check: in_helper may
450 // have written `int_len` chars into the caller's `to` buffer before
451 // hitting an invalid byte. Throwing without this update would leave
452 // m_accu_len inconsistent with the chars already produced.
453 m_accu_len += int_len;
454 total_size += int_len;
455
456 if (!succ)
457 throw cvt_error("code_cvt::get fail: invalid external sequence");
458
459 if (ext_cur == ptr + cur_size)
460 prev_rollback = 0;
461 else
462 {
463 prev_rollback = ptr + cur_size - ext_cur;
464 reader.rollback(prev_rollback);
465 }
466 }
467 return total_size;
468 }

479 void seek(size_t pos)
480 requires (cvt_cpt::support_positioning<KernelType>)
481 {
482 BT::assert_not_tainted();
483 if (this->tell() == pos) return;
484
485 const bool needs_state_reset =
486 m_cvt_kernel.is_var_length() || m_cvt_kernel.is_state_dep();
487
488 if (needs_state_reset && (pos != 0 || BT::m_io_status == io_status::output))
489 throw cvt_error("code_cvt::seek fail: cannot seek with dependent converter");
490
491 const unsigned epc = m_cvt_kernel.epc();
492 if (pos > std::numeric_limits<size_t>::max() / epc)
493 throw cvt_error("code_cvt::seek fail: position overflow");
494
495 // Commit lower layer first. If it throws, this object is unchanged
496 // (strong exception guarantee). After this point the remaining
497 // mutations are noexcept (mbstate_t reset / scalar assignment), so
498 // we cannot end up half-updated.
499 BT::m_kernel.seek(pos * epc);
500
501 if (needs_state_reset)
502 m_cvt_kernel.init_state();
503 m_accu_len = pos;
504 }

506 void rseek(size_t pos)
507 requires (cvt_cpt::support_positioning<KernelType>)
508 {
509 BT::assert_not_tainted();
510 if (m_cvt_kernel.is_var_length() || m_cvt_kernel.is_state_dep())
511 throw cvt_error("code_cvt::rseek fail: cannot seek with dependent converter");
512
513 const unsigned epc = m_cvt_kernel.epc();
514 if (pos > std::numeric_limits<size_t>::max() / epc)
515 throw cvt_error("code_cvt::rseek fail: position overflow");
516
517 const size_t saved_dev_pos = BT::m_kernel.tell();
518 BT::m_kernel.rseek(pos * epc);
519 const size_t new_dev_pos = BT::m_kernel.tell();
520
521 if (new_dev_pos % epc != 0)
522 {
523 // Restore device so this call appears atomic. If the restore
524 // itself throws, suppress it — the caller already gets a
525 // cvt_error and is expected to re-seek before further use;
526 // surfacing a different exception here would only obscure the
527 // failure mode.
528 try { BT::m_kernel.seek(saved_dev_pos); } catch (...) {} // NOLINT(bugprone-empty-catch)
529 throw cvt_error("code_cvt::rseek fail: partial sequence");
530 }
531
532 m_accu_len = new_dev_pos / epc;
533 m_cvt_kernel.init_state();
534 }
