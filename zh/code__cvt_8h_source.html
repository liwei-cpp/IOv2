313 std::pair<bool, size_t> in_helper(const char*& from, const char* from_end,
314 TInt*& to, TInt* to_end)
315 {
316 if (std::greater<>{}(from, from_end) || std::greater<>{}(to, to_end)) [[unlikely]]
317 throw cvt_error("codecvt_kernel::in_helper fail: invalid pointer range");
318 clocale_user guard(m_inter_locale);
319 wchar_t wch = 0;
320 size_t i_count = 0;
321
322 const size_t to_max = to_end - to;
323 while (from < from_end && (i_count < to_max))
324 {
325 auto tmp_state = m_state;
326 size_t conv = mbrtowc(&wch, from, from_end - from, &tmp_state);
327 if (conv == static_cast<size_t>(-1)) // NOLINT(modernize-use-integer-sign-comparison)
328 return std::pair{false, i_count};
329 else if (conv == static_cast<size_t>(-2)) // NOLINT(modernize-use-integer-sign-comparison)
330 {
331 from = from_end;
332 m_state = tmp_state;
333 break;
334 }
335 else if (conv == 0)
336 {
337 // Find the actual byte length of the null character encoding
338 // before writing to output buffer to ensure consistency
339 size_t n = 1;
340 const auto max_n = static_cast<size_t>(from_end - from);
341 for (; n <= max_n; ++n)
342 {
343 auto tmp_state2(m_state);
344 if (mbrtowc(nullptr, from, n, &tmp_state2) == 0)
345 {
346 m_state = tmp_state2;
347 break;
348 }
349 }
350 if (n > max_n)
351 return std::pair{false, i_count};
352
353 // Only write to output after validation succeeds
354 *to++ = static_cast<TInt>(0);
355 from += n;
356 ++i_count;
357 }
358 else
359 {
360 // mbrtowc returning > 0 guarantees a complete character was converted;
361 // no partial character can exist in m_state at this point
362 m_state = tmp_state;
363 *to++ = wch;
364 from += conv;
365 ++i_count;
366 }
367 }
368
369 return std::pair{true, i_count};
370 }

373 clocale_wrapper m_inter_locale;
374 unsigned m_epc = 0;
375 std::mbstate_t m_state{};
376 bool m_is_state_dep = false;

514 bool out_helper(TInt ch, char8_t*& to, char8_t* to_end)
515 {
516 if (std::greater<>{}(to, to_end)) [[unlikely]]
517 throw cvt_error("codecvt_kernel::out_helper fail: invalid pointer range");
518 // Check for maximum UTF-8 length (4 bytes) upfront
519 if (to_end - to < 4)
520 return false;
521 const auto c = static_cast<uint32_t>(ch);
522 if (0xD800U <= c && c <= 0xDFFFU) [[unlikely]]
523 return false;
524
525 if (c < 0x80U)
526 *to++ = static_cast<char8_t>(c);
527 else if (c <= 0x7ffU)
528 {
529 *to++ = static_cast<char8_t>((c >> 6) + 0xC0U);
530 *to++ = static_cast<char8_t>((c & 0x3FU) + 0x80U);
531 }
532 else if (c <= 0xFFFFU)
533 {
534 *to++ = static_cast<char8_t>((c >> 12) + 0xE0U);
535 *to++ = static_cast<char8_t>(((c >> 6) & 0x3FU) + 0x80U);

591 std::pair<bool, size_t> in_helper(const char8_t*& from, const char8_t* from_end,
592 TInt*& to, TInt* to_end)
593 {
594 if (std::greater<>{}(from, from_end) || std::greater<>{}(to, to_end)) [[unlikely]]
595 throw cvt_error("codecvt_kernel::in_helper fail: invalid pointer range");
596 const TInt* const ori_to = to;
597
598 while ((from != from_end) && (to != to_end))
599 {
600 auto c1 = static_cast<uint32_t>(*from);
601 if (c1 < 0x80U) [[likely]]
602 {
603 ++from;
604 *to++ = static_cast<TInt>(c1);
605 }
606 else if (c1 < 0xE0U)
607 {
608 if (c1 < 0xC0U) [[unlikely]]
609 return std::pair{false, static_cast<size_t>(to - ori_to)};
610 if (from_end - from < 2) break;
611 auto c2 = static_cast<uint32_t>(from[1]);
612 if ((c2 & 0xC0U) != 0x80U) [[unlikely]]
613 return std::pair{false, static_cast<size_t>(to - ori_to)};
614 auto c = (c1 << 6) + c2 - 0x3080U;
615 if (c < 0x80U) return std::pair{false, static_cast<size_t>(to - ori_to)};
616 *to++ = c;
617 from += 2;
618 }
619 else if (c1 < 0xF0U)
620 {
621 if (from_end - from < 3) break;
622 auto c2 = static_cast<uint32_t>(from[1]);
623 auto c3 = static_cast<uint32_t>(from[2]);
624 if (((c2 & 0xC0U) != 0x80U) || ((c3 & 0xC0U) != 0x80U)) [[unlikely]]
625 return std::pair{false, static_cast<size_t>(to - ori_to)};
626 auto c = (c1 << 12) + (c2 << 6) + c3 - 0xE2080U;
627 if (c < 0x800U) return std::pair{false, static_cast<size_t>(to - ori_to)};
628 if (c >= 0xD800U && c <= 0xDFFFU) [[unlikely]]
629 return std::pair{false, static_cast<size_t>(to - ori_to)};
630 *to++ = c;
631 from += 3;
632 }
633 else if (c1 < 0xF8U)
634 {
635 if (from_end - from < 4) break;
636 auto c2 = static_cast<uint32_t>(from[1]);
637 auto c3 = static_cast<uint32_t>(from[2]);
638 auto c4 = static_cast<uint32_t>(from[3]);
639 if (((c2 & 0xC0U) != 0x80U) || ((c3 & 0xC0U) != 0x80U) || ((c4 & 0xC0U) != 0x80U)) [[unlikely]]
640 return std::pair{false, static_cast<size_t>(to - ori_to)};
641 auto c = (c1 << 18) + (c2 << 12) + (c3 << 6) + c4 - 0x3C82080U;
642 if (c < 0x10000U || c > 0x10FFFFU) return std::pair{false, static_cast<size_t>(to - ori_to)};
643 *to++ = c;
644 from += 4;
645 }
646 else
647 return std::pair{false, static_cast<size_t>(to - ori_to)};
648 }
649
650 return std::pair{true, static_cast<size_t>(to - ori_to)};
651 }

875 void put_main(cvt_writer<KernelType>& writer, const internal_type* to, size_t to_size)
876 requires (cvt_cpt::support_put<KernelType>)
877 {
878 writer.reset(s_max_buf_size);
879 const size_t buf_len = m_cvt_kernel.epc();
880 for (size_t i = 0; i < to_size; ++i)
881 {
882 external_type* out_beg = writer.put_buf(buf_len);
883 external_type* out_next = out_beg;
884
885 internal_type ch = *to++;
886
887 if (!m_cvt_kernel.out_helper(ch, out_next, out_beg + buf_len))
888 throw cvt_error("code_cvt::put fail: input character cannot be encoded");
889
890 if (out_next < out_beg + buf_len)
891 writer.rollback(out_beg + buf_len - out_next);
892 }
893 // commit() is the responsibility of abs_cvt::put; m_accu_len is
894 // updated only on success so that a thrown encoding error leaves
895 // it untouched (the converter will be tainted by abs_cvt::put,
896 // so further reads of m_accu_len are blocked anyway).
897 m_accu_len += to_size;
898 }

914 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
915 requires (cvt_cpt::support_get<KernelType>)
916 {
917 if (to_max == 0) return 0;
918 reader.reset(s_max_buf_size);
919 size_t total_size = 0;
920
921 size_t prev_rollback = 0;
922 while (total_size < to_max)
923 {
924 size_t dest_size = std::min<size_t>(to_max - total_size, s_max_buf_size);
925 dest_size = std::max(dest_size, prev_rollback + 1);
926
927 if (dest_size > s_max_buf_size) [[unlikely]]
928 throw cvt_error("code_cvt::get fail: input sequence too long");
929
930 auto [ptr, cur_size] = reader.get_buf(dest_size);
931 if (cur_size == prev_rollback)
932 {
933 if (cur_size == 0) return total_size;
934 throw cvt_error("code_cvt::get fail: partial input sequence");
935 }
936
937 auto ext_cur = ptr;
938 auto [succ, int_len] = m_cvt_kernel.in_helper(ext_cur, ptr + cur_size, to, to + to_max - total_size);
939
940 // Update accumulated length BEFORE the failure check: in_helper may
941 // have written `int_len` chars into the caller's `to` buffer before
942 // hitting an invalid byte. Throwing without this update would leave
943 // m_accu_len inconsistent with the chars already produced.
944 m_accu_len += int_len;
945 total_size += int_len;
946
947 if (!succ)
948 throw cvt_error("code_cvt::get fail: invalid external sequence");
949
950 if (ext_cur == ptr + cur_size)
951 prev_rollback = 0;
952 else
953 {
954 prev_rollback = ptr + cur_size - ext_cur;
955 reader.rollback(prev_rollback);
956 }
957 }
958 return total_size;
959 }

1064 void seek_impl(size_t pos)
1065 requires (cvt_cpt::support_positioning<KernelType>)
1066 {
1067 const bool needs_state_reset =
1068 m_cvt_kernel.is_var_length() || m_cvt_kernel.is_state_dep();
1069
1070 if (needs_state_reset && (pos != 0 || BT::m_io_status == io_status::output))
1071 throw cvt_error("code_cvt::seek fail: cannot seek with dependent converter");
1072
1073 const unsigned epc = m_cvt_kernel.epc();
1074 if (pos > std::numeric_limits<size_t>::max() / epc)
1075 throw cvt_error("code_cvt::seek fail: position overflow");
1076
1077 // Commit lower layer first. If it throws, this object is unchanged
1078 // (strong exception guarantee). After this point the remaining
1079 // mutations are noexcept (mbstate_t reset / scalar assignment), so
1080 // we cannot end up half-updated.
1081 BT::m_kernel.seek(pos * epc);
1082
1083 if (needs_state_reset)
1084 m_cvt_kernel.init_state();
1085 m_accu_len = pos;

1112 void rseek_impl(size_t pos)
1113 requires (cvt_cpt::support_positioning<KernelType>)
1114 {
1115 if (m_cvt_kernel.is_var_length() || m_cvt_kernel.is_state_dep())
1116 throw cvt_error("code_cvt::rseek fail: cannot seek with dependent converter");
1117
1118 const unsigned epc = m_cvt_kernel.epc();
1119 if (pos > std::numeric_limits<size_t>::max() / epc)
1120 throw cvt_error("code_cvt::rseek fail: position overflow");
1121
1122 const size_t saved_dev_pos = BT::m_kernel.tell();
1123 BT::m_kernel.rseek(pos * epc);
1124 const size_t new_dev_pos = BT::m_kernel.tell();
1125
1126 if (new_dev_pos % epc != 0)
1127 {
1128 // Restore device so this call appears atomic. If the restore
1129 // itself throws, suppress it — the caller already gets a
1130 // cvt_error and is expected to re-seek before further use;
1131 // surfacing a different exception here would only obscure the
1132 // failure mode.
1133 try { BT::m_kernel.seek(saved_dev_pos); } catch (...) {} // NOLINT(bugprone-empty-catch)
1134 throw cvt_error("code_cvt::rseek fail: partial sequence");
1135 }
1136
