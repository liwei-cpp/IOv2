452 template <typename TIter, std::sentinel_for<TIter> TSent>
453 TIter get(TIter beg, TSent end, ios_base<char_type>& io, bool& v) const
454 {
455 bool success = true;
456
457 if (!(io.flags() & ios_defs::boolalpha))
458 {
459 // Parse bool values as long.
460 long l = -1;
461 std::tie(success, beg) = extract_int(beg, end, io, l);
462
463 if (l == 0 || l == 1) v = bool(l);
464 else
465 {
466 // _GLIBCXX_RESOLVE_LIB_DEFECTS
467 // 23. Num_get overflow result.
468 v = true;
469 success = false;
470 }
471 }
472 else
473 {
474 // Parse bool values as alphanumeric.
475 bool testf = true;
476 bool testt = true;
477 bool donef = (m_false_name.empty());
478 bool donet = (m_true_name.empty());
479 size_t n = 0;
480 while (!donef || !donet)
481 {
482 if (beg == end)
483 break;
484
485 const CharT c = *beg;
486
487 if (!donef)
488 testf = c == m_false_name[n];
489
490 if (!testf && donet) break;
491
492 if (!donet)
493 testt = c == m_true_name[n];
494
495 if (!testt && donef) break;
496
497 if (!testt && !testf) break;
498
499 ++n;
500 ++beg;
501
502 donef = !testf || n >= m_false_name.size();
503 donet = !testt || n >= m_true_name.size();
504 }
505 if (testf && n == m_false_name.size() && n)
506 {
507 v = false;
508 if (testt && n == m_true_name.size())
509 success = false;
510 }
511 else if (testt && n == m_true_name.size() && n)
512 {
513 v = true;
514 }
515 else
516 {
517 // _GLIBCXX_RESOLVE_LIB_DEFECTS
518 // 23. Num_get overflow result.
519 v = false;
520 success = false;
521 }
522 }
523
524 if (!success) throw stream_error("numeric::get fail: parse boolean fail");
525 return beg;
526 }

735 template <typename TIter, typename TValue>
736 TIter insert_float(TIter s, ios_base<char_type>& io, TValue v, char mod) const
737 {
738 // precision() is now a bounded uint8_t (0-255); there is no negative /
739 // out-of-range sentinel to normalise here.
740 const std::streamsize prec = io.precision();
741 const int max_digits = std::numeric_limits<TValue>::digits10;
742
743 // Consume the field width up front, before any allocation or conversion
744 // that can throw: width() is one-shot and a stale value must not survive
745 // onto the stream if we leave by an exception. Used for padding below.
746 const std::streamsize w = io.width();
747 io.width(0);
748
749 // [22.2.2.2.2] Stage 1, numeric conversion to character.
750 size_t len = 0;
751 std::array<char, 16> fbuf{};
752 format_float_(io.flags(), fbuf.data(), mod);
753
754 const ios_defs::fmtflags fltfield = io.flags() & ios_defs::floatfield;
755
756 // Initial buffer size estimate (GCC style).
757 // For non-fixed fields, max_digits * 3 is a safe heuristic.
758 size_t cs_size = static_cast<size_t>(max_digits) * 3 + 32;
759 if (fltfield == ios_defs::fixed)
760 cs_size = static_cast<size_t>(std::numeric_limits<TValue>::max_exponent10) + static_cast<size_t>(prec) + 32;
761
762 // Cap initial allocation to a reasonable size (e.g., 2048) to avoid huge initial pressure.
763 if (cs_size > 2048) cs_size = 2048;
764
765 std::vector<char> vec_cs(cs_size);
766
767 {
768 clocale_wrapper inter_locale("C");
769 clocale_user guard(inter_locale);
770
771 auto do_snprintf = [&](char* buf, size_t size) {
772 if (fltfield == (ios_defs::fixed | ios_defs::scientific))
773 return snprintf(buf, size, fbuf.data(), v);
774 else
775 return snprintf(buf, size, fbuf.data(), static_cast<int>(prec), v);
776 };
777
778 // snprintf returns int. Trap negative (encoding error) before
779 // promoting to size_t — otherwise -1 becomes SIZE_MAX and
780 // bypasses every downstream bound check.
781 int n = do_snprintf(vec_cs.data(), cs_size);
782 if (n < 0)
783 throw stream_error("numeric::put fail: floating-point conversion failed");
784 len = static_cast<size_t>(n);
785
786 // If buffer was too small, snprintf returns required length.
787 if (len >= cs_size)
788 {
789 cs_size = len + 1;
790 vec_cs.resize(cs_size);
791 n = do_snprintf(vec_cs.data(), cs_size);
792 if (n < 0)
793 throw stream_error("numeric::put fail: floating-point conversion failed");
794 len = static_cast<size_t>(n);
795 }
796 }
797
798 // len == 0 is also treated as failure: every well-formed numeric
799 // print (including 0, "inf", "nan") emits at least one character,
800 // and the downstream path assumes vec_ws(len) has at least one
801 // element before dereferencing vec_ws.data().
802 if (len == 0 || len >= cs_size)
803 throw stream_error("numeric::put fail: floating-point conversion failed");
804
805 const char* cs = vec_cs.data();
806
807 // [22.2.2.2.2] Stage 2, convert to char_type, using correct
808 // numeric_conf.decimal_point() values for '.' and adding grouping.
809 std::vector<CharT> vec_ws(len);
810 CharT* ws = vec_ws.data();
811 m_ctype->widen_seq(cs, cs + len, ws);
812
813 // Replace decimal point.
814 const char* p = std::find(cs, cs + len, '.');
815 CharT* wp = nullptr;
816 if (p != cs + len)
817 {
818 wp = ws + (p - cs);
819 *wp = m_decimal_point;
820 }
821
822 // Add grouping, if necessary.
823 if ((!m_grouping.empty())
824 && (wp || len < 3u || (cs[1] <= '9' && cs[2] <= '9' && cs[1] >= '0' && cs[2] >= '0')))
825 {
826 // Grouping can add (almost) as many separators as the
827 // number of digits, but no more.
828 std::vector<CharT> vec_ws2(len * 2);
829 CharT* ws2 = vec_ws2.data();
830
831 size_t off = 0;
832 if (cs[0] == '-' || cs[0] == '+')
833 {
834 off = 1;
835 ws2[0] = ws[0];
836 len -= 1;
837 }
838
839 group_float(m_grouping, m_thousands_sep, wp,
840 ws2 + off, ws + off, len);
841 len += off;
842 std::swap(vec_ws, vec_ws2);
843 ws = vec_ws.data();
844 }
845
846 // Pad.
847 if (std::cmp_greater(w, len))
848 {
849 std::vector<CharT> vec_ws3(w);
850 CharT* ws3 = vec_ws3.data();
851 bool startSign = (ws[0] == m_out_atoms[s_ominus]) || (ws[0] == m_out_atoms[s_oplus]);
852 bool start0x = (ws[0] == m_out_atoms[s_odigits]) && (len > 1u) &&
853 ((ws[1] == m_out_atoms[s_ox]) || (ws[1] == m_out_atoms[s_oX]));
854 const ios_defs::fmtflags adjust = io.flags() & ios_defs::adjustfield;
855 pad(io.fill(), w, adjust, ws3, ws, len, startSign, start0x);
856
857 std::swap(vec_ws, vec_ws3);
858 ws = vec_ws.data();
859 }
860
861 // [22.2.2.2.2] Stage 4.
862 // Write resulting, fully-formatted string to output iterator.
863 return std::copy(ws, ws + len, s);
864 }

897 template <typename TIter, typename TValue>
898 TIter insert_int(TIter s, ios_base<char_type>& io, TValue v) const
899 {
900 using unsigned_type = std::make_unsigned_t<TValue>;
901
902 const ios_defs::fmtflags flags = io.flags();
903
904 // Consume the field width up front, before any allocation that can throw:
905 // width() is one-shot and a stale value must not survive onto the stream
906 // if we leave by an exception. Used for padding below.
907 const std::streamsize w = io.width();
908 io.width(0);
909
910 // Long enough to hold hex, dec, and octal representations.
911 const auto ilen = 5 * sizeof(TValue);
912 std::vector<char_type> cs_vec(ilen);
913 char_type* cs = cs_vec.data();
914
915 // [22.2.2.2.2] Stage 1, numeric conversion to character.
916 // Result is returned right-justified in the buffer.
917 const ios_defs::fmtflags basefield = flags & ios_defs::basefield;
918 const bool dec = (basefield != ios_defs::oct && basefield != ios_defs::hex);
919 const unsigned_type u = ((v > 0 || !dec) ? unsigned_type(v)
920 : -unsigned_type(v));
921 auto len = static_cast<size_t>(int_to_char(cs + ilen, u, flags, dec));
922 cs += ilen - len;
923
924 // Add grouping, if necessary.
925 if (!m_grouping.empty())
926 {
927 // Grouping can add (almost) as many separators as the number
928 // of digits + space is reserved for numeric base or sign.
929 std::vector<char_type> cs_vec2((ilen + 1) * 2);
930 char_type* cs2 = cs_vec2.data() + 2;
931 len = static_cast<size_t>(FacetHelper::add_grouping(cs2, m_thousands_sep, m_grouping, cs, cs + len) - cs2);
932 std::swap(cs_vec, cs_vec2);
933 cs = cs_vec.data() + 2;
934 }
935
936 // Complete Stage 1, prepend numeric base or sign.
937 if (dec)
938 { // Decimal.
939 if (v >= 0)
940 {
941 if (bool(flags & ios_defs::showpos) && std::is_signed_v<TValue>)
942 *--cs = m_out_atoms[s_oplus], ++len;
943 }
944 else
945 *--cs = m_out_atoms[s_ominus], ++len;
946 }
947 else if (bool(flags & ios_defs::showbase) && v)
948 {
949 if (basefield == ios_defs::oct)
950 *--cs = m_out_atoms[s_odigits], ++len;
951 else
952 {
953 // 'x' or 'X'
954 const bool uppercase = flags & ios_defs::uppercase;
955 *--cs = m_out_atoms[s_ox + uppercase];
956 // '0'
957 *--cs = m_out_atoms[s_odigits];
958 len += 2;
959 }
960 }
961
962 // Pad.
963 if (std::cmp_greater(w, len))
964 {
965 std::vector<char_type> cs_vec3(w);
966 char_type* cs3 = cs_vec3.data();
967 bool startSign = (cs[0] == m_out_atoms[s_ominus]) || (cs[0] == m_out_atoms[s_oplus]);
968 bool start0x = (cs[0] == m_out_atoms[s_odigits]) && (len > 1u) &&
969 ((cs[1] == m_out_atoms[s_ox]) || (cs[1] == m_out_atoms[s_oX]));
970 const ios_defs::fmtflags adjust = io.flags() & ios_defs::adjustfield;
971 pad(io.fill(), w, adjust, cs3, cs, len, startSign, start0x);
972 std::swap(cs_vec, cs_vec3);
973 cs = cs_vec.data();
974 }
975
976 // [22.2.2.2.2] Stage 4.
977 // Write resulting, fully-formatted string to output iterator.
978 return std::copy(cs, cs + len, s);
979 }

1008 template <typename TValue>
1009 int int_to_char(CharT* bufend, TValue v, ios_defs::fmtflags flags, bool dec) const
1010 {
1011 auto buf = bufend;
1012 if (dec)
1013 { // Decimal.
1014 do // NOLINT(cppcoreguidelines-avoid-do-while)
1015 {
1016 *--buf = m_out_atoms[(v % 10) + s_odigits];
1017 v /= 10;
1018 } while (v != 0);
1019 }
1020 else if ((flags & ios_defs::basefield) == ios_defs::oct)
1021 { // Octal.
1022 do // NOLINT(cppcoreguidelines-avoid-do-while)
1023 {
1024 *--buf = m_out_atoms[(v & 0x7) + s_odigits];
1025 v >>= 3;
1026 } while (v != 0);
1027 }
1028 else
1029 { // Hex.
1030 const bool uppercase = flags & ios_defs::uppercase;
1031 const int case_offset = uppercase ? s_oudigits : s_odigits;
1032 do // NOLINT(cppcoreguidelines-avoid-do-while)
1033 {
1034 *--buf = m_out_atoms[(v & 0xf) + case_offset];
1035 v >>= 4;
1036 } while (v != 0);
1037 }
1038 return bufend - buf;
1039 }

1116 void pad_impl_(ios_defs::fmtflags adjust, char_type fill,
1117 char_type* news, const char_type* olds,
1118 std::streamsize newlen, std::streamsize oldlen,
1119 bool startSign, bool start0x) const
1120 {
1121
1122 const auto plen = static_cast<size_t>(newlen - oldlen);
1123
1124 // Padding last.
1125 if (adjust == ios_defs::left)
1126 {
1127 std::copy(olds, olds + oldlen, news);
1128 std::fill_n(news + oldlen, plen, fill);
1129 return;

1133 if (adjust == ios_defs::internal)

1147 news[1] = olds[1];

1149 news += 2;
1150 }
1151 // else Padding first.
1152 }
1153 std::fill_n(news, plen, fill);

1181 {
1182 *fptr++ = '%';
1183 // [22.2.2.2.2] Table 60
1184 if (flags & ios_defs::showpos)

1186 if (flags & ios_defs::showpoint)

1190 if (fltfield != (ios_defs::fixed | ios_defs::scientific))
1191 {

1193 *fptr++ = '.';

1196
1197 if (mod)
1198 *fptr++ = mod;
1199 // [22.2.2.2.2] Table 58
1200 if (fltfield == ios_defs::fixed)
1201 *fptr++ = 'f';

1337 {
1338 if ((!m_grouping.empty() && c == m_thousands_sep)
1339 || c == m_decimal_point)
1340 break; // NOLINT(bugprone-branch-clone)
1341 else if (c == m_in_atoms[s_izero] && (!found_zero || base == 10))

1343 found_zero = true;
1344 ++sep_pos;
1345 if (basefield == 0) base = 8;
1346 if (base == 8) sep_pos = 0;

1355 }

1357 break;

1359 else
1360 break;

1362 if (++beg != end)
1363 {
1364 c = *beg;
1365 if (!found_zero) break;
1366 }
1367 else

1372 // base digits as valid input.
1373 const size_t len = (base == 16 ? s_iend - s_izero : base);
1374
1375 // Extract.
1376 std::vector<uint8_t> found_grouping;
1377 if (!m_grouping.empty()) found_grouping.reserve(32);
1378 bool testfail = false;
1379 bool testoverflow = false;
1380 const unsigned_type max_val = (negative && std::is_signed_v<TValue>)
1381 ? -static_cast<unsigned_type>(std::numeric_limits<TValue>::min()) : std::numeric_limits<TValue>::max();
1382 const unsigned_type smax = max_val / base;
1383 unsigned_type result = 0;
1384 int digit = 0;
1385 const char_type* lit_zero = m_in_atoms.data() + s_izero;
1386
1387 while (!testeof)
1388 {
1389 // According to 22.2.2.1.2, p8-9, first look for thousands_sep
1390 // and decimal_point.
1391 if (!m_grouping.empty() && c == m_thousands_sep)
1392 {
1393 // NB: Thousands separator at the beginning of a string
1394 // is a no-no, as is two consecutive thousands separators.
1395 if (sep_pos)
1396 {
1397 // found_grouping stores each group's digit count in a
1398 // uint8_t. A single group whose size cannot fit in that
1399 // byte is rejected rather than silently truncated.
1400 if (std::cmp_greater(sep_pos, std::numeric_limits<uint8_t>::max()))
1401 {
1402 testfail = true;
1403 break;
1404 }
1405 found_grouping.push_back(static_cast<uint8_t>(sep_pos));
1406 sep_pos = 0;
1407 }
1408 else
1409 {
1410 testfail = true;
1411 break;
1412 }
1413 }
1414 else if (c == m_decimal_point) break;
1415 else
1416 {
1417 const char_type* q = std::find(lit_zero, lit_zero + len, c);
1418 if (q == lit_zero + len) break;

1420 digit = q - lit_zero;
1421 if (digit > 15) digit -= 6;
1422 if (result > smax) testoverflow = true;
1423 else
1424 {
1425 result *= base;
1426 testoverflow |= result > max_val - digit;
1427 result += digit;
1428 ++sep_pos;
1429 }
1430 }
1431
1432 if (++beg != end) c = *beg;
1433 else testeof = true;
1434 }

1438 // match, then get very very upset, and set failbit.
1439 if (!found_grouping.empty())
1440 {
1441 // Add the ending grouping. Reject inputs whose final group
1442 // exceeds the uint8_t storage in found_grouping.
1443 if (std::cmp_greater(sep_pos, std::numeric_limits<uint8_t>::max()))
1444 {
1445 testfail = true;
1446 }
1447 else
1448 {
1449 found_grouping.push_back(static_cast<uint8_t>(sep_pos));
1450 success = FacetHelper::verify_grouping(m_grouping, found_grouping);
1451 }
1452 }
1453
1454 // LWG 23 (Num_get overflow result): when the field overflows,
1455 // v must be set to numeric_limits::max() / min() with failbit.
1456 //
1457 // testoverflow is checked BEFORE testfail by design. Once the
1458 // digit-accumulation loop sets testoverflow, ++sep_pos is skipped
1459 // for every subsequent digit, so sep_pos freezes. A later, otherwise
1460 // well-formed thousands_sep then fails the grouping check and sets
1461 // testfail — but that testfail is a side effect of the overflow
1462 // short-circuit, not an independent structural error in the input.
1463 //
1464 // Letting testfail win in that overlap would map a structurally
1465 // valid, numerically out-of-range input (e.g. "12,345,678,901,234,567"
1466 // into uint32_t under grouping "\3") to v = 0 instead of v = max,
1467 // contradicting LWG 23. We diverge from libstdc++'s ordering here
1468 // (which has the same latent issue) to keep the overflow signal
1469 // dominant whenever it fires.
1470 if (testoverflow)
1471 {
1472 if (negative && std::is_signed_v<TValue>)
1473 v = std::numeric_limits<TValue>::min();
1474 else
1475 v = std::numeric_limits<TValue>::max();
1476 success = false;
1477 }
1478 else if ((!sep_pos && !found_zero && !found_grouping.size()) || testfail)
1479 {
1480 v = 0;
1481 success = false;
1482 }
1483 else
1484 v = negative ? -result : result;
1485
1486 return std::pair(success, beg);
1487 }
