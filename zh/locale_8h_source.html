608 locale involve_msg(const std::string& domain, const std::string& lang = "",
609 bool throw_if_fail = false) const requires (!std::is_same_v<TChar, char>)
610 {
611 // messages_conf<TChar> is only specialized for char8_t, char32_t and
612 // UTF-32 wchar_t (plus the char overload above); for char16_t or a
613 // non-UTF-32 wchar_t it stays incomplete. Reject those here with a clear
614 // message instead of an opaque "incomplete type" error from make_shared
615 // below. Safe as a completeness check because messages_conf<TChar>'s
616 // completeness is fixed program-wide (a type either has a specialization
617 // everywhere or nowhere).
618 static_assert(requires { sizeof(messages_conf<TChar>); },
619 "involve_msg: messages translation is unsupported for this character "
620 "type -- no messages_conf<TChar> specialization exists (e.g. char16_t, "
621 "or wchar_t on a non-UTF-32 platform).");
622 const std::string filtered_lang = base_ft<messages>::filter_lang(domain, lang);
623 const std::string dirname = base_ft<messages>::get_dirname(domain);
624 auto ft = s_ori_facet_buf.try_get_msg<TChar>(domain, filtered_lang, dirname);
625 if (!ft)
626 {
627 // Build strictly so a load failure surfaces as an exception, and intern only
628 // on success: a degraded (load-failed) facet is never cached, so a later
629 // throw_if_fail=true call re-attempts the load and can still throw.
630 try
631 {
632 ft = std::make_shared<messages_conf<TChar>>(domain, filtered_lang, dirname, true);
633 // Caching is separate from the load: a cache-insert failure (e.g. an
634 // allocation failure while interning) must never degrade or drop a
635 // successfully-loaded facet. On such a failure keep it, just uncached.
636 try { ft = s_ori_facet_buf.put_msg<TChar>(ft, domain, filtered_lang, dirname); }
637 catch (...) {} // NOLINT(bugprone-empty-catch)
638 }
639 catch (...)
640 {
641 if (throw_if_fail) throw;
642 ft = std::make_shared<messages_conf<TChar>>(domain, filtered_lang, dirname, false);
643 }
644 }
645
646 // At most one messages facet per locale (keyed by the single messages_conf
647 // type id). If *this already carries one, this replaces it in the returned
648 // copy -- matching involve()'s same-id replace semantics; *this is unchanged.
649 locale res(*this);
650 res.m_facet_confs[ft->id()] = std::move(ft);
651 res.m_facets.clear();
652 return res;
653 }
