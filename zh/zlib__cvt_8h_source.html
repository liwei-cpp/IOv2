413 zlib_cvt& operator=(const zlib_cvt& val)
414 {
415 if (this == &val) return *this;
416
417 // close_stream failure indicates the OLD output stream could not be
418 // finalized on its device; it leaves *this's in-memory zlib state
419 // self-consistent (guards inside close_stream reset m_strm / status).
420 // It does NOT damage the new assignment we are about to perform, so
421 // capture it and rethrow after the new state is fully installed.
422 std::exception_ptr close_err;
423 try { close_stream(); }
424 catch (...) { close_err = std::current_exception(); }
425
426 try
427 {
428 BT::operator=(val);
429 m_put_level = val.m_put_level;
430 m_sync_flush = val.m_sync_flush;
431 m_stream_ended = val.m_stream_ended;
432

434 {
435 m_strm = std::make_unique<z_stream>();
436 zerr("zlib_cvt copy assignment fail", deflateCopy(m_strm.get(), val.m_strm.get())); // NOLINT(cppcoreguidelines-pro-type-const-cast)
437 }
438 else if (BT::m_io_status == io_status::input)
439 {

554 bool is_eof()

556 {

646 io_status bos()
647 {
648 BT::bos();
649 // After BT::bos() succeeds, m_io_status is no longer neutral. Any failure
650 // from here on must restore the invariant "m_io_status != neutral implies
651 // m_strm is a fully-initialized zlib stream" before propagating, otherwise
652 // copy/assign, flush_impl, and the direction-switch branches in
653 // abs_cvt::get/put will dereference a null m_strm. Reset the status and
654 // taint the converter so subsequent get/put/flush refuse to run until
655 // the caller reattaches a device.
656 try
657 {
658 if (BT::m_io_status == io_status::input)
659 {
660 if constexpr (cvt_cpt::support_get<KernelType>)
661 {
662 m_strm = std::make_unique<z_stream>();
663 m_strm->zalloc = Z_NULL;
664 m_strm->zfree = Z_NULL;
665 m_strm->opaque = Z_NULL;
666 m_strm->avail_in = 0;
667 m_strm->next_in = Z_NULL;
668 m_strm->state = Z_NULL;
669 auto ret = inflateInit(m_strm.get());
670 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
671 inflate_guard zlib_guard(m_strm);
672
673 // Contract: kernel.get() on BOS path guarantees to return exactly the
674 // requested byte count, or throw an exception if insufficient data is
675 // available. The assert below is purely defensive, verifying this
676 // invariant during development. It is intentionally compiled out in
677 // release builds.
678 std::array<external_type, zlib_header_size> header_buf{};
679 [[maybe_unused]] const size_t n = BT::m_kernel.get(header_buf.data(), zlib_header_size);
680 assert(n == zlib_header_size);
681
682 m_strm->avail_in = zlib_header_size;
683 m_strm->next_in = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
684
685 unsigned char ch = 0;
686 m_strm->avail_out = 1;
687 m_strm->next_out = &ch;
688
689 ret = inflate(m_strm.get(), Z_NO_FLUSH);
690 if (ret == Z_NEED_DICT)
691 throw cvt_error("zlib_cvt::bos fail: preset dictionary not supported");
692 zerr("zlib_cvt::bos fail", ret);
693
694 if ((m_strm->avail_in != 0) || (m_strm->avail_out != 1)) throw cvt_error("zlib_cvt::bos fail: Invalid zlib header");
695 m_strm->next_in = nullptr;
696 m_strm->avail_out = 0;
697 m_strm->next_out = nullptr;
698
699 zlib_guard.release();
700 }
701 else
702 {
703 throw cvt_error("zlib_cvt::bos fail: BT returned input but KernelType lacks get support");
704 }
705 }
706 else if (BT::m_io_status == io_status::output)
707 {
708 if constexpr (cvt_cpt::support_put<KernelType>)
709 {
710 m_strm = std::make_unique<z_stream>();
711 m_strm->zalloc = Z_NULL;
712 m_strm->zfree = Z_NULL;
713 m_strm->opaque = Z_NULL;
714 m_strm->state = Z_NULL;
715
716 auto ret = deflateInit(m_strm.get(), static_cast<int>(m_put_level));
717 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
718 deflate_guard zlib_guard(m_strm);
719
720 // Contract enforced below: after this deflate() call we require
721 // exactly zlib_header_size bytes in the output buffer and zero
722 // bytes/bits still pending inside zlib. Z_NO_FLUSH on a fresh
723 // stream with no input is expected to flush the queued header
724 // and nothing else. The post-checks turn that expectation into
725 // a hard invariant: any future zlib behavior change (more
726 // bytes, fewer bytes, or bytes left buffered) surfaces here as
727 // a loud failure rather than silent framing corruption.
728 std::array<external_type, zlib_header_size + 1> header_buf{};
729
730 m_strm->avail_in = 0;
731 m_strm->next_in = nullptr;
732 m_strm->avail_out = zlib_header_size + 1;
733 m_strm->next_out = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
734 ret = deflate(m_strm.get(), Z_NO_FLUSH);
735 zerr("zlib_cvt::bos fail", ret);
736
737 if (m_strm->avail_out != 1)
738 throw cvt_error("zlib_cvt::bos fail: zlib did not emit exactly the header");
739
740 unsigned pending_bytes = 0;
741 int pending_bits = 0;
742 if (deflatePending(m_strm.get(), &pending_bytes, &pending_bits) != Z_OK ||
743 pending_bytes != 0 || pending_bits != 0)
744 throw cvt_error("zlib_cvt::bos fail: zlib has unflushed bytes after header");
745
746 BT::m_kernel.put(header_buf.data(), zlib_header_size);
747
748 m_strm->avail_out = 0;
749 m_strm->next_out = nullptr;
750
751 zlib_guard.release();
752 }
753 else
754 {
755 throw cvt_error("zlib_cvt::bos fail: BT returned output but KernelType lacks put support");
756 }
757 }
758 else
759 throw cvt_error("zlib_cvt::bos fail: invalid response value.");
760 }
761 catch (...)
762 {
763 // m_strm has already been reset to nullptr by inflate_guard /
764 // deflate_guard (or by the manual reset on inflateInit/deflateInit
765 // failure) before the exception reaches here.
766 BT::m_io_status = io_status::neutral;
767 BT::set_tainted();
768 throw;
769 }
770 return BT::m_io_status;
771 }

834 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
835 requires (cvt_cpt::support_get<KernelType>)
836 {
837 // Once inflate() has reported Z_STREAM_END on this stream, further
838 // inflate() calls on the same z_stream would either fail with
839 // Z_DATA_ERROR or attempt to interpret trailing bytes as a new
840 // concatenated zlib stream. Both are wrong for a single-stream
841 // contract. Surface end-of-stream as a zero-byte read so callers
842 // see a clean EOF instead. Reset by close_stream's state_guard.
843 if (m_stream_ended) return 0;
844
845 // Note: We read one byte at a time intentionally. Decompression has
846 // unpredictable expansion ratio - a few compressed bytes could expand
847 // to overflow the output buffer. This keeps the code simple and correct.
848 // The underlying converter already buffers, so this doesn't cause syscalls.
849 reader.reset(1);
850
851 // Ensures m_strm's in/out pointers and counts are cleared on every exit
852 // path, so a throw cannot leave next_in pointing into reader's buffer
853 // after the reader is destroyed.
854 io_buf_guard buf_guard{*m_strm};
855
856 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
857 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
858
859 auto aim_output = (max_chunk / sizeof(internal_type) > to_max)
860 ? static_cast<decltype(m_strm->avail_out)>(to_max * sizeof(internal_type))
861 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
862
863 m_strm->avail_out = aim_output;
864 m_strm->next_out = reinterpret_cast<unsigned char*>(to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
865 auto ret = inflate(m_strm.get(), Z_NO_FLUSH);
866 zerr<true>("zlib_cvt::get fail", ret);
867
868 while (m_strm->avail_out && ret != Z_STREAM_END)
869 {
870 auto [ptr, len] = reader.get_buf(1);
871 if (len == 0) break;
872 m_strm->next_in = const_cast<unsigned char*>(reinterpret_cast<const unsigned char*>(ptr)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast,cppcoreguidelines-pro-type-const-cast)
873 m_strm->avail_in = 1;
874 // Snapshot before the call so we can prove zlib made progress on at
875 // least one axis (input consumed or output produced). The Z_OK
876 // contract guarantees this, but enforcing it here turns a
877 // hypothetical infinite loop into an explicit failure if the
878 // library ever diverges from its contract.
879 const auto prev_avail_in = m_strm->avail_in;
880 const auto prev_avail_out = m_strm->avail_out;
881 ret = inflate(m_strm.get(), Z_NO_FLUSH);
882 // Invariant: When output space is available, inflate() always consumes
883 // all provided input into its internal state, even if no output is
884 // produced yet. This assert guards against zlib misbehavior or memory
885 // corruption during development.
886 assert(m_strm->avail_in == 0);
887 zerr("zlib_cvt::get fail", ret);
888 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
889 throw cvt_error("zlib_cvt::get fail: zlib made no progress");
890 }
891
892 // Latch end-of-stream as soon as inflate() observes it, before any
893 // post-condition checks below. If a check throws afterwards, this
894 // ensures is_eof() still reports true and any future get_main call
895 // short-circuits to 0 instead of re-entering inflate() on a finished
896 // stream. The latch is a pure value write and cannot throw.
897 if (ret == Z_STREAM_END)
898 m_stream_ended = true;
899
900 // Exiting the loop with output space still available means the underlying
901 // kernel ran out of bytes before zlib reached Z_STREAM_END. The compressed
902 // stream is truncated and partial output would silently corrupt caller data.
903 if (ret != Z_STREAM_END && m_strm->avail_out != 0)
904 throw cvt_error("zlib_cvt::get fail: compressed stream truncated");
905
906 auto res = aim_output - m_strm->avail_out;
907 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
908
909 if (res % sizeof(internal_type))
910 throw cvt_error("zlib_cvt::get fail: partial sequence");
911
912 return res / sizeof(internal_type);
913 }

941 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
942 requires (cvt_cpt::support_put<KernelType>)
943 {
944 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
945 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
946
947 auto to = reinterpret_cast<const unsigned char*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
948
949 writer.reset(CHUNK);
950
951 // Ensures m_strm's in/out pointers and counts are cleared on every exit
952 // path, so a throw cannot leave next_in / next_out pointing into the
953 // caller's input or writer's buffer after they go out of scope.
954 io_buf_guard buf_guard{*m_strm};
955
956 while (to_size > 0)
957 {
958 auto aim_output = (max_chunk / sizeof(internal_type) > to_size)
959 ? static_cast<decltype(m_strm->avail_out)>(to_size * sizeof(internal_type))
960 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
961
962 size_t write_size = 0;
963 while (aim_output != write_size)
964 {
965 m_strm->next_in = const_cast<unsigned char*>(to + write_size); // NOLINT(cppcoreguidelines-pro-type-const-cast)
966 auto cur_put_size = static_cast<size_t>(aim_output - write_size);
967 m_strm->avail_in = static_cast<decltype(m_strm->avail_in)>(cur_put_size);
968 m_strm->next_out = reinterpret_cast<unsigned char*>(writer.put_buf(CHUNK)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
969 m_strm->avail_out = CHUNK;
970
971 // put_buf reserved CHUNK bytes directly inside the kernel's
972 // persistent output buffer (root_cvt / mem_device specializations
973 // of cvt_writer commit at put_buf, not at commit()). If anything
974 // below throws, the trailing avail_out bytes are uninitialized
975 // memory still inside the kernel's cursor range and would be
976 // flushed to the device on the next put / close_stream / dtor.
977 // Roll the tail back on the failure path so the kernel buffer
978 // ends exactly at the last valid deflate output byte.
979 try
980 {
981 // Snapshot before the call so we can prove zlib made progress on
982 // at least one axis (input consumed or output produced). The
983 // Z_OK contract guarantees this, but enforcing it here turns a
984 // hypothetical infinite loop into an explicit failure if the
985 // library ever diverges from its contract.
986 const auto prev_avail_in = m_strm->avail_in;
987 const auto prev_avail_out = m_strm->avail_out;
988 auto ret = deflate(m_strm.get(), Z_NO_FLUSH);
989 zerr("zlib_cvt::put fail", ret);
990 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
991 throw cvt_error("zlib_cvt::put fail: zlib made no progress");
992 write_size += cur_put_size - m_strm->avail_in;
993 }
994 catch (...)
995 {
996 // rollback's throw conditions (len==0, len>reserved) are
997 // unreachable here: we just reserved CHUNK and avail_out<=CHUNK.
998 // The catch is defensive cleanup-path policy - swallow any
999 // future regression in rollback rather than mask the original
1000 // exception.
1001 if (m_strm->avail_out > 0)
1002 {
1003 try { writer.rollback(m_strm->avail_out); } catch (...) {} // NOLINT(bugprone-empty-catch)
1004 }
1005 throw;
1006 }
1007
1008 if (m_strm->avail_out)
1009 writer.rollback(m_strm->avail_out);
1010 }
1011 to += aim_output;
1012 to_size -= aim_output / sizeof(internal_type);
1013 }
1014 // commit() is the responsibility of abs_cvt::put.
1015 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
1016 }

1039 void flush_impl()
1040 requires (cvt_cpt::support_put<KernelType>)
1041 {
1042 if (m_sync_flush)
1043 {
1044 io_buf_guard buf_guard{*m_strm};
1045
1046 std::array<external_type, CHUNK> local_buf{};
1047 m_strm->next_in = nullptr;
1048 m_strm->avail_in = 0;
1049 while (true)
1050 {
1051 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1052 m_strm->avail_out = CHUNK;
1053 zerr("zlib_cvt::flush_impl fail", deflate(m_strm.get(), Z_SYNC_FLUSH));
1054 const size_t written = CHUNK - m_strm->avail_out;
1055 if (written > 0)
1056 BT::m_kernel.put(local_buf.data(), written);
1057 if (m_strm->avail_out)
1058 break;
1059 }
1060 }

1088 template <bool IgnoreBufError = false>
1089 void zerr(const char* info, int ret)
1090 {
1091 switch (ret)
1092 {
1093 case Z_ERRNO:
1094 throw cvt_error(std::string(info) + ": zlib i/o error");
1095 case Z_STREAM_ERROR:
1096 // Z_STREAM_ERROR has two meanings depending on the call site:
1097 // deflateInit returns it for an invalid compression level argument;
1098 // inflate / deflate / inflateEnd / deflateEnd return it when zlib
1099 // detects an inconsistent internal stream state (often a symptom
1100 // of memory corruption, ABI mismatch, or concurrent access).
1101 throw cvt_error(std::string(info) + ": invalid argument or inconsistent stream state (Z_STREAM_ERROR)");
1102 case Z_DATA_ERROR:
1103 throw cvt_error(std::string(info) + ": invalid or incomplete deflate data");
1104 case Z_MEM_ERROR:
1105 throw cvt_error(std::string(info) + ": out of memory");
1106 case Z_VERSION_ERROR:
1107 throw cvt_error(std::string(info) + ": zlib version mismatch");
1108 case Z_BUF_ERROR:
1109 if constexpr (!IgnoreBufError)
1110 throw cvt_error(std::string(info) + ": no progress possible (Z_BUF_ERROR)");
1111 break;
1112 default:
1113 if (ret < 0)
1114 throw cvt_error(std::string(info) + ": unknown zlib error " + std::to_string(ret));
1115 }
1116 }
1117
1155 void close_stream()
1156 {
1157 struct state_guard
1158 {
1159 zlib_cvt& self;
1160 ~state_guard()
1161 {
1162 self.BT::m_is_bos_done = false;
1163 self.BT::m_io_status = io_status::neutral;
1164 self.m_sync_flush = false;
1165 self.m_stream_ended = false;
1166 }
1167 } sg{*this};
1168
1169 // Skip cleanup if zlib was never successfully initialized (m_strm is
1170 // null until bos() calls inflateInit/deflateInit, and is reset to null
1171 // by the guards on both the success and failure paths).
1172 if (!m_strm)
1173 return;
1174
1175 if (BT::m_io_status == io_status::output)
1176 {
1177 if (!BT::m_is_bos_done)
1178 {
1179 // bos() ran (m_strm is valid) but main_cont_beg() was never called.
1180 // No user data was written, so skip the Z_FINISH loop and just free
1181 // the zlib state.
1182 deflateEnd(m_strm.get());
1183 m_strm.reset();
1184 }
1185 else
1186 {
1187 std::array<external_type, CHUNK> local_buf{};
1188 deflate_guard g(m_strm); // calls deflateEnd + resets m_strm on scope exit
1189 m_strm->next_in = nullptr;
1190 m_strm->avail_in = 0;
1191 while (true)
1192 {
1193 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1194 m_strm->avail_out = CHUNK;
1195 auto ret = deflate(m_strm.get(), Z_FINISH);
1196 zerr("zlib_cvt::close_stream fail", ret);
1197 const size_t written = CHUNK - m_strm->avail_out;
1198 if (written > 0)
1199 BT::m_kernel.put(local_buf.data(), written);
1200 if (ret == Z_STREAM_END)
1201 break;
1202 // Z_FINISH must reach Z_STREAM_END in finite iterations.
1203 // If deflate returned Z_OK and did not even fill CHUNK
1204 // (avail_out > 0), it is claiming "no more output right
1205 // now" without declaring end-of-stream. Since we feed
1206 // no input on this path, the next iteration will hit
1207 // the exact same state and loop forever. Surface the
1208 // inconsistency instead.
1209 if (m_strm->avail_out > 0)
1210 throw cvt_error("zlib_cvt::close_stream fail: Z_FINISH returned without stream end");
1211 }
1212 }
1213 }
1214 else if (BT::m_io_status == io_status::input)
1215 {
1216 // inflateEnd frees zlib's internal state regardless of the return
1217 // code, so reset m_strm BEFORE potentially throwing. Otherwise a
1218 // Z_STREAM_ERROR throw would leave m_strm non-null; the next entry
1219 // to close_stream would skip the early-return guard and call
1220 // inflateEnd again on already-freed state.
1221 auto ret = inflateEnd(m_strm.get());
1222 m_strm.reset();
1223 zerr("zlib_cvt::close_stream fail", ret);
1224 }
1225 }
1226private:
1227 unsigned m_put_level;
1228
1245 bool m_sync_flush{false};
1246
1264 bool m_stream_ended{false};
1265
1268 std::unique_ptr<z_stream> m_strm;
1269};
