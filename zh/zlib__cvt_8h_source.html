528 void attach(device_type&& dev = device_type{})
529 {
530 // close_stream failure only means the OLD output stream could not be
531 // finalized on its old device. Its state_guard resets m_io_status /
532 // m_is_bos_done / m_sync_flush, and the inflate_guard / deflate_guard
533 // inside close_stream have already released m_strm. So the failure
534 // does NOT prevent us from installing the new device. If we let it
535 // propagate here, the caller's `dev` (already moved into our parameter)
536 // would be destroyed during stack unwinding and silently lost. Capture
537 // the exception, install the new device first, then rethrow.
538 std::exception_ptr close_err;
539 try { close_stream(); }
540 catch (...) { close_err = std::current_exception(); }
541
542 BT::attach(std::move(dev));
543
544 if (close_err) std::rethrow_exception(close_err);
545 }

628 void main_cont_beg()
629 {
630 BT::main_cont_beg();
631
632 if (BT::m_io_status == io_status::output)

684 io_status bos()
685 {
686 BT::bos();
687 // After BT::bos() succeeds, m_io_status is no longer neutral. Any failure
688 // from here on must restore the invariant "m_io_status != neutral implies
689 // m_strm is a fully-initialized zlib stream" before propagating, otherwise
690 // copy/assign, flush_impl, and the direction-switch branches in
691 // abs_cvt::get/put will dereference a null m_strm. Reset the status and
692 // taint the converter so subsequent get/put/flush refuse to run until
693 // the caller reattaches a device.
694 try
695 {
696 if (BT::m_io_status == io_status::input)
697 {
698 if constexpr (cvt_cpt::support_get<KernelType>)
699 {
700 m_strm = std::make_unique<z_stream>();
701 m_strm->zalloc = Z_NULL;
702 m_strm->zfree = Z_NULL;
703 m_strm->opaque = Z_NULL;
704 m_strm->avail_in = 0;
705 m_strm->next_in = Z_NULL;
706 m_strm->state = Z_NULL;
707 auto ret = inflateInit(m_strm.get());
708 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
709 inflate_guard zlib_guard(m_strm);
710
711 // Contract: kernel.get() on BOS path guarantees to return exactly the
712 // requested byte count, or throw an exception if insufficient data is
713 // available. The assert below is purely defensive, verifying this
714 // invariant during development. It is intentionally compiled out in
715 // release builds.
716 std::array<external_type, zlib_header_size> header_buf{};
717 [[maybe_unused]] const size_t n = BT::m_kernel.get(header_buf.data(), zlib_header_size);
718 assert(n == zlib_header_size);
719
720 m_strm->avail_in = zlib_header_size;
721 m_strm->next_in = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
722
723 unsigned char ch = 0;
724 m_strm->avail_out = 1;
725 m_strm->next_out = &ch;
726
727 ret = inflate(m_strm.get(), Z_NO_FLUSH);
728 if (ret == Z_NEED_DICT)
729 throw cvt_error("zlib_cvt::bos fail: preset dictionary not supported");
730 zerr("zlib_cvt::bos fail", ret);
731
732 if ((m_strm->avail_in != 0) || (m_strm->avail_out != 1)) throw cvt_error("zlib_cvt::bos fail: Invalid zlib header");
733 m_strm->next_in = nullptr;
734 m_strm->avail_out = 0;
735 m_strm->next_out = nullptr;
736
737 zlib_guard.release();
738 }
739 else
740 {
741 throw cvt_error("zlib_cvt::bos fail: BT returned input but KernelType lacks get support");
742 }
743 }
744 else if (BT::m_io_status == io_status::output)
745 {
746 if constexpr (cvt_cpt::support_put<KernelType>)
747 {
748 m_strm = std::make_unique<z_stream>();
749 m_strm->zalloc = Z_NULL;
750 m_strm->zfree = Z_NULL;
751 m_strm->opaque = Z_NULL;
752 m_strm->state = Z_NULL;
753
754 auto ret = deflateInit(m_strm.get(), static_cast<int>(m_put_level));
755 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
756 deflate_guard zlib_guard(m_strm);
757
758 // Contract enforced below: after this deflate() call we require
759 // exactly zlib_header_size bytes in the output buffer and zero
760 // bytes/bits still pending inside zlib. Z_NO_FLUSH on a fresh
761 // stream with no input is expected to flush the queued header
762 // and nothing else. The post-checks turn that expectation into
763 // a hard invariant: any future zlib behavior change (more
764 // bytes, fewer bytes, or bytes left buffered) surfaces here as
765 // a loud failure rather than silent framing corruption.
766 std::array<external_type, zlib_header_size + 1> header_buf{};
767
768 m_strm->avail_in = 0;
769 m_strm->next_in = nullptr;
770 m_strm->avail_out = zlib_header_size + 1;
771 m_strm->next_out = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
772 ret = deflate(m_strm.get(), Z_NO_FLUSH);
773 zerr("zlib_cvt::bos fail", ret);
774
775 if (m_strm->avail_out != 1)
776 throw cvt_error("zlib_cvt::bos fail: zlib did not emit exactly the header");
777
778 unsigned pending_bytes = 0;
779 int pending_bits = 0;
780 if (deflatePending(m_strm.get(), &pending_bytes, &pending_bits) != Z_OK ||
781 pending_bytes != 0 || pending_bits != 0)
782 throw cvt_error("zlib_cvt::bos fail: zlib has unflushed bytes after header");
783
784 BT::m_kernel.put(header_buf.data(), zlib_header_size);
785
786 m_strm->avail_out = 0;
787 m_strm->next_out = nullptr;
788
789 zlib_guard.release();
790 }
791 else
792 {
793 throw cvt_error("zlib_cvt::bos fail: BT returned output but KernelType lacks put support");
794 }
795 }
796 else
797 throw cvt_error("zlib_cvt::bos fail: invalid response value.");
798 }
799 catch (...)
800 {
801 // m_strm has already been reset to nullptr by inflate_guard /
802 // deflate_guard (or by the manual reset on inflateInit/deflateInit
803 // failure) before the exception reaches here.
804 BT::m_io_status = io_status::neutral;
805 BT::set_tainted();
806 throw;
807 }
808 return BT::m_io_status;
809 }

872 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
873 requires (cvt_cpt::support_get<KernelType>)
874 {
875 // Once inflate() has reported Z_STREAM_END on this stream, further
876 // inflate() calls on the same z_stream would either fail with
877 // Z_DATA_ERROR or attempt to interpret trailing bytes as a new
878 // concatenated zlib stream. Both are wrong for a single-stream
879 // contract. Surface end-of-stream as a zero-byte read so callers
880 // see a clean EOF instead. Reset by close_stream's state_guard.
881 if (m_stream_ended) return 0;
882
883 // Note: We read one byte at a time intentionally. Decompression has
884 // unpredictable expansion ratio - a few compressed bytes could expand
885 // to overflow the output buffer. This keeps the code simple and correct.
886 // The underlying converter already buffers, so this doesn't cause syscalls.
887 reader.reset(1);
888
889 // Ensures m_strm's in/out pointers and counts are cleared on every exit
890 // path, so a throw cannot leave next_in pointing into reader's buffer
891 // after the reader is destroyed.
892 io_buf_guard buf_guard{*m_strm};
893
894 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
895 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
896
897 auto aim_output = (max_chunk / sizeof(internal_type) > to_max)
898 ? static_cast<decltype(m_strm->avail_out)>(to_max * sizeof(internal_type))
899 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
900
901 m_strm->avail_out = aim_output;
902 m_strm->next_out = reinterpret_cast<unsigned char*>(to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
903 auto ret = inflate(m_strm.get(), Z_NO_FLUSH);
904 zerr<true>("zlib_cvt::get fail", ret);
905
906 while (m_strm->avail_out && ret != Z_STREAM_END)
907 {
908 auto [ptr, len] = reader.get_buf(1);
909 if (len == 0) break;
910 m_strm->next_in = const_cast<unsigned char*>(reinterpret_cast<const unsigned char*>(ptr)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast,cppcoreguidelines-pro-type-const-cast)
911 m_strm->avail_in = 1;
912 // Snapshot before the call so we can prove zlib made progress on at
913 // least one axis (input consumed or output produced). The Z_OK
914 // contract guarantees this, but enforcing it here turns a
915 // hypothetical infinite loop into an explicit failure if the
916 // library ever diverges from its contract.
917 const auto prev_avail_in = m_strm->avail_in;
918 const auto prev_avail_out = m_strm->avail_out;
919 ret = inflate(m_strm.get(), Z_NO_FLUSH);
920 // Invariant: When output space is available, inflate() always consumes
921 // all provided input into its internal state, even if no output is
922 // produced yet. This assert guards against zlib misbehavior or memory
923 // corruption during development.
924 assert(m_strm->avail_in == 0);
925 zerr("zlib_cvt::get fail", ret);
926 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
927 throw cvt_error("zlib_cvt::get fail: zlib made no progress");
928 }
929
930 // Latch end-of-stream as soon as inflate() observes it, before any
931 // post-condition checks below. If a check throws afterwards, this
932 // ensures is_eof() still reports true and any future get_main call
933 // short-circuits to 0 instead of re-entering inflate() on a finished
934 // stream. The latch is a pure value write and cannot throw.
935 if (ret == Z_STREAM_END)
936 m_stream_ended = true;
937
938 // Exiting the loop with output space still available means the underlying
939 // kernel ran out of bytes before zlib reached Z_STREAM_END. The compressed
940 // stream is truncated and partial output would silently corrupt caller data.
941 if (ret != Z_STREAM_END && m_strm->avail_out != 0)
942 throw cvt_error("zlib_cvt::get fail: compressed stream truncated");
943
944 auto res = aim_output - m_strm->avail_out;
945 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
946
947 if (res % sizeof(internal_type))
948 throw cvt_error("zlib_cvt::get fail: partial sequence");
949
950 return res / sizeof(internal_type);
951 }

979 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
980 requires (cvt_cpt::support_put<KernelType>)
981 {
982 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
983 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
984
985 auto to = reinterpret_cast<const unsigned char*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
986
987 writer.reset(CHUNK);
988
989 // Ensures m_strm's in/out pointers and counts are cleared on every exit
990 // path, so a throw cannot leave next_in / next_out pointing into the
991 // caller's input or writer's buffer after they go out of scope.
992 io_buf_guard buf_guard{*m_strm};
993
994 while (to_size > 0)
995 {
996 auto aim_output = (max_chunk / sizeof(internal_type) > to_size)
997 ? static_cast<decltype(m_strm->avail_out)>(to_size * sizeof(internal_type))
998 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
999
1000 size_t write_size = 0;
1001 while (aim_output != write_size)
1002 {
1003 m_strm->next_in = const_cast<unsigned char*>(to + write_size); // NOLINT(cppcoreguidelines-pro-type-const-cast)
1004 auto cur_put_size = static_cast<size_t>(aim_output - write_size);
1005 m_strm->avail_in = static_cast<decltype(m_strm->avail_in)>(cur_put_size);
1006 m_strm->next_out = reinterpret_cast<unsigned char*>(writer.put_buf(CHUNK)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1007 m_strm->avail_out = CHUNK;
1008
1009 // put_buf reserved CHUNK bytes directly inside the kernel's
1010 // persistent output buffer (root_cvt / mem_device specializations
1011 // of cvt_writer commit at put_buf, not at commit()). If anything
1012 // below throws, the trailing avail_out bytes are uninitialized
1013 // memory still inside the kernel's cursor range and would be
1014 // flushed to the device on the next put / close_stream / dtor.
1015 // Roll the tail back on the failure path so the kernel buffer
1016 // ends exactly at the last valid deflate output byte.
1017 try
1018 {
1019 // Snapshot before the call so we can prove zlib made progress on
1020 // at least one axis (input consumed or output produced). The
1021 // Z_OK contract guarantees this, but enforcing it here turns a
1022 // hypothetical infinite loop into an explicit failure if the
1023 // library ever diverges from its contract.
1024 const auto prev_avail_in = m_strm->avail_in;
1025 const auto prev_avail_out = m_strm->avail_out;
1026 auto ret = deflate(m_strm.get(), Z_NO_FLUSH);
1027 zerr("zlib_cvt::put fail", ret);
1028 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
1029 throw cvt_error("zlib_cvt::put fail: zlib made no progress");
1030 write_size += cur_put_size - m_strm->avail_in;
1031 }
1032 catch (...)
1033 {
1034 // rollback's throw conditions (len==0, len>reserved) are
1035 // unreachable here: we just reserved CHUNK and avail_out<=CHUNK.
1036 // The catch is defensive cleanup-path policy - swallow any
1037 // future regression in rollback rather than mask the original
1038 // exception.
1039 if (m_strm->avail_out > 0)
1040 {
1041 try { writer.rollback(m_strm->avail_out); } catch (...) {} // NOLINT(bugprone-empty-catch)
1042 }
1043 throw;
1044 }
1045
1046 if (m_strm->avail_out)
1047 writer.rollback(m_strm->avail_out);
1048 }
1049 to += aim_output;
1050 to_size -= aim_output / sizeof(internal_type);
1051 }
1052 // commit() is the responsibility of abs_cvt::put.
1053 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
1054 }

1077 void flush_impl()
1078 requires (cvt_cpt::support_put<KernelType>)
1079 {
1080 if (m_sync_flush)
1081 {
1082 io_buf_guard buf_guard{*m_strm};
1083
1084 std::array<external_type, CHUNK> local_buf{};
1085 m_strm->next_in = nullptr;
1086 m_strm->avail_in = 0;
1087 while (true)
1088 {
1089 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)

1193 void close_stream()
1194 {
1195 struct state_guard
1196 {
1197 zlib_cvt& self;
1198 ~state_guard()
1199 {
1200 self.BT::m_is_bos_done = false;
1201 self.BT::m_io_status = io_status::neutral;
1202 self.m_sync_flush = false;
1203 self.m_stream_ended = false;
1204 }
1205 } sg{*this};
1206
1207 // Skip cleanup if zlib was never successfully initialized (m_strm is
1208 // null until bos() calls inflateInit/deflateInit, and is reset to null
1209 // by the guards on both the success and failure paths).
1210 if (!m_strm)
1211 return;
1212
1213 if (BT::m_io_status == io_status::output)
1214 {
1215 if (!BT::m_is_bos_done)
1216 {
1217 // bos() ran (m_strm is valid) but main_cont_beg() was never called.
1218 // No user data was written, so skip the Z_FINISH loop and just free
1219 // the zlib state.
1220 deflateEnd(m_strm.get());
1221 m_strm.reset();
1222 }
1223 else
1224 {
1225 std::array<external_type, CHUNK> local_buf{};
1226 deflate_guard g(m_strm); // calls deflateEnd + resets m_strm on scope exit
1227 m_strm->next_in = nullptr;
1228 m_strm->avail_in = 0;
1229 while (true)
1230 {
1231 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1232 m_strm->avail_out = CHUNK;
1233 auto ret = deflate(m_strm.get(), Z_FINISH);
1234 zerr("zlib_cvt::close_stream fail", ret);
1235 const size_t written = CHUNK - m_strm->avail_out;
1236 if (written > 0)
1237 BT::m_kernel.put(local_buf.data(), written);
1238 if (ret == Z_STREAM_END)
1239 break;
1240 // Z_FINISH must reach Z_STREAM_END in finite iterations.
1241 // If deflate returned Z_OK and did not even fill CHUNK
1242 // (avail_out > 0), it is claiming "no more output right
1243 // now" without declaring end-of-stream. Since we feed
1244 // no input on this path, the next iteration will hit
1245 // the exact same state and loop forever. Surface the
1246 // inconsistency instead.
1247 if (m_strm->avail_out > 0)
1248 throw cvt_error("zlib_cvt::close_stream fail: Z_FINISH returned without stream end");
1249 }
1250 }
1251 }
1252 else if (BT::m_io_status == io_status::input)
1253 {
1254 // inflateEnd frees zlib's internal state regardless of the return
1255 // code, so reset m_strm BEFORE potentially throwing. Otherwise a
1256 // Z_STREAM_ERROR throw would leave m_strm non-null; the next entry
1257 // to close_stream would skip the early-return guard and call
1258 // inflateEnd again on already-freed state.
1259 auto ret = inflateEnd(m_strm.get());
1260 m_strm.reset();
1261 zerr("zlib_cvt::close_stream fail", ret);
1262 }
1263 }
1264private:
1265 unsigned m_put_level;
1266
1283 bool m_sync_flush{false};
1284
1302 bool m_stream_ended{false};
1303
1306 std::unique_ptr<z_stream> m_strm;
1307};
