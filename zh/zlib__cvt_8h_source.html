413 zlib_cvt& operator=(const zlib_cvt& val)
414 {
415 if (this == &val) return *this;
416
417 // close_stream failure indicates the OLD output stream could not be
418 // finalized on its device; it leaves *this's in-memory zlib state
419 // self-consistent (guards inside close_stream reset m_strm / status).
420 // It does NOT damage the new assignment we are about to perform, so
421 // capture it and rethrow after the new state is fully installed.
422 std::exception_ptr close_err;
423 try { close_stream(); }
424 catch (...) { close_err = std::current_exception(); }
425
426 try
427 {
428 BT::operator=(val);
429 m_put_level = val.m_put_level;
430 m_sync_flush = val.m_sync_flush;
431 m_stream_ended = val.m_stream_ended;
432
433 if (BT::m_io_status == io_status::output)
434 {
435 m_strm = std::make_unique<z_stream>();
436 zerr("zlib_cvt copy assignment fail", deflateCopy(m_strm.get(), val.m_strm.get())); // NOLINT(cppcoreguidelines-pro-type-const-cast)
437 }
438 else if (BT::m_io_status == io_status::input)
439 {
440 m_strm = std::make_unique<z_stream>();
441 zerr("zlib_cvt copy assignment fail", inflateCopy(m_strm.get(), val.m_strm.get())); // NOLINT(cppcoreguidelines-pro-type-const-cast)
442 }
443 }
444 catch(...)
445 {
446 m_strm.reset();
447 m_stream_ended = false;
448 BT::set_tainted();
449 throw;
450 }
451
452 if (close_err) std::rethrow_exception(close_err);
453 return *this;
454 }

554 bool is_eof()
555 requires (cvt_cpt::support_get<KernelType>)
556 {
557 // m_stream_ended is the zlib-level EOF latch set by get_main when
558 // inflate() reports Z_STREAM_END. BT::is_eof() reflects the
559 // underlying kernel's bytes-available view, which can lag behind the
560 // zlib stream end (zlib's trailer/footer may sit inside the kernel
561 // buffer even after the decompressed payload is exhausted). Either
562 // condition is a true end-of-stream for the caller, so OR them.
563 return m_stream_ended || BT::is_eof();

681 void bos_impl()
682 {
683 if (BT::m_io_status == io_status::input)
684 {
685 if constexpr (cvt_cpt::support_get<KernelType>)
686 {
687 m_strm = std::make_unique<z_stream>();
688 m_strm->zalloc = Z_NULL;
689 m_strm->zfree = Z_NULL;
690 m_strm->opaque = Z_NULL;
691 m_strm->avail_in = 0;
692 m_strm->next_in = Z_NULL;
693 m_strm->state = Z_NULL;
694 auto ret = inflateInit(m_strm.get());
695 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
696 inflate_guard zlib_guard(m_strm);
697
698 // Contract: kernel.get() on BOS path guarantees to return exactly the
699 // requested byte count, or throw an exception if insufficient data is
700 // available. The assert below is purely defensive, verifying this
701 // invariant during development. It is intentionally compiled out in
702 // release builds.
703 std::array<external_type, zlib_header_size> header_buf{};
704 [[maybe_unused]] const size_t n = BT::m_kernel.get(header_buf.data(), zlib_header_size);
705 assert(n == zlib_header_size);
706
707 m_strm->avail_in = zlib_header_size;
708 m_strm->next_in = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
709
710 unsigned char ch = 0;
711 m_strm->avail_out = 1;
712 m_strm->next_out = &ch;
713
714 ret = inflate(m_strm.get(), Z_NO_FLUSH);
715 if (ret == Z_NEED_DICT)
716 throw cvt_error("zlib_cvt::bos fail: preset dictionary not supported");
717 zerr("zlib_cvt::bos fail", ret);
718
719 if ((m_strm->avail_in != 0) || (m_strm->avail_out != 1)) throw cvt_error("zlib_cvt::bos fail: Invalid zlib header");
720 m_strm->next_in = nullptr;
721 m_strm->avail_out = 0;
722 m_strm->next_out = nullptr;
723
724 zlib_guard.release();
725 }
726 else
727 throw cvt_error("zlib_cvt::bos fail: BT returned input but KernelType lacks get support");
728 }
729 else if (BT::m_io_status == io_status::output)
730 {
731 if constexpr (cvt_cpt::support_put<KernelType>)
732 {
733 m_strm = std::make_unique<z_stream>();
734 m_strm->zalloc = Z_NULL;
735 m_strm->zfree = Z_NULL;
736 m_strm->opaque = Z_NULL;
737 m_strm->state = Z_NULL;
738
739 auto ret = deflateInit(m_strm.get(), static_cast<int>(m_put_level));
740 if (ret != Z_OK) { m_strm.reset(); throw cvt_error("zlib_cvt::bos fail: Cannot initialize zlib."); }
741 deflate_guard zlib_guard(m_strm);
742
743 // Contract enforced below: after this deflate() call we require
744 // exactly zlib_header_size bytes in the output buffer and zero
745 // bytes/bits still pending inside zlib. Z_NO_FLUSH on a fresh
746 // stream with no input is expected to flush the queued header
747 // and nothing else. The post-checks turn that expectation into
748 // a hard invariant: any future zlib behavior change (more
749 // bytes, fewer bytes, or bytes left buffered) surfaces here as
750 // a loud failure rather than silent framing corruption.
751 std::array<external_type, zlib_header_size + 1> header_buf{};
752
753 m_strm->avail_in = 0;
754 m_strm->next_in = nullptr;
755 m_strm->avail_out = zlib_header_size + 1;
756 m_strm->next_out = reinterpret_cast<unsigned char*>(header_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
757 ret = deflate(m_strm.get(), Z_NO_FLUSH);
758 zerr("zlib_cvt::bos fail", ret);
759
760 if (m_strm->avail_out != 1)
761 throw cvt_error("zlib_cvt::bos fail: zlib did not emit exactly the header");
762
763 unsigned pending_bytes = 0;
764 int pending_bits = 0;
765 if (deflatePending(m_strm.get(), &pending_bytes, &pending_bits) != Z_OK ||
766 pending_bytes != 0 || pending_bits != 0)
767 throw cvt_error("zlib_cvt::bos fail: zlib has unflushed bytes after header");
768
769 BT::m_kernel.put(header_buf.data(), zlib_header_size);
770
771 m_strm->avail_out = 0;
772 m_strm->next_out = nullptr;
773
774 zlib_guard.release();
775 }
776 else
777 throw cvt_error("zlib_cvt::bos fail: BT returned output but KernelType lacks put support");
778 }
779 else
780 throw cvt_error("zlib_cvt::bos fail: invalid response value.");
781 }

812 size_t get_main(cvt_reader<KernelType>& reader, internal_type* to, size_t to_max)
813 requires (cvt_cpt::support_get<KernelType>)
814 {
815 // Once inflate() has reported Z_STREAM_END on this stream, further
816 // inflate() calls on the same z_stream would either fail with
817 // Z_DATA_ERROR or attempt to interpret trailing bytes as a new
818 // concatenated zlib stream. Both are wrong for a single-stream
819 // contract. Surface end-of-stream as a zero-byte read so callers
820 // see a clean EOF instead. Reset by close_stream's state_guard.
821 if (m_stream_ended) return 0;
822
823 // Note: We read one byte at a time intentionally. Decompression has
824 // unpredictable expansion ratio - a few compressed bytes could expand
825 // to overflow the output buffer. This keeps the code simple and correct.
826 // The underlying converter already buffers, so this doesn't cause syscalls.
827 reader.reset(1);
828
829 // Ensures m_strm's in/out pointers and counts are cleared on every exit
830 // path, so a throw cannot leave next_in pointing into reader's buffer
831 // after the reader is destroyed.
832 io_buf_guard buf_guard{*m_strm};
833
834 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
835 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
836
837 auto aim_output = (max_chunk / sizeof(internal_type) > to_max)
838 ? static_cast<decltype(m_strm->avail_out)>(to_max * sizeof(internal_type))
839 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
840
841 m_strm->avail_out = aim_output;
842 m_strm->next_out = reinterpret_cast<unsigned char*>(to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
843 auto ret = inflate(m_strm.get(), Z_NO_FLUSH);
844 zerr<true>("zlib_cvt::get fail", ret);
845
846 while (m_strm->avail_out && ret != Z_STREAM_END)
847 {
848 auto [ptr, len] = reader.get_buf(1);
849 if (len == 0) break;
850 m_strm->next_in = const_cast<unsigned char*>(reinterpret_cast<const unsigned char*>(ptr)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast,cppcoreguidelines-pro-type-const-cast)
851 m_strm->avail_in = 1;
852 // Snapshot before the call so we can prove zlib made progress on at
853 // least one axis (input consumed or output produced). The Z_OK
854 // contract guarantees this, but enforcing it here turns a
855 // hypothetical infinite loop into an explicit failure if the
856 // library ever diverges from its contract.
857 const auto prev_avail_in = m_strm->avail_in;
858 const auto prev_avail_out = m_strm->avail_out;
859 ret = inflate(m_strm.get(), Z_NO_FLUSH);
860 // Invariant: When output space is available, inflate() always consumes
861 // all provided input into its internal state, even if no output is
862 // produced yet. This assert guards against zlib misbehavior or memory
863 // corruption during development.
864 assert(m_strm->avail_in == 0);
865 zerr("zlib_cvt::get fail", ret);
866 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
867 throw cvt_error("zlib_cvt::get fail: zlib made no progress");
868 }
869
870 // Latch end-of-stream as soon as inflate() observes it, before any
871 // post-condition checks below. If a check throws afterwards, this
872 // ensures is_eof() still reports true and any future get_main call
873 // short-circuits to 0 instead of re-entering inflate() on a finished
874 // stream. The latch is a pure value write and cannot throw.
875 if (ret == Z_STREAM_END)
876 m_stream_ended = true;
877
878 // Exiting the loop with output space still available means the underlying
879 // kernel ran out of bytes before zlib reached Z_STREAM_END. The compressed
880 // stream is truncated and partial output would silently corrupt caller data.
881 if (ret != Z_STREAM_END && m_strm->avail_out != 0)
882 throw cvt_error("zlib_cvt::get fail: compressed stream truncated");
883
884 auto res = aim_output - m_strm->avail_out;
885 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
886
887 if (res % sizeof(internal_type))
888 throw cvt_error("zlib_cvt::get fail: partial sequence");
889
890 return res / sizeof(internal_type);
891 }

919 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
920 requires (cvt_cpt::support_put<KernelType>)
921 {
922 constexpr size_t max_type_limit = std::numeric_limits<decltype(m_strm->avail_out)>::max();
923 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
924
925 auto to = reinterpret_cast<const unsigned char*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
926
927 writer.reset(CHUNK);
928
929 // Ensures m_strm's in/out pointers and counts are cleared on every exit
930 // path, so a throw cannot leave next_in / next_out pointing into the
931 // caller's input or writer's buffer after they go out of scope.
932 io_buf_guard buf_guard{*m_strm};
933
934 while (to_size > 0)
935 {
936 auto aim_output = (max_chunk / sizeof(internal_type) > to_size)
937 ? static_cast<decltype(m_strm->avail_out)>(to_size * sizeof(internal_type))
938 : static_cast<decltype(m_strm->avail_out)>(max_chunk);
939
940 size_t write_size = 0;
941 while (aim_output != write_size)
942 {
943 m_strm->next_in = const_cast<unsigned char*>(to + write_size); // NOLINT(cppcoreguidelines-pro-type-const-cast)
944 auto cur_put_size = static_cast<size_t>(aim_output - write_size);
945 m_strm->avail_in = static_cast<decltype(m_strm->avail_in)>(cur_put_size);
946 m_strm->next_out = reinterpret_cast<unsigned char*>(writer.put_buf(CHUNK)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
947 m_strm->avail_out = CHUNK;
948
949 // put_buf reserved CHUNK bytes directly inside the kernel's
950 // persistent output buffer (root_cvt / mem_device specializations
951 // of cvt_writer commit at put_buf, not at commit()). If anything
952 // below throws, the trailing avail_out bytes are uninitialized
953 // memory still inside the kernel's cursor range and would be
954 // flushed to the device on the next put / close_stream / dtor.
955 // Roll the tail back on the failure path so the kernel buffer
956 // ends exactly at the last valid deflate output byte.
957 try
958 {
959 // Snapshot before the call so we can prove zlib made progress on
960 // at least one axis (input consumed or output produced). The
961 // Z_OK contract guarantees this, but enforcing it here turns a
962 // hypothetical infinite loop into an explicit failure if the
963 // library ever diverges from its contract.
964 const auto prev_avail_in = m_strm->avail_in;
965 const auto prev_avail_out = m_strm->avail_out;
966 auto ret = deflate(m_strm.get(), Z_NO_FLUSH);
967 zerr("zlib_cvt::put fail", ret);
968 if (m_strm->avail_in == prev_avail_in && m_strm->avail_out == prev_avail_out)
969 throw cvt_error("zlib_cvt::put fail: zlib made no progress");
970 write_size += cur_put_size - m_strm->avail_in;
971 }
972 catch (...)
973 {
974 // rollback's throw conditions (len==0, len>reserved) are
975 // unreachable here: we just reserved CHUNK and avail_out<=CHUNK.
976 // The catch is defensive cleanup-path policy - swallow any
977 // future regression in rollback rather than mask the original
978 // exception.
979 if (m_strm->avail_out > 0)
980 {
981 try { writer.rollback(m_strm->avail_out); } catch (...) {} // NOLINT(bugprone-empty-catch)
982 }
983 throw;
984 }
985
986 if (m_strm->avail_out)
987 writer.rollback(m_strm->avail_out);
988 }
989 to += aim_output;
990 to_size -= aim_output / sizeof(internal_type);
991 }
992 // commit() is the responsibility of abs_cvt::put.
993 // m_strm in/out fields are cleared by io_buf_guard on scope exit.
994 }

1017 void flush_impl()
1018 requires (cvt_cpt::support_put<KernelType>)
1019 {
1020 if (m_sync_flush)
1021 {
1022 io_buf_guard buf_guard{*m_strm};
1023
1024 std::array<external_type, CHUNK> local_buf{};
1025 m_strm->next_in = nullptr;
1026 m_strm->avail_in = 0;
1027 while (true)
1028 {
1029 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1030 m_strm->avail_out = CHUNK;
1031 zerr("zlib_cvt::flush_impl fail", deflate(m_strm.get(), Z_SYNC_FLUSH));
1032 const size_t written = CHUNK - m_strm->avail_out;
1033 if (written > 0)
1034 BT::m_kernel.put(local_buf.data(), written);
1035 if (m_strm->avail_out)
1036 break;
1037 }
1038 }

1066 template <bool IgnoreBufError = false>
1067 void zerr(const char* info, int ret)
1068 {
1069 switch (ret)
1070 {
1071 case Z_ERRNO:
1072 throw cvt_error(std::string(info) + ": zlib i/o error");
1073 case Z_STREAM_ERROR:
1074 // Z_STREAM_ERROR has two meanings depending on the call site:
1075 // deflateInit returns it for an invalid compression level argument;
1076 // inflate / deflate / inflateEnd / deflateEnd return it when zlib
1077 // detects an inconsistent internal stream state (often a symptom
1078 // of memory corruption, ABI mismatch, or concurrent access).
1079 throw cvt_error(std::string(info) + ": invalid argument or inconsistent stream state (Z_STREAM_ERROR)");
1080 case Z_DATA_ERROR:
1081 throw cvt_error(std::string(info) + ": invalid or incomplete deflate data");
1082 case Z_MEM_ERROR:
1083 throw cvt_error(std::string(info) + ": out of memory");
1084 case Z_VERSION_ERROR:
1085 throw cvt_error(std::string(info) + ": zlib version mismatch");
1086 case Z_BUF_ERROR:
1087 if constexpr (!IgnoreBufError)
1088 throw cvt_error(std::string(info) + ": no progress possible (Z_BUF_ERROR)");
1089 break;
1090 default:
1091 if (ret < 0)
1092 throw cvt_error(std::string(info) + ": unknown zlib error " + std::to_string(ret));
1093 }
1094 }

1133 void close_stream()
1134 {
1135 struct state_guard
1136 {
1137 zlib_cvt& self;
1138 ~state_guard()
1139 {
1140 self.BT::m_is_bos_done = false;
1141 self.BT::m_io_status = io_status::neutral;
1142 self.m_sync_flush = false;
1143 self.m_stream_ended = false;
1144 }
1145 } sg{*this};
1146
1147 // Skip cleanup if zlib was never successfully initialized (m_strm is
1148 // null until bos() calls inflateInit/deflateInit, and is reset to null
1149 // by the guards on both the success and failure paths).
1150 if (!m_strm)
1151 return;
1152
1153 if (BT::m_io_status == io_status::output)
1154 {
1155 if (!BT::m_is_bos_done)
1156 {
1157 // bos() ran (m_strm is valid) but main_cont_beg() was never called.
1158 // No user data was written, so skip the Z_FINISH loop and just free
1159 // the zlib state.
1160 deflateEnd(m_strm.get());
1161 m_strm.reset();
1162 }
1163 else
1164 {
1165 std::array<external_type, CHUNK> local_buf{};
1166 deflate_guard g(m_strm); // calls deflateEnd + resets m_strm on scope exit
1167 m_strm->next_in = nullptr;
1168 m_strm->avail_in = 0;
1169 while (true)
1170 {
1171 m_strm->next_out = reinterpret_cast<unsigned char*>(local_buf.data()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
1172 m_strm->avail_out = CHUNK;
1173 auto ret = deflate(m_strm.get(), Z_FINISH);
1174 zerr("zlib_cvt::close_stream fail", ret);
1175 const size_t written = CHUNK - m_strm->avail_out;
1176 if (written > 0)
1177 BT::m_kernel.put(local_buf.data(), written);
1178 if (ret == Z_STREAM_END)
1179 break;
1180 // Z_FINISH must reach Z_STREAM_END in finite iterations.
1181 // If deflate returned Z_OK and did not even fill CHUNK
1182 // (avail_out > 0), it is claiming "no more output right
1183 // now" without declaring end-of-stream. Since we feed
1184 // no input on this path, the next iteration will hit
1185 // the exact same state and loop forever. Surface the
1186 // inconsistency instead.
1187 if (m_strm->avail_out > 0)
1188 throw cvt_error("zlib_cvt::close_stream fail: Z_FINISH returned without stream end");
1189 }
1190 }
1191 }
1192 else if (BT::m_io_status == io_status::input)
1193 {
1194 // inflateEnd frees zlib's internal state regardless of the return
1195 // code, so reset m_strm BEFORE potentially throwing. Otherwise a
1196 // Z_STREAM_ERROR throw would leave m_strm non-null; the next entry
1197 // to close_stream would skip the early-return guard and call
1198 // inflateEnd again on already-freed state.
1199 auto ret = inflateEnd(m_strm.get());
1200 m_strm.reset();
1201 zerr("zlib_cvt::close_stream fail", ret);
1202 }
1203 }
1204private:
1205 unsigned m_put_level;
1206
1223 bool m_sync_flush{false};
1224
1242 bool m_stream_ended{false};
1243
1246 std::unique_ptr<z_stream> m_strm;
1247};
