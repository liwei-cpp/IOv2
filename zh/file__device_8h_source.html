161 explicit basic_file_device(const std::string& file_name, file_open_flag flags = file_open_flag::none)
162 {
163 const char* mode = fopen_mode(flags);
164 m_file.reset(std::fopen(file_name.c_str(), mode));
165 if (!m_file)
166 throw device_error("cannot open file \"" + file_name + "\" (errno=" + std::to_string(errno) + ")");
167
168 if (fseek_64(m_file.get(), 0, SEEK_END) != 0)
169 throw device_error("cannot get file length for \"" + file_name + "\"");
170
171 int64_t len = ftell_64(m_file.get());
172 if (len < 0)
173 throw device_error("cannot get file length for \"" + file_name + "\"");
174 else if (!std::in_range<size_t>(len))
175 throw device_error("file too large for this platform");
176
177 if (fseek_64(m_file.get(), 0, SEEK_SET) != 0)
178 throw device_error("cannot reset the file for \"" + file_name + "\"");
179
180 m_file_len = static_cast<size_t>(len);
181 }

405 void dseek(size_t v)
406 {
407 if (!is_open())
408 throw device_error("file_device::dseek fail: file is closed");
409
410 // In input-enabled modes, seeking is restricted to the current file length.
411 // This ensures that any subsequent read operation (dget) starts from a valid
412 // position within the existing data bounds.
413 if constexpr (IsIn)

468 void dput(const CharType* ch, size_t n)
469 requires (IsOut)
470 {
471 if (n == 0) return;
472 if (ch == nullptr)
473 throw device_error("file_device::dput fail: null buffer");
474 if (!is_open())
475 throw device_error("file_device::dput fail: file is closed");
476 size_t written = std::fwrite(ch, sizeof(CharType), n, m_file.get());
477
478 auto res = ftell_64(m_file.get());
479 if (res >= 0)
480 {
481 if (!std::in_range<size_t>(res))
482 throw device_error("file_device::dput fail: file size overflow");
483 if (static_cast<size_t>(res) > m_file_len)
484 m_file_len = static_cast<size_t>(res);
485 }
486 else
487 throw device_error("file_device::dput fail: cannot determine file position");
488
489 if (written != n)
490 throw device_error("file_device::dput fail: partial write");
491 }

517 static const char* fopen_mode(file_open_flag flags)
518 {
519 using enum file_open_flag;
520
521 file_open_flag checkres = flags & (trunc | binary | noreplace);
522
523 if constexpr (IsIn && IsOut)
524 {
525 if (checkres == none)
526 return "r+";
527 else if (checkres == trunc)
528 return "w+";
529 else if (checkres == binary)
530 return "r+b";
531 else if (checkres == (trunc | binary))
532 return "w+b";
533 // noreplace (O_EXCL) implies creating a new file, so it's always empty.
534 // Therefore, trunc is implicit in this mode.
535 else if (checkres == noreplace || checkres == (trunc | noreplace))
536 return "w+x";
537 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
538 return "w+bx";
539 else
540 throw device_error("Invalid open mode combination for read-write file");
541 }
542 else if constexpr (IsIn)
543 {
544 if (checkres == none)
545 return "r";
546 else if (checkres == binary)
547 return "rb";
548 else
549 throw device_error("Invalid open mode combination for read-only file");
550 }
551 else if constexpr (IsOut)
552 {
553 // In write-only mode, "w" already truncates by default if noreplace is not present.
554 if (checkres == none || checkres == trunc)
555 return "w";
556 else if (checkres == binary || checkres == (trunc | binary))
557 return "wb";
558 // noreplace implies creation of a new file.
559 else if (checkres == noreplace || checkres == (trunc | noreplace))
560 return "wx";
561 else if (checkres == (binary | noreplace) || checkres == (binary | trunc | noreplace))
562 return "wbx";
563 else
564 throw device_error("Invalid open mode combination for write-only file");
565 }
566 else
567 {
568 static_assert(dependent_false_nttp_v<IsIn, IsOut>, "file device cannot open with neither in nor out mode");
569 }
570 }

592 std::unique_ptr<FILE, file_deleter> m_file;

594 // Cached file length, in bytes.
595 //
596 // Type is `size_t` (not `uint64_t`) because the public API — `dsize()`,
597 // `dseek(size_t)`, `drseek(size_t)` — is `size_t`-typed; using `size_t`
598 // here keeps the value lossless across the boundary on every platform
599 // (including 32-bit, where a `uint64_t` field could hold values that
600 // `dsize()` cannot return without truncation).
601 //
602 // Invariant: `m_file_len <= INT64_MAX`. All writes to this field source
603 // from a non-negative `int64_t` (the result of `ftell_64()`), narrowed
604 // through `std::in_range<size_t>` in the constructor and `dput()`.
605 // This invariant is what makes `-static_cast<int64_t>(offset)` in
606 // `drseek()` safe: any `offset <= m_file_len` fits in `int64_t`'s
607 // positive range, so the negation cannot overflow. Do not introduce
608 // setters or alternate sources that bypass the `int64_t`-derived path.
609 size_t m_file_len = 0;
