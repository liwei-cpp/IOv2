295 std::lock_guard graph_lock(tie_graph_mutex());

297 if constexpr (std::derived_from<TSelf, abs_ostream>)

301 while (ptr != nullptr)
302 {
303 if (ptr == check)

305 auto* node = dynamic_cast<stream_common_operators*>(ptr);
