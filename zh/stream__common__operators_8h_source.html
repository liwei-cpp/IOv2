287 std::lock_guard graph_lock(tie_graph_mutex());

289 if constexpr (std::derived_from<TSelf, abs_ostream>)

293 while (ptr != nullptr)

297 auto* node = dynamic_cast<stream_common_operators*>(ptr);

300 ptr = node->tie();
301 }
302 }
303 self.m_tie_stream.store(str);
