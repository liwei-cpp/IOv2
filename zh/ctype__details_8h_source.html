72// The ctype lookup tables in this header and in ctype.h are sized as
73// std::numeric_limits<unsigned char>::max() + 1, on the assumption that a
74// byte is 8 bits and that the table covers all values an unsigned char can
75// hold. Exotic platforms with CHAR_BIT != 8 (some DSPs, historical mainframes)
76// would silently grow every per-byte table by 2^(CHAR_BIT-8)x. Reject them at
77// compile time rather than ship a non-obviously-broken binary.
78static_assert(CHAR_BIT == 8,
79 "facet/ctype tables assume an 8-bit byte; see ctype.h::s_len");

231 ctype_conf(const std::string& name)
232 : ft_basic<ctype<char>>()
233 , m_inter_locale(name.c_str())
234 {
235 // Build lookup tables using standard POSIX functions
236 for (unsigned c = 0; c <= std::numeric_limits<unsigned char>::max(); ++c)
237 {
238 m_toupper_table[c] = toupper_l(static_cast<int>(c), m_inter_locale.c_locale);
239 m_tolower_table[c] = tolower_l(static_cast<int>(c), m_inter_locale.c_locale);
240
241 mask m = 0;
242 if (isupper_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::upper;
243 if (islower_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::lower;
244 if (isalpha_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::alpha;
245 if (isdigit_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::digit;
246 if (isxdigit_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::xdigit;
247 if (isspace_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::space;
248 if (isprint_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::print;
249 if (iscntrl_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::cntrl;
250 if (ispunct_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::punct;
251 m_table[c] = m;
252 }

507 {

512 if (iswctype_l(c, m_wmask_upper, m_inter_locale.c_locale)) res |= base_ft<ctype>::upper;
