59 ctype_conf(const std::string& name)
60 : ft_basic<ctype<char>>()
61 , m_inter_locale(name.c_str())
62 {
63 // Build lookup tables using standard POSIX functions
64 for (unsigned c = 0; c <= std::numeric_limits<unsigned char>::max(); ++c)
65 {
66 m_toupper_table[c] = toupper_l(static_cast<int>(c), m_inter_locale.c_locale);
67 m_tolower_table[c] = tolower_l(static_cast<int>(c), m_inter_locale.c_locale);
68
69 mask m = 0;
70 if (isupper_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::upper;
71 if (islower_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::lower;
72 if (isalpha_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::alpha;
73 if (isdigit_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::digit;
74 if (isxdigit_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::xdigit;
75 if (isspace_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::space;
76 if (isprint_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::print;
77 if (iscntrl_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::cntrl;
78 if (ispunct_l(static_cast<int>(c), m_inter_locale.c_locale)) m |= base_ft<ctype>::punct;
79 m_table[c] = m;
80 }

177 // Matches std::ctype<wchar_t>::widen() semantics: only required to be
178 // correct for the basic source character set. For multi-byte locales
179 // (e.g. UTF-8), high bytes are not standalone characters and btowc()
180 // returns WEOF for them at table-build time; the corresponding entries
181 // in m_widen hold WEOF cast to CharT (a sentinel-looking but
182 // unspecified value). Callers must not widen() such bytes.
183 virtual CharT widen(char c) const
184 {
185 return static_cast<CharT>(m_widen[static_cast<unsigned char>(c)]);
186 }

188 virtual std::optional<char> narrow(CharT wc) const
189 {
190 if (out_of_wchar_range(wc)) return std::nullopt;
191 clocale_user guard(m_inter_locale);
192 const int c = wctob(wc);
193 if (c == EOF) return std::nullopt;
194 else return static_cast<char>(c);
195 }
