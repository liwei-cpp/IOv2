217 template <typename TIter>
218 void add(TIter b, TIter e, TValue v)
219 {
220 node* node_ptr = &m_root;
221 while (b != e)
222 {
223 CharT ch = *b++;
224
225 auto child = node_ptr->children.find(ch);
226 if (child == node_ptr->children.end())
227 {
228 bool insert_suc = false;
229 auto c = std::make_unique<node>(std::nullopt, node_ptr->depth + 1);
230 std::tie(child, insert_suc) = node_ptr->children.insert(std::pair(ch, std::move(c)));
231 if (!insert_suc)
232 throw std::runtime_error("prefix_tree insert fails");
233 }
234 node_ptr = child->second.get();
235 }
236
237 if (node_ptr->val.has_value() && *node_ptr->val != v)
238 throw std::runtime_error("duplicate items in prefix_tree");
239 node_ptr->val = v;
240 }

270 [[nodiscard]] TIter max_match(TIter b, TSent e, match_out_type& out) const
271 {
272 // Debug assertion: check for valid range (only for random access iterators)
273 if constexpr (std::random_access_iterator<TIter> && std::same_as<TIter, TSent>)
274 assert(b <= e && "max_match: invalid iterator range (b > e)");
275
276 if constexpr (is_small_type_v<TValue>)
277 out = std::nullopt;
278 else
279 out = nullptr;
280
281 if (m_root.val.has_value())
282 {
283 if constexpr (is_small_type_v<TValue>)
284 out = m_root.val;
285 else
286 out = &(*m_root.val);
287 }
288
289 size_t found_depth = 0;
290
291 const node* node_ptr = &m_root;
292 for (; b != e; ++b)
293 {
294 CharT ch = *b;
295 auto it = node_ptr->children.find(ch);
296 if (it == node_ptr->children.end()) break;
297 node_ptr = it->second.get();
298 if (node_ptr->val.has_value())
299 {
300 if constexpr (is_small_type_v<TValue>)
301 out = node_ptr->val;
302 else
303 out = &(*node_ptr->val);
304 found_depth = node_ptr->depth;
305 }
306 }
307
308 if (node_ptr->depth != found_depth)
309 {
310 const size_t steps_back = node_ptr->depth - found_depth;
311 std::advance(b, -static_cast<std::ptrdiff_t>(steps_back));
312 }
313 return b;
