96template <>
97struct hash<IOv2::detail::msg_key>
98{
99 std::size_t operator()(const IOv2::detail::msg_key& k) const noexcept
100 {
101 // Golden-ratio constant sized to std::size_t (2^bits / phi). The 64-bit
102 // 0x9e3779b97f4a7c15 mixes the high half of a 64-bit seed that the 32-bit
103 // 0x9e3779b9 leaves largely untouched; chosen at compile time so the mixer
104 // matches the actual width of std::size_t on every platform. (The unselected
105 // ternary branch is a well-defined explicit truncation, valid on either width.)
106 constexpr std::size_t golden = sizeof(std::size_t) >= 8
107 ? static_cast<std::size_t>(0x9e3779b97f4a7c15ULL)
108 : static_cast<std::size_t>(0x9e3779b9UL);
109
110 const std::hash<std::string> h;
111 std::size_t seed = h(k.domain);
112 seed ^= h(k.lang) + golden + (seed << 6) + (seed >> 2);
113 seed ^= h(k.dirname) + golden + (seed << 6) + (seed >> 2);
114 seed ^= h(k.cvt) + golden + (seed << 6) + (seed >> 2);
115 return seed;
116 }
117};
118} // namespace std

192 template <typename TF>
193 std::shared_ptr<abs_ft> try_get(const std::string& name)
194 {
195 const facet_id_t id = TF::id();
196
197 {
198 std::scoped_lock guard(m_mutex);
199 if (auto it = m_cache.find(id); it != m_cache.end())
200 {
201 if (auto cached = it->second.get(name))
202 return *cached;
203 }
204 }
205
206 // Construct the facet *outside* the lock: a facet constructor may be
207 // expensive and -- crucially -- may itself call back into ori_facet_buf, so
208 // building under m_mutex would both serialize all construction and risk a
209 // self-deadlock on this non-recursive mutex. Re-lock only to insert, and
210 // double-check: if another thread (or a re-entrant call) inserted the same
211 // (id, name) meanwhile, return that entry and discard the object built here.
212 auto obj = std::make_shared<TF>(name);
213
214 std::scoped_lock guard(m_mutex);
215 auto& sub_cache = m_cache[id];
216 if (auto cached = sub_cache.get(name))
217 return *cached;
218 // The get() above missed under this same lock, so the key is absent and put()
219 // simply inserts (no existing value to overwrite).
220 sub_cache.put(name, obj);
221 return obj;
222 }

456 static std::string resolve_locale(const char* category_var)
457 {
458 std::string name = "C";
459 if (const char* all = std::getenv("LC_ALL"); all && *all)
460 name = all;
461 else if (const char* cat = std::getenv(category_var); cat && *cat)
462 name = cat;
463 else if (const char* lang = std::getenv("LANG"); lang && *lang)
464 name = lang;
465
466 // Validate the resolved name against the C library: clocale_wrapper builds a
467 // locale_t via newlocale and throws if the name cannot be instantiated. An
468 // unusable name (e.g. a bogus LANG) falls back to the neutral "C". Done here,
469 // once per category at construction, so locale_name stays a plain O(1) read.
470 try
471 {
472 clocale_wrapper probe(name.c_str());
473 return name;
474 }
475 catch (const cvt_error&)
476 {
477 return "C";
478 }
479 }
