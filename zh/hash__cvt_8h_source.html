174 std::exception_ptr detach_impl() noexcept

176 std::exception_ptr local_err = nullptr;

215 {

272 void dump_stream()
273 {
274 if (!m_has_main_cont) return;
275 if (!m_hash) return;
276
277 // Use copy_state()->final() to preserve original state until put() succeeds.
278 // This provides strong exception safety: if put() fails, state is unchanged
279 // and the operation can be retried.
280 //
281 // Intentional: a failure inside `BT::m_kernel.put(...)` is the kernel's
282 // own concern — the kernel layer must taint itself per the abs_cvt
283 // contract. hash_cvt deliberately does NOT mirror that taint upward,
284 // because (a) hash_cvt's own invariants remain intact on a kernel-side
285 // partial write (`m_has_main_cont` stays true, `m_hash` is untouched
286 // via copy_state), and (b) leaving hash_cvt clean allows the caller to
287 // recover the kernel transiently and retry `dump_stream()` without
288 // losing the accumulated `update()` history.
289 auto cp_state = m_hash->copy_state();
290 if (!cp_state)
291 throw cvt_error("hash_cvt::dump_stream fail: cannot copy hash state");
292 auto digest = cp_state->final();
293 if (digest.empty())
294 throw cvt_error("hash_cvt::dump_stream fail: cannot get hash result");
295
296 switch (m_out_fmt)
297 {
298 case hash_fmt::binary:
299 BT::m_kernel.put(reinterpret_cast<const external_type*>(digest.data()), digest.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
300 break;
301 case hash_fmt::upper_hex:
302 {
303 std::string hex_string = Botan::hex_encode(digest);
304 BT::m_kernel.put(reinterpret_cast<const external_type*>(hex_string.data()), hex_string.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
305 }
306 break;
307 case hash_fmt::lower_hex:
308 {
309 std::string hex_string = Botan::hex_encode(digest, false);
310 BT::m_kernel.put(reinterpret_cast<const external_type*>(hex_string.data()), hex_string.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
311 }
312 break;
313 default:
314 throw cvt_error("hash_cvt::dump_stream fail: invalid output format");
315 }
316
317 // Only clear state after successful output
318 m_has_main_cont = false;
319 try
320 {
321 m_hash->clear();
322 }
323 catch (...)
324 {
325 BT::set_tainted();
326 throw;
327 }
328 }
329private:
330 std::unique_ptr<Botan::HashFunction> m_hash;
331 bool m_has_main_cont{false};
332 hash_fmt m_out_fmt{hash_fmt::lower_hex};
