338 std::exception_ptr detach_impl() noexcept
339 {
340 std::exception_ptr local_err = nullptr;
341 try { if (m_has_main_cont) dump_stream(); }
342 catch (...)
343 {
344 local_err = std::current_exception();

346 // Clear any residual hash state so a subsequent attach()+put()
347 // does not accumulate on top of half-finished data. If clear()

451 void put_main(cvt_writer<KernelType>& /*writer*/, const internal_type* to, size_t to_size)
452 {
453 if (to_size == 0) return;
454 if (!m_hash)
455 throw cvt_error("hash_cvt::put_main fail: hash not initialized (moved-from object?)");

461
462 m_hash->update(reinterpret_cast<const uint8_t*>(to), to_size * sizeof(internal_type)); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
463 m_has_main_cont = true;

491 void adjust_impl(const cvt_behavior& acc)
492 {
493 if (const auto* shf_ptr = dynamic_cast<const set_hash_fmt*>(&acc); shf_ptr)
494 m_out_fmt = shf_ptr->m_val;
495 else if (const auto* dh_ptr = dynamic_cast<const dump_hash*>(&acc); dh_ptr)
496 {
497 if (m_has_main_cont)

499 dump_stream();
500 if (dh_ptr->m_delim)
501 {
502 const uint8_t delim = *dh_ptr->m_delim;
503 // If the delimiter write fails, the digest is already on
504 // the kernel without its separator; subsequent dumps would
505 // produce concatenated digests with no boundary. Taint so
506 // the user is forced to detach/attach instead of silently
507 // emitting corrupted output.
508 try

511 }
512 catch (...)

594 void dump_stream()
595 {
596 if (!m_has_main_cont) return;
597 if (!m_hash) return;
598
599 // Use copy_state()->final() to preserve original state until put() succeeds.
600 // This provides strong exception safety: if put() fails, state is unchanged
601 // and the operation can be retried.
602 //
603 // Intentional: a failure inside `BT::m_kernel.put(...)` is the kernel's
604 // own concern — the kernel layer must taint itself per the abs_cvt
605 // contract. hash_cvt deliberately does NOT mirror that taint upward,
606 // because (a) hash_cvt's own invariants remain intact on a kernel-side
607 // partial write (`m_has_main_cont` stays true, `m_hash` is untouched
608 // via copy_state), and (b) leaving hash_cvt clean allows the caller to
609 // recover the kernel transiently and retry `dump_stream()` without
610 // losing the accumulated `update()` history.
611 auto cp_state = m_hash->copy_state();
612 if (!cp_state)
613 throw cvt_error("hash_cvt::dump_stream fail: cannot copy hash state");
614 auto digest = cp_state->final();
615 if (digest.empty())
616 throw cvt_error("hash_cvt::dump_stream fail: cannot get hash result");
617
618 switch (m_out_fmt)
619 {
620 case hash_fmt::binary:
621 BT::m_kernel.put(reinterpret_cast<const external_type*>(digest.data()), digest.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
622 break;
623 case hash_fmt::upper_hex:
624 {
625 std::string hex_string = Botan::hex_encode(digest);
626 BT::m_kernel.put(reinterpret_cast<const external_type*>(hex_string.data()), hex_string.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
627 }
628 break;
629 case hash_fmt::lower_hex:
630 {
631 std::string hex_string = Botan::hex_encode(digest, false);
632 BT::m_kernel.put(reinterpret_cast<const external_type*>(hex_string.data()), hex_string.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
633 }
634 break;
635 default:
636 throw cvt_error("hash_cvt::dump_stream fail: invalid output format");
637 }
638
639 // Only clear state after successful output
640 m_has_main_cont = false;
641 try
642 {
643 m_hash->clear();
644 }
645 catch (...)
646 {
647 BT::set_tainted();
648 throw;
649 }
650 }
651private:
652 std::unique_ptr<Botan::HashFunction> m_hash;
653 bool m_has_main_cont{false};
654 hash_fmt m_out_fmt{hash_fmt::lower_hex};
