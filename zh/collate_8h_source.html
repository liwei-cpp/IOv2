462 std::pair<TIter, size_t> transform(const CharT* low, const CharT* high, TIter dest, size_t mx_len = 0) const
463 {
464 size_t trans_count = 0;
465 std::vector<CharT> buf;
466 std::vector<CharT> buf2;
467
468 while ((low != high) && ((mx_len == 0) || (trans_count < mx_len)))
469 {
470 const CharT* cur = low;
471 if (const CharT* next = std::find(low, high, static_cast<CharT>(0)); next == high)
472 {
473 buf.resize(high - low);
474 std::copy(low, high, buf.data());
