32 collate_conf(const std::string& name)
33 : ft_basic<collate<CharT>>()
34 , m_inter_locale(name.c_str())
35 {
36 if constexpr (std::is_same_v<CharT, char8_t>)
37 {
38 // collate<char8_t> routes through the narrow strcoll/strxfrm path,
39 // which interprets its input as UTF-8 only when the inter locale's
40 // LC_CTYPE codeset is UTF-8. The codeset name from
41 // nl_langinfo(CODESET) is implementation-defined, so rather than
42 // string-matching it we probe the locale's own multibyte decoder on
43 // known code points: mbrtoc32 exercises the same LC_CTYPE codeset
44 // that strcoll/strxfrm use to parse their byte input, and because
45 // m_inter_locale is built from a single name with LC_ALL_MASK its
46 // CTYPE and COLLATE codesets are the same. Agreement here therefore
47 // means strcoll/strxfrm agree.
48 static constexpr struct { const char* mb; size_t len; char32_t cp; }
49 probes[] = {
50 {"\xC3\xA9", 2, U'é'}, // U+00E9 e-acute
51 {"\xE2\x82\xAC", 3, U'€'}, // U+20AC euro sign
52 {"\xF0\x9F\x98\x80", 4, U'\U0001F600'}, // U+1F600 grinning face
53 };
54 clocale_user guard(m_inter_locale);
55 for (const auto& p : probes)
56 {
57 char32_t c32 = 0;
58 std::mbstate_t st{};
59 size_t n = std::mbrtoc32(&c32, p.mb, p.len, &st);
60 if ((n != p.len) || (c32 != p.cp))
61 throw cvt_error("collate_conf<char8_t>: inter locale is not UTF-8");
62 }
63 }
64 }

67 virtual std::strong_ordering compare(const CharT* low1, const CharT* high1,
68 const CharT* low2, const CharT* high2) const
69 {
70 std::vector<CharT> buf1; bool extra_eos1 = false;
71 std::vector<CharT> buf2; bool extra_eos2 = false;
72
73 clocale_user guard(m_inter_locale);
74
75 while ((low1 != high1) && (low2 != high2))
76 {
77 const CharT* cl1 = low1;
78 if (auto ch1 = std::find(low1, high1, static_cast<CharT>(0)); ch1 == high1)
79 {
80 auto data_len = high1 - low1;
81 buf1.resize(data_len + 1 + SIMD_PADDING_BYTES / sizeof(CharT));
82 std::copy(low1, high1, buf1.data());
83 buf1[data_len] = static_cast<CharT>(0);
84 extra_eos1 = true;
85 cl1 = buf1.data();
86 low1 = high1;
87 }
88 else low1 = ch1 + 1;
89
90 const CharT* cl2 = low2;
91 if (auto ch2 = std::find(low2, high2, static_cast<CharT>(0)); ch2 == high2)
92 {
93 auto data_len = high2 - low2;
94 buf2.resize(data_len + 1 + SIMD_PADDING_BYTES / sizeof(CharT));
95 std::copy(low2, high2, buf2.data());
96 buf2[data_len] = static_cast<CharT>(0);
97 extra_eos2 = true;
98 cl2 = buf2.data();
99 low2 = high2;
100 }
101 else low2 = ch2 + 1;
102
103 int c_res = 0;
104 if constexpr (std::is_same_v<CharT, char>)
105 c_res = std::strcoll(cl1, cl2);
106 else if constexpr (std::is_same_v<CharT, wchar_t>)
107 c_res = std::wcscoll(cl1, cl2);
108 else if constexpr (std::is_same_v<CharT, char8_t>)
109 c_res = std::strcoll(reinterpret_cast<const char*>(cl1),
110 reinterpret_cast<const char*>(cl2));
111 else if constexpr (wchar_t_is_utf32)
112 {
113 if constexpr (std::is_same_v<CharT, char32_t>)
114 c_res = std::wcscoll(reinterpret_cast<const wchar_t*>(cl1),
115 reinterpret_cast<const wchar_t*>(cl2));
116 else
117 static_assert(dependent_false_v<CharT>, "collate_conf::compare is not implemented.");
118 }
119 else
120 static_assert(dependent_false_v<CharT>, "collate_conf::compare is not implemented.");
121
122 if (c_res < 0) return std::strong_ordering::less;
123 if (c_res > 0) return std::strong_ordering::greater;
124 }
125
126 if (low1 != high1) return std::strong_ordering::greater;
127 if (low2 != high2) return std::strong_ordering::less;
128 if (extra_eos1 && !extra_eos2) return std::strong_ordering::less;
129 if (!extra_eos1 && extra_eos2) return std::strong_ordering::greater;
130 return std::strong_ordering::equal;
131 }

182 virtual size_t transform(const CharT* low, const CharT* high, CharT* dest, size_t mx_len = 0) const
183 {
184 // transform() emits an opaque collation key, not text: the output bytes
185 // are order-preserving sort weights — strcmp/wcscmp on the result
186 // reproduces strcoll/wcscoll on the input — and are NOT a readable or
187 // valid char/UTF sequence. The key is only ever compared bytewise,
188 // never decoded back to characters.
189 size_t trans_count = 0;
190 // All buffers are hoisted out of the loop so resize() reuses their
191 // capacity instead of reallocating per segment.
192 // buf : input staging — null-terminated copy of a segment that has
193 // no embedded '\0' (so cur can point at a terminated string).
194 // buf2 : output staging for strxfrm/wcsxfrm.
195 std::vector<CharT> buf;
196 std::vector<CharT> buf2;
197 bool extra_eos = false;
198
199 clocale_user guard(m_inter_locale);
200 while ((low != high) && ((mx_len == 0) || (trans_count < mx_len)))
201 {
202 const CharT* cur = low;
203 if (auto next = std::find(low, high, static_cast<CharT>(0)); next == high)
204 {
205 auto data_len = high - low;
206 buf.resize(data_len + 1 + SIMD_PADDING_BYTES / sizeof(CharT));
207 std::copy(low, high, buf.data());
208 buf[data_len] = static_cast<CharT>(0);
209 cur = buf.data();
210 low = high;
211 extra_eos = true;
212 }
213 else
214 low = next + 1;
215
216 {
217 // strxfrm/wcsxfrm always append a terminating '\0' (writing
218 // cur_trans + 1 elements). Transform into the reused buffer
219 // first, then copy exactly cur_trans elements to dest, so the
220 // appended '\0' is discarded. Writing it straight into dest
221 // would overflow the caller's buffer by one on the trailing
222 // segment, for which transform_length only accounts for
223 // cur_trans (no terminator).
224 size_t trans_len = 0;
225 if constexpr (std::is_same_v<CharT, char>)
226 trans_len = strxfrm(nullptr, cur, 0);
227 else if constexpr (std::is_same_v<CharT, wchar_t>)
228 trans_len = wcsxfrm(nullptr, cur, 0);
229 else if constexpr (std::is_same_v<CharT, char8_t>)
230 trans_len = strxfrm(nullptr, reinterpret_cast<const char*>(cur), 0);
231 else if constexpr ((std::is_same_v<CharT, char32_t> &&
232 wchar_t_is_utf32))
233 trans_len = wcsxfrm(nullptr, reinterpret_cast<const wchar_t*>(cur), 0);
234 else
235 static_assert(dependent_false_v<CharT>, "collate_conf::transform is not implemented.");
236
237 if (trans_len == xfrm_failed)
238 throw cvt_error("collate_conf::transform: strxfrm/wcsxfrm failed");
239 buf2.resize(trans_len + 1);
240
241 size_t cur_trans = 0;
242 if constexpr (std::is_same_v<CharT, char>)
243 cur_trans = strxfrm(buf2.data(), cur, buf2.size());
244 else if constexpr (std::is_same_v<CharT, wchar_t>)
245 cur_trans = wcsxfrm(buf2.data(), cur, buf2.size());
246 else if constexpr (std::is_same_v<CharT, char8_t>)
247 cur_trans = strxfrm(reinterpret_cast<char*>(buf2.data()), reinterpret_cast<const char*>(cur), buf2.size());
248 else if constexpr ((std::is_same_v<CharT, char32_t> &&
249 wchar_t_is_utf32))
250 cur_trans = wcsxfrm(reinterpret_cast<wchar_t*>(buf2.data()), reinterpret_cast<const wchar_t*>(cur), buf2.size());
251 else
252 static_assert(dependent_false_v<CharT>, "collate_conf::transform is not implemented.");
253
254 if (cur_trans == xfrm_failed)
255 throw cvt_error("collate_conf::transform: strxfrm/wcsxfrm failed");
256 if (mx_len != 0)
257 cur_trans = std::min(cur_trans, mx_len - trans_count);
258 dest = std::copy(buf2.data(), buf2.data() + cur_trans, dest);
259 trans_count += cur_trans;
260 }
261
262 if ((!extra_eos) && ((mx_len == 0) || (trans_count < mx_len)))
263 {
264 *dest++ = '\0';
265 ++trans_count;
266 }
267 }
268 return trans_count;
269 }

271 // m_inter_locale is shared across compare / transform_length / transform,
272 // all of which are const. Each call constructs a clocale_user that calls
273 // uselocale(m_inter_locale.c_locale) to swap the calling thread's locale.
274 // POSIX does not explicitly guarantee that the same locale_t may be
275 // passed to uselocale() concurrently from multiple threads; IOv2 relies
276 // on the de-facto thread-safety provided by glibc and macOS libc on the
277 // target platforms (Linux / macOS). Porting to platforms with stricter
278 // semantics requires reevaluating this assumption.
279 clocale_wrapper m_inter_locale;
