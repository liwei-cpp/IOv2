50// ─── Facet type-identity key (single source of truth) ───────────────────────
51//
52// Every facet type gets a process-stable key of type `facet_id_t`, used to key
53// facets into the locale caches (`locale::m_facet_confs` / `m_facets`) and into
54// `ori_facet_buf`. Two modes:
55//
56// - Header-only (single binary): the key is the *address* of a per-type static
57// (`type_id<T>::s_id`). Zero collision, constant-initialised (safe to read
58// during static init), and effectively free. Within one binary the linker
59// merges all copies of that static, so every TU agrees on the address.
60//
61// - Shared library (`IOV2_SHARED`): the same facet type is instantiated in
62// several DSOs, and `-fvisibility=hidden` keeps each DSO's `s_id` local -- so
63// an address is per-DSO and breaks cross-DSO facet identity (that was the
64// "cannot get numeric facet" bug). Use `std::type_index` instead: identity is
65// by *type* (`type_info::operator==`), not by object address, so it is stable
66// across DSO / `dlopen(RTLD_LOCAL)` / DLL boundaries and collision-free
67// (`hash_code` only buckets; `operator==` decides the match).
68//
69// This relies on `type_info` comparison being correct across DSOs, which every
70// honest platform provides: on ELF/dlopen targets comparison is name-string
71// based (the default), and targets that instead compare type_info by address
72// do so only where the ABI guarantees a single merged type_info per type. The
73// only way to break it is to force pointer-only type_info comparison (e.g.
74// `-D__GXX_MERGED_TYPEINFO_NAMES=1`) on a platform that does not actually merge
75// names -- a deliberate misconfiguration that also breaks any other
76// `type_index`/RTTI-based code, so IOv2 does not police it here.
77#if defined(IOV2_SHARED)
78using facet_id_t = std::type_index;
79#else
80using facet_id_t = std::size_t;
81
82// Per-type storage whose *address* is the (header-only) type key. Only needed
83// when the key is an address; in shared mode the key comes from `typeid` instead.
84template <typename T>
85struct type_id
86{
87 inline static const void* s_id = nullptr;
88};
89
90// The address key is std::bit_cast<size_t>(pointer); bit_cast is well-formed only
91// when sizeof(To) == sizeof(From), so the pointer and size_t must be equal in size.
92static_assert(sizeof(void*) == sizeof(std::size_t),
93 "facet_id_t address keying uses std::bit_cast<size_t>(pointer), which "
94 "is well-formed only when sizeof(void*) == sizeof(size_t)");
95#endif

109template <typename T>
110inline facet_id_t type_id_v() noexcept
111{
112#if defined(IOV2_SHARED)
113 return facet_id_t{typeid(T)};
114#else
115 return std::bit_cast<std::size_t>(&type_id<T>::s_id);
116#endif
117}
