291 std::pair<device_type, std::exception_ptr> detach() noexcept
292 {
293 if constexpr (IsIn)
294 {
295 if (!m_read_buf.empty())
296 {
297 try
298 {
299 const size_t pos = tell();
300 m_cvt.seek(pos);
301 }
302 catch (...)
303 {
304 // Swallowed on purpose: a device that doesn't support positioning
305 // (e.g. a pipe/tty-backed stdin) can never honor this reposition,
306 // and that is an inherent property of the device, not an
307 // actionable failure - the caller could not have repositioned it
308 // either after getting it back. Reporting this error would make
309 // routine detach()/attach() cycles on such devices start throwing
310 // (e.g. sync_with_stdio() right after a formatted read has left
311 // one buffered lookahead character), even though nothing is
312 // actually broken; losing that lookahead character is the
313 // accepted, unavoidable cost for non-positionable devices.
314 }
315 m_read_buf.clear();
316 }
317 }
