342 size_t sgetn(char_type* s, size_t n, size_t* got = nullptr) requires (IsIn)
343 {
344 if (got) *got = 0;
345 if ((s == nullptr) || (n == 0)) return 0;
346
347 if constexpr (IsOut)
348 switch_to_get();
349
350 size_t res = 0;
351 while (!m_read_buf.empty())
352 {
353 *s++ = m_read_buf.front();
354 m_read_buf.pop_front();
355 if (++res == n) break;
356 }
357 if (got) *got = res;
358
359 while (res < n)
360 {
361 size_t c = m_cvt.get(s, n - res);
362 assert(c <= n - res);
363 if (c == 0) break;
364 res += c;
365 s += c;
366 if (got) *got = res;
367 }
368 return res;
369 }

760 std::pair<device_type, std::exception_ptr> detach() noexcept
761 {
762 if constexpr (IsIn)
763 {
764 if (!m_read_buf.empty())
765 {
766 try
767 {
768 // On over-pushback (more put-back than actually read), tell()
769 // saturates at 0, so the device is handed back positioned at the
770 // stream origin. That is consistent with the logical read cursor
771 // being at 0; see tell()/sputbackc() and switch_to_put().
772 const size_t pos = tell();
773 m_cvt.seek(pos);
774 }
775 catch (...) // NOLINT(bugprone-empty-catch)
776 {
777 // Swallowed on purpose: a device that doesn't support positioning
778 // (e.g. a pipe/tty-backed stdin) can never honor this reposition,
779 // and that is an inherent property of the device, not an
780 // actionable failure - the caller could not have repositioned it
781 // either after getting it back. Reporting this error would make
782 // routine detach()/attach() cycles on such devices start throwing
783 // (e.g. sync_with_stdio() right after a formatted read has left
784 // one buffered lookahead character), even though nothing is
785 // actually broken; losing that lookahead character is the
786 // accepted, unavoidable cost for non-positionable devices.
787 }
788 m_read_buf.clear();
789 }
790 }
791 return m_cvt.detach();
792 }
