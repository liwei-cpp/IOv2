186 static pattern s_construct_pattern(pattern_spec spec)
187 {
188 using enum part;
189 const auto [precedes, sp, posn] = spec;
190 pattern ret;
191
192 // This insanely complicated routine attempts to construct a valid
193 // pattern for use with moneypunct. A couple of invariants:
194
195 // if (precedes) symbol -> value
196 // else value -> symbol
197
198 // if (sp) space
199 // else none
200
201 // none == never first
202 // space never first or last
203 switch (posn)
204 {
205 case 0:
206 case 1:
207 // 1 The sign precedes the value and symbol.
208 ret[0] = sign;
209 if (sp)
210 {
211 // Pattern starts with sign.
212 if (precedes)
213 {
214 ret[1] = symbol;
215 ret[3] = value;
216 }
217 else
218 {
219 ret[1] = value;
220 ret[3] = symbol;
221 }
222 ret[2] = space;
223 }
224 else
225 {
226 // Pattern starts with sign and ends with none.
227 if (precedes)
228 {
229 ret[1] = symbol;
230 ret[2] = value;
231 }
232 else
233 {
234 ret[1] = value;
235 ret[2] = symbol;
236 }
237 ret[3] = none;
238 }
239 break;
240 case 2:
241 // 2 The sign follows the value and symbol.
242 if (sp)
243 {
244 // Pattern either ends with sign.
245 if (precedes)
246 {
247 ret[0] = symbol;
248 ret[2] = value;
249 }
250 else
251 {
252 ret[0] = value;
253 ret[2] = symbol;
254 }
255 ret[1] = space;
256 ret[3] = sign;
257 }
258 else
259 {
260 // Pattern ends with sign then none.
261 if (precedes)
262 {
263 ret[0] = symbol;
264 ret[1] = value;
265 }
266 else
267 {
268 ret[0] = value;
269 ret[1] = symbol;
270 }
271 ret[2] = sign;
272 ret[3] = none;
273 }
274 break;
275 case 3:
276 // 3 The sign immediately precedes the symbol.
277 if (precedes)
278 {
279 ret[0] = sign;
280 ret[1] = symbol;
281 if (sp)
282 {
283 ret[2] = space;
284 ret[3] = value;
285 }
286 else
287 {
288 ret[2] = value;
289 ret[3] = none;
290 }
291 }
292 else
293 {
294 ret[0] = value;
295 if (sp)
296 {
297 ret[1] = space;
298 ret[2] = sign;
299 ret[3] = symbol;
300 }
301 else
302 {
303 ret[1] = sign;
304 ret[2] = symbol;
305 ret[3] = none;
306 }
307 }
308 break;
309 case 4:
310 // 4 The sign immediately follows the symbol.
311 if (precedes)
312 {
313 ret[0] = symbol;
314 ret[1] = sign;
315 if (sp)
316 {
317 ret[2] = space;
318 ret[3] = value;
319 }
320 else
321 {
322 ret[2] = value;
323 ret[3] = none;
324 }
325 }
326 else
327 {
328 ret[0] = value;
329 if (sp)
330 {
331 ret[1] = space;
332 ret[2] = symbol;
333 ret[3] = sign;
334 }
335 else
336 {
337 ret[1] = symbol;
338 ret[2] = sign;
339 ret[3] = none;
340 }
341 }
342 break;
343 default:
344 ret = s_default_pattern;
345 }
346 return ret;
