93 // Snapshot all locale-dependent strings while the locale guard is
94 // active. The lconv* and nl_langinfo() pointers may be invalidated
95 // by any subsequent setlocale()/uselocale() call (e.g. inside
96 // FacetHelper::string_to_char_convert), so copy first
97 // and convert after.
98 std::string dp_raw, ts_raw, grp_raw, yes_raw, no_raw;
99 bool yes_set = false, no_set = false;
100 {
101 clocale_wrapper inter_locale(name.c_str());
102 clocale_user guard(inter_locale);
103 const lconv* lc = localeconv();
104 if (lc->decimal_point) dp_raw = lc->decimal_point;
105 if (lc->thousands_sep) ts_raw = lc->thousands_sep;
106 if (lc->grouping) grp_raw = lc->grouping;
107
108 // Treat empty YESSTR/NOSTR the same as a missing key: an empty
109 // m_true_name / m_false_name would make numeric::get(bool&) fail
110 // unconditionally (zero-length prefix never matches) and let
111 // numeric::put(bool) write nothing but padding.
112 if (auto* p = nl_langinfo(YESSTR); p && *p) { yes_raw = p; yes_set = true; }
113 if (auto* p = nl_langinfo(NOSTR); p && *p) { no_raw = p; no_set = true; }
114 }
115
116 m_decimal_point = FacetHelper::string_to_char_convert(dp_raw, name);
117 m_thousands_sep = FacetHelper::string_to_char_convert(ts_raw, name);
118
119 // string_to_char_convert hard-codes '\0' as its failure sentinel. A
120 // decimal point of '\0' would later be mistaken for an in-stream
121 // null byte during parsing; fall back to '.' so the contract that
122 // m_decimal_point is a real, distinguishable character is preserved.
123 if (m_decimal_point == '\0') m_decimal_point = '.';
124
125 if (m_thousands_sep != '\0' && !grp_raw.empty())
126 {
127 // Copy raw POSIX grouping bytes, then normalise into the
128 // internal convention here at the POSIX boundary.
129 // Downstream (numeric, user-derived numeric_conf) sees
130 // only the internal form.
131 m_grouping.resize(grp_raw.size());
132 for (size_t i = 0; i < grp_raw.size(); ++i)
133 m_grouping[i] = static_cast<uint8_t>(grp_raw[i]);
134 FacetHelper::adjust_grouping(m_grouping);
135 }
136
137 // If a locale (or a user-derived numeric_conf that overrides
138 // thousands_sep() / decimal_point()) ends up with the same
139 // character for both separators, downstream extract_int /
140 // extract_float in facet/numeric.h check thousands_sep BEFORE
141 // decimal_point against each input character. An equal pair
142 // makes the decimal-point branch unreachable and groups would
143 // be parsed where a decimal mark was intended. Drop grouping
144 // so parsing falls back to the no-grouping path. Mirrors the
145 // same guard in numeric_conf<char8_t>, which can hit this case
146 // after multi-byte UTF-8 separators collapse onto single-byte
147 // fallbacks.
148 if (m_thousands_sep == m_decimal_point)
149 m_grouping.clear();
150
151 if (yes_set) m_true_name = yes_raw;
152 else m_true_name = "true";
153 if (no_set) m_false_name = no_raw;
154 else m_false_name = "false";
155 }

531 "numeric_conf<char8_t> delegates to numeric_conf<char32_t>, which "
532 "requires wchar_t to be a 32-bit UTF-32 code unit. This platform "
533 "(e.g. wchar_t is 16-bit UTF-16) does not satisfy that assumption.");
534
535public:
545 using char_type = char8_t;
546
547public:
570 numeric_conf(const std::string& name)
571 : ft_basic<numeric<char8_t>>()
572 {
573 if ((name == "C") || (name == "POSIX"))
574 { // "C" locale
575 m_decimal_point = u8'.';
576 m_thousands_sep = u8',';
577 m_true_name = u8"true";
578 m_false_name = u8"false";
579 return;
580 }
581
582 numeric_conf<char32_t> numeric_temp(name);
583 {
584 const auto input = numeric_temp.decimal_point();
585 auto output = detail::to_u8string(input);
586 // Fall back to u8'.' when the UTF-8 encoding is not a single
587 // byte OR when it is a single u8'\0' byte. The latter would be
588 // indistinguishable from an in-stream null byte during parsing.
589 m_decimal_point = (output.size() != 1 || output[0] == u8'\0')
590 ? u8'.' : output[0];
591 }
592
593 bool ts_representable = false;
594 {
595 const auto input = numeric_temp.thousands_sep();
596 auto output = detail::to_u8string(input);
597 ts_representable = (output.size() == 1 && output[0] != u8'\0');
598 m_thousands_sep = ts_representable ? output[0] : u8',';
599 }
600
601 m_true_name = detail::to_u8string(numeric_temp.truename());
602 m_false_name = detail::to_u8string(numeric_temp.falsename());
603
604 m_grouping = numeric_temp.grouping();
605
606 if (!ts_representable)
607 m_grouping.clear();
608
609 // After the single-byte fallbacks above, m_thousands_sep may have
610 // collapsed onto u8',' while m_decimal_point also resolved to u8','
611 // (e.g. a locale where decimal_point is ',' and thousands_sep is a
612 // multi-byte UTF-8 codepoint such as U+202F NARROW NO-BREAK SPACE).
613 // When the two separators coincide, downstream extract_int /
614 // extract_float would compare each input byte against both fields
615 // with the same value, making grouping detection ambiguous. Drop
616 // grouping in that case so parsing falls back to the no-grouping
617 // path instead of misclassifying digits.
618 if (m_thousands_sep == m_decimal_point)
619 m_grouping.clear();
620 }

652 [[nodiscard]] virtual char8_t thousands_sep() const { return m_thousands_sep; }
653

666

679
