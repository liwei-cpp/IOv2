27inline Botan::secure_vector<uint8_t> key_gen(std::string_view key)
28{
29 // Reject empty input: hashing an empty string would silently yield the
30 // publicly known constant SHA-256("") and use it as the ChaCha20 key,
31 // providing zero confidentiality. The secure_vector-based constructor
32 // already rejects invalid key lengths; mirror that contract here.
33 if (key.empty())
34 throw cvt_error("chacha20 key generation fail: key must not be empty");
35
36 auto hash = Botan::HashFunction::create("SHA-256");
37 if (!hash)
38 throw cvt_error("chacha20 key generation fail: cannot create SHA-256 hash");
39
40 hash->update(reinterpret_cast<const uint8_t*>(key.data()), key.size()); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
41 return hash->final();
42}
43}

178 void bos_impl()
179 {
180 if (!m_cipher || m_key.empty())
181 throw cvt_error("chacha20_cvt::bos fail: cipher or key missing (moved-from object?)");
182

184 // fresh stream, so the leftover buffer from a prior main-content phase
185 // must not bleed across the boundary.

190 if (iv_len == 0)

199 if (n != iv_len)

201

204 }
205 else
206 throw cvt_error("chacha20_cvt::bos fail: input mode but kernel does not support get");
207 }

209 {
210 if constexpr (cvt_cpt::support_put<KernelType>)
211 {

213 rng.randomize(reinterpret_cast<uint8_t*>(iv_buf.data()), iv_len); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
214
215 m_cipher->set_key(m_key);
216 m_cipher->set_iv(reinterpret_cast<const uint8_t*>(iv_buf.data()), iv_len); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)

219 }

291 size_t get_main(cvt_reader<KernelType>& reader, internal_type* _to, size_t to_max)
292 requires (cvt_cpt::support_get<KernelType>)
293 {
294 if (!m_cipher)
295 throw cvt_error("chacha20_cvt::get_main fail: cipher not initialized (moved-from object?)");
296 if (to_max == 0) return 0;
297
298 constexpr size_t isize = sizeof(internal_type);
299 constexpr size_t bulk_chunk = (block_size / isize) * isize;
300 static_assert(bulk_chunk >= isize,
301 "block_size must accommodate at least one internal_type");
302
303 auto* to = reinterpret_cast<uint8_t*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
304 size_t elements_delivered = 0;
305
306 reader.reset(block_size);
307
308 while (elements_delivered < to_max)
309 {
310 if (m_leftover_len > 0)
311 {
312 // Splice path: finish the partial element carried over from a
313 // previous short read before consuming any new bulk data.
314 const size_t want = isize - m_leftover_len;
315 auto [ptr, cur_len] = reader.get_buf(want);
316 if (cur_len == 0) break;
317 std::memcpy(m_leftover.data() + m_leftover_len, ptr, cur_len);
318 m_leftover_len += cur_len;
319 if (m_leftover_len == isize)
320 {
321 try { m_cipher->cipher(m_leftover.data(), to, isize); }
322 catch (...) { BT::set_tainted(); throw; }
323 to += isize;
324 elements_delivered += 1;
325 m_leftover_len = 0;
326 }
327 }
328 else
329 {
330 // Bulk path: cipher only the aligned prefix into the user buffer;
331 // stash any trailing sub-element ciphertext for the next call.
332 const size_t elements_remaining = to_max - elements_delivered;
333 const size_t want = (elements_remaining > bulk_chunk / isize)
334 ? bulk_chunk
335 : elements_remaining * isize;
336 auto [ptr, cur_len] = reader.get_buf(want);
337 if (cur_len == 0) break;
338
339 const size_t complete_bytes = (cur_len / isize) * isize;
340 const size_t tail = cur_len - complete_bytes;
341
342 if (complete_bytes > 0)
343 {
344 try
345 {
346 m_cipher->cipher(reinterpret_cast<const uint8_t*>(ptr), to, complete_bytes); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
347 }
348 catch (...) { BT::set_tainted(); throw; }
349 to += complete_bytes;
350 elements_delivered += complete_bytes / isize;
351 }
352 if (tail > 0)
353 {
354 std::memcpy(m_leftover.data(), ptr + complete_bytes, tail);
355 m_leftover_len = tail;
356 }
357 }
358 }
359
360 // A short read with no real EOF is a recoverable state: m_leftover is
361 // preserved and the next get_main call will complete the partial
362 // element via the splice path above. Only flag taint when we could not
363 // satisfy the request, the stream is genuinely at EOF, and we still
364 // hold a partial element — that misalignment can never be resolved.
365 //
366 // Gating on `elements_delivered < to_max` is essential: when the loop
367 // exits because the request was fully satisfied, the leftover is an
368 // in-flight artifact of the final bulk read, not a failure indicator —
369 // throwing here would discard elements we just successfully decrypted.
370 if (elements_delivered < to_max && m_leftover_len > 0 && BT::is_eof())
371 {
372 BT::set_tainted();
373 throw cvt_error("chacha20_cvt::get_main fail: stream ended on a non-aligned boundary");
374 }
375
376 return elements_delivered;
377 }

379 void put_main(cvt_writer<KernelType>& writer, const internal_type* _to, size_t to_size)
380 requires (cvt_cpt::support_put<KernelType>)
381 {
382 if (!m_cipher)
383 throw cvt_error("chacha20_cvt::put_main fail: cipher not initialized (moved-from object?)");
384
385 constexpr size_t max_type_limit = std::numeric_limits<size_t>::max();
386 constexpr size_t max_chunk = max_type_limit - (max_type_limit % sizeof(internal_type));
387
388 const auto* to = reinterpret_cast<const uint8_t*>(_to); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
389
390 writer.reset(block_size);
391 while (to_size > 0)
392 {
393 size_t aim_output = (max_chunk / sizeof(internal_type) > to_size)
394 ? (to_size * sizeof(internal_type))
395 : max_chunk;
396 size_t to_i = 0;
397 while (to_i != aim_output)
398 {
399 size_t dest_size = std::min<size_t>(block_size, aim_output - to_i);
400 auto ptr = writer.put_buf(dest_size);
401 // No local try/catch here: abs_cvt::put already wraps put_main in
402 // a catch-all that sets the taint flag on any exception. If cipher
403 // ever throws, that outer guard correctly marks the cvt unusable.
404 // See the get_main doxygen for the full design rationale.
405 m_cipher->cipher(to + to_i, reinterpret_cast<uint8_t*>(ptr), dest_size); // NOLINT(cppcoreguidelines-pro-type-reinterpret-cast)
406 to_i += dest_size;
407 }
408 to += aim_output;
409 to_size -= aim_output / sizeof(internal_type);
410 }
411 // commit() is the responsibility of abs_cvt::put.
412 }
413private:
414 std::unique_ptr<Botan::StreamCipher> m_cipher;
415 Botan::secure_vector<uint8_t> m_key;
416 // Cross-call carry-over for sub-element ciphertext. Holds raw (un-ciphered)
417 // bytes from a short read that did not complete one internal_type; the next
418 // get_main call splices them with fresh bytes before feeding cipher() so the
419 // keystream stays aligned with the byte stream.
420 std::array<uint8_t, sizeof(internal_type)> m_leftover{};
421 size_t m_leftover_len{0};
422};
