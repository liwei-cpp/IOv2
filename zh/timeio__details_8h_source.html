464// This specialization delegates to timeio_conf<char32_t> to populate its data
465// (see init_from_u32 below), so it requires timeio_conf<char32_t> to be a
466// complete type when a char8_t conf is constructed. That delegation lives in a
467// member function template whose reference to timeio_conf<char32_t> is dependent
468// on a template parameter, so it is checked only when actually instantiated
469// (never eagerly, hence not IFNDR): merely naming or default-constructing
470// timeio_conf<char8_t> stays well-formed even on platforms where
471// timeio_conf<char32_t> is unavailable (wchar_t is not UTF-32); there, only the
472// construction path fails to compile, which is intended.
473// The completeness requirement is deliberately NOT encoded as a constraint on
474// this class: a non-dependent sizeof(timeio_conf<char32_t>) in the
475// requires-clause is a hard error (not a soft constraint failure) and would
476// break partial-specialization resolution for unrelated timeio_conf<T>.

481public:

483 : ft_basic<timeio<char8_t>>()

485 init_from_u32<>(name);
486 }

488private:
489 // T is defaulted/constrained to char32_t; templating it makes the
490 // timeio_conf<T> reference below dependent, so the whole body is checked
491 // only upon instantiation of this helper (i.e. only on the construction
492 // path), never eagerly.
493 template <typename T = char32_t>

514 }
