392 TIter put(TIter s, bool intl, ios_base<char_type>& io, TVal v) const
393 {
394 constexpr size_t buf_size = std::numeric_limits<TVal>::digits10 + 3;
395 std::array<char_type, buf_size> vec;
396
397 char_type* p = vec.data() + buf_size;
398 *--p = '\0';
399
400 using TU = std::make_unsigned_t<TVal>;
401 bool negative = false;
402 TU uv;
403
404 if constexpr(std::is_unsigned_v<TVal>)
405 uv = v;
406 else
407 {
408 negative = (v < 0);
409 uv = negative ? static_cast<TU>(-(v + 1)) + 1
410 : static_cast<TU>(v);
411 }
412
413 do { // NOLINT(cppcoreguidelines-avoid-do-while)
414 *--p = static_cast<char_type>('0' + (uv % 10));
415 uv /= 10;
416 } while (uv != 0);
417
418 if (negative)
419 *--p = '-';
420
421 return intl ? insert<true>(s, io, p)
422 : insert<false>(s, io, p);
423 }

567 std::basic_string<char_type> tmp;

569 {

574 tmp.push_back(s_atoms[0]);
575 else if ((ch >= '0') && (ch <= '9'))
576 tmp.push_back(s_atoms[ch - '0' + 1]);

641 template <bool isIntl, typename TIter>
642 TIter insert(TIter s, ios_base<char_type>& io, const std::basic_string<char_type>& digits) const
643 {
644 const split_info& info = isIntl ? m_int : m_nat;
645 using part = base_ft<monetary>::part;
646
647 // Capture and consume the field width up front. width() is one-shot, so
648 // it must be cleared exactly once per put; resetting it here — before any
649 // allocation or copy that could throw — guarantees no leftover width
650 // leaks into the next output operation on whichever path we leave by,
651 // including an exception thrown while formatting. The captured value is
652 // used for padding below.
653 const int width = static_cast<int>(io.width());
654 io.width(0);
655
656 // Determine if negative or positive formats are to be used, and
657 // discard leading negative_sign if it is present.
658 const char_type* beg = digits.data();
659
660 base_ft<monetary>::pattern p;
661 const std::basic_string<char_type>* sign_ptr = nullptr;
662 if (!(*beg == s_atoms[s_minus]))
663 {
664 p = info.m_pos_format;
665 sign_ptr = &(info.m_positive_sign);
666 }
667 else
668 {
669 p = info.m_neg_format;
670 sign_ptr = &(info.m_negative_sign);
671 if (digits.size())
672 ++beg;
673 }
674
675 // Look for valid numbers in input digits.
676 int len = 0;
677 for (auto i = beg; i != digits.data() + digits.size(); ++i)
678 {
679 char_type ch = *i;
680 int j = 1;
681 for (; j < 11; ++j)
682 if (ch == s_atoms[j]) break;
683
684 if (j == 11) break;
685 ++len;
686 }
687
688 if (len)
689 {
690 // Assume valid input, and attempt to format.
691 // Break down input numbers into base components, as follows:
692 // final_value = grouped units + (decimal point) + (digits)
693 std::basic_string<char_type> value;
694 value.reserve(2 * len);
695
696 // Add thousands separators to non-decimal digits, per
697 // grouping rules.
698 long paddec = len - info.m_frac_digits;
699 if (paddec > 0)
700 {
701 if (info.m_frac_digits < 0)
702 paddec = len;
703 if (!m_grouping.empty())
704 {
705 value.assign(2 * paddec, char_type());
706 char_type* vend = FacetHelper::add_grouping(&value[0], m_thousands_sep, m_grouping, beg, beg + paddec);
707 value.erase(vend - &value[0]);
708 }
709 else
710 value.assign(beg, paddec);
711 }
712
713 // Deal with decimal point, decimal digits.
714 if (info.m_frac_digits > 0)
715 {
716 value += m_decimal_point;
717 if (paddec >= 0)
718 value.append(beg + paddec, info.m_frac_digits);
719 else
720 {
721 // Have to pad zeros in the decimal position.
722 value.append(-paddec, s_atoms[s_zero]);
723 value.append(beg, len);
724 }
725 }
726
727 // Calculate length of resulting string.
728 const ios_defs::fmtflags f = io.flags() & ios_defs::adjustfield;
729 len = value.size() + sign_ptr->size();
730 len += ((io.flags() & ios_defs::showbase) ? info.m_curr_symbol.size() : 0);
731
732 std::basic_string<char_type> res;
733 res.reserve(2 * len);
734
735 const bool testipad = (f == ios_defs::internal && len < width);
736 // Fit formatted digits into the required pattern.
737 for (int i = 0; i < 4; ++i)
738 {
739 const part which = static_cast<part>(p[i]);
740 switch (which)
741 {
742 case part::symbol:
743 if (io.flags() & ios_defs::showbase)
744 res += info.m_curr_symbol;
745 break;
746 case part::sign:
747 // Sign might not exist, or be more than one
748 // character long. In that case, add in the rest
749 // below.
750 if (!sign_ptr->empty())
751 res += (*sign_ptr)[0];
752 break;
753 case part::value:
754 res += value;
755 break;
756 case part::space:
757 // At least one space is required, but if internal
758 // formatting is required, an arbitrary number of
759 // fill spaces will be necessary.
760 if (testipad)
761 res.append(width - len, io.fill());
762 else
763 res += io.fill();
764 break;
765 case part::none:
766 if (testipad)
767 res.append(width - len, io.fill());
768 break;
769 }
770 }
771
772 // Special case of multi-part sign parts.
773 if (sign_ptr->size() > 1)
774 res.append(sign_ptr->c_str() + 1, sign_ptr->size() - 1);
775
776 // Pad, if still necessary.
777 len = res.size();
778 if (width > len)
779 {
780 if (f == ios_defs::left) // After.
781 res.append(width - len, io.fill());
782 else // Before.
783 res.insert(0, width - len, io.fill());
784 len = width;
785 }
786
787 // Write resulting, fully-formatted string to output iterator.
788 s = std::copy(res.data(), res.data() + len, s);
789 }
790 return s;
791 }

861 template <bool isIntl, typename TIter, std::sentinel_for<TIter> TSent>
862 std::pair<bool, TIter> extract(TIter beg, TSent end, ios_base<char_type>& io, std::string& units) const
863 {
864 const split_info& info = isIntl ? m_int : m_nat;
865 using part = base_ft<monetary>::part;
866
867 // Deduced sign.
868 bool negative = false;
869 // Sign size.
870 int sign_size = 0;
871 // True if sign is mandatory.
872 const bool mandatory_sign = (!info.m_positive_sign.empty() && !info.m_negative_sign.empty());
873 // Vector of grouping info from thousands_sep plucked from units.
874 std::vector<uint8_t> grouping_tmp;
875 if (!m_grouping.empty())
876 grouping_tmp.reserve(32);
877 // Last position before the decimal point.
878 int last_pos = 0;
879 // Separator positions, then, possibly, fractional digits.
880 int n = 0;
881 // If input iterator is in a valid state.
882 bool testvalid = true;
883 // Flag marking when a decimal point is found.
884 bool testdecfound = false;
885
886 // The tentative returned string is stored here.
887 std::string res; res.reserve(32);
888
889 const char_type* lit_zero = s_atoms.data() + s_zero;
890 const base_ft<monetary>::pattern p = info.m_neg_format;
891
892 for (int i = 0; i < 4 && testvalid; ++i)
893 {
894 const part which = static_cast<part>(p[i]);
895 switch (which)
896 {
897 case part::symbol:
898 // According to 22.2.6.1.2, p2, symbol is required
899 // if (io.flags() & ios_base::showbase), otherwise
900 // is optional and consumed only if other characters
901 // are needed to complete the format.
902 if (io.flags() & ios_defs::showbase || sign_size > 1
903 || i == 0
904 || (i == 1 && (mandatory_sign
905 || (static_cast<part>(p[0])
906 == part::sign)
907 || (static_cast<part>(p[2])
908 == part::space)))
909 || (i == 2 && ((static_cast<part>(p[3])
910 == part::value)
911 || (mandatory_sign
912 && (static_cast<part>(p[3])
913 == part::sign)))))
914 {
915 const int len = info.m_curr_symbol.size();
916 int j = 0;
917 for (; beg != end && j < len && *beg == info.m_curr_symbol[j]; ++beg, (void)++j);
918 if (j != len && (j || io.flags() & ios_defs::showbase))
919 testvalid = false;
920 }
921 break;
922 case part::sign:
923 // Sign might not exist, or be more than one character long.
924 if (!info.m_positive_sign.empty() && beg != end && *beg == info.m_positive_sign[0])
925 {
926 sign_size = info.m_positive_sign.size();
927 ++beg;
928 }
929 else if (!info.m_negative_sign.empty() && beg != end && *beg == info.m_negative_sign[0])
930 {
931 negative = true;
932 sign_size = info.m_negative_sign.size();
933 ++beg;
934 }
935 else if (!info.m_positive_sign.empty() && info.m_negative_sign.empty())
936 // "... if no sign is detected, the result is given the sign
937 // that corresponds to the source of the empty string"
938 negative = true;
939 else if (mandatory_sign)
940 testvalid = false;
941 break;
942 case part::value:
943 // Extract digits, remove and stash away the
944 // grouping of found thousands separators.
945 for (; beg != end; ++beg)
946 {
947 const char_type c = *beg;
948 const char_type* q = std::find(lit_zero, lit_zero + 10, c);
949 if (q != lit_zero + 10)
950 {
951 res += '0' + (q - lit_zero);
952 ++n;
953 }
954 else if (c == m_decimal_point && !testdecfound)
955 {
956 if (info.m_frac_digits <= 0)
957 break;
958
959 last_pos = n;
960 n = 0;
961 testdecfound = true;
962 }
963 else if (!m_grouping.empty() && c == m_thousands_sep && !testdecfound)
964 {
965 // A separator with no preceding digits, or a group
966 // longer than the largest representable group size,
967 // can never satisfy any grouping spec: reject outright
968 // rather than truncating the count.
969 if (n == 0 || std::cmp_greater(n, std::numeric_limits<uint8_t>::max()))
970 {
971 testvalid = false;
972 break;
973 }
974 // Mark position for later analysis.
975 grouping_tmp.push_back(static_cast<uint8_t>(n));
976 n = 0;
977 }
978 else
979 break;
980 }
981 if (res.empty())
982 testvalid = false;
983 break;
984 case part::space:
985 // At least one space is required.
986 if (beg != end && (*beg == io.fill()))
987 ++beg;
988 else
989 testvalid = false;
990 [[fallthrough]];
991 case part::none:
992 // Only if not at the end of the pattern.
993 if (i != 3)
994 for (; beg != end && (*beg == io.fill()); ++beg);
995 break;
996 }
997 }
998
999 // Need to get the rest of the sign characters, if they exist.
1000 if (sign_size > 1 && testvalid)
1001 {
1002 const auto& sign_str = negative ? info.m_negative_sign
1003 : info.m_positive_sign;
1004 int i = 1;
1005 for (; beg != end && i < sign_size && *beg == sign_str[i]; ++beg, (void)++i);
1006 if (i != sign_size)
1007 testvalid = false;
1008 }
1009
1010 bool succ = true;
1011 if (testvalid)
1012 {
1013 // Strip leading zeros.
1014 if (res.size() > 1)
1015 {
1016 const auto first = res.find_first_not_of('0');
1017 const bool only_zeros = first == std::string::npos;
1018 if (first)
1019 res.erase(0, only_zeros ? res.size() - 1 : first);
1020 }
1021
1022 // 22.2.6.1.2, p4
1023 if (negative && !res.empty() && res[0] != '0')
1024 res.insert(res.begin(), '-');
1025
1026 // Test for grouping fidelity.
1027 if (!grouping_tmp.empty())
1028 {
1029 // Add the ending grouping. A final group longer than the
1030 // largest representable group size cannot satisfy any spec:
1031 // fail rather than truncating.
1032 const int last_group = testdecfound ? last_pos : n;
1033 if (std::cmp_greater(last_group, std::numeric_limits<uint8_t>::max()))
1034 succ = false;
1035 else
1036 {
1037 grouping_tmp.push_back(static_cast<uint8_t>(last_group));
1038 succ = FacetHelper::verify_grouping(m_grouping, grouping_tmp);
1039 }
1040 }
1041
1042 // Iff not enough digits were supplied after the decimal-point.
1043 if (testdecfound && n != info.m_frac_digits)
1044 testvalid = false;
1045 }
1046
1047 // Iff valid sequence is not recognized.
1048 if (!testvalid)
1049 succ = false;
1050 else
1051 units.swap(res);
1052
1053 return std::pair(succ, beg);
1054 }

1086 template <std::integral TVal>
1087 bool str_to_v(const std::string& s, TVal& value) const
1088 {
1089 if (s.empty()) return false;
1090
1091 size_t i = 0;
1092 value = 0;
1093 constexpr TVal max_value = std::numeric_limits<TVal>::max();
1094
1095 if constexpr (std::is_signed_v<TVal>)
1096 {
1097 bool negative = false;
1098 constexpr TVal min_value = std::numeric_limits<TVal>::min();
1099
1100 if (s[i] == '+' || s[i] == '-') {
1101 negative = (s[i] == '-');
1102 ++i;
1103 }
1104
1105 if (i == s.size()) return false;
1106
1107 for (; i < s.size(); ++i)
1108 {
1109 char c = s[i];
1110 if (c < '0' || c > '9') return false;
1111
1112 int digit = c - '0';
1113
1114 if (!negative)
1115 {
1116 if (value > (max_value - digit) / 10)
1117 return false;
1118 }
1119 else if (value < (min_value + digit) / 10)
1120 return false;
1121
1122 value = value * 10 + (negative ? -digit : digit);
1123 }
1124 }
1125 else
1126 {
1127 if (s[i] == '+')
1128 ++i;
1129 else if (s[i] == '-')
1130 return false;
1131
1132 if (i == s.size()) return false;
1133
1134 for (; i < s.size(); ++i)
1135 {
1136 char c = s[i];
1137 if (c < '0' || c > '9') return false;
1138
1139 TVal digit = static_cast<TVal>(c - '0');
1140 if (value > (max_value - digit) / 10)
1141 return false;
1142
1143 value = value * 10 + digit;
1144 }
1145 }
1146 return true;
1147 }
