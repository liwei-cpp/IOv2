template<typename TInt>
requires std::is_same_v<TInt, wchar_t> || (std::is_same_v<TInt, char32_t> && wchar_t_is_utf32)
class IOv2::code_cvt_creator< char, TInt >

基于区域设置（locale）的 code_cvt 工厂类（char <-> wchar_t/char32_t）。 持有一个区域设置名称，并通过 create() 方法为给定底层内核生成 code_cvt 实例。 满足 cvt_creator 概念。
