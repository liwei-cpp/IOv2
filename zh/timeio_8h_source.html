1253 template <typename OutIt>
1254 OutIt put(OutIt out, const std::tm& t, std::basic_string_view<CharT> fmt) const
1255 {
1256 using namespace std::chrono;
1257
1258 // Validate raw tm fields before constructing chrono types: year/month
1259 // truncate out-of-range values (short / unsigned char) and would silently
1260 // wrap, so range-check the integers first, then reject field combinations
1261 // that are individually in range but not a real calendar date (e.g. Feb 30).
1262 // tm_sec is intentionally limited to [0,59], NOT C's [0,60]: this facet
1263 // routes time-of-day through std::chrono::hh_mm_ss<seconds>, which cannot
1264 // represent a leap second (23:59:60 would normalise to 24:00:00 and be
1265 // mis-formatted), so a leap-second tm is rejected up front here rather
1266 // than silently corrupted on output.
1267 const int y = t.tm_year + 1900;
1268 if (t.tm_mon < 0 || t.tm_mon > 11 ||
1269 t.tm_mday < 1 || t.tm_mday > 31 ||
1270 t.tm_hour < 0 || t.tm_hour > 23 ||
1271 t.tm_min < 0 || t.tm_min > 59 ||
1272 t.tm_sec < 0 || t.tm_sec > 59 ||
1273 y < static_cast<int>(year::min()) || y > static_cast<int>(year::max()))
1274 throw stream_error("timeio put error: std::tm field out of range");
1275
1276 year_month_day ymd{ year{y}, month{static_cast<unsigned>(t.tm_mon) + 1}, day{static_cast<unsigned>(t.tm_mday)} };
1277 if (!ymd.ok())
1278 throw stream_error("timeio put error: std::tm is not a valid calendar date");
1279
1280 weekday wd{ymd};
1281
1282 seconds sec = hours{t.tm_hour} + minutes{t.tm_min} + seconds{t.tm_sec};
1283 std::chrono::hh_mm_ss hms{sec};
1284 return do_put(out, fmt, &ymd, &wd, &hms, nullptr);
1285 }

2305 {

2317 if ((a / 2) != (b / 2) && *names[a] == *names[b])
2318 throw std::runtime_error(std::string("timeio: duplicate ") + what + " name in locale data");
2319 }
